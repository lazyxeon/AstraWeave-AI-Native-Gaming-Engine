fn test_audio_engine_spatial_attenuation() {
    let mut engine = AudioEngine::new();

        "test.ogg",
        Vec3::new(10.0, 0.0, 0.0),
        max_distance: 20.0

    engine.set_listener_position(listener_pos);
    assert_approx_eq!(engine.get_source_volume(source_id), 1.0);

1//! Weather System — Biome-aware weather transitions and probabilities.
2//!
3//! Provides:
4//! - **`WeatherTransition`**: Smooth crossfade of fog/ambient multipliers and particle
5//!   density when weather changes (e.g., Rain → Snow during a biome transition).
6//! - **`BiomeWeatherMap`**: Per-biome weighted probability tables for weather selection.
7//! - **`BiomeWindProfile`**: Characteristic wind parameters per biome.
8//!
9//! # Usage
10//!
11//! ```rust,no_run
12//! # use astraweave_render::weather_system::{WeatherTransition, BiomeWeatherMap, BiomeWindProfile};
13//! # use astraweave_render::effects::WeatherKind;
14//! # use astraweave_terrain::biome::BiomeType;
15//! // Select weather from biome probability table
16//! let weather = BiomeWeatherMap::pick(BiomeType::Tundra, 0.3);
17//! assert_eq!(weather, WeatherKind::Snow);
18//!
19//! // Smooth weather transition
20//! let mut transition = WeatherTransition::new(3.0); // 3-second crossfade
21//! transition.start(WeatherKind::Rain, WeatherKind::Snow);
22//! transition.update(1.5); // half-way
23//! let (fog, ambient) = transition.current_multipliers();
24//!
25//! // Wind profile for biome
26//! let wind = BiomeWindProfile::for_biome(BiomeType::Mountain);
27//! assert!(wind.base_strength > 1.0); // mountains are windy
28//! ```

35// WeatherTransition — smooth crossfade between weather states
36// ═════════════════════════════════════════════════════════════════════════

38/// Fog/ambient multiplier pair for a weather type.
39#[derive(Debug, Clone, Copy)]
40struct WeatherMultipliers {
41    fog: f32,
42    ambient: f32,
43    particle_density: f32,
44}
45
46impl WeatherMultipliers {
47    /// Get the multiplier set for a given weather kind.
48    fn for_kind(kind: WeatherKind) -> Self {
49        match kind {
50            WeatherKind::None => Self {
51                fog: 1.0,
52                ambient: 1.0,
53                particle_density: 0.0,
54            },
55            WeatherKind::Rain => Self {
56                fog: 2.5,
57                ambient: 0.6,
58                particle_density: 1.0,
59            },
60            WeatherKind::Snow => Self {
61                fog: 1.8,
62                ambient: 0.75,
63                particle_density: 1.0,
64            },
65            WeatherKind::Sandstorm => Self {
66                fog: 4.0,
67                ambient: 0.4,
68                particle_density: 1.0,
69            },
70            WeatherKind::WindTrails => Self {
71                fog: 1.4,
72                ambient: 0.9,
73                particle_density: 0.6,
74            },
75        }
76    }
77
78    /// Linearly interpolate between two multiplier sets.
79    fn lerp(a: &Self, b: &Self, t: f32) -> Self {
80        let t = t.clamp(0.0, 1.0);
81        Self {
82            fog: a.fog + (b.fog - a.fog) * t,
83            ambient: a.ambient + (b.ambient - a.ambient) * t,
84            particle_density: a.particle_density + (b.particle_density - a.particle_density) * t,
85        }
86    }
87}

89/// Smooth crossfade between two weather states.
90///
91/// When weather changes (e.g. transitioning from a biome with Rain to one
92/// with Snow), this struct provides eased interpolation of:
93/// - Fog multiplier
94/// - Ambient multiplier
95/// - Particle density (for crossfading particle systems)
96///
97/// The particle crossfade allows both the outgoing and incoming weather
98/// particle effects to coexist briefly, preventing jarring visual pops.
99#[derive(Debug, Clone)]
100pub struct WeatherTransition {
101    /// Duration of the crossfade in seconds.
102    duration: f32,
103    /// Source weather.
104    from: WeatherKind,
105    /// Target weather.
106    to: WeatherKind,
107    /// Progress (0.0 … 1.0).
108    progress: f32,
109    /// Whether a transition is currently active.
110    active: bool,
111    /// Cached multipliers for source.
112    from_mul: WeatherMultipliers,
113    /// Cached multipliers for target.
114    to_mul: WeatherMultipliers,
115}
116
117impl WeatherTransition {
118    /// Create a new weather transition handler with a given crossfade
119    /// duration (seconds).
120    pub fn new(duration: f32) -> Self {
121        Self {
122            duration: duration.max(0.01),
123            from: WeatherKind::None,
124            to: WeatherKind::None,
125            progress: 1.0,
126            active: false,
127            from_mul: WeatherMultipliers::for_kind(WeatherKind::None),
128            to_mul: WeatherMultipliers::for_kind(WeatherKind::None),
129        }
130    }
131
132    /// Start a crossfade from one weather kind to another.
133    ///
134    /// If the same kind is requested, this is a no-op.
135    pub fn start(&mut self, from: WeatherKind, to: WeatherKind) {
136        if from == to {
137            return;
138        }
139        self.from = from;
140        self.to = to;
141        self.from_mul = WeatherMultipliers::for_kind(from);
142        self.to_mul = WeatherMultipliers::for_kind(to);
143        self.progress = 0.0;
144        self.active = true;
145    }
146
147    /// Advance the crossfade by `dt` seconds.
148    pub fn update(&mut self, dt: f32) {
149        if !self.active {
150            return;
151        }
152        let rate = 1.0 / self.duration;
153        self.progress += dt * rate;
154        if self.progress >= 1.0 {
155            self.progress = 1.0;
156            self.active = false;
157            self.from = self.to;
158            self.from_mul = self.to_mul;
159        }
160    }
161
162    /// Whether a transition is in progress.
163    pub fn is_active(&self) -> bool {
164        self.active
165    }
166
167    /// Raw linear progress (0.0 … 1.0).
168    pub fn progress(&self) -> f32 {
169        self.progress
170    }
171
172    /// Eased progress using SmoothStep for perceptual smoothness.
173    pub fn eased_progress(&self) -> f32 {
174        let t = self.progress.clamp(0.0, 1.0);
175        t * t * (3.0 - 2.0 * t) // smoothstep
176    }
177
178    /// Current interpolated fog and ambient multipliers.
179    ///
180    /// Returns `(fog_multiplier, ambient_multiplier)`.
181    pub fn current_multipliers(&self) -> (f32, f32) {
182        let t = self.eased_progress();
183        let m = WeatherMultipliers::lerp(&self.from_mul, &self.to_mul, t);
184        (m.fog, m.ambient)
185    }
186
187    /// Current interpolated particle density (0.0 … 1.0).
188    ///
189    /// Use this to scale the incoming weather particle count while
190    /// fading out the outgoing particles.
191    pub fn current_particle_density(&self) -> f32 {
192        let t = self.eased_progress();
193        WeatherMultipliers::lerp(&self.from_mul, &self.to_mul, t).particle_density
194    }
195
196    /// Outgoing weather particle fade-out factor (1.0 → 0.0).
197    ///
198    /// Multiply the outgoing particle system's density by this value
199    /// during the crossfade to smoothly retire it.
200    pub fn outgoing_particle_fade(&self) -> f32 {
201        if !self.active {
202            return 0.0;
203        }
204        1.0 - self.eased_progress()
205    }
206
207    /// Incoming weather particle fade-in factor (0.0 → 1.0).
208    ///
209    /// Multiply the incoming particle system's density by this value
210    /// during the crossfade.
211    pub fn incoming_particle_fade(&self) -> f32 {
212        if !self.active {
213            return 1.0;
214        }
215        self.eased_progress()
216    }
217
218    /// Get the source weather kind.
219    pub fn from_kind(&self) -> WeatherKind {
220        self.from
221    }
222
223    /// Get the target weather kind.
224    pub fn to_kind(&self) -> WeatherKind {
225        self.to
226    }
227
228    /// Current effective weather (the target once completed).
229    pub fn current_kind(&self) -> WeatherKind {
230        self.to
231    }
232
233    /// Set the crossfade duration.
234    pub fn set_duration(&mut self, duration: f32) {
235        self.duration = duration.max(0.01);
236    }
237
238    /// Get the crossfade duration.
239    pub fn duration(&self) -> f32 {
240        self.duration
241    }
242
243    /// Force-complete the current transition.
244    pub fn complete(&mut self) {
245        self.progress = 1.0;
246        self.active = false;
247        self.from = self.to;
248        self.from_mul = self.to_mul;
249    }
250}
251
252impl Default for WeatherTransition {
253    fn default() -> Self {
254        Self::new(3.0)
255    }
256}

432    /// Sample a weather type for a biome given a random roll in [0.0, 1.0).
433    ///
434    /// The weights are normalized internally, so they need not sum to 1.0.
435    /// If `roll` is out of range it is clamped.
436    pub fn pick(biome: BiomeType, roll: f32) -> WeatherKind {
437        let table = Self::weights(biome);
438        let total: f32 = table.iter().map(|w| w.weight).sum();
439        if total <= 0.0 {
440            return WeatherKind::None;
441        }
442        let roll = roll.clamp(0.0, 0.9999) * total;
443        let mut acc = 0.0;
444        for entry in table {
445            acc += entry.weight;
446            if roll < acc {
447                return entry.kind;
448            }
449        }
450        // Fallback (shouldn't reach here)
451        table.last().map_or(WeatherKind::None, |e| e.kind)
452    }
453
454    /// Get the probability (0.0 … 1.0) of a specific weather kind in a biome.
455    pub fn probability(biome: BiomeType, kind: WeatherKind) -> f32 {
456        let table = Self::weights(biome);
457        let total: f32 = table.iter().map(|w| w.weight).sum();
458        if total <= 0.0 {
459            return 0.0;
460        }
461        table
462            .iter()
463            .find(|w| w.kind == kind)
464            .map_or(0.0, |w| w.weight / total)
465    }
466
467    /// Get the most likely weather for a biome.
468    pub fn most_likely(biome: BiomeType) -> WeatherKind {
469        let table = Self::weights(biome);
470        table
471            .iter()
472            .max_by(|a, b| {
473                a.weight
474                    .partial_cmp(&b.weight)
475                    .unwrap_or(std::cmp::Ordering::Equal)
476            })
477            .map_or(WeatherKind::None, |w| w.kind)
478    }
479}

483// ═════════════════════════════════════════════════════════════════════════

485/// Per-biome wind characteristics fed into `WeatherFx::set_wind()`.
486#[derive(Debug, Clone, Copy)]
487pub struct BiomeWindProfile {
488    /// Base wind strength multiplier (1.0 = default).
489    pub base_strength: f32,
490    /// Whether wind direction shifts over time (gusty).
491    pub gusty: bool,
492    /// Gust frequency in Hz (how often direction shifts).
493    pub gust_frequency: f32,
494    /// Gust strength variance (added to base_strength during gusts).
495    pub gust_variance: f32,
496    /// Dominant wind direction (normalized XZ plane).
497    pub dominant_direction: Vec3,
498}
499
500impl Default for BiomeWindProfile {
501    fn default() -> Self {
502        Self {
503            base_strength: 1.0,
504            gusty: false,
505            gust_frequency: 0.0,
506            gust_variance: 0.0,
507            dominant_direction: Vec3::new(1.0, 0.0, 0.0),
508        }
509    }
510}
511
512impl BiomeWindProfile {
513    /// Get the wind profile for a specific biome.
514    pub fn for_biome(biome: BiomeType) -> Self {
515        match biome {
516            BiomeType::Forest => Self {
517                base_strength: 0.4, // Canopy blocks wind
518                gusty: true,
519                gust_frequency: 0.15, // Occasional gusts through clearings
520                gust_variance: 0.6,
521                dominant_direction: Vec3::new(0.8, 0.0, 0.6),
522            },
523            BiomeType::Desert => Self {
524                base_strength: 1.8, // Open terrain, strong winds
525                gusty: true,
526                gust_frequency: 0.3, // Frequent gusts
527                gust_variance: 1.5,  // Can spike during sandstorms
528                dominant_direction: Vec3::new(1.0, 0.0, 0.2),
529            },
530            BiomeType::Grassland => Self {
531                base_strength: 1.2, // Open terrain, moderate winds
532                gusty: true,
533                gust_frequency: 0.2,
534                gust_variance: 0.8,
535                dominant_direction: Vec3::new(0.9, 0.0, 0.4),
536            },
537            BiomeType::Mountain => Self {
538                base_strength: 2.5, // Strong altitude winds
539                gusty: true,
540                gust_frequency: 0.4,                          // Very gusty
541                gust_variance: 2.0,                           // Extreme gusts possible
542                dominant_direction: Vec3::new(0.6, 0.0, 0.8), // Cross-slope
543            },
544            BiomeType::Tundra => Self {
545                base_strength: 2.0, // Arctic winds
546                gusty: true,
547                gust_frequency: 0.25,
548                gust_variance: 1.2,
549                dominant_direction: Vec3::new(0.0, 0.0, 1.0), // From the pole
550            },
551            BiomeType::Swamp => Self {
552                base_strength: 0.3, // Stagnant air
553                gusty: false,
554                gust_frequency: 0.05,
555                gust_variance: 0.2,
556                dominant_direction: Vec3::new(0.5, 0.0, 0.5),
557            },
558            BiomeType::Beach => Self {
559                base_strength: 1.5, // Sea breeze
560                gusty: true,
561                gust_frequency: 0.2,
562                gust_variance: 0.9,
563                dominant_direction: Vec3::new(-1.0, 0.0, 0.0), // Onshore
564            },
565            BiomeType::River => Self {
566                base_strength: 0.8, // Follows the valley
567                gusty: false,
568                gust_frequency: 0.1,
569                gust_variance: 0.4,
570                dominant_direction: Vec3::new(0.7, 0.0, 0.7),
571            },
572            // Fallback
573            _ => Self::default(),
574        }
575    }
576
577    /// Compute the effective wind strength at a given time, accounting for gusts.
578    ///
579    /// `time` is total elapsed seconds; the gust pattern is periodic.
580    pub fn effective_strength(&self, time: f32) -> f32 {
581        if !self.gusty || self.gust_frequency <= 0.0 {
582            return self.base_strength;
583        }
584        // Sine-based gust modulation
585        let phase = time * self.gust_frequency * std::f32::consts::TAU;
586        let gust = phase.sin().max(0.0); // Only positive gusts
587        self.base_strength + gust * self.gust_variance
588    }
589
590    /// Compute the effective wind direction at a given time.
591    ///
592    /// When gusty, the direction oscillates ±30° around the dominant.
593    pub fn effective_direction(&self, time: f32) -> Vec3 {
594        if !self.gusty || self.gust_frequency <= 0.0 {
595            let len = self.dominant_direction.length();
596            return if len > 0.001 {
597                self.dominant_direction / len
598            } else {
599                Vec3::X
600            };
601        }
602        let angle_offset = (time * self.gust_frequency * std::f32::consts::TAU * 0.7).sin() * 0.52; // ±30° in radians
603        let cos_a = angle_offset.cos();
604        let sin_a = angle_offset.sin();
605        let d = self.dominant_direction;
606        // Rotate in XZ plane
607        let rotated = Vec3::new(d.x * cos_a - d.z * sin_a, 0.0, d.x * sin_a + d.z * cos_a);
608        let len = rotated.length();
609        if len > 0.001 {
610            rotated / len
611        } else {
612            Vec3::X
613        }
614    }
615}

8/// Decal instance on GPU
9#[repr(C)]
10#[derive(Copy, Clone, Debug, bytemuck::Pod, bytemuck::Zeroable)]
11pub struct GpuDecal {
12    /// Inverse projection matrix (world to decal space)
13    pub inv_projection: [[f32; 4]; 4],
14    /// Albedo color tint
15    pub albedo_tint: [f32; 4],
16    /// Normal strength and roughness/metallic override
17    pub params: [f32; 4], // x=normal_strength, y=roughness, z=metallic, w=blend_mode
18    /// Atlas UV offset and scale
19    pub atlas_uv: [f32; 4], // xy=offset, zw=scale
20}
21
22/// Decal blend mode
23#[derive(Debug, Clone, Copy, PartialEq, Eq)]
24pub enum DecalBlendMode {
25    /// Multiply albedo, blend normals
26    Multiply = 0,
27    /// Additive blend
28    Additive = 1,
29    /// Alpha blend
30    AlphaBlend = 2,
31    /// Stain (darken only)
32    Stain = 3,
33}
34
35/// Decal definition (CPU-side)
36#[derive(Debug, Clone)]
37pub struct Decal {
38    /// Position in world space
39    pub position: Vec3,
40    /// Rotation quaternion
41    pub rotation: Quat,
42    /// Scale (half-extents of decal box)
43    pub scale: Vec3,
44    /// Albedo tint color
45    pub albedo_tint: [f32; 4],
46    /// Normal strength (0-1)
47    pub normal_strength: f32,
48    /// Roughness override
49    pub roughness: f32,
50    /// Metallic override
51    pub metallic: f32,
52    /// Blend mode
53    pub blend_mode: DecalBlendMode,
54    /// Atlas UV coordinates (offset and scale)
55    pub atlas_uv: ([f32; 2], [f32; 2]),
56    /// Fade-out timer (0 = permanent, >0 = fade duration)
57    pub fade_duration: f32,
58    /// Current fade time
59    pub fade_time: f32,
60}
61
62impl Decal {
63    /// Create a new decal
64    pub fn new(
65        position: Vec3,
66        rotation: Quat,
67        scale: Vec3,
68        atlas_uv: ([f32; 2], [f32; 2]),
69    ) -> Self {
70        Self {
71            position,
72            rotation,
73            scale,
74            albedo_tint: [1.0, 1.0, 1.0, 1.0],
75            normal_strength: 1.0,
76            roughness: 0.5,
77            metallic: 0.0,
78            blend_mode: DecalBlendMode::AlphaBlend,
79            atlas_uv,
80            fade_duration: 0.0,
81            fade_time: 0.0,
82        }
83    }
84
85    /// Update fade-out
86    pub fn update(&mut self, dt: f32) -> bool {
87        if self.fade_duration > 0.0 {
88            self.fade_time += dt;
89            if self.fade_time >= self.fade_duration {
90                return false; // Decal should be removed
91            }
92            // Update alpha based on fade
93            let fade_alpha = 1.0 - (self.fade_time / self.fade_duration);
94            self.albedo_tint[3] = fade_alpha;
95        }
96        true
97    }
98
99    /// Convert to GPU representation
100    pub fn to_gpu(&self) -> GpuDecal {
101        // Build projection matrix (world to decal space)
102        let transform =
103            Mat4::from_scale_rotation_translation(self.scale, self.rotation, self.position);
104        let inv_projection = transform.inverse();
105
106        GpuDecal {
107            inv_projection: inv_projection.to_cols_array_2d(),
108            albedo_tint: self.albedo_tint,
109            params: [
110                self.normal_strength,
111                self.roughness,
112                self.metallic,
113                self.blend_mode as u32 as f32,
114            ],
115            atlas_uv: [
116                self.atlas_uv.0[0],
117                self.atlas_uv.0[1],
118                self.atlas_uv.1[0],
119                self.atlas_uv.1[1],
120            ],
121        }
122    }
123}

125/// Decal atlas manager
126pub struct DecalAtlas {
127    /// Atlas texture
128    pub texture: wgpu::Texture,
129    /// Atlas texture view
130    pub view: wgpu::TextureView,
131    /// Atlas sampler
132    pub sampler: wgpu::Sampler,
133    /// Atlas size
134    pub size: u32,
135    /// Grid size (decals per row/column)
136    pub grid_size: u32,
137}
138
139impl DecalAtlas {
140    /// Create a new decal atlas
141    pub fn new(device: &wgpu::Device, size: u32, grid_size: u32) -> Self {
142        let texture = device.create_texture(&wgpu::TextureDescriptor {
143            label: Some("Decal Atlas"),
144            size: wgpu::Extent3d {
145                width: size,
146                height: size,
147                depth_or_array_layers: 1,
148            },
149            mip_level_count: 1,
150            sample_count: 1,
151            dimension: wgpu::TextureDimension::D2,
152            format: wgpu::TextureFormat::Rgba8UnormSrgb,
153            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
154            view_formats: &[],
155        });
156
157        let view = texture.create_view(&wgpu::TextureViewDescriptor::default());
158
159        let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
160            label: Some("Decal Atlas Sampler"),
161            address_mode_u: wgpu::AddressMode::ClampToEdge,
162            address_mode_v: wgpu::AddressMode::ClampToEdge,
163            address_mode_w: wgpu::AddressMode::ClampToEdge,
164            mag_filter: wgpu::FilterMode::Linear,
165            min_filter: wgpu::FilterMode::Linear,
166            mipmap_filter: wgpu::FilterMode::Linear,
167            ..Default::default()
168        });
169
170        Self {
171            texture,
172            view,
173            sampler,
174            size,
175            grid_size,
176        }
177    }
178
179    /// Get UV coordinates for a decal in the atlas
180    pub fn get_uv(&self, atlas_x: u32, atlas_y: u32) -> ([f32; 2], [f32; 2]) {
181        let cell_size = 1.0 / self.grid_size as f32;
182        let offset = [atlas_x as f32 * cell_size, atlas_y as f32 * cell_size];
183        let scale = [cell_size, cell_size];
184        (offset, scale)
185    }
186}

188/// Decal system manager
189pub struct DecalSystem {
190    /// Active decals
191    decals: Vec<Decal>,
192    /// GPU buffer for decal instances
193    decal_buffer: wgpu::Buffer,
194    /// Decal atlas
195    atlas: DecalAtlas,
196    /// Maximum decals
197    max_decals: usize,
198}
199
200impl DecalSystem {
201    /// Create a new decal system
202    pub fn new(device: &wgpu::Device, max_decals: usize, atlas_size: u32, atlas_grid: u32) -> Self {
203        let decal_buffer = device.create_buffer(&wgpu::BufferDescriptor {
204            label: Some("Decal Buffer"),
205            size: (max_decals * std::mem::size_of::<GpuDecal>()) as u64,
206            usage: wgpu::BufferUsages::STORAGE | wgpu::BufferUsages::COPY_DST,
207            mapped_at_creation: false,
208        });
209
210        let atlas = DecalAtlas::new(device, atlas_size, atlas_grid);
211
212        Self {
213            decals: Vec::with_capacity(max_decals),
214            decal_buffer,
215            atlas,
216            max_decals,
217        }
218    }
219
220    /// Add a decal
221    pub fn add_decal(&mut self, decal: Decal) -> Result<()> {
222        anyhow::ensure!(
223            self.decals.len() < self.max_decals,
224            "Maximum decal count reached"
225        );
226        self.decals.push(decal);
227        Ok(())
228    }
229
230    /// Update all decals
231    pub fn update(&mut self, queue: &wgpu::Queue, dt: f32) {
232        // Update and remove faded decals
233        self.decals.retain_mut(|decal| decal.update(dt));
234
235        // Upload to GPU
236        let gpu_decals: Vec<GpuDecal> = self.decals.iter().map(|d| d.to_gpu()).collect();
237        queue.write_buffer(&self.decal_buffer, 0, bytemuck::cast_slice(&gpu_decals));
238    }
239
240    /// Get decal count
241    pub fn count(&self) -> usize {
242        self.decals.len()
243    }
244
245    /// Get decal buffer
246    pub fn buffer(&self) -> &wgpu::Buffer {
247        &self.decal_buffer
248    }
249
250    /// Get atlas
251    pub fn atlas(&self) -> &DecalAtlas {
252        &self.atlas
253    }
254}

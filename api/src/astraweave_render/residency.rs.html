6/// Tracks residency of assets in GPU memory for streaming.
7/// Evicts least-recently-used assets when memory pressure is high.
8pub struct ResidencyManager {
9    db: Arc<Mutex<AssetDatabase>>,
10    loaded_assets: HashMap<String, ResidencyInfo>, // GUID -> info
11    lru_queue: VecDeque<String>,                   // GUIDs in LRU order
12    max_memory_mb: usize,
13    current_memory_mb: usize,
14    hot_reload_rx: watch::Receiver<()>,
15}
16
17#[derive(Debug, Clone)]
18pub struct ResidencyInfo {
19    pub kind: AssetKind,
20    pub memory_mb: usize,
21    pub last_used: std::time::Instant,
22    pub gpu_handle: Option<String>, // Placeholder for actual GPU handle
23}

25impl ResidencyManager {
26    pub fn new(db: Arc<Mutex<AssetDatabase>>, max_memory_mb: usize) -> Self {
27        let (_tx, rx) = watch::channel(());
28        Self {
29            db,
30            loaded_assets: HashMap::new(),
31            lru_queue: VecDeque::new(),
32            max_memory_mb,
33            current_memory_mb: 0,
34            hot_reload_rx: rx,
35        }
36    }
37
38    pub fn with_hot_reload(

52
53    /// Load an asset into residency if not already loaded.
54    pub fn load_asset(&mut self, guid: &str) -> Result<(), anyhow::Error> {
55        if self.loaded_assets.contains_key(guid) {
56            // Already loaded, update LRU
57            self.touch_asset(guid);
58            return Ok(());
59        }
60
61        // Get asset metadata
62        let (meta, memory_mb) = {
63            let db = self
64                .db
65                .lock()
66                .expect("Database mutex poisoned - cannot recover");
67            if let Some(meta) = db.get_asset(guid) {
68                let memory_mb = (meta.size_bytes / (1024 * 1024)) as usize + 1;
69                (meta.clone(), memory_mb)
70            } else {
71                return Err(anyhow::anyhow!("Asset {} not found in database", guid));
72            }
73        }; // db lock dropped here
74
75        // Evict if necessary
76        while self.current_memory_mb + memory_mb > self.max_memory_mb {
77            self.evict_lru()?;
78        }
79
80        // Load asset (placeholder: in real impl, upload to GPU)
81        let info = ResidencyInfo {
82            kind: meta.kind,
83            memory_mb,
84            last_used: std::time::Instant::now(),
85            gpu_handle: Some(format!("gpu_{}", guid)),
86        };
87
88        self.loaded_assets.insert(guid.to_string(), info);
89        self.lru_queue.push_back(guid.to_string());
90        self.current_memory_mb += memory_mb;
91
92        Ok(())
93    }
94
95    /// Mark asset as recently used.
96    pub fn touch_asset(&mut self, guid: &str) {
97        if let Some(info) = self.loaded_assets.get_mut(guid) {
98            info.last_used = std::time::Instant::now();
99            // Move to back of LRU queue
100            if let Some(pos) = self.lru_queue.iter().position(|g| g == guid) {
101                self.lru_queue.remove(pos);
102                self.lru_queue.push_back(guid.to_string());
103            }
104        }
105    }
106
107    /// Evict the least recently used asset.
108    pub fn evict_lru(&mut self) -> Result<(), anyhow::Error> {
109        if let Some(guid) = self.lru_queue.pop_front() {
110            if let Some(info) = self.loaded_assets.remove(&guid) {
111                self.current_memory_mb = self.current_memory_mb.saturating_sub(info.memory_mb);
112                // Placeholder: unload from GPU
113                println!("Evicted asset {}", guid);
114            }
115        }
116        Ok(())
117    }
118
119    /// Get loaded assets.
120    pub fn get_loaded_assets(&self) -> Vec<String> {
121        self.loaded_assets.keys().cloned().collect()
122    }
123
124    /// Check for hot-reload notifications and invalidate affected assets.
125    pub fn check_hot_reload(&mut self) {
126        if self.hot_reload_rx.has_changed().unwrap_or(false) {
127            // Mark hot-reload notification as seen
128            let _ = self.hot_reload_rx.borrow_and_update();
129            // Clear all loaded assets on hot-reload signal
130            self.loaded_assets.clear();
131            self.lru_queue.clear();
132            self.current_memory_mb = 0;
133        }
134    }
135}

138mod tests {
139    use super::*;
140    use astraweave_asset::AssetMetadata;
141
142    #[test]
143    fn test_residency_load_and_evict() {
144        let db = Arc::new(Mutex::new(AssetDatabase::new()));
145        let mut rm = ResidencyManager::new(db.clone(), 10); // 10 MB limit
146
147        // Mock asset
148        let guid = "test_guid".to_string();
149        {
150            let mut db = db.lock().unwrap();
151            db.assets.insert(
152                guid.clone(),
153                AssetMetadata {
154                    guid: guid.clone(),
155                    path: "test".to_string(),
156                    kind: AssetKind::Texture,
157                    hash: "hash".to_string(),
158                    dependencies: vec![],
159                    last_modified: 0,
160                    size_bytes: 5 * 1024 * 1024, // 5 MB
161                },
162            );
163        }
164
165        // Load asset
166        rm.load_asset(&guid).unwrap();
167        assert!(rm.loaded_assets.contains_key(&guid));
168        assert_eq!(rm.current_memory_mb, 6); // 5MB rounds up to 6MB (size/MB + 1)
169
170        // Touch
171        rm.touch_asset(&guid);
172
173        // Load another to trigger eviction
174        let guid2 = "test_guid2".to_string();
175        {
176            let mut db = db.lock().unwrap();
177            db.assets.insert(
178                guid2.clone(),
179                AssetMetadata {
180                    guid: guid2.clone(),
181                    path: "test2".to_string(),
182                    kind: AssetKind::Texture,
183                    hash: "hash2".to_string(),
184                    dependencies: vec![],
185                    last_modified: 0,
186                    size_bytes: 6 * 1024 * 1024, // 6 MB
187                },
188            );
189        }
190
191        rm.load_asset(&guid2).unwrap();
192        // Should have evicted guid
193        assert!(!rm.loaded_assets.contains_key(&guid));
194        assert!(rm.loaded_assets.contains_key(&guid2));
195        assert_eq!(rm.current_memory_mb, 7); // 6MB rounds up to 7MB (size/MB + 1)
196    }

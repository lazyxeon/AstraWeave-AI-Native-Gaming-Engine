23fn cluster_index(ix: u32, iy: u32, iz: u32, dims: ClusterDims) -> usize {

27/// Compute per-cluster light lists on CPU using simple sphere-screen bounds.
28/// Inputs:
29/// - lights: positions are in camera/view space with +Z forward and in meters
30/// - screen: (width,height)
31/// - near/far: camera near/far planes (positive distances)
32/// - fov_y: vertical field of view in radians
33///   Returns: (counts, indices, offsets) where offsets is exclusive scan over counts
34pub fn bin_lights_cpu(
35    lights: &[CpuLight],
36    dims: ClusterDims,
37    screen: (u32, u32),
38    near: f32,
39    far: f32,
40    fov_y: f32,
41) -> (Vec<u32>, Vec<u32>, Vec<u32>) {
42    let (width, height) = screen;
43    let aspect = width.max(1) as f32 / height.max(1) as f32;
44    let fy = 1.0 / (0.5 * fov_y).tan();
45    let fx = fy / aspect;
46    let tile_w = (width as f32 / dims.x as f32).max(1.0);
47    let tile_h = (height as f32 / dims.y as f32).max(1.0);
48
49    let clusters = (dims.x * dims.y * dims.z) as usize;
50    let mut counts = vec![0u32; clusters];
51
52    // First pass: counts
53    for l in lights.iter() {
54        let z = l.pos.z.max(near + 1e-4);
55        if z - l.radius > far {
56            continue;
57        }
58        // Project center to NDC
59        let ndc_x = (l.pos.x / z) * fx;
60        let ndc_y = (l.pos.y / z) * fy;
61        // Convert to pixel space
62        let px = (ndc_x * 0.5 + 0.5) * width as f32;
63        let py = (ndc_y * 0.5 + 0.5) * height as f32;
64        // Screen-space radius in pixels (approx)
65        let rpx_x = (l.radius / z) * fx * (width as f32 * 0.5);
66        let rpx_y = (l.radius / z) * fy * (height as f32 * 0.5);
67        let min_px = (px - rpx_x).floor() as i32;
68        let max_px = (px + rpx_x).ceil() as i32;
69        let min_py = (py - rpx_y).floor() as i32;
70        let max_py = (py + rpx_y).ceil() as i32;
71        let ix0 = clamp_u32(min_px / tile_w as i32, 0, dims.x as i32 - 1);
72        let ix1 = clamp_u32(max_px / tile_w as i32, 0, dims.x as i32 - 1);
73        let iy0 = clamp_u32(min_py / tile_h as i32, 0, dims.y as i32 - 1);
74        let iy1 = clamp_u32(max_py / tile_h as i32, 0, dims.y as i32 - 1);
75        // Z slices
76        let zmin = (z - l.radius).max(near);
77        let zmax = (z + l.radius).min(far);
78        if zmin >= zmax {
79            continue;
80        }
81        let iz0 = (((zmin - near) / (far - near)) * dims.z as f32)
82            .floor()
83            .clamp(0.0, dims.z as f32 - 1.0) as u32;
84        let iz1 = (((zmax - near) / (far - near)) * dims.z as f32)
85            .floor()
86            .clamp(0.0, dims.z as f32 - 1.0) as u32;
87        for iz in iz0..=iz1 {
88            for iy in iy0..=iy1 {
89                for ix in ix0..=ix1 {
90                    let ci = cluster_index(ix, iy, iz, dims);
91                    counts[ci] += 1;
92                }
93            }
94        }
95    }
96
97    // Exclusive scan to offsets
98    let mut offsets = vec![0u32; clusters + 1];
99    for i in 0..clusters {
100        offsets[i + 1] = offsets[i] + counts[i];
101    }
102
103    // Second pass: fill indices using running write cursors
104    let total = offsets[clusters] as usize;
105    let mut indices = vec![0u32; total];
106    let mut cursors = vec![0u32; clusters];
107    for (li, l) in lights.iter().enumerate() {
108        let z = l.pos.z.max(near + 1e-4);
109        if z - l.radius > far {
110            continue;
111        }
112        let ndc_x = (l.pos.x / z) * fx;
113        let ndc_y = (l.pos.y / z) * fy;
114        let px = (ndc_x * 0.5 + 0.5) * width as f32;
115        let py = (ndc_y * 0.5 + 0.5) * height as f32;
116        let rpx_x = (l.radius / z) * fx * (width as f32 * 0.5);
117        let rpx_y = (l.radius / z) * fy * (height as f32 * 0.5);
118        let min_px = (px - rpx_x).floor() as i32;
119        let max_px = (px + rpx_x).ceil() as i32;
120        let min_py = (py - rpx_y).floor() as i32;
121        let max_py = (py + rpx_y).ceil() as i32;
122        let ix0 = clamp_u32(min_px / tile_w as i32, 0, dims.x as i32 - 1);
123        let ix1 = clamp_u32(max_px / tile_w as i32, 0, dims.x as i32 - 1);
124        let iy0 = clamp_u32(min_py / tile_h as i32, 0, dims.y as i32 - 1);
125        let iy1 = clamp_u32(max_py / tile_h as i32, 0, dims.y as i32 - 1);
126        let zmin = (z - l.radius).max(near);
127        let zmax = (z + l.radius).min(far);
128        if zmin >= zmax {
129            continue;
130        }
131        let iz0 = (((zmin - near) / (far - near)) * dims.z as f32)
132            .floor()
133            .clamp(0.0, dims.z as f32 - 1.0) as u32;
134        let iz1 = (((zmax - near) / (far - near)) * dims.z as f32)
135            .floor()
136            .clamp(0.0, dims.z as f32 - 1.0) as u32;
137        for iz in iz0..=iz1 {
138            for iy in iy0..=iy1 {
139                for ix in ix0..=ix1 {
140                    let ci = cluster_index(ix, iy, iz, dims);
141                    let off = offsets[ci] + cursors[ci];
142                    indices[off as usize] = li as u32;
143                    cursors[ci] += 1;
144                }
145            }
146        }
147    }
148
149    (counts, indices, offsets)
150}

152pub const WGSL_CLUSTER_BIN: &str = r#"
153struct Light { pos_radius: vec4<f32> };
154struct Params { screen: vec4<u32>, clusters: vec4<u32>, val: vec4<f32> };
155@group(0) @binding(0) var<storage, read> lights: array<Light>;
156@group(0) @binding(1) var<uniform> params: Params;
157// Offsets: length = clusters+1 (exclusive scan). Counts: length = clusters. Indices: variable length
158@group(0) @binding(2) var<storage, read> offsets: array<u32>;
159@group(0) @binding(3) var<storage, read_write> counts: array<atomic<u32>>;
160@group(0) @binding(4) var<storage, read_write> indices: array<u32>;
161
162fn cluster_index(ix: u32, iy: u32, iz: u32, dims: vec4<u32>) -> u32 {
163    return ix + iy * dims.x + iz * (dims.x * dims.y);
164}
165
166@compute @workgroup_size(1,1,1)
167fn cs_main(@builtin(global_invocation_id) gid: vec3<u32>) {
168    // One invocation per light
169    let li = gid.x;
170    if (li >= arrayLength(&lights)) { return; }
171    let L = lights[li].pos_radius;
172    let pos = L.xyz;
173    let radius = L.w;
174    let width = f32(params.screen.x);
175    let height = f32(params.screen.y);
176    let near = params.val.x;
177    let far = params.val.y;
178    let fov_y = params.val.z;
179    let aspect = width / max(height, 1.0);
180    let fy = 1.0 / tan(0.5 * fov_y);
181    let fx = fy / aspect;
182    let z = max(pos.z, near + 1e-4);
183    if (z - radius > far) { return; }
184    let ndc_x = (pos.x / z) * fx;
185    let ndc_y = (pos.y / z) * fy;
186    let px = (ndc_x * 0.5 + 0.5) * width;
187    let py = (ndc_y * 0.5 + 0.5) * height;
188    let rpx_x = (radius / z) * fx * (width * 0.5);
189    let rpx_y = (radius / z) * fy * (height * 0.5);
190    let tile_w = max(1.0, width / f32(params.clusters.x));
191    let tile_h = max(1.0, height / f32(params.clusters.y));
192    let min_px = floor(px - rpx_x);
193    let max_px = ceil(px + rpx_x);
194    let min_py = floor(py - rpx_y);
195    let max_py = ceil(py + rpx_y);
196    let ix0 = u32(clamp(min_px / tile_w, 0.0, f32(params.clusters.x - 1u)));
197    let ix1 = u32(clamp(max_px / tile_w, 0.0, f32(params.clusters.x - 1u)));
198    let iy0 = u32(clamp(min_py / tile_h, 0.0, f32(params.clusters.y - 1u)));
199    let iy1 = u32(clamp(max_py / tile_h, 0.0, f32(params.clusters.y - 1u)));
200    let zmin = max(z - radius, near);
201    let zmax = min(z + radius, far);
202    if (zmin >= zmax) { return; }
203    let iz0 = u32(clamp(floor(((zmin - near) / (far - near)) * f32(params.clusters.z)), 0.0, f32(params.clusters.z - 1u)));
204    let iz1 = u32(clamp(floor(((zmax - near) / (far - near)) * f32(params.clusters.z)), 0.0, f32(params.clusters.z - 1u)));
205    for (var iz: u32 = iz0; iz <= iz1; iz = iz + 1u) {
206        for (var iy: u32 = iy0; iy <= iy1; iy = iy + 1u) {
207            for (var ix: u32 = ix0; ix <= ix1; ix = ix + 1u) {
208                let ci = cluster_index(ix, iy, iz, params.clusters);
209                let write_idx = offsets[ci] + atomicAdd(&counts[ci], 1u);
210                indices[write_idx] = li;
211            }
212        }
213    }
214}

231    fn cpu_binning_produces_consistent_counts() {
232        let dims = ClusterDims { x: 8, y: 4, z: 8 };
233        let (w, h) = (640u32, 360u32);
234        let near = 0.1f32;
235        let far = 100.0f32;
236        let fov_y = std::f32::consts::FRAC_PI_3; // 60 deg
237        let mut rng = rand::rngs::StdRng::seed_from_u64(42);
238        let mut lights = Vec::new();
239        for _ in 0..64 {
240            let x = rand::Rng::random_range(&mut rng, -10.0..10.0);
241            let y = rand::Rng::random_range(&mut rng, -5.0..5.0);
242            let z = rand::Rng::random_range(&mut rng, 1.0..50.0);
243            let r = rand::Rng::random_range(&mut rng, 0.25..3.0);
244            lights.push(CpuLight {
245                pos: Vec3::new(x, y, z),
246                radius: r,
247            });
248        }
249        let (counts, indices, offsets) = bin_lights_cpu(&lights, dims, (w, h), near, far, fov_y);
250        // Basic invariants
251        assert_eq!(offsets.last().copied().unwrap_or(0), indices.len() as u32);
252        assert_eq!(
253            counts.iter().map(|&c| c as usize).sum::<usize>(),
254            indices.len()
255        );
256        // Re-derive counts from indices vector
257        let mut rec = vec![0u32; (dims.x * dims.y * dims.z) as usize];
258        for iz in 0..dims.z {
259            for iy in 0..dims.y {
260                for ix in 0..dims.x {
261                    let ci = cluster_index(ix, iy, iz, dims);
262                    rec[ci] = offsets[ci + 1] - offsets[ci];
263                }
264            }
265        }
266        assert_eq!(rec, counts);
267    }

272        let lights = vec![
273            CpuLight {
274                pos: Vec3::new(0.0, 0.0, 5.0),
275                radius: 1.0,

279                radius: 2.0,
280            },
281        ];
282        let (counts, indices, offsets) = bin_lights_cpu(
283            &lights,
284            dims,
285            (320, 180),
286            0.1,
287            50.0,
288            std::f32::consts::FRAC_PI_3,
289        );
290        // Offsets monotonic and last equals indices length
291        assert!(offsets.windows(2).all(|w| w[1] >= w[0]));
292        assert_eq!(*offsets.last().unwrap() as usize, indices.len());
293        // Indices in range
294        for &idx in &indices {
295            assert!((idx as usize) < lights.len());
296        }
297        // Counts match offsets delta
298        for i in 0..(counts.len()) {
299            assert_eq!(counts[i], offsets[i + 1] - offsets[i]);
300        }
301    }

14fn cluster_index(ix: u32, iy: u32, iz: u32, dims: ClusterDims) -> usize {
15    (ix + iy * dims.x + iz * (dims.x * dims.y)) as usize

18/// Compute per-cluster light lists on CPU using simple sphere-screen bounds.
19/// Inputs:
20/// - lights: positions are in camera/view space with +Z forward and in meters
21/// - screen: (width,height)
22/// - near/far: camera near/far planes (positive distances)
23/// - fov_y: vertical field of view in radians
24/// Returns: (counts, indices, offsets) where offsets is exclusive scan over counts
25pub fn bin_lights_cpu(
26    lights: &[CpuLight],
27    dims: ClusterDims,
28    screen: (u32, u32),
29    near: f32,
30    far: f32,
31    fov_y: f32,
32) -> (Vec<u32>, Vec<u32>, Vec<u32>) {
33    let (width, height) = screen;
34    let aspect = width.max(1) as f32 / height.max(1) as f32;
35    let fy = 1.0 / (0.5 * fov_y).tan();
36    let fx = fy / aspect;
37    let tile_w = (width as f32 / dims.x as f32).max(1.0);
38    let tile_h = (height as f32 / dims.y as f32).max(1.0);
39
40    let clusters = (dims.x * dims.y * dims.z) as usize;
41    let mut counts = vec![0u32; clusters];
42
43    // First pass: counts
44    for l in lights.iter() {
45        let z = l.pos.z.max(near + 1e-4);
46        if z - l.radius > far { continue; }
47        // Project center to NDC
48        let ndc_x = (l.pos.x / z) * fx;
49        let ndc_y = (l.pos.y / z) * fy;
50        // Convert to pixel space
51        let px = (ndc_x * 0.5 + 0.5) * width as f32;
52        let py = (ndc_y * 0.5 + 0.5) * height as f32;
53        // Screen-space radius in pixels (approx)
54        let rpx_x = (l.radius / z) * fx * (width as f32 * 0.5);
55        let rpx_y = (l.radius / z) * fy * (height as f32 * 0.5);
56        let min_px = (px - rpx_x).floor() as i32;
57        let max_px = (px + rpx_x).ceil() as i32;
58        let min_py = (py - rpx_y).floor() as i32;
59        let max_py = (py + rpx_y).ceil() as i32;
60        let ix0 = clamp_u32(min_px / tile_w as i32, 0, dims.x as i32 - 1);
61        let ix1 = clamp_u32(max_px / tile_w as i32, 0, dims.x as i32 - 1);
62        let iy0 = clamp_u32(min_py / tile_h as i32, 0, dims.y as i32 - 1);
63        let iy1 = clamp_u32(max_py / tile_h as i32, 0, dims.y as i32 - 1);
64        // Z slices
65        let zmin = (z - l.radius).max(near);
66        let zmax = (z + l.radius).min(far);
67        if zmin >= zmax { continue; }
68        let iz0 = (((zmin - near) / (far - near)) * dims.z as f32).floor().clamp(0.0, dims.z as f32 - 1.0) as u32;
69        let iz1 = (((zmax - near) / (far - near)) * dims.z as f32).floor().clamp(0.0, dims.z as f32 - 1.0) as u32;
70        for iz in iz0..=iz1 {
71            for iy in iy0..=iy1 {
72                for ix in ix0..=ix1 {
73                    let ci = cluster_index(ix, iy, iz, dims);
74                    counts[ci] += 1;
75                }
76            }
77        }
78    }
79
80    // Exclusive scan to offsets
81    let mut offsets = vec![0u32; clusters + 1];
82    for i in 0..clusters { offsets[i + 1] = offsets[i] + counts[i]; }
83
84    // Second pass: fill indices using running write cursors
85    let total = offsets[clusters] as usize;
86    let mut indices = vec![0u32; total];
87    let mut cursors = vec![0u32; clusters];
88    for (li, l) in lights.iter().enumerate() {
89        let z = l.pos.z.max(near + 1e-4);
90        if z - l.radius > far { continue; }
91        let ndc_x = (l.pos.x / z) * fx;
92        let ndc_y = (l.pos.y / z) * fy;
93        let px = (ndc_x * 0.5 + 0.5) * width as f32;
94        let py = (ndc_y * 0.5 + 0.5) * height as f32;
95        let rpx_x = (l.radius / z) * fx * (width as f32 * 0.5);
96        let rpx_y = (l.radius / z) * fy * (height as f32 * 0.5);
97        let min_px = (px - rpx_x).floor() as i32;
98        let max_px = (px + rpx_x).ceil() as i32;
99        let min_py = (py - rpx_y).floor() as i32;
100        let max_py = (py + rpx_y).ceil() as i32;
101        let ix0 = clamp_u32(min_px / tile_w as i32, 0, dims.x as i32 - 1);
102        let ix1 = clamp_u32(max_px / tile_w as i32, 0, dims.x as i32 - 1);
103        let iy0 = clamp_u32(min_py / tile_h as i32, 0, dims.y as i32 - 1);
104        let iy1 = clamp_u32(max_py / tile_h as i32, 0, dims.y as i32 - 1);
105        let zmin = (z - l.radius).max(near);
106        let zmax = (z + l.radius).min(far);
107        if zmin >= zmax { continue; }
108        let iz0 = (((zmin - near) / (far - near)) * dims.z as f32).floor().clamp(0.0, dims.z as f32 - 1.0) as u32;
109        let iz1 = (((zmax - near) / (far - near)) * dims.z as f32).floor().clamp(0.0, dims.z as f32 - 1.0) as u32;
110        for iz in iz0..=iz1 {
111            for iy in iy0..=iy1 {
112                for ix in ix0..=ix1 {
113                    let ci = cluster_index(ix, iy, iz, dims);
114                    let off = offsets[ci] + cursors[ci];
115                    indices[off as usize] = li as u32;
116                    cursors[ci] += 1;
117                }
118            }
119        }
120    }
121
122    (counts, indices, offsets)
123}

125pub const WGSL_CLUSTER_BIN: &str = r#"
126struct Light { pos_radius: vec4<f32> };
127struct Params { screen: vec2<u32>, clusters: vec3<u32>, near: f32, far: f32, fov_y: f32 };
128@group(0) @binding(0) var<storage, read> lights: array<Light>;
129@group(0) @binding(1) var<uniform> params: Params;
130// Offsets: length = clusters+1 (exclusive scan). Counts: length = clusters. Indices: variable length
131@group(0) @binding(2) var<storage, read> offsets: array<u32>;
132@group(0) @binding(3) var<storage, read_write> counts: array<atomic<u32>>;
133@group(0) @binding(4) var<storage, read_write> indices: array<u32>;
134
135fn cluster_index(ix: u32, iy: u32, iz: u32, dims: vec3<u32>) -> u32 {
136    return ix + iy * dims.x + iz * (dims.x * dims.y);
137}
138
139@compute @workgroup_size(1,1,1)
140fn cs_main(@builtin(global_invocation_id) gid: vec3<u32>) {
141    // One invocation per light
142    let li = gid.x;
143    if (li >= arrayLength(&lights)) { return; }
144    let L = lights[li].pos_radius;
145    let pos = L.xyz;
146    let radius = L.w;
147    let width = f32(params.screen.x);
148    let height = f32(params.screen.y);
149    let aspect = width / max(height, 1.0);
150    let fy = 1.0 / tan(0.5 * params.fov_y);
151    let fx = fy / aspect;
152    let z = max(pos.z, params.near + 1e-4);
153    if (z - radius > params.far) { return; }
154    let ndc_x = (pos.x / z) * fx;
155    let ndc_y = (pos.y / z) * fy;
156    let px = (ndc_x * 0.5 + 0.5) * width;
157    let py = (ndc_y * 0.5 + 0.5) * height;
158    let rpx_x = (radius / z) * fx * (width * 0.5);
159    let rpx_y = (radius / z) * fy * (height * 0.5);
160    let tile_w = max(1.0, width / f32(params.clusters.x));
161    let tile_h = max(1.0, height / f32(params.clusters.y));
162    let min_px = floor(px - rpx_x);
163    let max_px = ceil(px + rpx_x);
164    let min_py = floor(py - rpx_y);
165    let max_py = ceil(py + rpx_y);
166    let ix0 = u32(clamp(min_px / tile_w, 0.0, f32(params.clusters.x - 1u)));
167    let ix1 = u32(clamp(max_px / tile_w, 0.0, f32(params.clusters.x - 1u)));
168    let iy0 = u32(clamp(min_py / tile_h, 0.0, f32(params.clusters.y - 1u)));
169    let iy1 = u32(clamp(max_py / tile_h, 0.0, f32(params.clusters.y - 1u)));
170    let zmin = max(z - radius, params.near);
171    let zmax = min(z + radius, params.far);
172    if (zmin >= zmax) { return; }
173    let iz0 = u32(clamp(floor(((zmin - params.near) / (params.far - params.near)) * f32(params.clusters.z)), 0.0, f32(params.clusters.z - 1u)));
174    let iz1 = u32(clamp(floor(((zmax - params.near) / (params.far - params.near)) * f32(params.clusters.z)), 0.0, f32(params.clusters.z - 1u)));
175    for (var iz: u32 = iz0; iz <= iz1; iz = iz + 1u) {
176        for (var iy: u32 = iy0; iy <= iy1; iy = iy + 1u) {
177            for (var ix: u32 = ix0; ix <= ix1; ix = ix + 1u) {
178                let ci = cluster_index(ix, iy, iz, params.clusters);
179                let write_idx = offsets[ci] + atomicAdd(&counts[ci], 1u);
180                indices[write_idx] = li;
181            }
182        }
183    }
184}

199    #[test]
200    fn cpu_binning_produces_consistent_counts() {
201        let dims = ClusterDims{ x: 8, y: 4, z: 8 };
202        let (w,h) = (640u32, 360u32);
203        let near = 0.1f32; let far = 100.0f32; let fov_y = std::f32::consts::FRAC_PI_3; // 60 deg
204        let mut rng = rand::rngs::StdRng::seed_from_u64(42);
205        let mut lights = Vec::new();
206        for _ in 0..64 {
207            let x = rand::Rng::random_range(&mut rng, -10.0..10.0);
208            let y = rand::Rng::random_range(&mut rng, -5.0..5.0);
209            let z = rand::Rng::random_range(&mut rng, 1.0..50.0);
210            let r = rand::Rng::random_range(&mut rng, 0.25..3.0);
211            lights.push(CpuLight{ pos: Vec3::new(x,y,z), radius: r });
212        }
213        let (counts, indices, offsets) = bin_lights_cpu(&lights, dims, (w,h), near, far, fov_y);
214        // Basic invariants
215        assert_eq!(offsets.last().copied().unwrap_or(0), indices.len() as u32);
216        assert_eq!(counts.iter().map(|&c| c as usize).sum::<usize>(), indices.len());
217        // Re-derive counts from indices vector
218        let mut rec = vec![0u32; (dims.x * dims.y * dims.z) as usize];
219        for iz in 0..dims.z { for iy in 0..dims.y { for ix in 0..dims.x {
220            let ci = cluster_index(ix, iy, iz, dims);
221            rec[ci] = offsets[ci+1] - offsets[ci];
222        } } }
223        assert_eq!(rec, counts);
224    }

226    #[test]
227    fn cpu_binning_indices_valid() {
228        let dims = ClusterDims{ x: 4, y: 4, z: 4 };
229        let lights = vec![
230            CpuLight{ pos: Vec3::new(0.0,0.0,5.0), radius: 1.0 },
231            CpuLight{ pos: Vec3::new(1.0,0.0,7.0), radius: 2.0 },
232        ];
233        let (counts, indices, offsets) = bin_lights_cpu(&lights, dims, (320,180), 0.1, 50.0, std::f32::consts::FRAC_PI_3);
234        // Offsets monotonic and last equals indices length
235        assert!(offsets.windows(2).all(|w| w[1] >= w[0]));
236        assert_eq!(*offsets.last().unwrap() as usize, indices.len());
237        // Indices in range
238        for &idx in &indices { assert!((idx as usize) < lights.len()); }
239        // Counts match offsets delta
240        for i in 0..(counts.len()) {
241            assert_eq!(counts[i], offsets[i+1] - offsets[i]);
242        }
243    }

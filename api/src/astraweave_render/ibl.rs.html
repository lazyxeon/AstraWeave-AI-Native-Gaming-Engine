1//! Image-Based Lighting (IBL) manager and prefilter pipelines
2//!
3//! Minimal but complete IBL pipeline that can be refined later. Provides:
4//! - Procedural sky capture into an environment cubemap
5//! - Irradiance convolution for diffuse IBL (small cube)
6//! - Specular prefilter for GGX (mip chain encodes roughness)
7//! - BRDF LUT generation (split-sum), baked once at init
8//!
9//! The module is renderer-agnostic: it exposes bind group layout helpers and texture views
10//! that consumers can bind into their shading pipelines.
11

19/// Quality presets for IBL resource sizes
20#[derive(Clone, Copy, Debug)]
21pub enum IblQuality {
22    Low,
23    Medium,
24    High,
25}
26
27impl IblQuality {
28    fn env_size(self) -> u32 {
29        match self {
30            IblQuality::Low => 256,
31            IblQuality::Medium => 512,
32            IblQuality::High => 1024,
33        }
34    }
35    fn spec_size(self) -> u32 {
36        (self.env_size() / 2).max(128)
37    }
38    fn irradiance_size(self) -> u32 {
39        64
40    }
41    fn brdf_lut_size(self) -> u32 {
42        256
43    }
44    fn spec_mips(self) -> u32 {
45        let s = self.spec_size();
46        (s.max(1) as f32).log2().floor() as u32 + 1
47    }
48}

456    /// Ensure BRDF LUT is baked (one-time cost)
457    /// Returns a stable TextureView for binding
458    #[cfg(feature = "ibl")]
459    pub fn ensure_brdf_lut(
460        &mut self,
461        device: &wgpu::Device,
462        queue: &wgpu::Queue,
463        quality: IblQuality,
464    ) -> Result<wgpu::TextureView> {
465        // If textures are not baked, bake them
466        if self.textures.is_none() {
467            let _ = self.bake_environment(device, queue, quality)?;
468        }
469        let view = self
470            .textures
471            .as_ref()
472            .ok_or_else(|| anyhow::anyhow!("IBL textures not baked"))?
473            ._brdf_lut
474            .create_view(&wgpu::TextureViewDescriptor::default());
475        Ok(view)
476    }

527    /// Ensure environment and prefiltered outputs exist for the given mode/quality
528    pub fn bake_environment(
529        &mut self,
530        device: &wgpu::Device,
531        queue: &wgpu::Queue,
532        quality: IblQuality,
533    ) -> Result<IblResources> {
534        // Allocate textures
535        let env_size = quality.env_size();
536        let irr_size = quality.irradiance_size();
537        let spec_size = quality.spec_size();
538        let spec_mips = quality.spec_mips();
539
540        let env_tex = device.create_texture(&wgpu::TextureDescriptor {
541            label: Some("ibl-env-cube"),
542            size: wgpu::Extent3d {
543                width: env_size,
544                height: env_size,
545                depth_or_array_layers: 6,
546            },
547            mip_level_count: 1,
548            sample_count: 1,
549            dimension: wgpu::TextureDimension::D2,
550            format: wgpu::TextureFormat::Rgba16Float,
551            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
552            view_formats: &[],
553        });
554        let irr_tex = device.create_texture(&wgpu::TextureDescriptor {
555            label: Some("ibl-irr-cube"),
556            size: wgpu::Extent3d {
557                width: irr_size,
558                height: irr_size,
559                depth_or_array_layers: 6,
560            },
561            mip_level_count: 1,
562            sample_count: 1,
563            dimension: wgpu::TextureDimension::D2,
564            format: wgpu::TextureFormat::Rgba16Float,
565            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
566            view_formats: &[],
567        });
568        let spec_tex = device.create_texture(&wgpu::TextureDescriptor {
569            label: Some("ibl-spec-cube"),
570            size: wgpu::Extent3d {
571                width: spec_size,
572                height: spec_size,
573                depth_or_array_layers: 6,
574            },
575            mip_level_count: spec_mips,
576            sample_count: 1,
577            dimension: wgpu::TextureDimension::D2,
578            format: wgpu::TextureFormat::Rgba16Float,
579            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
580            view_formats: &[],
581        });
582        let brdf_tex = device.create_texture(&wgpu::TextureDescriptor {
583            label: Some("ibl-brdf-lut"),
584            size: wgpu::Extent3d {
585                width: quality.brdf_lut_size(),
586                height: quality.brdf_lut_size(),
587                depth_or_array_layers: 1,
588            },
589            mip_level_count: 1,
590            sample_count: 1,
591            dimension: wgpu::TextureDimension::D2,
592            format: wgpu::TextureFormat::Rg16Float,
593            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
594            view_formats: &[],
595        });
596
597        // Views
598        let env_view = env_tex.create_view(&wgpu::TextureViewDescriptor {
599            usage: None,
600            label: Some("ibl-env-view"),
601            dimension: Some(wgpu::TextureViewDimension::Cube),
602            ..Default::default()
603        });
604        let irr_view = irr_tex.create_view(&wgpu::TextureViewDescriptor {
605            usage: None,
606            label: Some("ibl-irr-view"),
607            dimension: Some(wgpu::TextureViewDimension::Cube),
608            ..Default::default()
609        });
610        let spec_view = spec_tex.create_view(&wgpu::TextureViewDescriptor {
611            usage: None,
612            label: Some("ibl-spec-view"),
613            dimension: Some(wgpu::TextureViewDimension::Cube),
614            base_mip_level: 0,
615            mip_level_count: Some(spec_mips),
616            ..Default::default()
617        });
618        let brdf_view = brdf_tex.create_view(&wgpu::TextureViewDescriptor::default());
619
620        // BRDF LUT bake
621        {
622            let mut enc = device.create_command_encoder(&wgpu::CommandEncoderDescriptor {
623                label: Some("ibl-brdf-enc"),
624            });
625            let mut rp = enc.begin_render_pass(&wgpu::RenderPassDescriptor {
626                label: Some("ibl-brdf-pass"),
627                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
628                    view: &brdf_view,
629                    resolve_target: None,
630                    ops: wgpu::Operations {
631                        load: wgpu::LoadOp::Clear(wgpu::Color::BLACK),
632                        store: wgpu::StoreOp::Store,
633                    },
634                })],
635                depth_stencil_attachment: None,
636                timestamp_writes: None,

1010// Irradiance convolution (Lambertian diffuse): proper cosine-weighted hemisphere sampling
1011// Note: This shader is executed per-face, so we derive the normal from clip-space position
1012const IRRADIANCE_WGSL: &str = r#"
1013struct VsOut { @builtin(position) pos: vec4<f32>, @location(0) clip_pos: vec2<f32> };
1014@vertex fn vs(@builtin(vertex_index) vi: u32) -> VsOut {
1015    var out: VsOut;
1016    var p = array<vec2<f32>, 3>(vec2<f32>(-1.0,-1.0), vec2<f32>(3.0,-1.0), vec2<f32>(-1.0,3.0));
1017    let xy = p[vi];
1018    out.pos = vec4<f32>(xy, 0.0, 1.0);
1019    out.clip_pos = xy;
1020    return out;
1021}
1022@group(0) @binding(0) var env_cube: texture_cube<f32>;
1023@group(0) @binding(1) var samp: sampler;
1024
1025@fragment fn fs(in: VsOut) -> @location(0) vec4<f32> {
1026    // Normal derived from clip-space coordinates (-1 to 1)
1027    // This works because we render one face at a time
1028    // The Z component will be 1.0 (facing outward from cube center)
1029    let N = normalize(vec3<f32>(in.clip_pos.x, in.clip_pos.y, 1.0));
1030    
1031    // Build orthonormal basis for tangent space
1032    let up = select(vec3<f32>(1.0, 0.0, 0.0), vec3<f32>(0.0, 1.0, 0.0), abs(N.z) < 0.999);
1033    let T = normalize(cross(up, N));
1034    let B = cross(N, T);
1035    
1036    // Integrate over hemisphere with cosine-weighted sampling
1037    var irradiance = vec3<f32>(0.0, 0.0, 0.0);
1038    let PHI_STEPS = 60u; // 6 degree steps
1039    let THETA_STEPS = 30u; // 3 degree steps
1040    let delta_phi = (2.0 * 3.14159265) / f32(PHI_STEPS);
1041    let delta_theta = (0.5 * 3.14159265) / f32(THETA_STEPS);
1042    
1043    var sample_count = 0.0;
1044    for (var i_phi = 0u; i_phi < PHI_STEPS; i_phi++) {
1045        for (var i_theta = 0u; i_theta < THETA_STEPS; i_theta++) {
1046            let phi = f32(i_phi) * delta_phi;
1047            let theta = f32(i_theta) * delta_theta;
1048            
1049            // Spherical to cartesian (in tangent space)
1050            let sample_vec_tangent = vec3<f32>(
1051                sin(theta) * cos(phi),
1052                sin(theta) * sin(phi),
1053                cos(theta)
1054            );
1055            // Transform to world space
1056            let sample_vec = normalize(
1057                T * sample_vec_tangent.x + 
1058                B * sample_vec_tangent.y + 
1059                N * sample_vec_tangent.z
1060            );
1061            
1062            // Sample environment with cosine weighting (NdotL)
1063            let sample_color = textureSample(env_cube, samp, sample_vec).rgb;
1064            irradiance += sample_color * cos(theta) * sin(theta);
1065            sample_count += 1.0;
1066        }
1067    }
1068    irradiance = irradiance * 3.14159265 / sample_count;
1069    
1070    return vec4<f32>(irradiance, 1.0);
1071}

1074// Specular prefilter (GGX): properly samples environment with importance sampling encoding roughness per mip
1075const SPECULAR_PREFILTER_WGSL: &str = r#"
1076struct VsOut { @builtin(position) pos: vec4<f32>, @location(0) uv: vec2<f32>, @location(1) face_idx: u32 };
1077struct PrefilterParams { roughness: f32, face_idx: u32, sample_count: u32, _pad: u32 };

1079@vertex fn vs(@builtin(vertex_index) vi: u32) -> VsOut {
1080    var out: VsOut;

1082    let xy = p[vi];

1086    return out;

1088@group(0) @binding(0) var env_cube: texture_cube<f32>;
1089@group(0) @binding(1) var samp: sampler;
1090fn radicalInverseVdC(bitsIn: u32) -> f32 { var bits = bitsIn; bits = (bits << 16u) | (bits >> 16u); bits = ((bits & 0x55555555u) << 1u) | ((bits & 0xAAAAAAAAu) >> 1u); bits = ((bits & 0x33333333u) << 2u) | ((bits & 0xCCCCCCCCu) >> 2u); bits = ((bits & 0x0F0F0F0Fu) << 4u) | ((bits & 0xF0F0F0F0u) >> 4u); bits = ((bits & 0x00FF00FFu) << 8u) | ((bits & 0xFF00FF00u) >> 8u); return f32(bits) * 2.3283064365386963e-10; }
1091fn hammersley(i: u32, n: u32) -> vec2<f32> { return vec2<f32>(f32(i)/f32(n), radicalInverseVdC(i)); }
1092fn importanceSampleGGX(Xi: vec2<f32>, N: vec3<f32>, roughness: f32) -> vec3<f32> { 
1093    let a = roughness*roughness; 
1094    let phi = 6.2831853*Xi.x; 
1095    let cosTheta = sqrt((1.0 - Xi.y) / (1.0 + (a*a - 1.0) * Xi.y)); 
1096    let sinTheta = sqrt(1.0 - cosTheta*cosTheta); 
1097    let H_tangent = vec3<f32>(cos(phi)*sinTheta, sin(phi)*sinTheta, cosTheta);
1098    // Build TBN
1099    let up = select(vec3<f32>(1.0,0.0,0.0), vec3<f32>(0.0,1.0,0.0), abs(N.z) < 0.999);
1100    let T = normalize(cross(up, N));
1101    let B = cross(N, T);
1102    return normalize(T*H_tangent.x + B*H_tangent.y + N*H_tangent.z);
1103}
1104fn uv_to_cube_dir(face: u32, uv: vec2<f32>) -> vec3<f32> {
1105    let tc = uv*2.0 - 1.0;
1106    if (face == 0u) { return normalize(vec3<f32>( 1.0,   -tc.y,  -tc.x)); }
1107    if (face == 1u) { return normalize(vec3<f32>(-1.0,   -tc.y,   tc.x)); }
1108    if (face == 2u) { return normalize(vec3<f32>( tc.x,   1.0,    tc.y)); }
1109    if (face == 3u) { return normalize(vec3<f32>( tc.x,  -1.0,   -tc.y)); }
1110    if (face == 4u) { return normalize(vec3<f32>( tc.x,  -tc.y,   1.0)); }
1111    return normalize(vec3<f32>(-tc.x,  -tc.y,  -1.0));
1112}
1113@fragment fn fs(in: VsOut) -> @location(0) vec4<f32> {
1114    let roughness = params.roughness;
1115    let N = uv_to_cube_dir(in.face_idx, in.uv);
1116    let R = N;
1117    let V = R;
1118    let SAMPLE_COUNT = params.sample_count;
1119    var acc = vec3<f32>(0.0, 0.0, 0.0);
1120    var w: f32 = 0.0;
1121    for (var i: u32 = 0u; i < SAMPLE_COUNT; i = i + 1u) {
1122        let Xi = hammersley(i, SAMPLE_COUNT);
1123        let H = importanceSampleGGX(Xi, N, roughness);
1124        let L = normalize(2.0 * dot(V,H) * H - V);
1125        let NdotL = max(dot(N,L), 0.0);
1126        if (NdotL > 0.0) {
1127            // Sample with appropriate mip level based on roughness for better filtering
1128            let D = ((roughness*roughness - 1.0)*roughness*roughness + 1.0);
1129            let pdf = max(D / (4.0 * 3.14159265), 1e-6);
1130            let texel_solid_angle = 4.0 * 3.14159265 / (6.0 * 512.0 * 512.0); // Assume 512 env resolution
1131            let sample_solid_angle = 1.0 / (f32(SAMPLE_COUNT) * pdf);
1132            let mip_level = select(0.0, 0.5 * log2(sample_solid_angle / texel_solid_angle), roughness > 0.0);
1133            acc += textureSampleLevel(env_cube, samp, L, mip_level).rgb * NdotL;
1134            w += NdotL;
1135        }
1136    }
1137    let outc = acc / max(w, 1e-4);
1138    return vec4<f32>(outc, 1.0);
1139}

1153fn radicalInverseVdC(bitsIn: u32) -> f32 { var bits = bitsIn; bits = (bits << 16u) | (bits >> 16u); bits = ((bits & 0x55555555u) << 1u) | ((bits & 0xAAAAAAAAu) >> 1u); bits = ((bits & 0x33333333u) << 2u) | ((bits & 0xCCCCCCCCu) >> 2u); bits = ((bits & 0x0F0F0F0Fu) << 4u) | ((bits & 0xF0F0F0F0u) >> 4u); bits = ((bits & 0x00FF00FFu) << 8u) | ((bits & 0xFF00FF00u) >> 8u); return f32(bits) * 2.3283064365386963e-10; }
1154fn hammersley(i: u32, n: u32) -> vec2<f32> { return vec2<f32>(f32(i)/f32(n), radicalInverseVdC(i)); }
1155fn geometrySchlickGGX(NdotV: f32, roughness: f32) -> f32 { let r = roughness + 1.0; let k = (r*r)/8.0; return NdotV / (NdotV * (1.0 - k) + k); }
1156fn geometrySmith(N: vec3<f32>, V: vec3<f32>, L: vec3<f32>, roughness: f32) -> f32 { let NdotV = max(dot(N,V),0.0); let NdotL = max(dot(N,L),0.0); let ggx2 = geometrySchlickGGX(NdotV, roughness); let ggx1 = geometrySchlickGGX(NdotL, roughness); return ggx1*ggx2; }
1157fn importanceSampleGGX(Xi: vec2<f32>, N: vec3<f32>, roughness: f32) -> vec3<f32> { let a = roughness*roughness; let phi = 6.2831853*Xi.x; let cosTheta = sqrt((1.0 - Xi.y) / (1.0 + (a*a - 1.0) * Xi.y)); let sinTheta = sqrt(1.0 - cosTheta*cosTheta); let H = vec3<f32>(cos(phi)*sinTheta, sin(phi)*sinTheta, cosTheta); let up = vec3<f32>(0.0,1.0,0.0); let T = normalize(cross(up, N)); let B = cross(N, T); let sampleVec = normalize(T*H.x + B*H.y + N*H.z); return sampleVec; }
1158@fragment fn fs(in: VsOut) -> @location(0) vec4<f32> {
1159    let N = vec3<f32>(0.0, 0.0, 1.0);
1160    let V = vec3<f32>(sqrt(1.0 - in.uv.x*in.uv.x), 0.0, in.uv.x);
1161    let roughness = clamp(in.uv.y, 0.0, 1.0);
1162    var A = 0.0; var B = 0.0; let SAMPLE_COUNT: u32 = 128u;
1163    for (var i: u32 = 0u; i < SAMPLE_COUNT; i = i + 1u) {
1164        let Xi = hammersley(i, SAMPLE_COUNT);
1165        let H = importanceSampleGGX(Xi, N, roughness);
1166        let L = normalize(2.0 * dot(V,H) * H - V);
1167        let NdotL = max(L.z, 0.0);
1168        if (NdotL > 0.0) {
1169            let NdotH = max(H.z, 0.0);
1170            let VdotH = max(dot(V,H), 0.0);
1171            let G = geometrySmith(N, V, L, roughness);
1172            let G_Vis = (G * VdotH) / max(NdotH * max(V.z, 1e-4), 1e-4);
1173            let Fc = pow(1.0 - VdotH, 5.0);
1174            A = A + (1.0 - Fc) * G_Vis;
1175            B = B + Fc * G_Vis;
1176        }
1177    }
1178    A = A / f32(SAMPLE_COUNT); B = B / f32(SAMPLE_COUNT);
1179    return vec4<f32>(A, B, 0.0, 1.0);
1180}

1254                    p[1] as f32 / 255.0,

1256                    p[3] as f32 / 255.0,
1257                ]
1258            })
1259            .collect(),

1282            texture: &tex,
1283            mip_level: 0,
1284            origin: wgpu::Origin3d::ZERO,
1285            aspect: wgpu::TextureAspect::All,
1286        },
1287        &rgba_f16,
1288        wgpu::TexelCopyBufferLayout {
1289            offset: 0,
1290            bytes_per_row: Some(w * 8),
1291            rows_per_image: Some(h),
1292        },
1293        wgpu::Extent3d {
1294            width: w,
1295            height: h,
1296            depth_or_array_layers: 1,

1300    let samp = device.create_sampler(&wgpu::SamplerDescriptor {

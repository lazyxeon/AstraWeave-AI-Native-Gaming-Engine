1//! Image-Based Lighting (IBL) manager and prefilter pipelines
2//!
3//! Minimal but complete IBL pipeline that can be refined later. Provides:
4//! - Procedural sky capture into an environment cubemap
5//! - Irradiance convolution for diffuse IBL (small cube)
6//! - Specular prefilter for GGX (mip chain encodes roughness)
7//! - BRDF LUT generation (split-sum), baked once at init
8//!
9//! The module is renderer-agnostic: it exposes bind group layout helpers and texture views
10//! that consumers can bind into their shading pipelines.
11
12use anyhow::{Context, Result};
13use image::GenericImageView;
14use std::{borrow::Cow, collections::HashMap, path::Path};
15
16/// Quality presets for IBL resource sizes
17#[derive(Clone, Copy, Debug)]
18pub enum IblQuality {
19    Low,
20    Medium,
21    High,
22}
23
24impl IblQuality {
25    fn env_size(self) -> u32 {
26        match self {
27            IblQuality::Low => 256,
28            IblQuality::Medium => 512,
29            IblQuality::High => 1024,
30        }
31    }
32    fn spec_size(self) -> u32 {
33        (self.env_size() / 2).max(128)
34    }
35    fn irradiance_size(self) -> u32 {
36        64
37    }
38    fn brdf_lut_size(self) -> u32 {
39        256
40    }
41    fn spec_mips(self) -> u32 {
42        let s = self.spec_size();
43        (s.max(1) as f32).log2().floor() as u32 + 1
44    }
45}

47/// Sky sources supported by the manager
48#[derive(Clone, Debug)]
49pub enum SkyMode {
50    /// Load an equirectangular HDR and convert to a cubemap
51    HdrPath { biome: String, path: String },
52    /// Render a simple procedural sky into the cubemap
53    Procedural {
54        last_capture_time: f32,
55        recapture_interval: f32,
56    },
57}

59/// Public handles to IBL resources (texture views suited for binding)
60pub struct IblResources {
61    pub env_cube: wgpu::TextureView, // optional to keep for debug
62    pub irradiance_cube: wgpu::TextureView,
63    pub specular_cube: wgpu::TextureView, // mip chain encodes roughness
64    pub brdf_lut: wgpu::TextureView,      // 2D LUT
65    pub mips_specular: u32,
66}

68/// Internal textures owned by the manager (kept to control lifetime)
69struct IblTextures {
70    _env: wgpu::Texture, // Kept alive for views
71    _irradiance: wgpu::Texture,
72    _specular: wgpu::Texture,
73    _brdf_lut: wgpu::Texture,
74    _spec_mips: u32,
75}

77pub struct IblManager {
78    pub enabled: bool,
79    pub mode: SkyMode,
80    pub sun_elevation: f32,
81    pub sun_azimuth: f32,
82
83    // GPU objects
84    sampler: wgpu::Sampler,
85    ibl_bgl: wgpu::BindGroupLayout,
86    // Bind group layout for convolution shaders (env cube + sampler)
87    env_bgl: wgpu::BindGroupLayout,
88    // Keep textures alive across frames/bind group creations
89    textures: Option<IblTextures>,
90    // Pipelines
91    sky_pipeline: wgpu::RenderPipeline,
92    irr_pipeline: wgpu::RenderPipeline,
93    spec_pipeline: wgpu::RenderPipeline,

99    // Cache decoded HDR equirectangular images by path to avoid repeated IO/decoding

427    /// Ensure BRDF LUT is baked (one-time cost)
428    /// Returns a stable TextureView for binding
429    #[cfg(feature = "ibl")]
430    pub fn ensure_brdf_lut(
431        &mut self,
432        device: &wgpu::Device,
433        queue: &wgpu::Queue,
434        quality: IblQuality,
435    ) -> Result<wgpu::TextureView> {
436        // If textures are not baked, bake them
437        if self.textures.is_none() {
438            let _ = self.bake_environment(device, queue, quality)?;
439        }
440        let view = self
441            .textures
442            .as_ref()
443            .ok_or_else(|| anyhow::anyhow!("IBL textures not baked"))?
444            .brdf_lut
445            .create_view(&wgpu::TextureViewDescriptor::default());
446        Ok(view)
447    }

475        &mut self,
476        device: &wgpu::Device,

491            ..Default::default()
492        });
493        Ok(view)
494    }

496    /// Ensure environment and prefiltered outputs exist for the given mode/quality
497    pub fn bake_environment(
498        &mut self,
499        device: &wgpu::Device,
500        queue: &wgpu::Queue,
501        quality: IblQuality,
502    ) -> Result<IblResources> {
503        // Allocate textures
504        let env_size = quality.env_size();
505        let irr_size = quality.irradiance_size();
506        let spec_size = quality.spec_size();
507        let spec_mips = quality.spec_mips();
508
509        let env_tex = device.create_texture(&wgpu::TextureDescriptor {
510            label: Some("ibl-env-cube"),
511            size: wgpu::Extent3d {
512                width: env_size,
513                height: env_size,
514                depth_or_array_layers: 6,
515            },
516            mip_level_count: 1,
517            sample_count: 1,
518            dimension: wgpu::TextureDimension::D2,
519            format: wgpu::TextureFormat::Rgba16Float,
520            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
521            view_formats: &[],
522        });
523        let irr_tex = device.create_texture(&wgpu::TextureDescriptor {
524            label: Some("ibl-irr-cube"),
525            size: wgpu::Extent3d {
526                width: irr_size,
527                height: irr_size,
528                depth_or_array_layers: 6,
529            },
530            mip_level_count: 1,
531            sample_count: 1,
532            dimension: wgpu::TextureDimension::D2,
533            format: wgpu::TextureFormat::Rgba16Float,
534            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
535            view_formats: &[],
536        });
537        let spec_tex = device.create_texture(&wgpu::TextureDescriptor {
538            label: Some("ibl-spec-cube"),
539            size: wgpu::Extent3d {
540                width: spec_size,
541                height: spec_size,
542                depth_or_array_layers: 6,
543            },
544            mip_level_count: spec_mips,
545            sample_count: 1,
546            dimension: wgpu::TextureDimension::D2,
547            format: wgpu::TextureFormat::Rgba16Float,
548            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
549            view_formats: &[],
550        });
551        let brdf_tex = device.create_texture(&wgpu::TextureDescriptor {
552            label: Some("ibl-brdf-lut"),
553            size: wgpu::Extent3d {
554                width: quality.brdf_lut_size(),
555                height: quality.brdf_lut_size(),
556                depth_or_array_layers: 1,
557            },
558            mip_level_count: 1,
559            sample_count: 1,
560            dimension: wgpu::TextureDimension::D2,
561            format: wgpu::TextureFormat::Rg16Float,
562            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
563            view_formats: &[],
564        });
565
566        // Views
567        let env_view = env_tex.create_view(&wgpu::TextureViewDescriptor {
568            usage: None,
569            label: Some("ibl-env-view"),
570            dimension: Some(wgpu::TextureViewDimension::Cube),
571            ..Default::default()
572        });
573        let irr_view = irr_tex.create_view(&wgpu::TextureViewDescriptor {
574            usage: None,
575            label: Some("ibl-irr-view"),
576            dimension: Some(wgpu::TextureViewDimension::Cube),
577            ..Default::default()
578        });
579        let spec_view = spec_tex.create_view(&wgpu::TextureViewDescriptor {
580            usage: None,
581            label: Some("ibl-spec-view"),
582            dimension: Some(wgpu::TextureViewDimension::Cube),
583            base_mip_level: 0,
584            mip_level_count: Some(spec_mips),
585            ..Default::default()
586        });
587        let brdf_view = brdf_tex.create_view(&wgpu::TextureViewDescriptor::default());
588
589        // BRDF LUT bake
590        {
591            let mut enc = device.create_command_encoder(&wgpu::CommandEncoderDescriptor {
592                label: Some("ibl-brdf-enc"),
593            });
594            let mut rp = enc.begin_render_pass(&wgpu::RenderPassDescriptor {
595                label: Some("ibl-brdf-pass"),
596                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
597                    view: &brdf_view,
598                    resolve_target: None,
599                    ops: wgpu::Operations {
600                        load: wgpu::LoadOp::Clear(wgpu::Color::BLACK),
601                        store: wgpu::StoreOp::Store,
602                    },
603                })],
604                depth_stencil_attachment: None,
605                timestamp_writes: None,
606                occlusion_query_set: None,
607            });
608            rp.set_pipeline(&self.brdf_pipeline);
609            rp.draw(0..3, 0..1);
610            drop(rp);
611            queue.submit(Some(enc.finish()));
612        }
613
614        // Sky capture into env cube (procedural or HDR-equirect conversion)
615        match &self.mode {
616            SkyMode::Procedural { .. } => {
617                let mut enc = device.create_command_encoder(&wgpu::CommandEncoderDescriptor {
618                    label: Some("ibl-sky-enc"),
619                });
620                for face in 0..6u32 {
621                    let face_view = env_tex.create_view(&wgpu::TextureViewDescriptor {
622                        usage: None,
623                        label: Some("ibl-env-face"),
624                        format: Some(wgpu::TextureFormat::Rgba16Float),
625                        dimension: Some(wgpu::TextureViewDimension::D2),
626                        base_mip_level: 0,
627                        mip_level_count: Some(1),
628                        base_array_layer: face,
629                        array_layer_count: Some(1),
630                        aspect: wgpu::TextureAspect::All,
631                    });
632                    let mut rp = enc.begin_render_pass(&wgpu::RenderPassDescriptor {
633                        label: Some("ibl-sky-face"),
634                        color_attachments: &[Some(wgpu::RenderPassColorAttachment {
635                            view: &face_view,
636                            resolve_target: None,
637                            ops: wgpu::Operations {

952fn radicalInverseVdC(bitsIn: u32) -> f32 { var bits = bitsIn; bits = (bits << 16u) | (bits >> 16u); bits = ((bits & 0x55555555u) << 1u) | ((bits & 0xAAAAAAAAu) >> 1u); bits = ((bits & 0x33333333u) << 2u) | ((bits & 0xCCCCCCCCu) >> 2u); bits = ((bits & 0x0F0F0F0Fu) << 4u) | ((bits & 0xF0F0F0F0u) >> 4u); bits = ((bits & 0x00FF00FFu) << 8u) | ((bits & 0xFF00FF00u) >> 8u); return f32(bits) * 2.3283064365386963e-10; }

978// Split-sum BRDF LUT (A, B) in RG channels
979const BRDF_LUT_WGSL: &str = r#"
980struct VsOut { @builtin(position) pos: vec4<f32>, @location(0) uv: vec2<f32> };
981@vertex fn vs(@builtin(vertex_index) vi: u32) -> VsOut {
982    var out: VsOut;
983    var p = array<vec2<f32>, 3>(vec2<f32>(-1.0,-1.0), vec2<f32>(3.0,-1.0), vec2<f32>(-1.0,3.0));
984    let xy = p[vi];
985    out.pos = vec4<f32>(xy, 0.0, 1.0);
986    out.uv = (xy+1.0)*0.5;
987    return out;
988}
989fn radicalInverseVdC(bitsIn: u32) -> f32 { var bits = bitsIn; bits = (bits << 16u) | (bits >> 16u); bits = ((bits & 0x55555555u) << 1u) | ((bits & 0xAAAAAAAAu) >> 1u); bits = ((bits & 0x33333333u) << 2u) | ((bits & 0xCCCCCCCCu) >> 2u); bits = ((bits & 0x0F0F0F0Fu) << 4u) | ((bits & 0xF0F0F0F0u) >> 4u); bits = ((bits & 0x00FF00FFu) << 8u) | ((bits & 0xFF00FF00u) >> 8u); return f32(bits) * 2.3283064365386963e-10; }
990fn hammersley(i: u32, n: u32) -> vec2<f32> { return vec2<f32>(f32(i)/f32(n), radicalInverseVdC(i)); }
991fn geometrySchlickGGX(NdotV: f32, roughness: f32) -> f32 { let r = roughness + 1.0; let k = (r*r)/8.0; return NdotV / (NdotV * (1.0 - k) + k); }
992fn geometrySmith(N: vec3<f32>, V: vec3<f32>, L: vec3<f32>, roughness: f32) -> f32 { let NdotV = max(dot(N,V),0.0); let NdotL = max(dot(N,L),0.0); let ggx2 = geometrySchlickGGX(NdotV, roughness); let ggx1 = geometrySchlickGGX(NdotL, roughness); return ggx1*ggx2; }
993fn importanceSampleGGX(Xi: vec2<f32>, N: vec3<f32>, roughness: f32) -> vec3<f32> { let a = roughness*roughness; let phi = 6.2831853*Xi.x; let cosTheta = sqrt((1.0 - Xi.y) / (1.0 + (a*a - 1.0) * Xi.y)); let sinTheta = sqrt(1.0 - cosTheta*cosTheta); let H = vec3<f32>(cos(phi)*sinTheta, sin(phi)*sinTheta, cosTheta); let up = vec3<f32>(0.0,1.0,0.0); let T = normalize(cross(up, N)); let B = cross(N, T); let sampleVec = normalize(T*H.x + B*H.y + N*H.z); return sampleVec; }
994@fragment fn fs(in: VsOut) -> @location(0) vec4<f32> {
995    let N = vec3<f32>(0.0, 0.0, 1.0);
996    let V = vec3<f32>(sqrt(1.0 - in.uv.x*in.uv.x), 0.0, in.uv.x);
997    let roughness = clamp(in.uv.y, 0.0, 1.0);
998    var A = 0.0; var B = 0.0; let SAMPLE_COUNT: u32 = 128u;
999    for (var i: u32 = 0u; i < SAMPLE_COUNT; i = i + 1u) {
1000        let Xi = hammersley(i, SAMPLE_COUNT);
1001        let H = importanceSampleGGX(Xi, N, roughness);
1002        let L = normalize(2.0 * dot(V,H) * H - V);
1003        let NdotL = max(L.z, 0.0);
1004        if (NdotL > 0.0) {
1005            let NdotH = max(H.z, 0.0);
1006            let VdotH = max(dot(V,H), 0.0);
1007            let G = geometrySmith(N, V, L, roughness);
1008            let G_Vis = (G * VdotH) / max(NdotH * max(V.z, 1e-4), 1e-4);
1009            let Fc = pow(1.0 - VdotH, 5.0);
1010            A = A + (1.0 - Fc) * G_Vis;
1011            B = B + Fc * G_Vis;
1012        }
1013    }
1014    A = A / f32(SAMPLE_COUNT); B = B / f32(SAMPLE_COUNT);
1015    return vec4<f32>(A, B, 0.0, 1.0);
1016}

1020const EQUIRECT_TO_CUBE_WGSL: &str = r#"
1021struct VsOut { @builtin(position) pos: vec4<f32>, @location(0) uv: vec2<f32> };
1022struct FaceIndex { idx: u32, _pad: vec3<u32> };
1023@group(1) @binding(0) var<uniform> u_face: FaceIndex;
1024@vertex fn vs(@builtin(vertex_index) vi: u32) -> VsOut {
1025    var out: VsOut;
1026    var p = array<vec2<f32>, 3>(vec2<f32>(-1.0,-1.0), vec2<f32>(3.0,-1.0), vec2<f32>(-1.0,3.0));
1027    let xy = p[vi];
1028    out.pos = vec4<f32>(xy, 0.0, 1.0);
1029    out.uv = (xy+1.0)*0.5;
1030    return out;
1031}
1032@group(0) @binding(0) var hdr_equirect: texture_2d<f32>;
1033@group(0) @binding(1) var samp: sampler;
1034fn uv_to_dir(face: i32, uv: vec2<f32>) -> vec3<f32> {
1035    let a = uv*2.0 - 1.0;
1036    if (face == 0) { return normalize(vec3<f32>( 1.0,    -a.y,   -a.x)); }
1037    if (face == 1) { return normalize(vec3<f32>(-1.0,    -a.y,    a.x)); }
1038    if (face == 2) { return normalize(vec3<f32>( a.x,     1.0,    a.y)); }
1039    if (face == 3) { return normalize(vec3<f32>( a.x,    -1.0,   -a.y)); }
1040    if (face == 4) { return normalize(vec3<f32>( a.x,    -a.y,    1.0)); }
1041    return normalize(vec3<f32>(-a.x,   -a.y,   -1.0));
1042}
1043fn dir_to_equirect_uv(dir: vec3<f32>) -> vec2<f32> {
1044    let n = normalize(dir);
1045    let phi = atan2(n.z, n.x);
1046    let theta = acos(clamp(n.y, -1.0, 1.0));
1047    let u = (phi / 6.2831853 + 0.5);
1048    let v = theta / 3.14159265;
1049    return vec2<f32>(u, v);
1050}
1051@fragment fn fs(in: VsOut) -> @location(0) vec4<f32> {
1052    // CPU passes face index via uniform
1053    let dir = uv_to_dir(i32(u_face.idx), in.uv);
1054    let uv = dir_to_equirect_uv(dir);
1055    let c = textureSample(hdr_equirect, samp, uv);
1056    return vec4<f32>(c.rgb, 1.0);
1057}

1072    let (w, h) = img.dimensions();

1083            .to_rgba8()
1084            .pixels()
1085            .flat_map(|p| {
1086                vec![
1087                    p[0] as f32 / 255.0,
1088                    p[1] as f32 / 255.0,
1089                    p[2] as f32 / 255.0,
1090                    p[3] as f32 / 255.0,
1091                ]
1092            })
1093            .collect(),
1094    };
1095    let mut rgba_f16 = Vec::with_capacity((w * h * 4) as usize * 2);
1096    for f in rgba_f32.into_iter() {
1097        let h = half::f16::from_f32(f);
1098        rgba_f16.extend_from_slice(&h.to_le_bytes());
1099    }
1100    let tex = device.create_texture(&wgpu::TextureDescriptor {
1101        label: Some("ibl-hdr2d"),
1102        size: wgpu::Extent3d {
1103            width: w,
1104            height: h,
1105            depth_or_array_layers: 1,
1106        },
1107        mip_level_count: 1,
1108        sample_count: 1,
1109        dimension: wgpu::TextureDimension::D2,
1110        format: wgpu::TextureFormat::Rgba16Float,
1111        usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
1112        view_formats: &[],
1113    });
1114    queue.write_texture(
1115        wgpu::TexelCopyTextureInfo {
1116            texture: &tex,
1117            mip_level: 0,
1118            origin: wgpu::Origin3d::ZERO,
1119            aspect: wgpu::TextureAspect::All,
1120        },
1121        &rgba_f16,
1122        wgpu::TexelCopyBufferLayout {
1123            offset: 0,
1124            bytes_per_row: Some(w * 8),
1125            rows_per_image: Some(h),
1126        },
1127        wgpu::Extent3d {
1128            width: w,
1129            height: h,
1130            depth_or_array_layers: 1,
1131        },
1132    );
1133    let view = tex.create_view(&wgpu::TextureViewDescriptor::default());
1134    let samp = device.create_sampler(&wgpu::SamplerDescriptor {
1135        label: Some("ibl-hdr2d-sampler"),
1136        address_mode_u: wgpu::AddressMode::ClampToEdge,
1137        address_mode_v: wgpu::AddressMode::ClampToEdge,
1138        address_mode_w: wgpu::AddressMode::ClampToEdge,
1139        mag_filter: wgpu::FilterMode::Linear,
1140        min_filter: wgpu::FilterMode::Linear,
1141        mipmap_filter: wgpu::FilterMode::Linear,
1142        ..Default::default()
1143    });
1144    Ok((tex, view, samp))

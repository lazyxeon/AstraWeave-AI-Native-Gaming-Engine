1//! Image-Based Lighting (IBL) manager and prefilter pipelines
2//!
3//! Minimal but complete IBL pipeline that can be refined later. Provides:
4//! - Procedural sky capture into an environment cubemap
5//! - Irradiance convolution for diffuse IBL (small cube)
6//! - Specular prefilter for GGX (mip chain encodes roughness)
7//! - BRDF LUT generation (split-sum), baked once at init
8//!
9//! The module is renderer-agnostic: it exposes bind group layout helpers and texture views
10//! that consumers can bind into their shading pipelines.
11
12use anyhow::{Context, Result};
13use image::GenericImageView;
14use std::{borrow::Cow, collections::HashMap, path::Path};
15
16/// Quality presets for IBL resource sizes
17#[derive(Clone, Copy, Debug)]
18pub enum IblQuality {
19    Low,
20    Medium,
21    High,
22}
23
24impl IblQuality {
25    fn env_size(self) -> u32 {
26        match self {
27            IblQuality::Low => 256,
28            IblQuality::Medium => 512,
29            IblQuality::High => 1024,
30        }
31    }
32    fn spec_size(self) -> u32 {
33        (self.env_size() / 2).max(128)
34    }
35    fn irradiance_size(self) -> u32 {
36        64
37    }
38    fn brdf_lut_size(self) -> u32 {
39        256
40    }
41    fn spec_mips(self) -> u32 {
42        let s = self.spec_size();
43        (s.max(1) as f32).log2().floor() as u32 + 1
44    }
45}

47/// Sky sources supported by the manager
48#[derive(Clone, Debug)]
49pub enum SkyMode {
50    /// Load an equirectangular HDR and convert to a cubemap
51    HdrPath { biome: String, path: String },
52    /// Render a simple procedural sky into the cubemap
53    Procedural {
54        last_capture_time: f32,
55        recapture_interval: f32,
56    },
57}

59/// Public handles to IBL resources (texture views suited for binding)
60pub struct IblResources {
61    pub env_cube: wgpu::TextureView, // optional to keep for debug
62    pub irradiance_cube: wgpu::TextureView,
63    pub specular_cube: wgpu::TextureView, // mip chain encodes roughness
64    pub brdf_lut: wgpu::TextureView,      // 2D LUT
65    pub mips_specular: u32,
66}

68/// Internal textures owned by the manager (kept to control lifetime)
69struct IblTextures {
70    _env: wgpu::Texture,
71    _irradiance: wgpu::Texture,
72    _specular: wgpu::Texture,
73    _brdf_lut: wgpu::Texture,
74}

76pub struct IblManager {
77    pub enabled: bool,
78    pub mode: SkyMode,
79    pub sun_elevation: f32,
80    pub sun_azimuth: f32,
81
82    // GPU objects
83    sampler: wgpu::Sampler,
84    ibl_bgl: wgpu::BindGroupLayout,
85    // Bind group layout for convolution shaders (env cube + sampler)
86    env_bgl: wgpu::BindGroupLayout,
87    // Keep textures alive across frames/bind group creations
88    textures: Option<IblTextures>,
89    // Pipelines
90    sky_pipeline: wgpu::RenderPipeline,
91    irr_pipeline: wgpu::RenderPipeline,
92    spec_pipeline: wgpu::RenderPipeline,

96    eqr_face_bgl: wgpu::BindGroupLayout,
97    eqr_pipeline: wgpu::RenderPipeline,
98    // Cache decoded HDR equirectangular images by path to avoid repeated IO/decoding
99    hdr_cache: HashMap<String, image::DynamicImage>,

421    /// Ensure environment and prefiltered outputs exist for the given mode/quality
422    pub fn bake_environment(
423        &mut self,
424        device: &wgpu::Device,
425        queue: &wgpu::Queue,
426        quality: IblQuality,
427    ) -> Result<IblResources> {
428        // Allocate textures
429        let env_size = quality.env_size();
430        let irr_size = quality.irradiance_size();
431        let spec_size = quality.spec_size();
432        let spec_mips = quality.spec_mips();
433
434        let env_tex = device.create_texture(&wgpu::TextureDescriptor {
435            label: Some("ibl-env-cube"),
436            size: wgpu::Extent3d {
437                width: env_size,
438                height: env_size,
439                depth_or_array_layers: 6,
440            },
441            mip_level_count: 1,
442            sample_count: 1,
443            dimension: wgpu::TextureDimension::D2,
444            format: wgpu::TextureFormat::Rgba16Float,
445            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
446            view_formats: &[],
447        });
448        let irr_tex = device.create_texture(&wgpu::TextureDescriptor {
449            label: Some("ibl-irr-cube"),
450            size: wgpu::Extent3d {
451                width: irr_size,
452                height: irr_size,
453                depth_or_array_layers: 6,
454            },
455            mip_level_count: 1,
456            sample_count: 1,
457            dimension: wgpu::TextureDimension::D2,
458            format: wgpu::TextureFormat::Rgba16Float,
459            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
460            view_formats: &[],
461        });
462        let spec_tex = device.create_texture(&wgpu::TextureDescriptor {
463            label: Some("ibl-spec-cube"),
464            size: wgpu::Extent3d {
465                width: spec_size,
466                height: spec_size,
467                depth_or_array_layers: 6,
468            },
469            mip_level_count: spec_mips,
470            sample_count: 1,
471            dimension: wgpu::TextureDimension::D2,
472            format: wgpu::TextureFormat::Rgba16Float,
473            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
474            view_formats: &[],
475        });
476        let brdf_tex = device.create_texture(&wgpu::TextureDescriptor {
477            label: Some("ibl-brdf-lut"),
478            size: wgpu::Extent3d {
479                width: quality.brdf_lut_size(),
480                height: quality.brdf_lut_size(),
481                depth_or_array_layers: 1,
482            },
483            mip_level_count: 1,
484            sample_count: 1,
485            dimension: wgpu::TextureDimension::D2,
486            format: wgpu::TextureFormat::Rg16Float,
487            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
488            view_formats: &[],
489        });
490
491        // Views
492        let env_view = env_tex.create_view(&wgpu::TextureViewDescriptor {
493            label: Some("ibl-env-view"),
494            dimension: Some(wgpu::TextureViewDimension::Cube),
495            ..Default::default()
496        });
497        let irr_view = irr_tex.create_view(&wgpu::TextureViewDescriptor {
498            label: Some("ibl-irr-view"),
499            dimension: Some(wgpu::TextureViewDimension::Cube),
500            ..Default::default()
501        });
502        let spec_view = spec_tex.create_view(&wgpu::TextureViewDescriptor {
503            label: Some("ibl-spec-view"),
504            dimension: Some(wgpu::TextureViewDimension::Cube),
505            base_mip_level: 0,
506            mip_level_count: Some(spec_mips),
507            ..Default::default()
508        });
509        let brdf_view = brdf_tex.create_view(&wgpu::TextureViewDescriptor::default());
510
511        // BRDF LUT bake
512        {
513            let mut enc = device.create_command_encoder(&wgpu::CommandEncoderDescriptor {
514                label: Some("ibl-brdf-enc"),
515            });
516            let mut rp = enc.begin_render_pass(&wgpu::RenderPassDescriptor {
517                label: Some("ibl-brdf-pass"),
518                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
519                    view: &brdf_view,
520                    resolve_target: None,
521                    ops: wgpu::Operations {
522                        load: wgpu::LoadOp::Clear(wgpu::Color::BLACK),
523                        store: wgpu::StoreOp::Store,
524                    },
525                })],
526                depth_stencil_attachment: None,
527                timestamp_writes: None,
528                occlusion_query_set: None,
529            });
530            rp.set_pipeline(&self.brdf_pipeline);
531            rp.draw(0..3, 0..1);
532            drop(rp);
533            queue.submit(Some(enc.finish()));
534        }
535
536        // Sky capture into env cube (procedural or HDR-equirect conversion)
537        match &self.mode {
538            SkyMode::Procedural { .. } => {
539                let mut enc = device.create_command_encoder(&wgpu::CommandEncoderDescriptor {
540                    label: Some("ibl-sky-enc"),
541                });
542                for face in 0..6u32 {
543                    let face_view = env_tex.create_view(&wgpu::TextureViewDescriptor {
544                        label: Some("ibl-env-face"),
545                        format: Some(wgpu::TextureFormat::Rgba16Float),
546                        dimension: Some(wgpu::TextureViewDimension::D2),
547                        base_mip_level: 0,
548                        mip_level_count: Some(1),
549                        base_array_layer: face,
550                        array_layer_count: Some(1),
551                        aspect: wgpu::TextureAspect::All,
552                    });
553                    let mut rp = enc.begin_render_pass(&wgpu::RenderPassDescriptor {
554                        label: Some("ibl-sky-face"),
555                        color_attachments: &[Some(wgpu::RenderPassColorAttachment {
556                            view: &face_view,
557                            resolve_target: None,
558                            ops: wgpu::Operations {
559                                load: wgpu::LoadOp::Clear(wgpu::Color {
560                                    r: 0.0,
561                                    g: 0.0,
562                                    b: 0.0,
563                                    a: 1.0,
564                                }),
565                                store: wgpu::StoreOp::Store,
566                            },
567                        })],
568                        depth_stencil_attachment: None,
569                        timestamp_writes: None,
570                        occlusion_query_set: None,
571                    });
572                    rp.set_pipeline(&self.sky_pipeline);
573                    rp.draw(0..3, 0..1);
574                    drop(rp);
575                }
576                queue.submit(Some(enc.finish()));
577            }
578            SkyMode::HdrPath { biome: _, path } => {
579                let img = if let Some(img) = self.hdr_cache.get(path) {
580                    img.clone()
581                } else {
582                    let img = load_hdr_equirectangular(Path::new(path))
583                        .with_context(|| format!("load HDR {}", path))?;
584                    self.hdr_cache.insert(path.clone(), img.clone());
585                    img
586                };
587                let (_hdr_tex, hdr_view, hdr_samp) = create_hdr2d(device, queue, &img)?;
588                let eqr_bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
589                    label: Some("ibl-eqr-bg"),
590                    layout: &self.eqr_bgl,
591                    entries: &[
592                        wgpu::BindGroupEntry {
593                            binding: 0,
594                            resource: wgpu::BindingResource::TextureView(&hdr_view),
595                        },
596                        wgpu::BindGroupEntry {
597                            binding: 1,
598                            resource: wgpu::BindingResource::Sampler(&hdr_samp),
599                        },
600                    ],
601                });
602                // Uniform buffer for face index (aligned to 16 bytes)
603                let face_buf = device.create_buffer(&wgpu::BufferDescriptor {
604                    label: Some("ibl-eqr-face-ub"),
605                    size: 16,
606                    usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
607                    mapped_at_creation: false,
608                });
609                let eqr_face_bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
610                    label: Some("ibl-eqr-face-bg"),
611                    layout: &self.eqr_face_bgl,
612                    entries: &[wgpu::BindGroupEntry {
613                        binding: 0,
614                        resource: face_buf.as_entire_binding(),
615                    }],
616                });
617                let mut enc = device.create_command_encoder(&wgpu::CommandEncoderDescriptor {
618                    label: Some("ibl-eqr-enc"),
619                });
620                for face in 0..6u32 {
621                    let face_view = env_tex.create_view(&wgpu::TextureViewDescriptor {
622                        label: Some("ibl-env-face"),
623                        format: Some(wgpu::TextureFormat::Rgba16Float),
624                        dimension: Some(wgpu::TextureViewDimension::D2),
625                        base_mip_level: 0,
626                        mip_level_count: Some(1),
627                        base_array_layer: face,

631                    // Update face index uniform
632                    let data: [u32; 4] = [face, 0, 0, 0];

653                }

655            }
656        }

658        // Irradiance convolution
659        {

677            });

681                    format: Some(wgpu::TextureFormat::Rgba16Float),

684                    array_layer_count: Some(1),
685                    base_mip_level: 0,
686                    mip_level_count: Some(1),
687                    aspect: wgpu::TextureAspect::All,
688                });
689                let mut rp = enc.begin_render_pass(&wgpu::RenderPassDescriptor {
690                    label: Some("ibl-irr-pass"),
691                    color_attachments: &[Some(wgpu::RenderPassColorAttachment {
692                        view: &dst_face,
693                        resolve_target: None,
694                        ops: wgpu::Operations {
695                            load: wgpu::LoadOp::Clear(wgpu::Color::BLACK),
696                            store: wgpu::StoreOp::Store,
697                        },
698                    })],
699                    depth_stencil_attachment: None,
700                    timestamp_writes: None,
701                    occlusion_query_set: None,
702                });
703                rp.set_pipeline(&self.irr_pipeline);
704                rp.set_bind_group(0, &env_bg, &[]);
705                rp.draw(0..3, 0..1);
706                drop(rp);
707            }
708            queue.submit(Some(enc.finish()));
709        }
710
711        // Specular prefilter for each mip and face
712        {
713            let env_bg = device.create_bind_group(&wgpu::BindGroupDescriptor {
714                label: Some("ibl-env-bg"),
715                layout: &self.env_bgl,
716                entries: &[
717                    wgpu::BindGroupEntry {
718                        binding: 0,
719                        resource: wgpu::BindingResource::TextureView(&env_view),
720                    },
721                    wgpu::BindGroupEntry {

723                        resource: wgpu::BindingResource::Sampler(&self.sampler),
724                    },
725                ],
726            });
727            let mut enc = device.create_command_encoder(&wgpu::CommandEncoderDescriptor {
728                label: Some("ibl-spec-enc"),
729            });
730            for mip in 0..spec_mips {
731                for face in 0..6u32 {
732                    let dst = spec_tex.create_view(&wgpu::TextureViewDescriptor {
733                        label: Some("ibl-spec-sub"),
734                        format: Some(wgpu::TextureFormat::Rgba16Float),
735                        dimension: Some(wgpu::TextureViewDimension::D2),
736                        base_array_layer: face,
737                        array_layer_count: Some(1),
738                        base_mip_level: mip,

741                    });
742                    let mut rp = enc.begin_render_pass(&wgpu::RenderPassDescriptor {
743                        label: Some("ibl-spec-pass"),
744                        color_attachments: &[Some(wgpu::RenderPassColorAttachment {
745                            view: &dst,
746                            resolve_target: None,
747                            ops: wgpu::Operations {
748                                load: wgpu::LoadOp::Clear(wgpu::Color::BLACK),
749                                store: wgpu::StoreOp::Store,
750                            },
751                        })],
752                        depth_stencil_attachment: None,
753                        timestamp_writes: None,
754                        occlusion_query_set: None,
755                    });
756                    rp.set_pipeline(&self.spec_pipeline);
757                    rp.set_bind_group(0, &env_bg, &[]);
758                    rp.draw(0..3, 0..1);
759                    drop(rp);
760                }
761            }
762            queue.submit(Some(enc.finish()));
763        }
764
765        // Hold textures so views remain valid for the lifetime of the manager
766        self.textures = Some(IblTextures {
767            _env: env_tex,
768            _irradiance: irr_tex,
769            _specular: spec_tex,
770            _brdf_lut: brdf_tex,
771        });
772        let resources = IblResources {
773            env_cube: env_view,
774            irradiance_cube: irr_view,
775            specular_cube: spec_view,
776            brdf_lut: brdf_view,
777            mips_specular: spec_mips,
778        };
779        Ok(resources)
780    }

856// Specular prefilter (GGX) minimal: roughness is derived from mip during write (encoded by pipeline state in a real version). Here we pick a fixed roughness per draw; caller draws per mip.
857const SPECULAR_PREFILTER_WGSL: &str = r#"
858struct VsOut { @builtin(position) pos: vec4<f32>, @location(0) uv: vec2<f32> };
859@vertex fn vs(@builtin(vertex_index) vi: u32) -> VsOut {
860    var out: VsOut;

862    let xy = p[vi];
863    out.pos = vec4<f32>(xy, 0.0, 1.0);
864    out.uv = (xy+1.0)*0.5;
865    return out;
866}
867@group(0) @binding(0) var env_cube: texture_cube<f32>;
868@group(0) @binding(1) var samp: sampler;
869fn radicalInverseVdC(bitsIn: u32) -> f32 { var bits = bitsIn; bits = (bits << 16u) | (bits >> 16u); bits = ((bits & 0x55555555u) << 1u) | ((bits & 0xAAAAAAAAu) >> 1u); bits = ((bits & 0x33333333u) << 2u) | ((bits & 0xCCCCCCCCu) >> 2u); bits = ((bits & 0x0F0F0F0Fu) << 4u) | ((bits & 0xF0F0F0F0u) >> 4u); bits = ((bits & 0x00FF00FFu) << 8u) | ((bits & 0xFF00FF00u) >> 8u); return f32(bits) * 2.3283064365386963e-10; }
870fn hammersley(i: u32, n: u32) -> vec2<f32> { return vec2<f32>(f32(i)/f32(n), radicalInverseVdC(i)); }
871fn importanceSampleGGX(Xi: vec2<f32>, roughness: f32) -> vec3<f32> { let a = roughness*roughness; let phi = 6.2831853*Xi.x; let cosTheta = sqrt((1.0 - Xi.y) / (1.0 + (a*a - 1.0) * Xi.y)); let sinTheta = sqrt(1.0 - cosTheta*cosTheta); return vec3<f32>(cos(phi)*sinTheta, sin(phi)*sinTheta, cosTheta); }
872@fragment fn fs(in: VsOut) -> @location(0) vec4<f32> {
873    // Derive roughness from UV.y to vary across mips surrogate; a real path passes roughness per-mip.
874    let roughness = clamp(in.uv.y, 0.0, 1.0);
875    let N = normalize(vec3<f32>(in.uv.x*2.0-1.0, 1.0, in.uv.y*2.0-1.0));
876    let V = N;
877    let SAMPLE_COUNT: u32 = 64u;
878    var acc = vec3<f32>(0.0, 0.0, 0.0);
879    var w: f32 = 0.0;
880    for (var i: u32 = 0u; i < SAMPLE_COUNT; i = i + 1u) {
881        let Xi = hammersley(i, SAMPLE_COUNT);
882        let H = importanceSampleGGX(Xi, roughness);
883        let L = normalize(2.0 * dot(V,H) * H - V);
884        let NdotL = max(dot(N,L), 0.0);
885        if (NdotL > 0.0) {
886            acc += textureSample(env_cube, samp, L).rgb * NdotL;
887            w += NdotL;
888        }
889    }
890    let outc = acc / max(w, 1e-4);
891    return vec4<f32>(outc, 1.0);
892}

895// Split-sum BRDF LUT (A, B) in RG channels
896const BRDF_LUT_WGSL: &str = r#"
897struct VsOut { @builtin(position) pos: vec4<f32>, @location(0) uv: vec2<f32> };
898@vertex fn vs(@builtin(vertex_index) vi: u32) -> VsOut {
899    var out: VsOut;
900    var p = array<vec2<f32>, 3>(vec2<f32>(-1.0,-1.0), vec2<f32>(3.0,-1.0), vec2<f32>(-1.0,3.0));
901    let xy = p[vi];
902    out.pos = vec4<f32>(xy, 0.0, 1.0);
903    out.uv = (xy+1.0)*0.5;
904    return out;
905}
906fn radicalInverseVdC(bitsIn: u32) -> f32 { var bits = bitsIn; bits = (bits << 16u) | (bits >> 16u); bits = ((bits & 0x55555555u) << 1u) | ((bits & 0xAAAAAAAAu) >> 1u); bits = ((bits & 0x33333333u) << 2u) | ((bits & 0xCCCCCCCCu) >> 2u); bits = ((bits & 0x0F0F0F0Fu) << 4u) | ((bits & 0xF0F0F0F0u) >> 4u); bits = ((bits & 0x00FF00FFu) << 8u) | ((bits & 0xFF00FF00u) >> 8u); return f32(bits) * 2.3283064365386963e-10; }
907fn hammersley(i: u32, n: u32) -> vec2<f32> { return vec2<f32>(f32(i)/f32(n), radicalInverseVdC(i)); }
908fn geometrySchlickGGX(NdotV: f32, roughness: f32) -> f32 { let r = roughness + 1.0; let k = (r*r)/8.0; return NdotV / (NdotV * (1.0 - k) + k); }
909fn geometrySmith(N: vec3<f32>, V: vec3<f32>, L: vec3<f32>, roughness: f32) -> f32 { let NdotV = max(dot(N,V),0.0); let NdotL = max(dot(N,L),0.0); let ggx2 = geometrySchlickGGX(NdotV, roughness); let ggx1 = geometrySchlickGGX(NdotL, roughness); return ggx1*ggx2; }
910fn importanceSampleGGX(Xi: vec2<f32>, N: vec3<f32>, roughness: f32) -> vec3<f32> { let a = roughness*roughness; let phi = 6.2831853*Xi.x; let cosTheta = sqrt((1.0 - Xi.y) / (1.0 + (a*a - 1.0) * Xi.y)); let sinTheta = sqrt(1.0 - cosTheta*cosTheta); let H = vec3<f32>(cos(phi)*sinTheta, sin(phi)*sinTheta, cosTheta); let up = vec3<f32>(0.0,1.0,0.0); let T = normalize(cross(up, N)); let B = cross(N, T); let sampleVec = normalize(T*H.x + B*H.y + N*H.z); return sampleVec; }
911@fragment fn fs(in: VsOut) -> @location(0) vec4<f32> {
912    let N = vec3<f32>(0.0, 0.0, 1.0);
913    let V = vec3<f32>(sqrt(1.0 - in.uv.x*in.uv.x), 0.0, in.uv.x);
914    let roughness = clamp(in.uv.y, 0.0, 1.0);
915    var A = 0.0; var B = 0.0; let SAMPLE_COUNT: u32 = 128u;
916    for (var i: u32 = 0u; i < SAMPLE_COUNT; i = i + 1u) {
917        let Xi = hammersley(i, SAMPLE_COUNT);
918        let H = importanceSampleGGX(Xi, N, roughness);
919        let L = normalize(2.0 * dot(V,H) * H - V);
920        let NdotL = max(L.z, 0.0);
921        if (NdotL > 0.0) {
922            let NdotH = max(H.z, 0.0);
923            let VdotH = max(dot(V,H), 0.0);
924            let G = geometrySmith(N, V, L, roughness);
925            let G_Vis = (G * VdotH) / max(NdotH * max(V.z, 1e-4), 1e-4);
926            let Fc = pow(1.0 - VdotH, 5.0);
927            A = A + (1.0 - Fc) * G_Vis;
928            B = B + Fc * G_Vis;
929        }
930    }
931    A = A / f32(SAMPLE_COUNT); B = B / f32(SAMPLE_COUNT);
932    return vec4<f32>(A, B, 0.0, 1.0);
933}

936// Equirectangular to cubemap conversion shader (minimal placeholder)
937const EQUIRECT_TO_CUBE_WGSL: &str = r#"
938struct VsOut { @builtin(position) pos: vec4<f32>, @location(0) uv: vec2<f32> };
939struct FaceIndex { idx: u32, _pad: vec3<u32> };
940@group(1) @binding(0) var<uniform> u_face: FaceIndex;
941@vertex fn vs(@builtin(vertex_index) vi: u32) -> VsOut {
942    var out: VsOut;
943    var p = array<vec2<f32>, 3>(vec2<f32>(-1.0,-1.0), vec2<f32>(3.0,-1.0), vec2<f32>(-1.0,3.0));
944    let xy = p[vi];
945    out.pos = vec4<f32>(xy, 0.0, 1.0);
946    out.uv = (xy+1.0)*0.5;

948}
949@group(0) @binding(0) var hdr_equirect: texture_2d<f32>;
950@group(0) @binding(1) var samp: sampler;
951fn uv_to_dir(face: i32, uv: vec2<f32>) -> vec3<f32> {
952    let a = uv*2.0 - 1.0;
953    if (face == 0) { return normalize(vec3<f32>( 1.0,    -a.y,   -a.x)); }
954    if (face == 1) { return normalize(vec3<f32>(-1.0,    -a.y,    a.x)); }
955    if (face == 2) { return normalize(vec3<f32>( a.x,     1.0,    a.y)); }
956    if (face == 3) { return normalize(vec3<f32>( a.x,    -1.0,   -a.y)); }
957    if (face == 4) { return normalize(vec3<f32>( a.x,    -a.y,    1.0)); }
958    return normalize(vec3<f32>(-a.x,   -a.y,   -1.0));
959}
960fn dir_to_equirect_uv(dir: vec3<f32>) -> vec2<f32> {
961    let n = normalize(dir);
962    let phi = atan2(n.z, n.x);
963    let theta = acos(clamp(n.y, -1.0, 1.0));
964    let u = (phi / 6.2831853 + 0.5);
965    let v = theta / 3.14159265;
966    return vec2<f32>(u, v);
967}
968@fragment fn fs(in: VsOut) -> @location(0) vec4<f32> {
969    // CPU passes face index via uniform
970    let dir = uv_to_dir(i32(u_face.idx), in.uv);
971    let uv = dir_to_equirect_uv(dir);
972    let c = textureSample(hdr_equirect, samp, uv);
973    return vec4<f32>(c.rgb, 1.0);
974}

984fn create_hdr2d(
985    device: &wgpu::Device,
986    queue: &wgpu::Queue,
987    img: &image::DynamicImage,
988) -> Result<(wgpu::Texture, wgpu::TextureView, wgpu::Sampler)> {
989    let (w, h) = img.dimensions();
990    let rgba_f32: Vec<f32> = match img {
991        image::DynamicImage::ImageRgb32F(buf) => buf
992            .pixels()
993            .flat_map(|p| vec![p[0], p[1], p[2], 1.0])
994            .collect(),
995        image::DynamicImage::ImageRgba32F(buf) => buf
996            .pixels()
997            .flat_map(|p| vec![p[0], p[1], p[2], p[3]])
998            .collect(),
999        _ => img
1000            .to_rgba8()
1001            .pixels()
1002            .flat_map(|p| {
1003                vec![
1004                    p[0] as f32 / 255.0,
1005                    p[1] as f32 / 255.0,
1006                    p[2] as f32 / 255.0,
1007                    p[3] as f32 / 255.0,
1008                ]
1009            })
1010            .collect(),
1011    };
1012    let mut rgba_f16 = Vec::with_capacity((w * h * 4) as usize * 2);
1013    for f in rgba_f32.into_iter() {
1014        let h = half::f16::from_f32(f);
1015        rgba_f16.extend_from_slice(&h.to_le_bytes());
1016    }
1017    let tex = device.create_texture(&wgpu::TextureDescriptor {
1018        label: Some("ibl-hdr2d"),
1019        size: wgpu::Extent3d {
1020            width: w,
1021            height: h,
1022            depth_or_array_layers: 1,
1023        },
1024        mip_level_count: 1,
1025        sample_count: 1,
1026        dimension: wgpu::TextureDimension::D2,
1027        format: wgpu::TextureFormat::Rgba16Float,
1028        usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
1029        view_formats: &[],
1030    });
1031    queue.write_texture(
1032        wgpu::ImageCopyTexture {
1033            texture: &tex,
1034            mip_level: 0,
1035            origin: wgpu::Origin3d::ZERO,
1036            aspect: wgpu::TextureAspect::All,
1037        },
1038        &rgba_f16,
1039        wgpu::ImageDataLayout {
1040            offset: 0,
1041            bytes_per_row: Some(w * 8),
1042            rows_per_image: Some(h),
1043        },
1044        wgpu::Extent3d {
1045            width: w,
1046            height: h,
1047            depth_or_array_layers: 1,
1048        },
1049    );
1050    let view = tex.create_view(&wgpu::TextureViewDescriptor::default());
1051    let samp = device.create_sampler(&wgpu::SamplerDescriptor {
1052        label: Some("ibl-hdr2d-sampler"),
1053        address_mode_u: wgpu::AddressMode::ClampToEdge,
1054        address_mode_v: wgpu::AddressMode::ClampToEdge,
1055        address_mode_w: wgpu::AddressMode::ClampToEdge,
1056        mag_filter: wgpu::FilterMode::Linear,
1057        min_filter: wgpu::FilterMode::Linear,
1058        mipmap_filter: wgpu::FilterMode::Linear,
1059        ..Default::default()
1060    });
1061    Ok((tex, view, samp))
1062}

159pub struct GpuMesh {
160    // Full interleaved vertex buffer (MeshVertex layout)
161    pub vertex_full: Buffer,
162    // Position-only vertex buffer (Float32x3 stride) for pipelines that only consume positions
163    pub vertex_pos: Buffer,
164    pub index: Buffer,
165    pub index_count: u32,
166    pub aabb: Option<(Vec3, Vec3)>,
167}
168
169impl MeshRegistry {
170    pub fn new() -> Self {
171        Self {
172            next_id: 1,
173            map: HashMap::new(),
174            uploads: HashMap::new(),
175        }
176    }
177
178    pub fn get(&self, key: &MeshKey) -> Option<MeshHandle> {
179        self.map.get(key).copied()
180    }
181
182    pub fn fetch_or_upload(
183        &mut self,
184        device: &Device,
185        _queue: &Queue,
186        key: MeshKey,
187        mesh: &CpuMesh,
188    ) -> Result<MeshHandle> {
189        if let Some(h) = self.map.get(&key).copied() {
190            return Ok(h);
191        }
192        let handle = MeshHandle(self.next_id);
193        self.next_id += 1;
194
195        let vertex_full = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
196            label: Some(&format!("mesh-vertex-full-{}", handle.0)),
197            contents: bytemuck::cast_slice(&mesh.vertices),
198            usage: BufferUsages::VERTEX | BufferUsages::COPY_DST,
199        });
200        let positions: Vec<[f32; 3]> = mesh.vertices.iter().map(|v| v.position).collect();
201        let vertex_pos = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
202            label: Some(&format!("mesh-vertex-pos-{}", handle.0)),
203            contents: bytemuck::cast_slice(&positions),
204            usage: BufferUsages::VERTEX | BufferUsages::COPY_DST,
205        });
206        let index = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
207            label: Some(&format!("mesh-index-{}", handle.0)),
208            contents: bytemuck::cast_slice(&mesh.indices),
209            usage: BufferUsages::INDEX | BufferUsages::COPY_DST,
210        });
211        let gpu = GpuMesh {
212            vertex_full,
213            vertex_pos,
214            index,
215            index_count: mesh.indices.len() as u32,
216            aabb: mesh.aabb(),
217        };
218        self.map.insert(key, handle);
219        self.uploads.insert(handle, gpu);
220        Ok(handle)
221    }
222
223    pub fn get_gpu(&self, handle: MeshHandle) -> Option<&GpuMesh> {
224        self.uploads.get(&handle)
225    }

152pub struct GpuMesh {
153    // Full interleaved vertex buffer (MeshVertex layout)
154    pub vertex_full: Buffer,
155    // Position-only vertex buffer (Float32x3 stride) for pipelines that only consume positions
156    pub vertex_pos: Buffer,
157    pub index: Buffer,
158    pub index_count: u32,
159    pub aabb: Option<(Vec3, Vec3)>,
160}
161
162impl MeshRegistry {
163    pub fn new() -> Self {
164        Self {
165            next_id: 1,
166            map: HashMap::new(),
167            uploads: HashMap::new(),
168        }
169    }
170
171    pub fn get(&self, key: &MeshKey) -> Option<MeshHandle> {
172        self.map.get(key).copied()
173    }
174
175    pub fn fetch_or_upload(
176        &mut self,
177        device: &Device,
178        _queue: &Queue,
179        key: MeshKey,
180        mesh: &CpuMesh,
181    ) -> Result<MeshHandle> {
182        if let Some(h) = self.map.get(&key).copied() {
183            return Ok(h);
184        }
185        let handle = MeshHandle(self.next_id);
186        self.next_id += 1;
187
188        let vertex_full = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
189            label: Some(&format!("mesh-vertex-full-{}", handle.0)),
190            contents: bytemuck::cast_slice(&mesh.vertices),
191            usage: BufferUsages::VERTEX | BufferUsages::COPY_DST,
192        });
193        let positions: Vec<[f32; 3]> = mesh.vertices.iter().map(|v| v.position).collect();
194        let vertex_pos = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
195            label: Some(&format!("mesh-vertex-pos-{}", handle.0)),
196            contents: bytemuck::cast_slice(&positions),
197            usage: BufferUsages::VERTEX | BufferUsages::COPY_DST,
198        });
199        let index = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
200            label: Some(&format!("mesh-index-{}", handle.0)),
201            contents: bytemuck::cast_slice(&mesh.indices),
202            usage: BufferUsages::INDEX | BufferUsages::COPY_DST,
203        });
204        let gpu = GpuMesh {
205            vertex_full,
206            vertex_pos,
207            index,
208            index_count: mesh.indices.len() as u32,
209            aabb: mesh.aabb(),
210        };
211        self.map.insert(key, handle);
212        self.uploads.insert(handle, gpu);
213        Ok(handle)
214    }
215
216    pub fn get_gpu(&self, handle: MeshHandle) -> Option<&GpuMesh> {
217        self.uploads.get(&handle)
218    }

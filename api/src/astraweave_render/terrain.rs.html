25/// Terrain rendering system that integrates with WorldGenerator
26pub struct TerrainRenderer {
27    world_generator: WorldGenerator,
28    loaded_meshes: std::collections::HashMap<astraweave_terrain::ChunkId, TerrainMesh>,
29    chunk_size: f32,
30}

32impl TerrainRenderer {
33    /// Create a new terrain renderer
34    pub fn new(world_config: WorldConfig) -> Self {
35        let chunk_size = world_config.chunk_size;
36        let world_generator = WorldGenerator::new(world_config);
37
38        Self {
39            world_generator,
40            loaded_meshes: std::collections::HashMap::new(),
41            chunk_size,
42        }
43    }

45    /// Generate or get a terrain mesh for the given chunk
46    pub fn get_or_generate_chunk_mesh(
47        &mut self,
48        chunk_id: astraweave_terrain::ChunkId,
49    ) -> anyhow::Result<&TerrainMesh> {
50        if !self.loaded_meshes.contains_key(&chunk_id) {
51            let chunk = self.world_generator.generate_chunk(chunk_id)?;
52            let mesh = self.create_terrain_mesh(&chunk)?;
53            self.loaded_meshes.insert(chunk_id, mesh);
54        }
55
56        Ok(self
57            .loaded_meshes
58            .get(&chunk_id)
59            .expect("BUG: chunk should exist after insert above"))
60    }
61
62    /// Generate a complete chunk with vegetation and resources
63    pub fn generate_chunk_complete(
64        &mut self,
65        chunk_id: astraweave_terrain::ChunkId,
66    ) -> anyhow::Result<(TerrainMesh, ScatterResult)> {
67        let (chunk, scatter_result) = self.world_generator.generate_chunk_with_scatter(chunk_id)?;
68        let mesh = self.create_terrain_mesh(&chunk)?;
69        Ok((mesh, scatter_result))
70    }

72    /// Create a terrain mesh from a terrain chunk
73    fn create_terrain_mesh(&self, chunk: &TerrainChunk) -> anyhow::Result<TerrainMesh> {
74        let heightmap = chunk.heightmap();
75        let resolution = heightmap.resolution();
76        let chunk_origin = chunk.id().to_world_pos(self.chunk_size);
77
78        let mut vertices = Vec::new();
79        let step = self.chunk_size / (resolution - 1) as f32;
80
81        // Generate vertices
82        for z in 0..resolution {
83            for x in 0..resolution {
84                let world_x = chunk_origin.x + x as f32 * step;
85                let world_z = chunk_origin.z + z as f32 * step;
86                let height = heightmap.get_height(x, z);
87
88                let position = [world_x, height, world_z];
89
90                // Calculate normal
91                let normal = heightmap.calculate_normal(x, z, step);
92                let normal_array = [normal.x, normal.y, normal.z];
93
94                // UV coordinates
95                let u = x as f32 / (resolution - 1) as f32;
96                let v = z as f32 / (resolution - 1) as f32;
97                let uv = [u, v];
98
99                // Get biome at this position
100                let biome_index = z as usize * resolution as usize + x as usize;
101                let biome = chunk
102                    .biome_map()
103                    .get(biome_index)
104                    .copied()
105                    .unwrap_or(astraweave_terrain::BiomeType::Grassland);
106                let biome_id = Self::biome_to_id(biome);
107
108                vertices.push(TerrainVertex {
109                    position,
110                    normal: normal_array,
111                    uv,
112                    biome_id,
113                });
114            }
115        }
116
117        // Generate indices
118        let indices = heightmap.generate_indices();
119
120        Ok(TerrainMesh {
121            vertices,
122            indices,
123            chunk_id: chunk.id(),
124        })
125    }

127    /// Convert biome type to numeric ID for shading
128    fn biome_to_id(biome: astraweave_terrain::BiomeType) -> u32 {
129        match biome {
130            astraweave_terrain::BiomeType::Grassland => 0,
131            astraweave_terrain::BiomeType::Desert => 1,
132            astraweave_terrain::BiomeType::Forest => 2,
133            astraweave_terrain::BiomeType::Mountain => 3,
134            astraweave_terrain::BiomeType::Tundra => 4,
135            astraweave_terrain::BiomeType::Swamp => 5,
136            astraweave_terrain::BiomeType::Beach => 6,
137            astraweave_terrain::BiomeType::River => 7,
138        }
139    }

141    /// Get chunks in a radius around a center position
142    pub fn get_chunks_in_radius(
143        &mut self,
144        center: Vec3,
145        radius: u32,
146    ) -> anyhow::Result<Vec<astraweave_terrain::ChunkId>> {
147        let chunks_needed =
148            astraweave_terrain::ChunkId::get_chunks_in_radius(center, radius, self.chunk_size);
149
150        // Generate all needed chunks first
151        for chunk_id in &chunks_needed {
152            if !self.loaded_meshes.contains_key(chunk_id) {
153                self.get_or_generate_chunk_mesh(*chunk_id)?;
154            }
155        }
156
157        Ok(chunks_needed)
158    }

162        self.loaded_meshes.get(&chunk_id)

165    /// Create GPU buffers for a terrain mesh
166    pub fn create_gpu_buffers(
167        device: &wgpu::Device,
168        mesh: &TerrainMesh,
169    ) -> (wgpu::Buffer, wgpu::Buffer) {
170        let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
171            label: Some(&format!("Terrain Vertex Buffer {:?}", mesh.chunk_id)),
172            contents: bytemuck::cast_slice(&mesh.vertices),
173            usage: wgpu::BufferUsages::VERTEX,
174        });
175
176        let index_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
177            label: Some(&format!("Terrain Index Buffer {:?}", mesh.chunk_id)),
178            contents: bytemuck::cast_slice(&mesh.indices),
179            usage: wgpu::BufferUsages::INDEX,
180        });
181
182        (vertex_buffer, index_buffer)
183    }

185    /// Create simple cube instances for vegetation (placeholder until proper models)
186    pub fn create_vegetation_instances(
187        scatter_result: &ScatterResult,
188    ) -> Vec<VegetationRenderInstance> {
189        scatter_result
190            .vegetation
191            .iter()
192            .map(|veg| {
193                let transform_matrix = Mat4::from_scale_rotation_translation(
194                    Vec3::splat(veg.scale),
195                    glam::Quat::from_rotation_y(veg.rotation),
196                    veg.position,
197                );
198                VegetationRenderInstance {
199                    transform: transform_matrix.to_cols_array(),
200                    vegetation_type: Self::vegetation_type_to_id(&veg.vegetation_type),
201                }
202            })
203            .collect()
204    }

244/// Simple terrain preview generator for testing
245pub fn generate_terrain_preview(
246    world_config: &WorldConfig,
247    center: Vec3,
248    size: u32,
249) -> anyhow::Result<Vec<f32>> {
250    let mut generator = WorldGenerator::new(world_config.clone());
251    let chunk_id = astraweave_terrain::ChunkId::from_world_pos(center, world_config.chunk_size);
252    let chunk = generator.generate_chunk(chunk_id)?;
253
254    let heightmap = chunk.heightmap();
255    let mut preview = Vec::with_capacity((size * size) as usize);
256
257    let step = world_config.chunk_size / size as f32;
258    for z in 0..size {
259        for x in 0..size {
260            let u = x as f32 * step / world_config.chunk_size * (heightmap.resolution() - 1) as f32;
261            let v = z as f32 * step / world_config.chunk_size * (heightmap.resolution() - 1) as f32;
262            let height = heightmap.sample_bilinear(u, v);
263            preview.push(height);
264        }
265    }
266
267    Ok(preview)
268}

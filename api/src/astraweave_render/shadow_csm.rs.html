1//! Cascaded Shadow Mapping (CSM) Implementation
2//!
3//! This module implements a 4-cascade shadow mapping system for directional lights
4//! (e.g., sun/moon). Each cascade covers a different depth range from the camera,
5//! providing high detail near the camera and broader coverage far away.
6//!
7//! # Architecture
8//!
9//! - **Shadow Atlas**: Single 8192×8192 texture containing all 4 cascades (2048×2048 each)
10//! - **Cascade Splitting**: Logarithmic distribution based on view frustum depth
11//! - **Sampling**: PCF (Percentage Closer Filtering) for soft shadows
12//! - **Bias**: Depth bias to prevent shadow acne
13//!
14//! # Performance Targets (60 FPS = 16.67ms budget)
15//!
16//! - Shadow map rendering: <2.0ms (4 cascades × 0.5ms each)
17//! - Cascade selection: <0.01ms (per-pixel, branchless)
18//! - PCF sampling: <0.5ms (5×5 kernel, optimized)
19//! - Total shadow budget: <2.5ms (15% of frame)
20

80/// Shadow cascade configuration
81///
82/// Each cascade covers a depth range [near, far] in view space.
83/// The view-projection matrix transforms world coords to shadow map space.
84#[derive(Debug, Clone, Copy)]
85pub struct ShadowCascade {
86    /// Near plane distance (view space)
87    pub near: f32,
88    /// Far plane distance (view space)
89    pub far: f32,
90    /// View matrix (light space)
91    pub view_matrix: Mat4,
92    /// Projection matrix (orthographic, light space)
93    pub proj_matrix: Mat4,
94    /// Combined view-projection matrix (for shader upload)
95    pub view_proj_matrix: Mat4,
96    /// Atlas offset (UV coords: 0.0-0.5 for each quadrant)
97    pub atlas_offset: Vec4, // (offset_x, offset_y, scale_x, scale_y)
98}

100/// GPU-compatible shadow cascade data (uploaded to shader)
101#[repr(C)]

103pub struct GpuShadowCascade {

438    /// Update cascade splits and matrices based on camera frustum
439    ///
440    /// # Arguments
441    ///
442    /// - `camera_pos`: Camera world position
443    /// - `camera_view`: Camera view matrix
444    /// - `camera_proj`: Camera projection matrix
445    /// - `light_dir`: Directional light direction (normalized, pointing TOWARD light)
446    /// - `near`: Camera near plane
447    /// - `far`: Camera far plane
448    ///
449    /// # Algorithm
450    ///
451    /// Uses logarithmic splitting with λ=0.5 (balance between uniform and logarithmic):
452    /// ```text
453    /// split[i] = λ * (near * (far/near)^(i/N)) + (1-λ) * (near + (far-near) * i/N)
454    /// ```
455    pub fn update_cascades(
456        &mut self,
457        _camera_pos: Vec3,
458        _camera_view: Mat4,
459        _camera_proj: Mat4,
460        light_dir: Vec3,
461        near: f32,
462        far: f32,
463    ) {
464        // Calculate cascade split distances (logarithmic distribution)
465        let lambda = 0.5; // Balance between uniform (0.0) and logarithmic (1.0)
466        let mut split_distances = [0.0f32; CASCADE_COUNT + 1];
467        split_distances[0] = near;
468        split_distances[CASCADE_COUNT] = far;
469
470        for i in 1..CASCADE_COUNT {
471            let i_f = i as f32;
472            let n_f = CASCADE_COUNT as f32;
473
474            // Logarithmic split
475            let log_split = near * (far / near).powf(i_f / n_f);
476
477            // Uniform split
478            let uniform_split = near + (far - near) * (i_f / n_f);
479
480            // Blend
481            split_distances[i] = lambda * log_split + (1.0 - lambda) * uniform_split;
482        }
483
484        // Update each cascade
485        for (i, cascade) in self.cascades.iter_mut().enumerate() {
486            cascade.near = split_distances[i];
487            cascade.far = split_distances[i + 1];
488
489            // 🎯 BEVY-STYLE: Compute light view matrix from direction
490            // Directional lights need a view matrix that looks FROM light TO scene
491            let light_distance = 50.0; // How far back to place the light
492            let scene_center = Vec3::ZERO; // Look at origin
493            let light_pos = scene_center - (light_dir.normalize() * light_distance);
494
495            // Choose up vector perpendicular to light direction
496            // If light is mostly vertical (|light_dir.y| > 0.9), use X as up
497            // Otherwise use Y as up (standard)
498            let up = if light_dir.y.abs() > 0.9 {
499                Vec3::X
500            } else {
501                Vec3::Y
502            };
503
504            cascade.view_matrix = Mat4::look_at_rh(light_pos, scene_center, up);
505
506            // FIXED: Use large enough orthographic bounds to cover entire test scene
507            // Scene bounds: Ground is 20×20 at origin, cubes at ±5, ±15, ±25
508            // Need at least -30 to +30 to capture everything
509            let ortho_size = 35.0; // Large enough for test scene (can optimize per-cascade later)
510
511            // DEBUG: Print ortho_size on first call
512            if i == 0 {
513                static FIRST_UPDATE: std::sync::atomic::AtomicBool =
514                    std::sync::atomic::AtomicBool::new(true);
515                if FIRST_UPDATE.swap(false, std::sync::atomic::Ordering::Relaxed) {
516                    println!(
517                        "🔍 Light frustum: ortho_size = {}, covers [{}, {}] in X and Z",
518                        ortho_size, -ortho_size, ortho_size
519                    );
520                }
521            }
522
523            cascade.proj_matrix = Mat4::orthographic_rh(
524                -ortho_size,
525                ortho_size,
526                -ortho_size,
527                ortho_size,
528                0.1,   // Near (light space)
529                100.0, // Far (light space)
530            );
531
532            cascade.view_proj_matrix = cascade.proj_matrix * cascade.view_matrix;
533        }
534    }

536    /// Upload cascade data to GPU
537    ///
538    /// Call this after `update_cascades()` and before rendering.
539    pub fn upload_to_gpu(&mut self, queue: &wgpu::Queue, device: &wgpu::Device) {
540        // Convert to GPU format
541        let gpu_cascades: Vec<GpuShadowCascade> =
542            self.cascades.iter().map(GpuShadowCascade::from).collect();
543
544        // Upload buffer
545        queue.write_buffer(&self.cascade_buffer, 0, bytemuck::cast_slice(&gpu_cascades));
546
547        // Create/update bind group
548        self.bind_group = Some(device.create_bind_group(&wgpu::BindGroupDescriptor {
549            label: Some("CSM Bind Group"),
550            layout: &self.bind_group_layout,
551            entries: &[
552                wgpu::BindGroupEntry {
553                    binding: 0,
554                    resource: wgpu::BindingResource::TextureView(&self.atlas_view),
555                },
556                wgpu::BindGroupEntry {
557                    binding: 1,
558                    resource: wgpu::BindingResource::Sampler(&self.shadow_sampler),
559                },
560                wgpu::BindGroupEntry {
561                    binding: 2,
562                    resource: self.cascade_buffer.as_entire_binding(),
563                },
564            ],
565        }));
566
567        // Create/update shadow-only bind group (just cascades for depth pass)
568        self.shadow_bind_group = Some(device.create_bind_group(&wgpu::BindGroupDescriptor {
569            label: Some("Shadow Depth Bind Group"),
570            layout: &self.shadow_bind_group_layout,
571            entries: &[wgpu::BindGroupEntry {
572                binding: 0,
573                resource: self.cascade_buffer.as_entire_binding(),
574            }],
575        }));
576    }

625                depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
626                    view: &self.cascade_views[cascade_idx], // Render to individual layer
627                    depth_ops: Some(wgpu::Operations {
628                        load: wgpu::LoadOp::Clear(1.0), // Clear to far plane
629                        store: wgpu::StoreOp::Store,
630                    }),
631                    stencil_ops: None,
632                }),
633                timestamp_writes: None,
634                occlusion_query_set: None,
635            });
636
637            render_pass.set_pipeline(&self.shadow_pipeline);
638
639            // Bind cascade data (group 0 for shadow pipeline)
640            if let Some(ref bind_group) = self.shadow_bind_group {
641                render_pass.set_bind_group(0, bind_group, &[]);
642            }
643
644            render_pass.set_vertex_buffer(0, vertex_buffer.slice(..));
645            render_pass.set_index_buffer(index_buffer.slice(..), wgpu::IndexFormat::Uint32);
646
647            // Draw with instance_index = cascade_idx (for shader cascade selection)
648            render_pass.draw_indexed(
649                0..index_count,
650                0,
651                cascade_idx as u32..(cascade_idx as u32 + 1),
652            );
653        }
654    }
655}

672    #[test]
673    fn test_cascade_split_distribution() {
674        // Test logarithmic split calculation
675        let near = 0.1f32;
676        let far = 1000.0f32;
677        let lambda = 0.5;
678
679        let mut splits = [0.0f32; CASCADE_COUNT + 1];
680        splits[0] = near;
681        splits[CASCADE_COUNT] = far;
682
683        for i in 1..CASCADE_COUNT {
684            let i_f = i as f32;
685            let n_f = CASCADE_COUNT as f32;
686            let log_split = near * (far / near).powf(i_f / n_f);
687            let uniform_split = near + (far - near) * (i_f / n_f);
688            splits[i] = lambda * log_split + (1.0 - lambda) * uniform_split;
689        }
690
691        // Verify splits are monotonically increasing
692        for i in 0..CASCADE_COUNT {
693            assert!(splits[i] < splits[i + 1]);
694        }
695
696        // Verify first and last splits
697        assert_eq!(splits[0], near);
698        assert_eq!(splits[CASCADE_COUNT], far);
699    }

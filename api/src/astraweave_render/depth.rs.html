7impl Depth {
8    pub fn create(device: &wgpu::Device, config: &wgpu::SurfaceConfiguration) -> Self {
9        let format = wgpu::TextureFormat::Depth32Float;
10        let size = wgpu::Extent3d {
11            width: config.width,
12            height: config.height,
13            depth_or_array_layers: 1,
14        };
15        let desc = wgpu::TextureDescriptor {
16            label: Some("depth"),
17            size,
18            mip_level_count: 1,
19            sample_count: 1,
20            dimension: wgpu::TextureDimension::D2,
21            format,
22            usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::TEXTURE_BINDING,
23            view_formats: &[],
24        };
25        let texture = device.create_texture(&desc);
26        let view = texture.create_view(&wgpu::TextureViewDescriptor::default());
27        Self {
28            texture,
29            view,
30            format,
31        }
32    }
33}

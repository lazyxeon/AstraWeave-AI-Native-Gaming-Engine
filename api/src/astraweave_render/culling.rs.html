10/// Per-instance data for culling compute shader
11/// Represents an axis-aligned bounding box (AABB) in world space
12/// Layout must match WGSL std140: vec3 aligned to 16 bytes
13#[repr(C)]
14#[derive(Copy, Clone, Debug, Pod, Zeroable)]
15pub struct InstanceAABB {
16    /// Center of the AABB in world space (vec3<f32> in WGSL = 16 bytes with padding)
17    pub center: [f32; 3],
18    pub _pad0: u32, // Padding to align extent to 16-byte boundary
19    /// Half-extents (radius) from center (vec3<f32> in WGSL = 16 bytes with padding)
20    pub extent: [f32; 3],
21    /// Original instance index in the instance buffer
22    pub instance_index: u32,
23}
24
25impl InstanceAABB {
26    pub fn new(center: Vec3, extent: Vec3, instance_index: u32) -> Self {
27        Self {
28            center: center.to_array(),
29            _pad0: 0,
30            extent: extent.to_array(),
31            instance_index,
32        }
33    }
34
35    /// Compute AABB from world transform matrix and local bounds
36    pub fn from_transform(
37        transform: &Mat4,
38        local_min: Vec3,
39        local_max: Vec3,
40        instance_index: u32,
41    ) -> Self {
42        let local_center = (local_min + local_max) * 0.5;
43        let local_extent = (local_max - local_min) * 0.5;
44
45        // Transform center to world space
46        let world_center = transform.transform_point3(local_center);
47
48        // For extent, transform all 8 corners and compute new AABB
49        // This handles rotation properly
50        let corners = [
51            Vec3::new(-1.0, -1.0, -1.0),
52            Vec3::new(1.0, -1.0, -1.0),
53            Vec3::new(-1.0, 1.0, -1.0),
54            Vec3::new(1.0, 1.0, -1.0),
55            Vec3::new(-1.0, -1.0, 1.0),
56            Vec3::new(1.0, -1.0, 1.0),
57            Vec3::new(-1.0, 1.0, 1.0),
58            Vec3::new(1.0, 1.0, 1.0),
59        ];
60
61        let mut world_min = Vec3::splat(f32::MAX);
62        let mut world_max = Vec3::splat(f32::MIN);
63
64        for corner in &corners {
65            let local_point = local_center + *corner * local_extent;
66            let world_point = transform.transform_point3(local_point);
67            world_min = world_min.min(world_point);
68            world_max = world_max.max(world_point);
69        }
70
71        let world_extent = (world_max - world_min) * 0.5;
72
73        Self {
74            center: world_center.to_array(),
75            _pad0: 0,
76            extent: world_extent.to_array(),
77            instance_index,
78        }
79    }
80}

82/// Frustum planes in world space for culling
83/// Planes are stored as (nx, ny, nz, d) where dot(normal, point) + d = 0 defines the plane
84#[repr(C)]
85#[derive(Copy, Clone, Debug, Pod, Zeroable)]
86pub struct FrustumPlanes {
87    /// Six planes: left, right, bottom, top, near, far
88    /// Each plane: [nx, ny, nz, d]
89    pub planes: [[f32; 4]; 6],
90}
91
92impl FrustumPlanes {
93    /// Extract frustum planes from view-projection matrix
94    /// Uses Gribb-Hartmann method
95    pub fn from_view_proj(view_proj: &Mat4) -> Self {
96        let m = view_proj.to_cols_array();
97
98        // Extract planes from matrix rows
99        let left = Self::normalize_plane([m[3] + m[0], m[7] + m[4], m[11] + m[8], m[15] + m[12]]);
100        let right = Self::normalize_plane([m[3] - m[0], m[7] - m[4], m[11] - m[8], m[15] - m[12]]);
101        let bottom = Self::normalize_plane([m[3] + m[1], m[7] + m[5], m[11] + m[9], m[15] + m[13]]);
102        let top = Self::normalize_plane([m[3] - m[1], m[7] - m[5], m[11] - m[9], m[15] - m[13]]);
103        let near = Self::normalize_plane([m[3] + m[2], m[7] + m[6], m[11] + m[10], m[15] + m[14]]);
104        let far = Self::normalize_plane([m[3] - m[2], m[7] - m[6], m[11] - m[10], m[15] - m[14]]);
105
106        Self {
107            planes: [left, right, bottom, top, near, far],
108        }
109    }
110
111    fn normalize_plane(plane: [f32; 4]) -> [f32; 4] {
112        let len = (plane[0] * plane[0] + plane[1] * plane[1] + plane[2] * plane[2]).sqrt();
113        if len > 0.0 {
114            [
115                plane[0] / len,
116                plane[1] / len,
117                plane[2] / len,
118                plane[3] / len,
119            ]
120        } else {
121            plane
122        }
123    }
124
125    /// CPU-based frustum culling (fallback path)
126    /// Returns true if AABB intersects frustum
127    pub fn test_aabb(&self, center: Vec3, extent: Vec3) -> bool {
128        for plane in &self.planes {
129            let normal = Vec3::new(plane[0], plane[1], plane[2]);
130            let d = plane[3];
131
132            // Compute signed distance from center to plane
133            let dist = normal.dot(center) + d;
134
135            // Compute the effective radius along the plane normal
136            let radius = extent.x.abs() * normal.x.abs()
137                + extent.y.abs() * normal.y.abs()
138                + extent.z.abs() * normal.z.abs();
139
140            // If center is more than radius away on the negative side, AABB is outside
141            if dist < -radius {
142                return false;
143            }
144        }
145        true
146    }
147}

222/// CPU fallback culling implementation
223/// Returns list of visible instance indices
224pub fn cpu_frustum_cull(instances: &[InstanceAABB], frustum: &FrustumPlanes) -> Vec<u32> {
225    instances
226        .iter()
227        .filter(|inst| {
228            let center = Vec3::from(inst.center);
229            let extent = Vec3::from(inst.extent);
230            frustum.test_aabb(center, extent)
231        })
232        .map(|inst| inst.instance_index)
233        .collect()
234}

252/// Group visible instances into batches by mesh/material (CPU path)
253/// This is a simplified version - real implementation would use scene data
254pub fn batch_visible_instances(
255    visible_indices: &[u32],
256    get_batch_id: impl Fn(u32) -> BatchId,
257    get_mesh_info: impl Fn(BatchId) -> (u32, u32), // Returns (vertex_count, first_vertex)
258) -> Vec<DrawBatch> {
259    use std::collections::BTreeMap;
260
261    let mut batch_map: BTreeMap<BatchId, DrawBatch> = BTreeMap::new();
262
263    for &instance_idx in visible_indices {
264        let batch_id = get_batch_id(instance_idx);
265
266        batch_map
267            .entry(batch_id)
268            .or_insert_with(|| {
269                let (vertex_count, first_vertex) = get_mesh_info(batch_id);
270                DrawBatch::new(batch_id, vertex_count, first_vertex)
271            })
272            .add_instance(instance_idx);
273    }
274
275    batch_map.into_values().collect()
276}

278/// Compute shader source for frustum culling
279pub const CULLING_SHADER: &str = r#"
280// Per-instance AABB data (std140 layout: vec3 is 16-byte aligned)
281struct InstanceAABB {
282    center: vec3<f32>,      // offset 0, size 12, align 16
283    _pad0: u32,             // offset 12
284    extent: vec3<f32>,      // offset 16, size 12, align 16
285    instance_index: u32,    // offset 28
286}
287// Total size: 32 bytes
288
289// Frustum planes (6 planes)
290struct FrustumPlanes {
291    planes: array<vec4<f32>, 6>,
292}
293
294// Input/output bindings
295@group(0) @binding(0) var<storage, read> instance_aabbs: array<InstanceAABB>;
296@group(0) @binding(1) var<uniform> frustum: FrustumPlanes;
297@group(0) @binding(2) var<storage, read_write> visible_instances: array<u32>;
298@group(0) @binding(3) var<storage, read_write> visible_count: atomic<u32>;
299
300// Test if AABB is visible against frustum
301fn is_aabb_visible(center: vec3<f32>, extent: vec3<f32>, frustum: FrustumPlanes) -> bool {
302    // Test AABB against all 6 frustum planes (manually unrolled - WGSL requires constant indices)
303    // Plane 0 (left)
304    var plane = frustum.planes[0];
305    var normal = plane.xyz;
306    var d = plane.w;
307    var dist = dot(center, normal) + d;
308    var radius = abs(extent.x) * abs(normal.x) + abs(extent.y) * abs(normal.y) + abs(extent.z) * abs(normal.z);
309    if (dist < -radius) { return false; }
310    
311    // Plane 1 (right)
312    plane = frustum.planes[1];
313    normal = plane.xyz;
314    d = plane.w;
315    dist = dot(center, normal) + d;
316    radius = abs(extent.x) * abs(normal.x) + abs(extent.y) * abs(normal.y) + abs(extent.z) * abs(normal.z);
317    if (dist < -radius) { return false; }
318    
319    // Plane 2 (bottom)
320    plane = frustum.planes[2];
321    normal = plane.xyz;
322    d = plane.w;
323    dist = dot(center, normal) + d;
324    radius = abs(extent.x) * abs(normal.x) + abs(extent.y) * abs(normal.y) + abs(extent.z) * abs(normal.z);
325    if (dist < -radius) { return false; }
326    
327    // Plane 3 (top)
328    plane = frustum.planes[3];
329    normal = plane.xyz;
330    d = plane.w;
331    dist = dot(center, normal) + d;
332    radius = abs(extent.x) * abs(normal.x) + abs(extent.y) * abs(normal.y) + abs(extent.z) * abs(normal.z);
333    if (dist < -radius) { return false; }
334    
335    // Plane 4 (near)
336    plane = frustum.planes[4];
337    normal = plane.xyz;
338    d = plane.w;
339    dist = dot(center, normal) + d;
340    radius = abs(extent.x) * abs(normal.x) + abs(extent.y) * abs(normal.y) + abs(extent.z) * abs(normal.z);
341    if (dist < -radius) { return false; }
342    
343    // Plane 5 (far)
344    plane = frustum.planes[5];
345    normal = plane.xyz;
346    d = plane.w;
347    dist = dot(center, normal) + d;
348    radius = abs(extent.x) * abs(normal.x) + abs(extent.y) * abs(normal.y) + abs(extent.z) * abs(normal.z);
349    if (dist < -radius) { return false; }
350    
351    return true;
352}
353
354@compute @workgroup_size(64)
355fn main(@builtin(global_invocation_id) gid: vec3<u32>) {
356    let idx = gid.x;
357    if (idx >= arrayLength(&instance_aabbs)) { 
358        return; 
359    }
360    
361    let aabb = instance_aabbs[idx];
362    if (is_aabb_visible(aabb.center, aabb.extent, frustum)) {
363        let slot = atomicAdd(&visible_count, 1u);
364        visible_instances[slot] = aabb.instance_index;
365    }
366}
367"#;

369/// GPU culling manager
370pub struct CullingPipeline {
371    pipeline: wgpu::ComputePipeline,
372    bind_group_layout: wgpu::BindGroupLayout,
373}
374
375impl CullingPipeline {
376    pub fn new(device: &wgpu::Device) -> Self {
377        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
378            label: Some("frustum_culling_shader"),
379            source: wgpu::ShaderSource::Wgsl(CULLING_SHADER.into()),
380        });
381
382        let bind_group_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
383            label: Some("culling_bind_group_layout"),
384            entries: &[
385                // Instance AABBs (storage buffer)
386                wgpu::BindGroupLayoutEntry {
387                    binding: 0,
388                    visibility: wgpu::ShaderStages::COMPUTE,
389                    ty: wgpu::BindingType::Buffer {
390                        ty: wgpu::BufferBindingType::Storage { read_only: true },
391                        has_dynamic_offset: false,
392                        min_binding_size: None,
393                    },
394                    count: None,
395                },
396                // Frustum planes (uniform buffer)
397                wgpu::BindGroupLayoutEntry {
398                    binding: 1,
399                    visibility: wgpu::ShaderStages::COMPUTE,
400                    ty: wgpu::BindingType::Buffer {
401                        ty: wgpu::BufferBindingType::Uniform,
402                        has_dynamic_offset: false,
403                        min_binding_size: None,
404                    },
405                    count: None,
406                },
407                // Visible instances output (storage buffer)
408                wgpu::BindGroupLayoutEntry {
409                    binding: 2,
410                    visibility: wgpu::ShaderStages::COMPUTE,
411                    ty: wgpu::BindingType::Buffer {
412                        ty: wgpu::BufferBindingType::Storage { read_only: false },
413                        has_dynamic_offset: false,
414                        min_binding_size: None,
415                    },
416                    count: None,
417                },
418                // Visible count (atomic storage buffer)
419                wgpu::BindGroupLayoutEntry {
420                    binding: 3,
421                    visibility: wgpu::ShaderStages::COMPUTE,
422                    ty: wgpu::BindingType::Buffer {
423                        ty: wgpu::BufferBindingType::Storage { read_only: false },
424                        has_dynamic_offset: false,
425                        min_binding_size: None,
426                    },
427                    count: None,
428                },
429            ],
430        });
431
432        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
433            label: Some("culling_pipeline_layout"),
434            bind_group_layouts: &[&bind_group_layout],
435            push_constant_ranges: &[],
436        });
437
438        let pipeline = device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
439            label: Some("frustum_culling_pipeline"),
440            layout: Some(&pipeline_layout),
441            module: &shader,
442            entry_point: Some("main"),
443            compilation_options: Default::default(),
444            cache: None,
445        });
446
447        Self {
448            pipeline,
449            bind_group_layout,
450        }
451    }
452
453    pub fn bind_group_layout(&self) -> &wgpu::BindGroupLayout {
454        &self.bind_group_layout
455    }
456
457    /// Create buffers and bind group for culling
458    pub fn create_culling_resources(
459        &self,
460        device: &wgpu::Device,
461        instances: &[InstanceAABB],
462        frustum: &FrustumPlanes,
463    ) -> CullingResources {
464        // Handle empty instance list with minimum buffer size
465        let instance_data = if instances.is_empty() {
466            vec![InstanceAABB::new(Vec3::ZERO, Vec3::ZERO, 0)]
467        } else {
468            instances.to_vec()
469        };
470
471        let instance_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
472            label: Some("culling_instance_buffer"),
473            contents: bytemuck::cast_slice(&instance_data),
474            usage: wgpu::BufferUsages::STORAGE,
475        });
476
477        let frustum_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
478            label: Some("culling_frustum_buffer"),
479            contents: bytemuck::bytes_of(frustum),
480            usage: wgpu::BufferUsages::UNIFORM,
481        });
482
483        // Allocate buffer for visible instances (max size = input size, min 1 element)
484        let buffer_size = (instances.len().max(1) * std::mem::size_of::<u32>()) as u64;
485        let visible_buffer = device.create_buffer(&wgpu::BufferDescriptor {
486            label: Some("culling_visible_buffer"),
487            size: buffer_size,
488            usage: wgpu::BufferUsages::STORAGE | wgpu::BufferUsages::COPY_SRC,
489            mapped_at_creation: false,
490        });
491
492        // Counter for visible instances (atomic)
493        let count_buffer = device.create_buffer(&wgpu::BufferDescriptor {
494            label: Some("culling_count_buffer"),
495            size: std::mem::size_of::<u32>() as u64,
496            usage: wgpu::BufferUsages::STORAGE
497                | wgpu::BufferUsages::COPY_SRC
498                | wgpu::BufferUsages::COPY_DST,
499            mapped_at_creation: false,
500        });
501
502        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
503            label: Some("culling_bind_group"),
504            layout: &self.bind_group_layout,
505            entries: &[
506                wgpu::BindGroupEntry {
507                    binding: 0,
508                    resource: instance_buffer.as_entire_binding(),
509                },
510                wgpu::BindGroupEntry {
511                    binding: 1,
512                    resource: frustum_buffer.as_entire_binding(),
513                },
514                wgpu::BindGroupEntry {
515                    binding: 2,
516                    resource: visible_buffer.as_entire_binding(),
517                },
518                wgpu::BindGroupEntry {
519                    binding: 3,
520                    resource: count_buffer.as_entire_binding(),
521                },
522            ],
523        });
524
525        CullingResources {
526            instance_buffer,
527            frustum_buffer,
528            visible_buffer,
529            count_buffer,
530            bind_group,
531        }
532    }
533
534    /// Execute culling compute pass
535    /// Note: Caller must ensure count_buffer is cleared to 0 before calling this
536    pub fn execute(
537        &self,
538        encoder: &mut wgpu::CommandEncoder,
539        bind_group: &wgpu::BindGroup,
540        instance_count: u32,
541    ) {
542        let mut cpass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
543            label: Some("frustum_culling_pass"),
544            timestamp_writes: None,
545        });
546        cpass.set_pipeline(&self.pipeline);
547        cpass.set_bind_group(0, bind_group, &[]);
548        let workgroup_count = instance_count.div_ceil(64);
549        cpass.dispatch_workgroups(workgroup_count, 1, 1);
550    }
551
552    /// Execute culling with automatic buffer setup (clears count buffer first)
553    pub fn execute_with_clear(
554        &self,
555        encoder: &mut wgpu::CommandEncoder,
556        resources: &CullingResources,
557        instance_count: u32,
558    ) {
559        // Clear count buffer to 0
560        encoder.clear_buffer(&resources.count_buffer, 0, None);
561
562        // Execute culling
563        self.execute(encoder, &resources.bind_group, instance_count);
564    }
565}

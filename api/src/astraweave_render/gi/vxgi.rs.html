1//! Voxel Global Illumination (VXGI) using Voxel Cone Tracing
2//!
3//! This module implements VXGI for dynamic global illumination on voxel terrain.
4//! It uses voxel cone tracing to sample indirect lighting from a sparse voxel
5//! radiance field built from the terrain SVO.
6
7/// Configuration for VXGI
8#[derive(Debug, Clone, Copy)]
9pub struct VxgiConfig {
10    /// Voxel grid resolution (power of 2)
11    pub voxel_resolution: u32,
12    /// World space size covered by voxel grid
13    pub world_size: f32,
14    /// Number of cone samples per pixel
15    pub cone_count: u32,
16    /// Maximum cone tracing distance
17    pub max_trace_distance: f32,
18    /// Cone aperture angle in radians
19    pub cone_aperture: f32,
20}
21
22impl Default for VxgiConfig {
23    fn default() -> Self {
24        Self {
25            voxel_resolution: 256,
26            world_size: 1000.0,
27            cone_count: 6,
28            max_trace_distance: 100.0,
29            cone_aperture: 0.577, // ~33 degrees
30        }
31    }
32}

62impl VxgiRenderer {
63    /// Create a new VXGI renderer
64    pub fn new(device: &wgpu::Device, config: VxgiConfig) -> Self {
65        // Create 3D texture for voxel radiance
66        let voxel_texture = device.create_texture(&wgpu::TextureDescriptor {
67            label: Some("VXGI Voxel Texture"),
68            size: wgpu::Extent3d {
69                width: config.voxel_resolution,
70                height: config.voxel_resolution,
71                depth_or_array_layers: config.voxel_resolution,
72            },
73            mip_level_count: (config.voxel_resolution as f32).log2() as u32 + 1,
74            sample_count: 1,
75            dimension: wgpu::TextureDimension::D3,
76            format: wgpu::TextureFormat::Rgba16Float,
77            usage: wgpu::TextureUsages::STORAGE_BINDING
78                | wgpu::TextureUsages::TEXTURE_BINDING
79                | wgpu::TextureUsages::COPY_DST,
80            view_formats: &[],
81        });
82
83        let voxel_texture_view = voxel_texture.create_view(&wgpu::TextureViewDescriptor {
84            usage: None,
85            label: Some("VXGI Voxel Texture View"),
86            format: Some(wgpu::TextureFormat::Rgba16Float),
87            dimension: Some(wgpu::TextureViewDimension::D3),
88            aspect: wgpu::TextureAspect::All,
89            base_mip_level: 0,
90            mip_level_count: None,
91            base_array_layer: 0,
92            array_layer_count: None,
93        });
94
95        let voxel_sampler = device.create_sampler(&wgpu::SamplerDescriptor {
96            label: Some("VXGI Voxel Sampler"),
97            address_mode_u: wgpu::AddressMode::ClampToEdge,
98            address_mode_v: wgpu::AddressMode::ClampToEdge,
99            address_mode_w: wgpu::AddressMode::ClampToEdge,
100            mag_filter: wgpu::FilterMode::Linear,
101            min_filter: wgpu::FilterMode::Linear,
102            mipmap_filter: wgpu::FilterMode::Linear,
103            ..Default::default()
104        });
105
106        // Create bind group layout
107        let vxgi_bind_group_layout =
108            device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
109                label: Some("VXGI Bind Group Layout"),
110                entries: &[
111                    wgpu::BindGroupLayoutEntry {
112                        binding: 0,
113                        visibility: wgpu::ShaderStages::FRAGMENT | wgpu::ShaderStages::COMPUTE,
114                        ty: wgpu::BindingType::Texture {
115                            sample_type: wgpu::TextureSampleType::Float { filterable: true },
116                            view_dimension: wgpu::TextureViewDimension::D3,
117                            multisampled: false,
118                        },
119                        count: None,
120                    },
121                    wgpu::BindGroupLayoutEntry {
122                        binding: 1,
123                        visibility: wgpu::ShaderStages::FRAGMENT | wgpu::ShaderStages::COMPUTE,
124                        ty: wgpu::BindingType::Sampler(wgpu::SamplerBindingType::Filtering),
125                        count: None,
126                    },
127                ],
128            });
129
130        let vxgi_bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
131            label: Some("VXGI Bind Group"),
132            layout: &vxgi_bind_group_layout,
133            entries: &[
134                wgpu::BindGroupEntry {
135                    binding: 0,
136                    resource: wgpu::BindingResource::TextureView(&voxel_texture_view),
137                },
138                wgpu::BindGroupEntry {
139                    binding: 1,
140                    resource: wgpu::BindingResource::Sampler(&voxel_sampler),
141                },
142            ],
143        });
144
145        // Create voxelization compute pipeline
146        let voxelization_shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
147            label: Some("VXGI Voxelization Shader"),
148            source: wgpu::ShaderSource::Wgsl(VOXELIZATION_SHADER.into()),
149        });
150
151        let voxelization_pipeline =
152            device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
153                label: Some("VXGI Voxelization Pipeline"),
154                layout: None,
155                module: &voxelization_shader,
156                entry_point: Some("voxelize"),
157                compilation_options: Default::default(),
158                cache: None,
159            });
160
161        Self {
162            config,
163            _voxel_texture: voxel_texture,
164            _voxel_texture_view: voxel_texture_view,
165            _voxel_sampler: voxel_sampler,
166            vxgi_bind_group_layout,
167            vxgi_bind_group,
168            voxelization_pipeline,
169            needs_update: true,
170        }
171    }
172
173    /// Mark voxel grid as needing update
174    pub fn mark_dirty(&mut self) {
175        self.needs_update = true;
176    }
177
178    /// Update voxel radiance field from terrain
179    pub fn update_voxel_field(&mut self, encoder: &mut wgpu::CommandEncoder) {
180        if !self.needs_update {
181            return;
182        }
183
184        // Run voxelization compute shader
185        let mut compute_pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
186            label: Some("VXGI Voxelization Pass"),
187            timestamp_writes: None,
188        });
189
190        compute_pass.set_pipeline(&self.voxelization_pipeline);
191
192        let workgroup_size = 8;
193        let dispatch_size = (self.config.voxel_resolution + workgroup_size - 1) / workgroup_size;
194        compute_pass.dispatch_workgroups(dispatch_size, dispatch_size, dispatch_size);
195
196        drop(compute_pass);
197
198        self.needs_update = false;
199    }
200
201    /// Get bind group layout
202    pub fn bind_group_layout(&self) -> &wgpu::BindGroupLayout {
203        &self.vxgi_bind_group_layout
204    }
205
206    /// Get bind group
207    pub fn bind_group(&self) -> &wgpu::BindGroup {
208        &self.vxgi_bind_group
209    }
210
211    /// Get configuration
212    pub fn config(&self) -> &VxgiConfig {
213        &self.config
214    }
215}

217/// WGSL shader for voxelization
218const VOXELIZATION_SHADER: &str = r#"
219@group(0) @binding(0) var voxel_texture: texture_storage_3d<rgba16float, write>;
220
221@compute @workgroup_size(8, 8, 8)
222fn voxelize(@builtin(global_invocation_id) global_id: vec3<u32>) {
223    let voxel_pos = vec3<f32>(global_id);
224    
225    // Sample terrain density and material at this voxel position
226    // This would integrate with the terrain voxel system
227    
228    // For now, initialize with ambient lighting
229    let radiance = vec4<f32>(0.1, 0.1, 0.1, 1.0);
230    
231    textureStore(voxel_texture, global_id, radiance);
232}

235/// WGSL shader for cone tracing
236pub const CONE_TRACING_SHADER: &str = r#"
237struct VxgiConfig {
238    voxel_resolution: u32,
239    world_size: f32,
240    cone_count: u32,
241    max_trace_distance: f32,
242    cone_aperture: f32,
243}
244
245@group(3) @binding(0) var voxel_texture: texture_3d<f32>;
246@group(3) @binding(1) var voxel_sampler: sampler;
247
248// Cone directions for diffuse sampling (6 cones)
249const CONE_DIRECTIONS = array<vec3<f32>, 6>(
250    vec3<f32>(0.0, 1.0, 0.0),
251    vec3<f32>(0.0, 0.5, 0.866),
252    vec3<f32>(0.823, 0.5, 0.267),
253    vec3<f32>(0.509, 0.5, -0.7),
254    vec3<f32>(-0.509, 0.5, -0.7),
255    vec3<f32>(-0.823, 0.5, 0.267),
256);
257
258fn world_to_voxel(world_pos: vec3<f32>, config: VxgiConfig) -> vec3<f32> {
259    return (world_pos / config.world_size + 0.5) * f32(config.voxel_resolution);
260}
261
262fn trace_cone(
263    origin: vec3<f32>,
264    direction: vec3<f32>,
265    aperture: f32,
266    config: VxgiConfig
267) -> vec4<f32> {
268    var accumulated_radiance = vec3<f32>(0.0);
269    var accumulated_opacity = 0.0;
270    
271    let step_size = 1.0;
272    let max_steps = u32(config.max_trace_distance / step_size);
273    
274    var current_pos = origin;
275    var diameter = 0.0;
276    
277    for (var i = 0u; i < max_steps; i = i + 1u) {
278        if (accumulated_opacity >= 0.95) {
279            break;
280        }
281        
282        // Calculate mip level based on cone diameter
283        diameter = diameter + aperture * step_size;
284        let mip_level = log2(diameter + 1.0);
285        
286        // Sample voxel texture
287        let voxel_coord = world_to_voxel(current_pos, config) / f32(config.voxel_resolution);
288        let sample = textureSampleLevel(voxel_texture, voxel_sampler, voxel_coord, mip_level);
289        
290        // Accumulate radiance
291        let opacity = sample.a * (1.0 - accumulated_opacity);
292        accumulated_radiance = accumulated_radiance + sample.rgb * opacity;
293        accumulated_opacity = accumulated_opacity + opacity;
294        
295        // Step forward
296        current_pos = current_pos + direction * step_size;
297    }
298    
299    return vec4<f32>(accumulated_radiance, accumulated_opacity);
300}
301
302fn calculate_vxgi_lighting(
303    world_pos: vec3<f32>,
304    normal: vec3<f32>,
305    config: VxgiConfig
306) -> vec3<f32> {
307    var total_radiance = vec3<f32>(0.0);
308    
309    // Trace multiple cones for diffuse indirect lighting
310    for (var i = 0u; i < config.cone_count; i = i + 1u) {
311        // Transform cone direction to world space aligned with normal
312        let cone_dir = normalize(CONE_DIRECTIONS[i]);
313        
314        // Create tangent space
315        let up = select(vec3<f32>(0.0, 1.0, 0.0), vec3<f32>(1.0, 0.0, 0.0), abs(normal.y) > 0.9);
316        let tangent = normalize(cross(up, normal));
317        let bitangent = cross(normal, tangent);
318        
319        let world_cone_dir = tangent * cone_dir.x + normal * cone_dir.y + bitangent * cone_dir.z;
320        
321        // Trace cone
322        let cone_result = trace_cone(
323            world_pos + normal * 0.1, // Offset to avoid self-intersection
324            world_cone_dir,
325            config.cone_aperture,
326            config
327        );
328        
329        total_radiance = total_radiance + cone_result.rgb;
330    }
331    
332    // Average over all cones
333    return total_radiance / f32(config.cone_count);
334}

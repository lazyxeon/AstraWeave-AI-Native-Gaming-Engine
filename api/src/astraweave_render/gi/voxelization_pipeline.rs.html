1//! Voxelization Pipeline - Converts Voxel Terrain Meshes to Radiance Field
2//!
3//! This module implements conservative rasterization-based voxelization of
4//! voxel terrain meshes (from Marching Cubes) into a 3D radiance texture
5//! for use with VXGI cone tracing.
6
7use glam::Vec3;
8use wgpu::util::DeviceExt;

10/// Configuration for voxelization
11#[repr(C)]
12#[derive(Debug, Clone, Copy, bytemuck::Pod, bytemuck::Zeroable)]
13pub struct VoxelizationConfig {
14    /// Voxel grid resolution (power of 2)
15    pub voxel_resolution: u32,
16    /// World space size covered by voxel grid
17    pub world_size: f32,
18    /// Number of triangles to voxelize
19    pub triangle_count: u32,
20    /// Intensity of direct lighting
21    pub light_intensity: f32,
22}
23
24impl Default for VoxelizationConfig {
25    fn default() -> Self {
26        Self {
27            voxel_resolution: 256,
28            world_size: 1000.0,
29            triangle_count: 0,
30            light_intensity: 1.0,
31        }
32    }
33}

54/// Material data for voxelization
55#[repr(C)]
56#[derive(Debug, Clone, Copy, bytemuck::Pod, bytemuck::Zeroable)]
57pub struct VoxelMaterial {
58    /// Base color (RGB)
59    pub albedo: [f32; 3],
60    /// Metallic factor
61    pub metallic: f32,
62    /// Roughness factor
63    pub roughness: f32,
64    /// Emissive radiance (RGB)
65    pub emissive: [f32; 3],
66}
67
68impl Default for VoxelMaterial {
69    fn default() -> Self {

71            albedo: [0.8, 0.8, 0.8],

73            roughness: 0.8,
74            emissive: [0.0, 0.0, 0.0],

76    }
77}

131    config_buffer: wgpu::Buffer,
132    vertex_buffer: Option<wgpu::Buffer>,
133    index_buffer: Option<wgpu::Buffer>,
134    material_buffer: wgpu::Buffer,
135
136    // Statistics
137    stats: VoxelizationStats,

140#[derive(Debug, Default, Clone, Copy)]
141pub struct VoxelizationStats {
142    pub total_triangles: u32,
143    pub total_vertices: u32,
144    pub voxelization_time_ms: f32,
145    pub clear_time_ms: f32,

148impl VoxelizationPipeline {
149    /// Create a new voxelization pipeline
150    pub fn new(device: &wgpu::Device, config: VoxelizationConfig) -> Self {
151        // Load shader
152        let shader_source = include_str!("../shaders/vxgi_voxelize.wgsl");
153        let shader_module = device.create_shader_module(wgpu::ShaderModuleDescriptor {
154            label: Some("VXGI Voxelization Shader"),
155            source: wgpu::ShaderSource::Wgsl(shader_source.into()),
156        });
157
158        // Create bind group layout
159        let bind_group_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
160            label: Some("Voxelization Bind Group Layout"),
161            entries: &[
162                // Config uniform buffer
163                wgpu::BindGroupLayoutEntry {
164                    binding: 0,
165                    visibility: wgpu::ShaderStages::COMPUTE,
166                    ty: wgpu::BindingType::Buffer {
167                        ty: wgpu::BufferBindingType::Uniform,
168                        has_dynamic_offset: false,
169                        min_binding_size: None,
170                    },
171                    count: None,
172                },
173                // Vertex storage buffer
174                wgpu::BindGroupLayoutEntry {
175                    binding: 1,
176                    visibility: wgpu::ShaderStages::COMPUTE,
177                    ty: wgpu::BindingType::Buffer {
178                        ty: wgpu::BufferBindingType::Storage { read_only: true },
179                        has_dynamic_offset: false,
180                        min_binding_size: None,
181                    },
182                    count: None,
183                },
184                // Index storage buffer
185                wgpu::BindGroupLayoutEntry {
186                    binding: 2,
187                    visibility: wgpu::ShaderStages::COMPUTE,
188                    ty: wgpu::BindingType::Buffer {
189                        ty: wgpu::BufferBindingType::Storage { read_only: true },
190                        has_dynamic_offset: false,
191                        min_binding_size: None,
192                    },
193                    count: None,
194                },
195                // Material storage buffer
196                wgpu::BindGroupLayoutEntry {
197                    binding: 3,
198                    visibility: wgpu::ShaderStages::COMPUTE,
199                    ty: wgpu::BindingType::Buffer {
200                        ty: wgpu::BufferBindingType::Storage { read_only: true },
201                        has_dynamic_offset: false,
202                        min_binding_size: None,
203                    },
204                    count: None,
205                },
206                // Voxel texture (read-write storage)
207                wgpu::BindGroupLayoutEntry {
208                    binding: 4,
209                    visibility: wgpu::ShaderStages::COMPUTE,
210                    ty: wgpu::BindingType::StorageTexture {
211                        access: wgpu::StorageTextureAccess::ReadWrite,
212                        format: wgpu::TextureFormat::Rgba16Float,
213                        view_dimension: wgpu::TextureViewDimension::D3,
214                    },
215                    count: None,
216                },
217            ],
218        });
219
220        // Create compute pipeline for voxelization
221        let voxelize_pipeline_layout =
222            device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
223                label: Some("Voxelization Pipeline Layout"),
224                bind_group_layouts: &[&bind_group_layout],
225                push_constant_ranges: &[],
226            });
227
228        let voxelize_pipeline = device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
229            label: Some("Voxelization Compute Pipeline"),
230            layout: Some(&voxelize_pipeline_layout),
231            module: &shader_module,
232            entry_point: Some("voxelize"),
233            compilation_options: Default::default(),
234            cache: None,
235        });
236
237        // Create compute pipeline for clearing voxels
238        let clear_pipeline = device.create_compute_pipeline(&wgpu::ComputePipelineDescriptor {
239            label: Some("Voxel Clear Compute Pipeline"),
240            layout: Some(&voxelize_pipeline_layout),
241            module: &shader_module,
242            entry_point: Some("clear_voxels"),
243            compilation_options: Default::default(),
244            cache: None,
245        });
246
247        // Create config buffer
248        let config_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
249            label: Some("Voxelization Config Buffer"),
250            contents: bytemuck::cast_slice(&[config]),
251            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
252        });
253
254        // Create material buffer with default material
255        let default_material = VoxelMaterial::default();
256        let material_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
257            label: Some("Voxelization Material Buffer"),
258            contents: bytemuck::cast_slice(&[default_material]),
259            usage: wgpu::BufferUsages::STORAGE | wgpu::BufferUsages::COPY_DST,
260        });
261
262        Self {
263            config,
264            _shader_module: shader_module,
265            voxelize_pipeline,
266            clear_pipeline,
267            bind_group_layout,
268            config_buffer,
269            vertex_buffer: None,
270            index_buffer: None,
271            material_buffer,
272            stats: VoxelizationStats::default(),
273        }
274    }

276    /// Update configuration
277    pub fn update_config(&mut self, queue: &wgpu::Queue, config: VoxelizationConfig) {
278        self.config = config;
279        queue.write_buffer(&self.config_buffer, 0, bytemuck::cast_slice(&[config]));
280    }

282    /// Upload mesh data to GPU
283    fn upload_mesh(&mut self, device: &wgpu::Device, queue: &wgpu::Queue, mesh: &VoxelizationMesh) {
284        // Create or recreate vertex buffer
285        let vertex_size = (mesh.vertices.len() * std::mem::size_of::<VoxelVertex>()) as u64;
286        match &self.vertex_buffer {
287            Some(buffer) if buffer.size() >= vertex_size => {
288                // Reuse existing buffer
289                queue.write_buffer(buffer, 0, bytemuck::cast_slice(&mesh.vertices));
290            }
291            _ => {
292                // Create new buffer (either None or too small)
293                self.vertex_buffer = Some(device.create_buffer_init(
294                    &wgpu::util::BufferInitDescriptor {
295                        label: Some("Voxelization Vertex Buffer"),
296                        contents: bytemuck::cast_slice(&mesh.vertices),
297                        usage: wgpu::BufferUsages::STORAGE | wgpu::BufferUsages::COPY_DST,
298                    },
299                ));
300            }
301        }
302
303        // Create or recreate index buffer
304        let index_size = (mesh.indices.len() * std::mem::size_of::<u32>()) as u64;
305        match &self.index_buffer {
306            Some(buffer) if buffer.size() >= index_size => {
307                // Reuse existing buffer
308                queue.write_buffer(buffer, 0, bytemuck::cast_slice(&mesh.indices));
309            }
310            _ => {
311                // Create new buffer (either None or too small)
312                self.index_buffer = Some(device.create_buffer_init(
313                    &wgpu::util::BufferInitDescriptor {
314                        label: Some("Voxelization Index Buffer"),
315                        contents: bytemuck::cast_slice(&mesh.indices),
316                        usage: wgpu::BufferUsages::STORAGE | wgpu::BufferUsages::COPY_DST,
317                    },
318                ));
319            }
320        }
321
322        // Update material buffer
323        queue.write_buffer(
324            &self.material_buffer,
325            0,
326            bytemuck::cast_slice(&[mesh.material]),
327        );
328
329        // Update config with triangle count
330        let mut updated_config = self.config;
331        updated_config.triangle_count = mesh.triangle_count();
332        self.update_config(queue, updated_config);
333
334        // Update stats
335        self.stats.total_triangles = mesh.triangle_count();
336        self.stats.total_vertices = mesh.vertices.len() as u32;
337    }

339    /// Clear voxel texture to prepare for voxelization
340    pub fn clear_voxels(
341        &self,
342        device: &wgpu::Device,
343        encoder: &mut wgpu::CommandEncoder,
344        voxel_texture_view: &wgpu::TextureView,
345    ) {
346        let _start_time = std::time::Instant::now();
347
348        // Create bind group for clear pass
349        let bind_group = self.create_bind_group_for_clear(device, voxel_texture_view);
350
351        let mut compute_pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
352            label: Some("Voxel Clear Pass"),
353            timestamp_writes: None,
354        });
355
356        compute_pass.set_pipeline(&self.clear_pipeline);
357        compute_pass.set_bind_group(0, &bind_group, &[]);
358
359        // Dispatch clear shader (8x8x8 workgroups)
360        let workgroup_size = 8;
361        let dispatch_size = self.config.voxel_resolution.div_ceil(workgroup_size);
362        compute_pass.dispatch_workgroups(dispatch_size, dispatch_size, dispatch_size);
363
364        drop(compute_pass);
365    }

367    /// Voxelize a mesh into the voxel texture
368    pub fn voxelize_mesh(
369        &mut self,
370        device: &wgpu::Device,
371        queue: &wgpu::Queue,
372        encoder: &mut wgpu::CommandEncoder,
373        mesh: &VoxelizationMesh,
374        voxel_texture_view: &wgpu::TextureView,
375    ) {
376        let _start_time = std::time::Instant::now();
377
378        // Upload mesh data
379        self.upload_mesh(device, queue, mesh);
380
381        // Create bind group
382        let bind_group = self.create_bind_group(device, voxel_texture_view);
383
384        // Run voxelization compute pass
385        let mut compute_pass = encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
386            label: Some("VXGI Voxelization Pass"),
387            timestamp_writes: None,
388        });
389
390        compute_pass.set_pipeline(&self.voxelize_pipeline);
391        compute_pass.set_bind_group(0, &bind_group, &[]);
392
393        // Dispatch one thread per triangle (workgroup size 64x1x1)
394        let triangle_count = mesh.triangle_count();
395        let workgroup_size = 64;
396        let dispatch_size = triangle_count.div_ceil(workgroup_size);
397        compute_pass.dispatch_workgroups(dispatch_size, 1, 1);
398
399        drop(compute_pass);
400
401        self.stats.voxelization_time_ms = _start_time.elapsed().as_secs_f32() * 1000.0;
402    }

404    /// Create bind group for voxelization
405    fn create_bind_group(
406        &self,
407        device: &wgpu::Device,
408        voxel_texture_view: &wgpu::TextureView,
409    ) -> wgpu::BindGroup {
410        // Week 3 Action 10: Safe buffer access (buffers guaranteed to exist after upload_mesh)
411        let vertex_buffer = self.vertex_buffer.as_ref().expect(
412            "vertex_buffer must be initialized before create_bind_group (call upload_mesh first)",
413        );
414        let index_buffer = self.index_buffer.as_ref().expect(
415            "index_buffer must be initialized before create_bind_group (call upload_mesh first)",
416        );
417
418        device.create_bind_group(&wgpu::BindGroupDescriptor {
419            label: Some("Voxelization Bind Group"),
420            layout: &self.bind_group_layout,
421            entries: &[
422                wgpu::BindGroupEntry {
423                    binding: 0,
424                    resource: self.config_buffer.as_entire_binding(),
425                },
426                wgpu::BindGroupEntry {
427                    binding: 1,
428                    resource: vertex_buffer.as_entire_binding(),
429                },
430                wgpu::BindGroupEntry {
431                    binding: 2,
432                    resource: index_buffer.as_entire_binding(),
433                },
434                wgpu::BindGroupEntry {
435                    binding: 3,
436                    resource: self.material_buffer.as_entire_binding(),
437                },
438                wgpu::BindGroupEntry {
439                    binding: 4,
440                    resource: wgpu::BindingResource::TextureView(voxel_texture_view),
441                },
442            ],
443        })
444    }

446    /// Create bind group for clear pass (minimal bindings)
447    fn create_bind_group_for_clear(
448        &self,
449        device: &wgpu::Device,
450        voxel_texture_view: &wgpu::TextureView,
451    ) -> wgpu::BindGroup {
452        // For clear pass, we need dummy buffers for unused bindings
453        let dummy_vertex_buffer = self.vertex_buffer.as_ref().unwrap_or(&self.config_buffer);
454        let dummy_index_buffer = self.index_buffer.as_ref().unwrap_or(&self.config_buffer);
455
456        device.create_bind_group(&wgpu::BindGroupDescriptor {
457            label: Some("Voxel Clear Bind Group"),
458            layout: &self.bind_group_layout,
459            entries: &[
460                wgpu::BindGroupEntry {
461                    binding: 0,
462                    resource: self.config_buffer.as_entire_binding(),
463                },
464                wgpu::BindGroupEntry {
465                    binding: 1,
466                    resource: dummy_vertex_buffer.as_entire_binding(),
467                },
468                wgpu::BindGroupEntry {
469                    binding: 2,
470                    resource: dummy_index_buffer.as_entire_binding(),
471                },
472                wgpu::BindGroupEntry {
473                    binding: 3,
474                    resource: self.material_buffer.as_entire_binding(),
475                },
476                wgpu::BindGroupEntry {
477                    binding: 4,
478                    resource: wgpu::BindingResource::TextureView(voxel_texture_view),
479                },
480            ],
481        })
482    }

1//! GPU Memory Budgeter
2//!
3//! Tracks GPU memory allocations by category and enforces configurable budgets
4//! to prevent OOM conditions and enable intelligent streaming decisions.
5
6use std::collections::HashMap;
7use std::sync::atomic::{AtomicU64, Ordering};
8use std::sync::{Arc, RwLock};
9
10/// Memory allocation categories
11#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
12pub enum MemoryCategory {
13    /// Vertex/index buffers
14    Geometry,
15    /// Texture data (color, normal, etc.)
16    Textures,
17    /// Render targets and framebuffers
18    RenderTargets,
19    /// Uniform/storage buffers
20    Uniforms,
21    /// Staging buffers (transient)
22    Staging,
23    /// Shadow maps
24    Shadows,
25    /// IBL/environment maps
26    Environment,
27    /// Other/uncategorized
28    Other,
29}
30
31impl MemoryCategory {
32    pub fn all() -> &'static [MemoryCategory] {
33        &[
34            MemoryCategory::Geometry,
35            MemoryCategory::Textures,
36            MemoryCategory::RenderTargets,
37            MemoryCategory::Uniforms,
38            MemoryCategory::Staging,
39            MemoryCategory::Shadows,
40            MemoryCategory::Environment,
41            MemoryCategory::Other,
42        ]
43    }
44}
45
46/// Budget configuration for a memory category
47#[derive(Debug, Clone)]
48pub struct CategoryBudget {
49    /// Soft limit - trigger warnings/streaming when exceeded
50    pub soft_limit: u64,
51    /// Hard limit - refuse allocations when exceeded
52    pub hard_limit: u64,
53    /// Current allocation
54    pub current: u64,
55}
56
57impl Default for CategoryBudget {
58    fn default() -> Self {
59        Self {
60            soft_limit: 256 * 1024 * 1024, // 256 MB soft
61            hard_limit: 512 * 1024 * 1024, // 512 MB hard
62            current: 0,
63        }
64    }
65}
66
67/// Callback for budget events
68pub type BudgetCallback = Arc<dyn Fn(BudgetEvent) + Send + Sync>;
69
70/// Budget event types
71#[derive(Debug, Clone)]
72pub enum BudgetEvent {
73    /// Soft limit exceeded
74    SoftLimitExceeded {
75        category: MemoryCategory,
76        current: u64,
77        limit: u64,
78    },
79    /// Hard limit would be exceeded (allocation rejected)
80    HardLimitBlocked {
81        category: MemoryCategory,
82        requested: u64,
83        available: u64,
84    },
85    /// Memory pressure warning (total budget usage high)
86    MemoryPressure {
87        total_used: u64,
88        total_budget: u64,
89        percentage: f32,
90    },
91}
92
93/// GPU Memory Budget Manager
94pub struct GpuMemoryBudget {
95    /// Per-category budgets
96    budgets: RwLock<HashMap<MemoryCategory, CategoryBudget>>,
97
98    /// Total memory used (atomic for fast queries)
99    total_used: AtomicU64,
100
101    /// Total budget across all categories
102    total_budget: AtomicU64,
103
104    /// Event callbacks
105    callbacks: RwLock<Vec<BudgetCallback>>,
106
107    /// Memory pressure threshold (0.0 - 1.0)
108    pressure_threshold: f32,
109}
110
111impl Default for GpuMemoryBudget {
112    fn default() -> Self {
113        Self::new()
114    }
115}
116
117impl GpuMemoryBudget {
118    /// Create a new budget manager with default limits
119    pub fn new() -> Self {
120        let mut budgets = HashMap::new();
121        for &cat in MemoryCategory::all() {
122            budgets.insert(cat, CategoryBudget::default());
123        }
124
125        Self {
126            budgets: RwLock::new(budgets),
127            total_used: AtomicU64::new(0),
128            total_budget: AtomicU64::new(2 * 1024 * 1024 * 1024), // 2 GB default total
129            callbacks: RwLock::new(Vec::new()),
130            pressure_threshold: 0.85,
131        }
132    }
133
134    /// Create with custom total budget
135    pub fn with_total_budget(total_bytes: u64) -> Self {
136        let mut mgr = Self::new();
137        mgr.total_budget.store(total_bytes, Ordering::SeqCst);
138
139        // Distribute budget proportionally
140        let per_category = total_bytes / 8;
141        let mut budgets = mgr.budgets.write().unwrap();
142        for budget in budgets.values_mut() {
143            budget.soft_limit = (per_category as f64 * 0.75) as u64;
144            budget.hard_limit = per_category;
145        }
146
147        // Give extra to textures (most memory-hungry)
148        if let Some(tex_budget) = budgets.get_mut(&MemoryCategory::Textures) {
149            tex_budget.soft_limit = (total_bytes as f64 * 0.3) as u64;
150            tex_budget.hard_limit = (total_bytes as f64 * 0.4) as u64;
151        }
152
153        drop(budgets);
154        mgr
155    }
156
157    /// Register a callback for budget events
158    pub fn on_event(&self, callback: BudgetCallback) {
159        self.callbacks.write().unwrap().push(callback);
160    }
161
162    /// Attempt to allocate memory in a category
163    /// Returns true if allocation succeeded, false if blocked
164    pub fn try_allocate(&self, category: MemoryCategory, bytes: u64) -> bool {
165        let mut budgets = self.budgets.write().unwrap();
166
167        if let Some(budget) = budgets.get_mut(&category) {
168            let new_total = budget.current + bytes;
169
170            // Check hard limit
171            if new_total > budget.hard_limit {
172                self.fire_event(BudgetEvent::HardLimitBlocked {
173                    category,
174                    requested: bytes,
175                    available: budget.hard_limit.saturating_sub(budget.current),
176                });
177                return false;
178            }
179
180            // Perform allocation
181            budget.current = new_total;
182            self.total_used.fetch_add(bytes, Ordering::SeqCst);
183
184            // Check soft limit
185            if new_total > budget.soft_limit {
186                self.fire_event(BudgetEvent::SoftLimitExceeded {
187                    category,
188                    current: new_total,
189                    limit: budget.soft_limit,
190                });
191            }
192
193            // Check total pressure
194            self.check_pressure();
195
196            true
197        } else {
198            false
199        }
200    }
201
202    /// Record a deallocation
203    pub fn deallocate(&self, category: MemoryCategory, bytes: u64) {
204        let mut budgets = self.budgets.write().unwrap();
205
206        if let Some(budget) = budgets.get_mut(&category) {
207            budget.current = budget.current.saturating_sub(bytes);
208            self.total_used.fetch_sub(bytes, Ordering::SeqCst);
209        }
210    }
211
212    /// Get current usage for a category
213    pub fn get_usage(&self, category: MemoryCategory) -> u64 {
214        self.budgets
215            .read()
216            .unwrap()
217            .get(&category)
218            .map(|b| b.current)
219            .unwrap_or(0)
220    }
221
222    /// Get total memory usage
223    pub fn total_usage(&self) -> u64 {
224        self.total_used.load(Ordering::SeqCst)
225    }
226
227    /// Get usage as percentage of total budget
228    pub fn usage_percentage(&self) -> f32 {
229        let used = self.total_used.load(Ordering::SeqCst) as f64;
230        let total = self.total_budget.load(Ordering::SeqCst) as f64;
231        if total > 0.0 {
232            (used / total) as f32
233        } else {
234            0.0
235        }
236    }
237
238    /// Get snapshot of all category usage
239    pub fn snapshot(&self) -> Vec<(MemoryCategory, u64, u64)> {
240        self.budgets
241            .read()
242            .unwrap()
243            .iter()
244            .map(|(&cat, budget)| (cat, budget.current, budget.hard_limit))
245            .collect()
246    }
247
248    /// Set budget for a specific category
249    pub fn set_category_budget(&self, category: MemoryCategory, soft: u64, hard: u64) {
250        let mut budgets = self.budgets.write().unwrap();
251        if let Some(budget) = budgets.get_mut(&category) {
252            budget.soft_limit = soft;
253            budget.hard_limit = hard;
254        }
255    }
256
257    fn check_pressure(&self) {
258        let percentage = self.usage_percentage();
259        if percentage > self.pressure_threshold {
260            self.fire_event(BudgetEvent::MemoryPressure {
261                total_used: self.total_used.load(Ordering::SeqCst),
262                total_budget: self.total_budget.load(Ordering::SeqCst),
263                percentage,
264            });
265        }
266    }
267
268    fn fire_event(&self, event: BudgetEvent) {
269        let callbacks = self.callbacks.read().unwrap();
270        for callback in callbacks.iter() {
271            callback(event.clone());
272        }
273    }
274}

280#[cfg(test)]
281mod tests {
282    use super::*;
283    use std::sync::atomic::AtomicBool;
284
285    #[test]
286    fn test_basic_allocation() {
287        let budget = GpuMemoryBudget::new();
288
289        assert!(budget.try_allocate(MemoryCategory::Textures, 1024));
290        assert_eq!(budget.get_usage(MemoryCategory::Textures), 1024);
291        assert_eq!(budget.total_usage(), 1024);
292
293        budget.deallocate(MemoryCategory::Textures, 512);
294        assert_eq!(budget.get_usage(MemoryCategory::Textures), 512);
295        assert_eq!(budget.total_usage(), 512);
296    }
297
298    #[test]
299    fn test_hard_limit_blocking() {
300        let budget = GpuMemoryBudget::with_total_budget(1024 * 1024); // 1 MB total
301        budget.set_category_budget(MemoryCategory::Geometry, 1024, 2048);
302
303        // Should succeed
304        assert!(budget.try_allocate(MemoryCategory::Geometry, 1024));
305
306        // Should succeed (at limit)
307        assert!(budget.try_allocate(MemoryCategory::Geometry, 1024));
308
309        // Should fail (over limit)
310        assert!(!budget.try_allocate(MemoryCategory::Geometry, 1));
311    }
312
313    #[test]
314    fn test_soft_limit_callback() {
315        let budget = GpuMemoryBudget::new();
316        budget.set_category_budget(MemoryCategory::Textures, 512, 1024);
317
318        let triggered = Arc::new(AtomicBool::new(false));
319        let triggered_clone = triggered.clone();
320
321        budget.on_event(Arc::new(move |event| {
322            if matches!(event, BudgetEvent::SoftLimitExceeded { .. }) {
323                triggered_clone.store(true, Ordering::SeqCst);
324            }
325        }));
326
327        // Below soft limit
328        assert!(budget.try_allocate(MemoryCategory::Textures, 256));
329        assert!(!triggered.load(Ordering::SeqCst));
330
331        // Above soft limit
332        assert!(budget.try_allocate(MemoryCategory::Textures, 512));
333        assert!(triggered.load(Ordering::SeqCst));
334    }
335
336    #[test]
337    fn test_usage_percentage() {
338        let budget = GpuMemoryBudget::new();
339        // Default has 2GB total budget and 512MB per category (hard limit)
340        // Allocate 400MB which is within Textures hard limit
341        assert!(budget.try_allocate(MemoryCategory::Textures, 400 * 1024 * 1024));
342        let pct = budget.usage_percentage();
343        // 400MB / 2GB = 0.195... (~20%)
344        assert!((pct - 0.195).abs() < 0.02, "Expected ~20%, got {}", pct);
345    }
346
347    #[test]
348    fn test_snapshot() {
349        let budget = GpuMemoryBudget::new();
350
351        budget.try_allocate(MemoryCategory::Textures, 1000);
352        budget.try_allocate(MemoryCategory::Geometry, 500);
353
354        let snapshot = budget.snapshot();
355        assert!(snapshot.len() >= 2);
356
357        let tex_entry = snapshot
358            .iter()
359            .find(|(cat, _, _)| *cat == MemoryCategory::Textures);
360        assert!(tex_entry.is_some());
361        assert_eq!(tex_entry.unwrap().1, 1000);
362    }

17            array_stride: size_of::<Vertex>() as wgpu::BufferAddress,
18            step_mode: wgpu::VertexStepMode::Vertex,
19            attributes: &[
20                // position
21                wgpu::VertexAttribute {
22                    offset: 0,
23                    shader_location: 0,
24                    format: wgpu::VertexFormat::Float32x3,

26                // normal

30                    format: wgpu::VertexFormat::Float32x3,

32                // tangent at location 12 (to match skinned variant convention)
33                wgpu::VertexAttribute {

37                },

184#[derive(Clone, Debug)]
185pub struct Instance {
186    pub transform: Mat4,
187    pub color: [f32; 4],
188    pub material_id: u32,
189}
190
191impl Instance {
192    pub fn raw(&self) -> InstanceRaw {
193        let model = self.transform;
194        let normal = model.inverse().transpose();
195        InstanceRaw {
196            model: model.to_cols_array_2d(),
197            normal_matrix: [
198                normal.x_axis.truncate().to_array(),
199                normal.y_axis.truncate().to_array(),
200                normal.z_axis.truncate().to_array(),
201            ],
202            color: self.color,
203            material_id: self.material_id,
204            _padding: [0; 3],
205        }
206    }
207
208    pub fn from_pos_scale_color(pos: Vec3, scale: Vec3, color: [f32; 4]) -> Self {
209        let transform = Mat4::from_scale_rotation_translation(scale, glam::Quat::IDENTITY, pos);
210        Self {
211            transform,
212            color,
213            material_id: 0,
214        }
215    }
216}

220pub struct ClusterDims {
221    pub x: u32,
222    pub y: u32,
223    pub z: u32,
224}
225
226/// Compute cluster index for a screen-space pixel and linear depth in view space.
227/// Inputs:
228/// - screen coords in [0,width) x [0,height)
229/// - near/far planes
230/// - dims: number of clusters in x/y/z
231pub fn cluster_index(
232    px: u32,
233    py: u32,
234    width: u32,
235    height: u32,
236    depth: f32,
237    near: f32,
238    far: f32,
239    dims: ClusterDims,
240) -> u32 {
241    let sx = (px as f32 * dims.x as f32 / width as f32).clamp(0.0, dims.x as f32 - 1.0) as u32;
242    let sy = (py as f32 * dims.y as f32 / height as f32).clamp(0.0, dims.y as f32 - 1.0) as u32;
243    // Logarithmic z slicing improves distribution
244    let z_lin = ((depth - near) / (far - near)).clamp(0.0, 0.99999);
245    let z_log = (z_lin * (dims.z as f32)).floor() as u32;
246    sx + sy * dims.x + z_log * dims.x * dims.y
247}

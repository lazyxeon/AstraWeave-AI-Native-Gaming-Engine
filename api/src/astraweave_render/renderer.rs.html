108// Simple Cook-Torrance PBR with single directional light, no IBL
109fn fresnel_schlick(cos_theta: f32, F0: vec3<f32>) -> vec3<f32> {
110    return F0 + (vec3<f32>(1.0,1.0,1.0) - F0) * pow(1.0 - cos_theta, 5.0);
111}

113fn distribution_ggx(N: vec3<f32>, H: vec3<f32>, roughness: f32) -> f32 {
114    let a = roughness * roughness;
115    let a2 = a * a;
116    let NdotH = max(dot(N, H), 0.0);
117    let NdotH2 = NdotH * NdotH;
118    let denom = (NdotH2 * (a2 - 1.0) + 1.0);
119    return a2 / (3.14159 * denom * denom + 1e-5);
120}

122fn geometry_smith(N: vec3<f32>, V: vec3<f32>, L: vec3<f32>, roughness: f32) -> f32 {
123    let r = (roughness + 1.0);
124    let k = (r * r) / 8.0;
125    let NdotV = max(dot(N, V), 0.0);
126    let NdotL = max(dot(N, L), 0.0);
127    let ggx1 = NdotV / (NdotV * (1.0 - k) + k + 1e-5);
128    let ggx2 = NdotL / (NdotL * (1.0 - k) + k + 1e-5);
129    return ggx1 * ggx2;
130}

468    /// Compose a standalone fragment shader from a `MaterialPackage` for validation/pipeline creation.
469    /// Returns a `ShaderModule` ready to be used in a pipeline (caller wires layouts/bindings).
470    pub fn shader_from_material_package(&self, pkg: &MaterialPackage) -> wgpu::ShaderModule {
471        // Declare group(0) bindings based on `bindings` ids collected by the compiler (tex/sampler pairs)
472        let mut decls = String::new();
473        let mut idx = 0u32;
474        for id in pkg.bindings.iter() {
475            decls.push_str(&format!("@group(0) @binding({}) var tex_{}: texture_2d<f32>;\n", idx, id));
476            idx += 1;
477            decls.push_str(&format!("@group(0) @binding({}) var samp_{}: sampler;\n", idx, id));
478            idx += 1;
479        }
480        // Compose WGSL: eval_material + a tiny VS/FS pair.
481        let full = format!(

1313fn fresnel_schlick(cos_theta: f32, F0: vec3<f32>) -> vec3<f32> {
1314    return F0 + (vec3<f32>(1.0,1.0,1.0) - F0) * pow(1.0 - cos_theta, 5.0);

1316fn distribution_ggx(N: vec3<f32>, H: vec3<f32>, roughness: f32) -> f32 {
1317    let a = roughness * roughness;
1318    let a2 = a * a;
1319    let NdotH = max(dot(N, H), 0.0);
1320    let NdotH2 = NdotH * NdotH;
1321    let denom = (NdotH2 * (a2 - 1.0) + 1.0);
1322    return a2 / (3.14159 * denom * denom + 1e-5);
1323}
1324fn geometry_smith(N: vec3<f32>, V: vec3<f32>, L: vec3<f32>, roughness: f32) -> f32 {
1325    let r = (roughness + 1.0);
1326    let k = (r * r) / 8.0;
1327    let NdotV = max(dot(N, V), 0.0);
1328    let NdotL = max(dot(N, L), 0.0);
1329    let ggx1 = NdotV / (NdotV * (1.0 - k) + k + 1e-5);
1330    let ggx2 = NdotL / (NdotL * (1.0 - k) + k + 1e-5);
1331    return ggx1 * ggx2;
1332}

1654    pub fn tick_cinematics(&mut self, dt: f32, camera: &mut Camera) -> Vec<awc::SequencerEvent> {
1655        let mut out = Vec::new();

1658            if let Ok(evs) = self.cin_seq.step(dt, tl) {
1659                for e in &evs {
1660                    match e {
1661                        awc::SequencerEvent::CameraKey(k) => Self::apply_camera_key(camera, k),
1662                        awc::SequencerEvent::FxTrigger { name, params } => {
1663                            // Minimal FX: support fade-in by instantly clearing letterbox/fade
1664                            if name == "fade-in" {
1665                                let _ = params; // reserved
1666                                self.overlay_params.fade = 0.0;
1667                            }
1668                        }
1669                        _ => {}
1670                    }
1671                }
1672                out = evs;
1673            }
1674        }
1675        out

1759    pub fn update_camera(&mut self, camera: &Camera) {
1760        self.camera_ubo.view_proj = camera.vp().to_cols_array_2d();
1761        // Update light dir from time-of-day system (simple linkage for Phase 0)
1762        let light_dir = self.sky.time_of_day().get_light_direction();
1763        self.camera_ubo.light_dir_pad = [light_dir.x, light_dir.y, light_dir.z, 0.0];
1764        self.queue
1765            .write_buffer(&self.camera_buf, 0, bytemuck::bytes_of(&self.camera_ubo));
1766        // Compute splits from camera frustum with lambda blend
1767        let n = camera.znear.max(0.01);
1768        let f = camera.zfar.max(n + 0.1);
1769        let c = 2.0; // cascades count (fixed to 2)
1770        let i = 1.0f32; // boundary between 0 and 1
1771        let u = n + (f - n) * (i / c);
1772        let l = n * (f / n).powf(i / c);
1773        let lambda = self.cascade_lambda.clamp(0.0, 1.0);
1774        let split = l * lambda + u * (1.0 - lambda);
1775        self.split0 = split;
1776        self.split1 = f;
1777
1778        // Build frustum corners per range in world space
1779        let frustum0 = frustum_corners_ws(camera, n, self.split0);
1780        let frustum1 = frustum_corners_ws(camera, self.split0, f);
1781        // Build a light view looking towards the cascade centers
1782        let up = glam::Vec3::Y;
1783        let center0 = frustum_center(&frustum0);
1784        let center1 = frustum_center(&frustum1);
1785        let light_dist = 80.0f32;
1786        let view0 = glam::Mat4::look_to_rh(center0 - light_dir * light_dist, light_dir, up);
1787        let view1 = glam::Mat4::look_to_rh(center1 - light_dir * light_dist, light_dir, up);
1788        // Fit orthographic bounds to cascade frusta in light space
1789        let (min0, max0) = aabb_in_view_space(&view0, &frustum0);
1790        let (min1, max1) = aabb_in_view_space(&view1, &frustum1);
1791        let margin = 5.0f32;
1792        let proj0 = glam::Mat4::orthographic_rh(min0.x - margin, max0.x + margin, min0.y - margin, max0.y + margin, (-max0.z + 0.1).max(0.1), (-min0.z + margin + 0.1).max(1.0));
1793        let proj1 = glam::Mat4::orthographic_rh(min1.x - margin, max1.x + margin, min1.y - margin, max1.y + margin, (-max1.z + 0.1).max(0.1), (-min1.z + margin + 0.1).max(1.0));
1794        self.cascade0 = proj0 * view0;
1795        self.cascade1 = proj1 * view1;
1796    // Pack data for main pass buffer: [mat0, mat1, vec2(splits), vec2(extras)]
1797        let mut data: Vec<f32> = Vec::with_capacity(36);
1798        data.extend_from_slice(&self.cascade0.to_cols_array());
1799        data.extend_from_slice(&self.cascade1.to_cols_array());
1800        data.push(self.split0);
1801        data.push(self.split1);
1802    // extras: pack pcf radius in x, depth_bias in y
1803    data.push(self.shadow_pcf_radius_px);
1804        data.push(self.shadow_depth_bias);
1805        self.queue.write_buffer(&self.light_buf, 0, bytemuck::cast_slice(&data));
1806    }

1937    pub fn render(&mut self) -> Result<()> {
1938        let frame = self.surface.get_current_texture()?;
1939        let view = frame
1940            .texture
1941            .create_view(&wgpu::TextureViewDescriptor::default());
1942
1943        let mut enc = self
1944            .device
1945            .create_command_encoder(&wgpu::CommandEncoderDescriptor {
1946                label: Some("encoder"),
1947            });
1948
1949        // Create plane buffer before render pass
1950        let plane_scale = glam::Mat4::from_scale(vec3(50.0, 1.0, 50.0));
1951        let plane_inst = Instance {
1952            transform: plane_scale,
1953            color: [0.1, 0.12, 0.14, 1.0],
1954        }
1955        .raw();

1957            .device
1958            .create_buffer_init(&wgpu::util::BufferInitDescriptor {
1959                label: Some("plane inst"),
1960                contents: bytemuck::bytes_of(&plane_inst),

1962            });
1963
1964    // Render sky first into HDR
1965    self.sky.render(&mut enc, &self.hdr_view, &self.depth.view, Mat4::from_cols_array_2d(&self.camera_ubo.view_proj), &self.queue)?;
1966
1967        {
1968            // Prepare clustered lighting for this frame: simple demo lights around origin
1969            if self.point_lights.is_empty() {
1970                self.point_lights.push(CpuLight{ pos: glam::Vec3::new(2.0, 2.0, 3.0), radius: 6.0 });
1971                self.point_lights.push(CpuLight{ pos: glam::Vec3::new(-3.0, 1.0, 8.0), radius: 5.0 });
1972            }
1973            // CPU pre-pass builds offsets array (exclusive scan) we share to GPU
1974            let (_counts_cpu, _indices_cpu, offsets_cpu) = bin_lights_cpu(&self.point_lights, self.clustered_dims, (self.config.width, self.config.height), 0.1, 200.0, std::f32::consts::FRAC_PI_3);
1975            // Upload lights and offsets; zero counts and indices
1976            #[repr(C)]
1977            #[derive(Clone, Copy, bytemuck::Pod, bytemuck::Zeroable)]
1978            struct GpuLight { pos_radius: [f32;4] }
1979            let glights: Vec<GpuLight> = self.point_lights.iter().map(|l| GpuLight{ pos_radius: [l.pos.x, l.pos.y, l.pos.z, l.radius] }).collect();
1980            if !glights.is_empty() {
1981                self.queue.write_buffer(&self.clustered_lights_buf, 0, bytemuck::cast_slice(&glights));
1982            }
1983            self.queue.write_buffer(&self.clustered_offsets_buf, 0, bytemuck::cast_slice(&offsets_cpu));
1984            // Zero counts
1985            let clusters = (self.clustered_dims.x * self.clustered_dims.y * self.clustered_dims.z) as usize;
1986            let zero_counts = vec![0u32; clusters];
1987            self.queue.write_buffer(&self.clustered_counts_buf, 0, bytemuck::cast_slice(&zero_counts));
1988            // Run compute to fill counts/indices
1989            #[cfg(feature = "gpu-tests")]
1990            { enc.write_timestamp(&self.timestamp_query_set, 0); }
1991            let mut cpass = enc.begin_compute_pass(&wgpu::ComputePassDescriptor{ label: Some("cluster bin"), timestamp_writes: None });
1992            cpass.set_pipeline(&self.clustered_comp_pipeline);
1993            cpass.set_bind_group(0, &self.clustered_comp_bg, &[]);
1994            cpass.dispatch_workgroups(glights.len() as u32, 1, 1);
1995            drop(cpass);
1996            #[cfg(feature = "gpu-tests")]
1997            {
1998                enc.write_timestamp(&self.timestamp_query_set, 1);
1999                enc.resolve_query_set(&self.timestamp_query_set, 0..2, &self.timestamp_buf, 0);
2000            }
2001        }
2002            // Prepare external mesh single-instance buffer if needed
2003            let ext_ibuf: Option<wgpu::Buffer> = if self.mesh_external.is_some() {
2004                let inst = Instance{ transform: Mat4::IDENTITY, color: [1.0,1.0,1.0,1.0] }.raw();

2006                    label: Some("ext inst"),
2007                    contents: bytemuck::bytes_of(&inst),
2008                    usage: wgpu::BufferUsages::VERTEX,
2009                }))
2010            } else { None };
2011            // Frustum cull instances
2012            let (vis_raws, vis_count) = self.build_visible_instances();
2013            if vis_count > 0 {
2014                self.queue.write_buffer(&self.instance_buf, 0, bytemuck::cast_slice(&vis_raws));

2016            // Shadow passes (depth only) - one per cascade layer
2017            // Write cascade0 matrix, render to layer0; then cascade1, render to layer1
2018            for (idx, layer_view) in [&self.shadow_layer0_view, &self.shadow_layer1_view].iter().enumerate() {
2019                let mat = if idx == 0 { self.cascade0 } else { self.cascade1 };
2020                let arr = mat.to_cols_array();
2021                self.queue.write_buffer(&self.light_buf, 0, bytemuck::cast_slice(&arr));

2023                    label: Some("shadow pass"),

2025                    depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment{
2026                        view: layer_view,
2027                        depth_ops: Some(wgpu::Operations{ load: wgpu::LoadOp::Clear(1.0), store: wgpu::StoreOp::Store }),

2030                    timestamp_writes: None,
2031                    occlusion_query_set: None,

2038                sp.set_vertex_buffer(1, plane_buf.slice(..));

2060                data.push(self.split1);
2061                data.push(self.shadow_pcf_radius_px);
2062                data.push(self.shadow_depth_bias);

2068                label: Some("main pass"),
2069                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
2070                    view: &self.hdr_view,
2071                    resolve_target: None,

2080                    depth_ops: Some(wgpu::Operations {

2082                        store: wgpu::StoreOp::Store,

2084                    stencil_ops: None,
2085                }),
2086                timestamp_writes: None,
2087                occlusion_query_set: None,
2088            });
2089
2090            rp.set_pipeline(&self.pipeline);
2091            rp.set_bind_group(0, &self.camera_bind_group, &[]);
2092            rp.set_bind_group(1, &self.material_bg, &[]);
2093            rp.set_bind_group(2, &self.light_bg, &[]);
2094            rp.set_bind_group(3, &self.tex_bg, &[]);
2095            rp.set_bind_group(4, &self.extra_tex_bg, &[]);
2096            rp.set_bind_group(5, &self.clustered_bg, &[]);
2097
2098            // Ground plane (scaled)
2099            rp.set_vertex_buffer(0, self.mesh_plane.vertex_buf.slice(..));
2100            rp.set_index_buffer(
2101                self.mesh_plane.index_buf.slice(..),
2102                wgpu::IndexFormat::Uint32,
2103            );
2104            rp.set_vertex_buffer(1, plane_buf.slice(..));
2105            rp.draw_indexed(0..self.mesh_plane.index_count, 0, 0..1);

2109            rp.set_index_buffer(
2110                self.mesh_cube.index_buf.slice(..),
2111                wgpu::IndexFormat::Uint32,
2112            );
2113            rp.set_vertex_buffer(1, self.instance_buf.slice(..));
2114            let inst_count = vis_count as u32;
2115            if inst_count > 0 {
2116                rp.draw_indexed(0..self.mesh_cube.index_count, 0, 0..inst_count);
2117            }
2118
2119            // External mesh if present
2120            if let (Some(mesh), Some(ibuf)) = (&self.mesh_external, ext_ibuf.as_ref()) {
2121                rp.set_vertex_buffer(0, mesh.vertex_buf.slice(..));
2122                rp.set_index_buffer(mesh.index_buf.slice(..), wgpu::IndexFormat::Uint32);
2123                rp.set_vertex_buffer(1, ibuf.slice(..));
2124                rp.draw_indexed(0..mesh.index_count, 0, 0..1);

2126            }

2129        #[cfg(feature = "postfx")]
2130        {
2131            let mut ssp = enc.begin_render_pass(&wgpu::RenderPassDescriptor{
2132                label: Some("ssr pass"),
2133                color_attachments: &[Some(wgpu::RenderPassColorAttachment{ view: &self.hdr_aux_view, resolve_target: None, ops: wgpu::Operations{ load: wgpu::LoadOp::Clear(wgpu::Color::BLACK), store: wgpu::StoreOp::Store } })],
2134                depth_stencil_attachment: None,
2135                timestamp_writes: None,
2136                occlusion_query_set: None,
2137            });
2138            ssp.set_pipeline(&self.ssr_pipeline);
2139            ssp.set_bind_group(0, &self.ssr_bind_group, &[]);

2143            let mut ao = enc.begin_render_pass(&wgpu::RenderPassDescriptor{ label: Some("ssao pass"), color_attachments: &[Some(wgpu::RenderPassColorAttachment{ view: &self.fx_ao_view, resolve_target: None, ops: wgpu::Operations{ load: wgpu::LoadOp::Clear(wgpu::Color::BLACK), store: wgpu::StoreOp::Store } })], depth_stencil_attachment: None, timestamp_writes: None, occlusion_query_set: None });

2583            .iter()

2585                let n = glam::Vec3::new(p.x, p.y, p.z);
2586                let len = n.length().max(1e-6);

2589            .collect();
2590
2591        let mut out = Vec::with_capacity(self.instances.len());
2592        for inst in &self.instances {
2593            let center = inst.transform.w_axis.truncate();
2594            // approximate radius from basis vectors length (half-extents length)
2595            let sx = inst.transform.x_axis.truncate().length();
2596            let sy = inst.transform.y_axis.truncate().length();
2597            let sz = inst.transform.z_axis.truncate().length();
2598            let radius = 0.5 * glam::Vec3::new(sx, sy, sz).length();
2599            if inside_frustum_sphere(center, radius, &norm_planes) {
2600                out.push(inst.raw());
2601            }
2602        }

2734fn frustum_corners_ws(cam: &crate::camera::Camera, near: f32, far: f32) -> [glam::Vec3; 8] {
2735    let dir = crate::camera::Camera::dir(cam.yaw, cam.pitch);
2736    let right = dir.cross(glam::Vec3::Y).normalize();
2737    let up = glam::Vec3::Y;
2738    let h_near = (cam.fovy * 0.5).tan() * near;
2739    let w_near = h_near * cam.aspect.max(0.01);
2740    let h_far = (cam.fovy * 0.5).tan() * far;
2741    let w_far = h_far * cam.aspect.max(0.01);
2742    let c_near = cam.position + dir * near;
2743    let c_far = cam.position + dir * far;
2744    [
2745        c_near + up * h_near - right * w_near, // near TL
2746        c_near + up * h_near + right * w_near, // near TR
2747        c_near - up * h_near - right * w_near, // near BL
2748        c_near - up * h_near + right * w_near, // near BR
2749        c_far + up * h_far - right * w_far,    // far TL
2750        c_far + up * h_far + right * w_far,    // far TR
2751        c_far - up * h_far - right * w_far,    // far BL
2752        c_far - up * h_far + right * w_far,    // far BR
2753    ]
2754}

2757    let mut acc = glam::Vec3::ZERO;

2762fn aabb_in_view_space(view: &glam::Mat4, corners_ws: &[glam::Vec3; 8]) -> (glam::Vec3, glam::Vec3) {
2763    let mut min = glam::Vec3::splat(f32::INFINITY);
2764    let mut max = glam::Vec3::splat(f32::NEG_INFINITY);
2765    for &c in corners_ws.iter() {
2766        let v = *view * glam::Vec4::new(c.x, c.y, c.z, 1.0);
2767        let p = v.xyz();
2768        min = min.min(p);
2769        max = max.max(p);
2770    }
2771    (min, max)

95// Distance-based fog (linear + exponential blend)
96fn apply_scene_fog(color: vec3<f32>, dist: f32) -> vec3<f32> {
97    // Linear fog component
98    let linear_fog = clamp((dist - uScene.fog_start) / max(uScene.fog_end - uScene.fog_start, 0.001), 0.0, 1.0);

131// Simple Cook-Torrance PBR with single directional light, no IBL
132fn fresnel_schlick(cos_theta: f32, F0: vec3<f32>) -> vec3<f32> {
133    return F0 + (vec3<f32>(1.0,1.0,1.0) - F0) * pow(1.0 - cos_theta, 5.0);
134}

136fn distribution_ggx(N: vec3<f32>, H: vec3<f32>, roughness: f32) -> f32 {
137    let a = roughness * roughness;
138    let a2 = a * a;
139    let NdotH = max(dot(N, H), 0.0);
140    let NdotH2 = NdotH * NdotH;
141    let denom = (NdotH2 * (a2 - 1.0) + 1.0);
142    return a2 / (3.14159 * denom * denom + 1e-5);
143}

145fn geometry_smith(N: vec3<f32>, V: vec3<f32>, L: vec3<f32>, roughness: f32) -> f32 {
146    let r = (roughness + 1.0);
147    let k = (r * r) / 8.0;
148    let NdotV = max(dot(N, V), 0.0);
149    let NdotL = max(dot(N, L), 0.0);
150    let ggx1 = NdotV / (NdotV * (1.0 - k) + k + 1e-5);
151    let ggx2 = NdotL / (NdotL * (1.0 - k) + k + 1e-5);
152    return ggx1 * ggx2;
153}

155@fragment
156fn fs(input: VSOut) -> @location(0) vec4<f32> {
157    let V = normalize(-input.world_pos); // fake view dir from origin camera
158    let L = normalize(-uCamera.light_dir);
159    let H = normalize(V + L);
160    // Base normal from geometry
161    var N = normalize(input.normal);
162    // Normal map sample using real UVs and TBN
163    let nrm_rgb = textureSample(normal_tex, normal_samp, input.uv).rgb;
164    let nrm_ts = normalize(nrm_rgb * 2.0 - vec3<f32>(1.0,1.0,1.0));
165    let T = input.tbn0; let B = input.tbn1; let NN = input.tbn2;
166    N = normalize(T * nrm_ts.x + B * nrm_ts.y + NN * nrm_ts.z);
167    let NdotL = max(dot(N, L), 0.0);
168
169    var base_color = (uMaterial.base_color.rgb * input.color.rgb);
170    let tex = textureSample(albedo_tex, albedo_samp, input.uv);
171    base_color = base_color * tex.rgb;
172    var metallic = clamp(uMaterial.metallic, 0.0, 1.0);
173    var roughness = clamp(uMaterial.roughness, 0.04, 1.0);
174    let mr = textureSample(mr_tex, mr_samp, input.uv);
175    metallic = clamp(max(metallic, mr.r), 0.0, 1.0);
176    roughness = clamp(min(roughness, max(mr.g, 0.04)), 0.04, 1.0);
177
178    let F0 = mix(vec3<f32>(0.04, 0.04, 0.04), base_color, metallic);
179    let F = fresnel_schlick(max(dot(H, V), 0.0), F0);
180    let D = distribution_ggx(N, H, roughness);
181    let G = geometry_smith(N, V, L, roughness);
182
183    let numerator = D * G * F;
184    let denom = 4.0 * max(dot(N, V), 0.0) * NdotL + 1e-5;
185    let specular = numerator / denom;
186
187    let kd = (vec3<f32>(1.0,1.0,1.0) - F) * (1.0 - metallic);
188    let diffuse = kd * base_color / 3.14159;
189
190    let radiance = vec3<f32>(2.0, 1.96, 1.8); // sun radiance (HDR, ACES compresses)
191        // Shadow sampling
192        // Cascaded shadow mapping (2 cascades)
193    let dist = length(input.world_pos);
194    let use_c0 = dist < uLight.splits.x;
195    var lvp: mat4x4<f32>;
196    if (use_c0) { lvp = uLight.view_proj0; } else { lvp = uLight.view_proj1; }
197        let lp = lvp * vec4<f32>(input.world_pos, 1.0);
198    let ndc_shadow = lp.xyz / lp.w;
199    let uv = ndc_shadow.xy * 0.5 + vec2<f32>(0.5, 0.5);
200    let depth = ndc_shadow.z;
201    let slope = max(0.0, 1.0 - dot(N, L));
202    let base_bias = uLight.extras.y;
203    let bias = max(base_bias /* + slope_scale * slope */ , 0.00001);
204        var shadow: f32 = 1.0;
205        if (uv.x >= 0.0 && uv.x <= 1.0 && uv.y >= 0.0 && uv.y <= 1.0) {
206            var layer: i32;
207            if (use_c0) { layer = 0; } else { layer = 1; }
208            // PCF 3x3 (scaled by pcf radius in texels from extras.x)
209            let dims = vec2<f32>(textureDimensions(shadow_tex).xy);
210            let texel = 1.0 / dims;
211            let r = max(0.0, uLight.extras.x);
212            var sum = 0.0;
213            for (var dx: i32 = -1; dx <= 1; dx = dx + 1) {
214                for (var dy: i32 = -1; dy <= 1; dy = dy + 1) {
215                    let o = vec2<f32>(f32(dx), f32(dy)) * texel * r;
216                    sum = sum + textureSampleCompare(shadow_tex, shadow_sampler, uv + o, layer, depth - bias);
217                }
218            }
219            shadow = sum / 9.0;
220        }
221        // Debug override: force shadow off when extras.x is negative (sentinel from CPU)
222        if (uLight.extras.x < 0.0) {
223            shadow = 1.0;
224        }
225
226        // Optional debug visualization: use uMaterial._pad.x > 0.5 to tint by cascade
227        if (uMaterial._pad.x > 0.5) {
228            var tint: vec3<f32>;
229            if (use_c0) { tint = vec3<f32>(1.0, 0.3, 0.0); } else { tint = vec3<f32>(0.0, 0.2, 1.0); }
230            base_color = mix(base_color, tint, 0.35);
231        }
232    // Add ambient from scene environment UBO (replaces hardcoded 0.2)
233    let ambient = uScene.ambient_color * uScene.ambient_intensity;
234    var lit_color = (diffuse + specular) * radiance * NdotL * shadow + base_color * ambient;
235        // Clustered point lights accumulation (Lambert + simple attenuation)
236    // Clustered lighting disabled for this example build; use lit_color directly
237
238    // Apply distance-based fog from biome scene environment
239    let frag_dist = length(input.world_pos);
240    lit_color = apply_scene_fog(lit_color, frag_dist);
241
242    // Apply screen tint overlay (peaks during biome transitions)
243    lit_color = apply_scene_tint(lit_color);
244
245    return vec4<f32>(lit_color, uMaterial.base_color.a * input.color.a);
246}

547    /// Compose a standalone fragment shader from a `MaterialPackage` for validation/pipeline creation.
548    /// Returns a `ShaderModule` ready to be used in a pipeline (caller wires layouts/bindings).
549    pub fn shader_from_material_package(&self, pkg: &MaterialPackage) -> wgpu::ShaderModule {
550        // Declare group(0) bindings based on `bindings` ids collected by the compiler (tex/sampler pairs)
551        let mut decls = String::new();
552        let mut idx = 0u32;
553        for id in pkg.bindings.iter() {
554            decls.push_str(&format!(
555                "@group(0) @binding({}) var tex_{}: texture_2d<f32>;\n",
556                idx, id
557            ));
558            idx += 1;
559            decls.push_str(&format!(
560                "@group(0) @binding({}) var samp_{}: sampler;\n",
561                idx, id
562            ));
563            idx += 1;
564        }
565        // Compose WGSL: eval_material + a tiny VS/FS pair.
566        let full = format!(
567            "{}\n{}\nstruct VSOut {{ @builtin(position) pos: vec4<f32>, @location(0) uv: vec2<f32> }};\n@vertex fn vs_main(@builtin(vertex_index) vid: u32) -> VSOut {{\n  var pos = array<vec2<f32>,3>(vec2<f32>(-1.0,-3.0), vec2<f32>(3.0,1.0), vec2<f32>(-1.0,1.0));\n  var o: VSOut; o.pos = vec4<f32>(pos[vid], 0.0, 1.0); o.uv = (pos[vid]+vec2<f32>(1.0,1.0))*0.5; return o; }}\n@fragment fn fs_main(i: VSOut) -> @location(0) vec4<f32> {{ let m = eval_material(i.uv); return vec4<f32>(m.base, 1.0); }}\n",
568            decls, pkg.wgsl
569        );
570        self.device
571            .create_shader_module(wgpu::ShaderModuleDescriptor {
572                label: Some("material composed shader"),
573                source: wgpu::ShaderSource::Wgsl(std::borrow::Cow::Owned(full)),
574            })
575    }

2008fn distribution_ggx(N: vec3<f32>, H: vec3<f32>, roughness: f32) -> f32 {
2009    let a = roughness * roughness;

2684    /// Transition to a new biome.
2685    ///
2686    /// This:
2687    /// 1. Checks if a transition is needed (same biome → no-op).
2688    /// 2. Resolves the best HDRI for the biome + current time-of-day.
2689    /// 3. Updates `IblManager` sky mode from the HDRI catalog.
2690    /// 4. Rebakes environment maps (irradiance + specular).
2691    /// 5. Marks the biome as loaded in the tracking state.
2692    ///
2693    /// For the full pipeline (HDRI + terrain textures), use
2694    /// [`transition_biome_full`] which also calls `MaterialManager::load_biome()`.
2695    ///
2696    /// Returns `true` if a transition occurred, `false` if already in that biome.
2697    pub fn transition_biome(
2698        &mut self,
2699        biome: astraweave_terrain::biome::BiomeType,
2700        quality: crate::ibl::IblQuality,
2701    ) -> Result<bool> {
2702        if !self.biome_system.needs_transition(biome) {
2703            return Ok(false);
2704        }
2705
2706        // Resolve HDRI for this biome + time
2707        let sky_mode = self.biome_system.resolve_sky_mode(biome)?;
2708        let hdri_path = self.biome_system.resolve_hdri_path(biome)?;
2709
2710        // Update IBL manager
2711        self.ibl.mode = sky_mode;
2712
2713        // Rebake environment
2714        let resources = self
2715            .ibl
2716            .bake_environment(&self.device, &self.queue, quality)?;
2717        self.ibl_resources = Some(resources);
2718
2719        // Track state
2720        self.biome_system.mark_loaded(biome, hdri_path);
2721
2722        log::info!(
2723            "Biome transition → {:?} (HDRI: {:?})",
2724            biome,
2725            self.biome_system.current_biome()
2726        );
2727
2728        Ok(true)
2729    }

2731    /// Full biome transition: HDRI environment + terrain material textures.
2732    ///
2733    /// This is the batteries-included version of [`transition_biome`]. It:
2734    /// 1. Resolves + loads the HDRI for the biome + time-of-day.
2735    /// 2. Rebakes the IBL environment maps.
2736    /// 3. Loads the biome's terrain material texture arrays via [`MaterialManager`].
2737    ///
2738    /// Returns `Ok(None)` if already in that biome (no-op).
2739    /// Returns `Ok(Some(stats))` with material load statistics on success.
2740    #[cfg(feature = "textures")]
2741    pub async fn transition_biome_full(
2742        &mut self,
2743        biome: astraweave_terrain::biome::BiomeType,
2744        quality: crate::ibl::IblQuality,
2745    ) -> Result<Option<crate::material::MaterialLoadStats>> {
2746        if !self.biome_system.needs_transition(biome) {
2747            return Ok(None);
2748        }
2749
2750        // 1. HDRI + IBL
2751        let sky_mode = self.biome_system.resolve_sky_mode(biome)?;
2752        let hdri_path = self.biome_system.resolve_hdri_path(biome)?;
2753        self.ibl.mode = sky_mode;
2754        let resources = self
2755            .ibl
2756            .bake_environment(&self.device, &self.queue, quality)?;
2757        self.ibl_resources = Some(resources);
2758
2759        // 2. Terrain material textures
2760        let mat_dir = self.biome_system.material_dir_for(biome);
2761        let stats = self

2764            .await?;

2766        // 3. Track state

2840    /// Update the player's world position and tick the biome transition
2841    /// pipeline. Call this once per frame (or whenever the player moves).
2842    ///
2843    /// This:
2844    /// 1. Feeds position to the `BiomeDetector` to check for biome changes.
2845    /// 2. On biome change: starts a `TransitionEffect` crossfade.
2846    /// 3. Advances the transition effect by `delta_time`.
2847    /// 4. Updates `SceneEnvironment` with interpolated fog/ambient.
2848    ///
2849    /// Returns `Some(biome)` if a new biome transition was started this frame.
2850    pub fn update_player_biome(
2851        &mut self,
2852        climate: &astraweave_terrain::climate::ClimateMap,
2853        x: f64,
2854        z: f64,
2855        height: f32,
2856        delta_time: f32,
2857    ) -> Option<astraweave_terrain::biome::BiomeType> {
2858        // 1. Check for biome transition
2859        let new_biome = if let Some(transition) = self.biome_detector.update(climate, x, z, height)
2860        {
2861            self.transition_effect
2862                .start(transition.old_biome, transition.new_biome);
2863            log::info!(
2864                "Biome transition detected: {:?} → {:?}",
2865                transition.old_biome,
2866                transition.new_biome,
2867            );
2868            // Trigger HDRI / IBL swap for the new biome (best-effort; non-fatal)
2869            if let Err(e) =
2870                self.transition_biome(transition.new_biome, crate::ibl::IblQuality::Medium)
2871            {
2872                log::warn!(
2873                    "IBL transition to {:?} failed (non-fatal): {e}",
2874                    transition.new_biome,
2875                );
2876            }
2877            Some(transition.new_biome)
2878        } else {
2879            None
2880        };
2881
2882        // 2. Tick transition effect
2883        if self.transition_effect.is_active() {
2884            self.transition_effect.update(delta_time);
2885        }
2886
2887        // 3. Update scene environment
2888        self.scene_env
2889            .update_from_transition(&self.transition_effect);
2890
2891        // 4. Sync sky + water colours from the interpolated BiomeVisuals
2892        self.sync_biome_sky_water();
2893
2894        new_biome
2895    }

3030    pub fn update_camera(&mut self, camera: &Camera) {
3031        self.cached_view = camera.view_matrix();
3032        self.cached_proj = camera.proj_matrix();
3033        self.camera_ubo.view_proj = camera.vp().to_cols_array_2d();
3034        // Update light dir from time-of-day system (simple linkage for Phase 0)
3035        let light_dir = self.sky.time_of_day().get_light_direction();
3036        self.camera_ubo.light_dir_pad = [light_dir.x, light_dir.y, light_dir.z, 0.0];
3037        self.queue
3038            .write_buffer(&self.camera_buf, 0, bytemuck::bytes_of(&self.camera_ubo));
3039        // Compute splits from camera frustum with lambda blend
3040        let n = camera.znear.max(0.01);
3041        let f = camera.zfar.max(n + 0.1);
3042        let c = 2.0; // cascades count (fixed to 2)
3043        let i = 1.0f32; // boundary between 0 and 1
3044        let u = n + (f - n) * (i / c);
3045        let l = n * (f / n).powf(i / c);
3046        let lambda = self.cascade_lambda.clamp(0.0, 1.0);
3047        let split = l * lambda + u * (1.0 - lambda);
3048        self.split0 = split;
3049        self.split1 = f;
3050
3051        // Build frustum corners per range in world space
3052        let frustum0 = frustum_corners_ws(camera, n, self.split0);
3053        let frustum1 = frustum_corners_ws(camera, self.split0, f);
3054        // Build a light view looking towards the cascade centers
3055        let up = glam::Vec3::Y;
3056        let center0 = frustum_center(&frustum0);
3057        let center1 = frustum_center(&frustum1);
3058        let light_dist = 80.0f32;
3059        let view0 = glam::Mat4::look_to_rh(center0 - light_dir * light_dist, light_dir, up);
3060        let view1 = glam::Mat4::look_to_rh(center1 - light_dir * light_dist, light_dir, up);
3061        // Fit orthographic bounds to cascade frusta in light space
3062        let (min0, max0) = aabb_in_view_space(&view0, &frustum0);
3063        let (min1, max1) = aabb_in_view_space(&view1, &frustum1);
3064        let margin = 5.0f32;
3065        let proj0 = glam::Mat4::orthographic_rh(
3066            min0.x - margin,
3067            max0.x + margin,
3068            min0.y - margin,
3069            max0.y + margin,
3070            (-max0.z + 0.1).max(0.1),
3071            (-min0.z + margin + 0.1).max(1.0),
3072        );
3073        let proj1 = glam::Mat4::orthographic_rh(
3074            min1.x - margin,
3075            max1.x + margin,
3076            min1.y - margin,
3077            max1.y + margin,
3078            (-max1.z + 0.1).max(0.1),
3079            (-min1.z + margin + 0.1).max(1.0),
3080        );
3081        self.cascade0 = proj0 * view0;
3082        self.cascade1 = proj1 * view1;
3083        // Pack data for main pass buffer: [mat0, mat1, vec2(splits), vec2(extras)]
3084        let mut data: Vec<f32> = Vec::with_capacity(36);
3085        data.extend_from_slice(&self.cascade0.to_cols_array());
3086        data.extend_from_slice(&self.cascade1.to_cols_array());
3087        data.push(self.split0);
3088        data.push(self.split1);
3089        // extras: pack pcf radius in x, depth_bias in y
3090        // If force_shadow_override is set, use -1.0 sentinel to disable shadows in shader.
3091        let extras_x = if self.force_shadow_override {
3092            -1.0
3093        } else {
3094            self.shadow_pcf_radius_px
3095        };
3096        data.push(extras_x);
3097        data.push(self.shadow_depth_bias);
3098        self.queue
3099            .write_buffer(&self.light_buf, 0, bytemuck::cast_slice(&data));
3100    }

4754fn frustum_corners_ws(cam: &crate::camera::Camera, near: f32, far: f32) -> [glam::Vec3; 8] {
4755    let dir = crate::camera::Camera::dir(cam.yaw, cam.pitch);
4756    let right = dir.cross(glam::Vec3::Y).normalize();
4757    let up = glam::Vec3::Y;
4758    let h_near = (cam.fovy * 0.5).tan() * near;
4759    let w_near = h_near * cam.aspect.max(0.01);
4760    let h_far = (cam.fovy * 0.5).tan() * far;
4761    let w_far = h_far * cam.aspect.max(0.01);
4762    let c_near = cam.position + dir * near;
4763    let c_far = cam.position + dir * far;
4764    [
4765        c_near + up * h_near - right * w_near, // near TL
4766        c_near + up * h_near + right * w_near, // near TR
4767        c_near - up * h_near - right * w_near, // near BL
4768        c_near - up * h_near + right * w_near, // near BR
4769        c_far + up * h_far - right * w_far,    // far TL
4770        c_far + up * h_far + right * w_far,    // far TR
4771        c_far - up * h_far - right * w_far,    // far BL
4772        c_far - up * h_far + right * w_far,    // far BR
4773    ]
4774}

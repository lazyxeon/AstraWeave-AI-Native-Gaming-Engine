93// Simple Cook-Torrance PBR with single directional light, no IBL
94fn fresnel_schlick(cos_theta: f32, F0: vec3<f32>) -> vec3<f32> {
95    return F0 + (vec3<f32>(1.0,1.0,1.0) - F0) * pow(1.0 - cos_theta, 5.0);
96}

98fn distribution_ggx(N: vec3<f32>, H: vec3<f32>, roughness: f32) -> f32 {
99    let a = roughness * roughness;
100    let a2 = a * a;
101    let NdotH = max(dot(N, H), 0.0);
102    let NdotH2 = NdotH * NdotH;
103    let denom = (NdotH2 * (a2 - 1.0) + 1.0);
104    return a2 / (3.14159 * denom * denom + 1e-5);
105}

107fn geometry_smith(N: vec3<f32>, V: vec3<f32>, L: vec3<f32>, roughness: f32) -> f32 {
108    let r = (roughness + 1.0);
109    let k = (r * r) / 8.0;
110    let NdotV = max(dot(N, V), 0.0);
111    let NdotL = max(dot(N, L), 0.0);
112    let ggx1 = NdotV / (NdotV * (1.0 - k) + k + 1e-5);
113    let ggx2 = NdotL / (NdotL * (1.0 - k) + k + 1e-5);
114    return ggx1 * ggx2;
115}

117@fragment
118fn fs(input: VSOut) -> @location(0) vec4<f32> {
119    let V = normalize(-input.world_pos); // fake view dir from origin camera
120    let L = normalize(-uCamera.light_dir);
121    let H = normalize(V + L);
122    // Base normal from geometry
123    var N = normalize(input.normal);
124    // Normal map sample using real UVs and TBN
125    let nrm_rgb = textureSample(normal_tex, normal_samp, input.uv).rgb;
126    let nrm_ts = normalize(nrm_rgb * 2.0 - vec3<f32>(1.0,1.0,1.0));
127    let T = input.tbn0; let B = input.tbn1; let NN = input.tbn2;
128    N = normalize(T * nrm_ts.x + B * nrm_ts.y + NN * nrm_ts.z);
129    let NdotL = max(dot(N, L), 0.0);
130
131    var base_color = (uMaterial.base_color.rgb * input.color.rgb);
132    let tex = textureSample(albedo_tex, albedo_samp, input.uv);
133    base_color = base_color * tex.rgb;
134    var metallic = clamp(uMaterial.metallic, 0.0, 1.0);
135    var roughness = clamp(uMaterial.roughness, 0.04, 1.0);
136    let mr = textureSample(mr_tex, mr_samp, input.uv);
137    metallic = clamp(max(metallic, mr.r), 0.0, 1.0);
138    roughness = clamp(min(roughness, max(mr.g, 0.04)), 0.04, 1.0);
139
140    let F0 = mix(vec3<f32>(0.04, 0.04, 0.04), base_color, metallic);
141    let F = fresnel_schlick(max(dot(H, V), 0.0), F0);
142    let D = distribution_ggx(N, H, roughness);
143    let G = geometry_smith(N, V, L, roughness);
144
145    let numerator = D * G * F;
146    let denom = 4.0 * max(dot(N, V), 0.0) * NdotL + 1e-5;
147    let specular = numerator / denom;
148
149    let kd = (vec3<f32>(1.0,1.0,1.0) - F) * (1.0 - metallic);
150    let diffuse = kd * base_color / 3.14159;
151
152    let radiance = vec3<f32>(1.0, 0.98, 0.9); // light color
153        // Shadow sampling
154        // Cascaded shadow mapping (2 cascades)
155        let dist = length(input.world_pos);
156        let use_c0 = dist < uLight.splits.x;
157        let lvp = select(uLight.view_proj1, uLight.view_proj0, use_c0);
158        let lp = lvp * vec4<f32>(input.world_pos, 1.0);
159        let ndc = lp.xyz / lp.w;
160        let uv = ndc.xy * 0.5 + vec2<f32>(0.5, 0.5);
161        let depth = ndc.z;
162        let bias = max(0.0008 * (1.0 - dot(N, L)), 0.00015);
163        var shadow: f32 = 1.0;
164        if (uv.x >= 0.0 && uv.x <= 1.0 && uv.y >= 0.0 && uv.y <= 1.0) {
165            let layer = i32(select(1, 0, use_c0));
166            shadow = textureSampleCompare(shadow_tex, shadow_sampler, uv, layer, depth - bias);
167        }
168        // Optional debug visualization: use uMaterial._pad.x > 0.5 to tint by cascade
169        if (uMaterial._pad.x > 0.5) {
170            let tint = select(vec3<f32>(0.0, 0.2, 1.0), vec3<f32>(1.0, 0.3, 0.0), use_c0);
171            base_color = mix(base_color, tint, 0.35);
172        }
173        let lit_color = (diffuse + specular) * radiance * NdotL * shadow + base_color * 0.02; // tiny ambient
174        let color = lit_color;
175    return vec4<f32>(color, uMaterial.base_color.a * input.color.a);
176}

197fn aces_tonemap(x: vec3<f32>) -> vec3<f32> {
198    let a = 2.51;

966fn distribution_ggx(N: vec3<f32>, H: vec3<f32>, roughness: f32) -> f32 {
967    let a = roughness * roughness;
968    let a2 = a * a;
969    let NdotH = max(dot(N, H), 0.0);
970    let NdotH2 = NdotH * NdotH;
971    let denom = (NdotH2 * (a2 - 1.0) + 1.0);
972    return a2 / (3.14159 * denom * denom + 1e-5);

974fn geometry_smith(N: vec3<f32>, V: vec3<f32>, L: vec3<f32>, roughness: f32) -> f32 {

980    let ggx2 = NdotL / (NdotL * (1.0 - k) + k + 1e-5);
981    return ggx1 * ggx2;

1184    pub fn update_camera(&mut self, camera: &Camera) {
1185        self.camera_ubo.view_proj = camera.vp().to_cols_array_2d();
1186        // Update light dir from time-of-day system (simple linkage for Phase 0)
1187        let light_dir = self.sky.time_of_day().get_light_direction();
1188        self.camera_ubo.light_dir_pad = [light_dir.x, light_dir.y, light_dir.z, 0.0];
1189        self.queue
1190            .write_buffer(&self.camera_buf, 0, bytemuck::bytes_of(&self.camera_ubo));
1191        // Update cascaded light view-proj for shadow mapping (very simple splits by distance to origin)
1192        let up = glam::Vec3::Y;
1193        let light_pos = -light_dir * 80.0; // position light looking at origin
1194        let view = glam::Mat4::look_to_rh(light_pos, light_dir, up);
1195        // Two cascades: tighter for near, wider for far (configurable extents)
1196        let e0 = self.cascade0_extent.max(1.0);
1197        let e1 = self.cascade1_extent.max(e0 + 1.0);
1198        let proj0 = glam::Mat4::orthographic_rh(-e0, e0, -e0, e0, 0.1, 120.0);
1199        let proj1 = glam::Mat4::orthographic_rh(-e1, e1, -e1, e1, 0.1, 220.0);
1200        self.cascade0 = proj0 * view;
1201        self.cascade1 = proj1 * view;
1202        // splits retained as prior configured values
1203        // Pack data for main pass buffer: [mat0, mat1, vec2(splits), vec2(pad)]
1204        let mut data: Vec<f32> = Vec::with_capacity(36);
1205        data.extend_from_slice(&self.cascade0.to_cols_array());
1206        data.extend_from_slice(&self.cascade1.to_cols_array());
1207        data.push(self.split0);
1208        data.push(self.split1);
1209        // pad to 4 floats
1210        data.push(0.0);
1211        data.push(0.0);
1212        self.queue.write_buffer(&self.light_buf, 0, bytemuck::cast_slice(&data));
1213    }

1334    pub fn render(&mut self) -> Result<()> {
1335        let frame = self.surface.get_current_texture()?;
1336        let view = frame
1337            .texture
1338            .create_view(&wgpu::TextureViewDescriptor::default());
1339
1340        let mut enc = self
1341            .device
1342            .create_command_encoder(&wgpu::CommandEncoderDescriptor {
1343                label: Some("encoder"),
1344            });
1345
1346        // Create plane buffer before render pass
1347        let plane_scale = glam::Mat4::from_scale(vec3(50.0, 1.0, 50.0));
1348        let plane_inst = Instance {
1349            transform: plane_scale,
1350            color: [0.1, 0.12, 0.14, 1.0],
1351        }
1352        .raw();
1353        let plane_buf = self
1354            .device
1355            .create_buffer_init(&wgpu::util::BufferInitDescriptor {
1356                label: Some("plane inst"),
1357                contents: bytemuck::bytes_of(&plane_inst),
1358                usage: wgpu::BufferUsages::VERTEX,
1359            });
1360
1361    // Render sky first into HDR
1362    self.sky.render(&mut enc, &self.hdr_view, &self.depth.view, Mat4::from_cols_array_2d(&self.camera_ubo.view_proj), &self.queue)?;
1363
1364        {
1365            // Prepare external mesh single-instance buffer if needed
1366            let ext_ibuf: Option<wgpu::Buffer> = if self.mesh_external.is_some() {
1367                let inst = Instance{ transform: Mat4::IDENTITY, color: [1.0,1.0,1.0,1.0] }.raw();
1368                Some(self.device.create_buffer_init(&wgpu::util::BufferInitDescriptor{
1369                    label: Some("ext inst"),
1370                    contents: bytemuck::bytes_of(&inst),
1371                    usage: wgpu::BufferUsages::VERTEX,
1372                }))
1373            } else { None };
1374            // Frustum cull instances
1375            let (vis_raws, vis_count) = self.build_visible_instances();
1376            if vis_count > 0 {
1377                self.queue.write_buffer(&self.instance_buf, 0, bytemuck::cast_slice(&vis_raws));
1378            }
1379            // Shadow passes (depth only) - one per cascade layer
1380            // Write cascade0 matrix, render to layer0; then cascade1, render to layer1
1381            for (idx, layer_view) in [&self.shadow_layer0_view, &self.shadow_layer1_view].iter().enumerate() {
1382                let mat = if idx == 0 { self.cascade0 } else { self.cascade1 };
1383                let arr = mat.to_cols_array();
1384                self.queue.write_buffer(&self.light_buf, 0, bytemuck::cast_slice(&arr));
1385                let mut sp = enc.begin_render_pass(&wgpu::RenderPassDescriptor{
1386                    label: Some("shadow pass"),
1387                    color_attachments: &[],
1388                    depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment{
1389                        view: layer_view,
1390                        depth_ops: Some(wgpu::Operations{ load: wgpu::LoadOp::Clear(1.0), store: wgpu::StoreOp::Store }),
1391                        stencil_ops: None,
1392                    }),
1393                    timestamp_writes: None,
1394                    occlusion_query_set: None,
1395                });
1396                sp.set_pipeline(&self.shadow_pipeline);
1397                sp.set_bind_group(0, &self.light_bg, &[]);
1398                // Draw plane
1399                sp.set_vertex_buffer(0, self.mesh_plane.vertex_buf.slice(..));
1400                sp.set_index_buffer(self.mesh_plane.index_buf.slice(..), wgpu::IndexFormat::Uint32);
1401                sp.set_vertex_buffer(1, plane_buf.slice(..));
1402                sp.draw_indexed(0..self.mesh_plane.index_count, 0, 0..1);
1403                // Draw cubes
1404                sp.set_vertex_buffer(0, self.mesh_cube.vertex_buf.slice(..));
1405                sp.set_index_buffer(self.mesh_cube.index_buf.slice(..), wgpu::IndexFormat::Uint32);
1406                sp.set_vertex_buffer(1, self.instance_buf.slice(..));
1407                let inst_count = vis_count as u32;
1408                if inst_count > 0 { sp.draw_indexed(0..self.mesh_cube.index_count, 0, 0..inst_count); }
1409                // External mesh
1410                if let (Some(mesh), Some(ibuf)) = (&self.mesh_external, ext_ibuf.as_ref()) {
1411                    sp.set_vertex_buffer(0, mesh.vertex_buf.slice(..));
1412                    sp.set_index_buffer(mesh.index_buf.slice(..), wgpu::IndexFormat::Uint32);
1413                    sp.set_vertex_buffer(1, ibuf.slice(..));
1414                    sp.draw_indexed(0..mesh.index_count, 0, 0..1);
1415                }
1416            }
1417            // After rendering shadow layers, restore full light buffer for main pass usage
1418            {
1419                let mut data: Vec<f32> = Vec::with_capacity(36);
1420                data.extend_from_slice(&self.cascade0.to_cols_array());
1421                data.extend_from_slice(&self.cascade1.to_cols_array());
1422                data.push(self.split0);
1423                data.push(self.split1);
1424                data.push(0.0);
1425                data.push(0.0);
1426                self.queue.write_buffer(&self.light_buf, 0, bytemuck::cast_slice(&data));
1427            }
1428
1429            let mut rp = enc.begin_render_pass(&wgpu::RenderPassDescriptor {
1430                label: Some("main pass"),
1431                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
1432                    view: &self.hdr_view,
1433                    resolve_target: None,
1434                    ops: wgpu::Operations {
1435                        // Sky already wrote color; load and draw geometry on top
1436                        load: wgpu::LoadOp::Load,
1437                        store: wgpu::StoreOp::Store,
1438                    },
1439                })],
1440                depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
1441                    view: &self.depth.view,
1442                    depth_ops: Some(wgpu::Operations {
1443                        load: wgpu::LoadOp::Clear(1.0),
1444                        store: wgpu::StoreOp::Store,
1445                    }),
1446                    stencil_ops: None,
1447                }),
1448                timestamp_writes: None,
1449                occlusion_query_set: None,
1450            });
1451
1452            rp.set_pipeline(&self.pipeline);
1453            rp.set_bind_group(0, &self.camera_bind_group, &[]);
1454            rp.set_bind_group(1, &self.material_bg, &[]);
1455            rp.set_bind_group(2, &self.light_bg, &[]);
1456            rp.set_bind_group(3, &self.tex_bg, &[]);
1457            rp.set_bind_group(4, &self.extra_tex_bg, &[]);
1458
1459            // Ground plane (scaled)
1460            rp.set_vertex_buffer(0, self.mesh_plane.vertex_buf.slice(..));
1461            rp.set_index_buffer(
1462                self.mesh_plane.index_buf.slice(..),
1463                wgpu::IndexFormat::Uint32,
1464            );
1465            rp.set_vertex_buffer(1, plane_buf.slice(..));
1466            rp.draw_indexed(0..self.mesh_plane.index_count, 0, 0..1);
1467
1468            // Cubes (instances)
1469            rp.set_vertex_buffer(0, self.mesh_cube.vertex_buf.slice(..));
1470            rp.set_index_buffer(
1471                self.mesh_cube.index_buf.slice(..),
1472                wgpu::IndexFormat::Uint32,
1473            );
1474            rp.set_vertex_buffer(1, self.instance_buf.slice(..));
1475            let inst_count = vis_count as u32;
1476            if inst_count > 0 {
1477                rp.draw_indexed(0..self.mesh_cube.index_count, 0, 0..inst_count);
1478            }
1479
1480            // External mesh if present
1481            if let (Some(mesh), Some(ibuf)) = (&self.mesh_external, ext_ibuf.as_ref()) {
1482                rp.set_vertex_buffer(0, mesh.vertex_buf.slice(..));
1483                rp.set_index_buffer(mesh.index_buf.slice(..), wgpu::IndexFormat::Uint32);
1484                rp.set_vertex_buffer(1, ibuf.slice(..));
1485                rp.draw_indexed(0..mesh.index_count, 0, 0..1);
1486            }
1487        }
1488
1489        // Postprocess HDR to surface
1490        {
1491            let mut pp = enc.begin_render_pass(&wgpu::RenderPassDescriptor{
1492                label: Some("post pass"),
1493                color_attachments: &[Some(wgpu::RenderPassColorAttachment{
1494                    view: &view,
1495                    resolve_target: None,
1496                    ops: wgpu::Operations{ load: wgpu::LoadOp::Clear(wgpu::Color::BLACK), store: wgpu::StoreOp::Store },
1497                })],
1498                depth_stencil_attachment: None,
1499                timestamp_writes: None,
1500                occlusion_query_set: None,
1501            });
1502            pp.set_pipeline(&self.post_pipeline);
1503            pp.set_bind_group(0, &self.post_bind_group, &[]);
1504            pp.draw(0..3, 0..1);
1505        }
1506
1507        self.queue.submit(Some(enc.finish()));
1508        frame.present();
1509        Ok(())
1510    }

1512    pub fn draw_into(&mut self, view: &wgpu::TextureView, enc: &mut wgpu::CommandEncoder) -> Result<()> {
1513        // Render sky/scene into HDR
1514        self.sky.render(enc, &self.hdr_view, &self.depth.view, Mat4::from_cols_array_2d(&self.camera_ubo.view_proj), &self.queue)?;
1515
1516        // Plane instance buffer
1517        let plane_scale = glam::Mat4::from_scale(vec3(50.0, 1.0, 50.0));
1518        let plane_inst = Instance {
1519            transform: plane_scale,
1520            color: [0.1, 0.12, 0.14, 1.0],
1521        }
1522        .raw();
1523        let plane_buf = self
1524            .device
1525            .create_buffer_init(&wgpu::util::BufferInitDescriptor {
1526                label: Some("plane inst"),
1527                contents: bytemuck::bytes_of(&plane_inst),
1528                usage: wgpu::BufferUsages::VERTEX,
1529            });
1530
1531        // Main pass (load color written by sky)
1532        // Prepare external mesh instance buffer if needed
1533        let ext_ibuf: Option<wgpu::Buffer> = if self.mesh_external.is_some() {
1534            let inst = Instance{ transform: Mat4::IDENTITY, color: [1.0,1.0,1.0,1.0] }.raw();
1535            Some(self.device.create_buffer_init(&wgpu::util::BufferInitDescriptor{
1536                label: Some("ext inst"),
1537                contents: bytemuck::bytes_of(&inst),
1538                usage: wgpu::BufferUsages::VERTEX,
1539            }))
1540        } else { None };
1541
1542        // Frustum cull
1543        let (vis_raws, vis_count) = self.build_visible_instances();
1544        if vis_count > 0 {
1545            self.queue.write_buffer(&self.instance_buf, 0, bytemuck::cast_slice(&vis_raws));
1546        }
1547
1548        // Shadow passes (depth only) - one per cascade layer
1549        for (idx, layer_view) in [&self.shadow_layer0_view, &self.shadow_layer1_view].iter().enumerate() {
1550            let mat = if idx == 0 { self.cascade0 } else { self.cascade1 };
1551            let arr = mat.to_cols_array();
1552            self.queue.write_buffer(&self.light_buf, 0, bytemuck::cast_slice(&arr));
1553            let mut sp = enc.begin_render_pass(&wgpu::RenderPassDescriptor{
1554                label: Some("shadow pass"),
1555                color_attachments: &[],
1556                depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment{
1557                    view: layer_view,
1558                    depth_ops: Some(wgpu::Operations{ load: wgpu::LoadOp::Clear(1.0), store: wgpu::StoreOp::Store }),
1559                    stencil_ops: None,
1560                }),
1561                timestamp_writes: None,
1562                occlusion_query_set: None,
1563            });
1564            sp.set_pipeline(&self.shadow_pipeline);
1565            sp.set_bind_group(0, &self.light_bg, &[]);
1566            sp.set_vertex_buffer(0, self.mesh_plane.vertex_buf.slice(..));
1567            sp.set_index_buffer(self.mesh_plane.index_buf.slice(..), wgpu::IndexFormat::Uint32);
1568            sp.set_vertex_buffer(1, plane_buf.slice(..));
1569            sp.draw_indexed(0..self.mesh_plane.index_count, 0, 0..1);
1570            sp.set_vertex_buffer(0, self.mesh_cube.vertex_buf.slice(..));
1571            sp.set_index_buffer(self.mesh_cube.index_buf.slice(..), wgpu::IndexFormat::Uint32);
1572            sp.set_vertex_buffer(1, self.instance_buf.slice(..));
1573            let inst_count = vis_count as u32;
1574            if inst_count > 0 { sp.draw_indexed(0..self.mesh_cube.index_count, 0, 0..inst_count); }
1575            if let (Some(mesh), Some(ibuf)) = (&self.mesh_external, ext_ibuf.as_ref()) {
1576                sp.set_vertex_buffer(0, mesh.vertex_buf.slice(..));
1577                sp.set_index_buffer(mesh.index_buf.slice(..), wgpu::IndexFormat::Uint32);
1578                sp.set_vertex_buffer(1, ibuf.slice(..));
1579                sp.draw_indexed(0..mesh.index_count, 0, 0..1);
1580            }
1581        }
1582        // Restore full light buffer for main pass
1583        {
1584            let mut data: Vec<f32> = Vec::with_capacity(36);
1585            data.extend_from_slice(&self.cascade0.to_cols_array());
1586            data.extend_from_slice(&self.cascade1.to_cols_array());
1587            data.push(self.split0);
1588            data.push(self.split1);
1589            data.push(0.0);
1590            data.push(0.0);
1591            self.queue.write_buffer(&self.light_buf, 0, bytemuck::cast_slice(&data));
1592        }
1593
1594        // Prepare a temporary instance buffer for skinned mesh if needed
1595        let sk_inst_buf: Option<wgpu::Buffer> = if self.skinned_mesh.is_some() {
1596            let inst = Instance{ transform: Mat4::IDENTITY, color: [1.0,1.0,1.0,1.0] }.raw();
1597            Some(self.device.create_buffer_init(&wgpu::util::BufferInitDescriptor{
1598                label: Some("skinned inst"),
1599                contents: bytemuck::bytes_of(&inst),
1600                usage: wgpu::BufferUsages::VERTEX,
1601            }))
1602        } else { None };
1603
1604        let mut rp = enc.begin_render_pass(&wgpu::RenderPassDescriptor {
1605            label: Some("main pass"),
1606            color_attachments: &[Some(wgpu::RenderPassColorAttachment {
1607                view: &self.hdr_view,
1608                resolve_target: None,
1609                ops: wgpu::Operations {
1610                    load: wgpu::LoadOp::Load,
1611                    store: wgpu::StoreOp::Store,
1612                },
1613            })],
1614            depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
1615                view: &self.depth.view,
1616                depth_ops: Some(wgpu::Operations {
1617                    load: wgpu::LoadOp::Clear(1.0),
1618                    store: wgpu::StoreOp::Store,
1619                }),
1620                stencil_ops: None,
1621            }),
1622            timestamp_writes: None,
1623            occlusion_query_set: None,
1624        });
1625
1626    rp.set_pipeline(&self.pipeline);
1627    rp.set_bind_group(0, &self.camera_bind_group, &[]);
1628    rp.set_bind_group(1, &self.material_bg, &[]);
1629    rp.set_bind_group(2, &self.light_bg, &[]);
1630    rp.set_bind_group(3, &self.tex_bg, &[]);
1631    rp.set_bind_group(4, &self.extra_tex_bg, &[]);
1632        // Ground plane
1633        rp.set_vertex_buffer(0, self.mesh_plane.vertex_buf.slice(..));
1634        rp.set_index_buffer(self.mesh_plane.index_buf.slice(..), wgpu::IndexFormat::Uint32);
1635        rp.set_vertex_buffer(1, plane_buf.slice(..));
1636        rp.draw_indexed(0..self.mesh_plane.index_count, 0, 0..1);
1637        // Cubes
1638        rp.set_vertex_buffer(0, self.mesh_cube.vertex_buf.slice(..));
1639        rp.set_index_buffer(self.mesh_cube.index_buf.slice(..), wgpu::IndexFormat::Uint32);
1640        rp.set_vertex_buffer(1, self.instance_buf.slice(..));
1641        let inst_count = vis_count as u32;
1642        if inst_count > 0 {
1643            rp.draw_indexed(0..self.mesh_cube.index_count, 0, 0..inst_count);
1644        }
1645
1646        // External mesh
1647        if let (Some(mesh), Some(ibuf)) = (&self.mesh_external, ext_ibuf.as_ref()) {
1648            rp.set_vertex_buffer(0, mesh.vertex_buf.slice(..));
1649            rp.set_index_buffer(mesh.index_buf.slice(..), wgpu::IndexFormat::Uint32);
1650            rp.set_vertex_buffer(1, ibuf.slice(..));
1651            rp.draw_indexed(0..mesh.index_count, 0, 0..1);
1652        }
1653        // Skinned mesh (if present)
1654        if let (Some((vbuf, ibuf, index_count)), Some(sk_inst)) = (&self.skinned_mesh, sk_inst_buf.as_ref()) {
1655            rp.set_pipeline(&self.skinned_pipeline);
1656            rp.set_bind_group(0, &self.camera_bind_group, &[]);
1657            rp.set_bind_group(1, &self.material_bg, &[]);
1658            rp.set_bind_group(2, &self.light_bg, &[]);
1659            rp.set_bind_group(3, &self.tex_bg, &[]);
1660            rp.set_bind_group(4, &self.skin_bg, &[]);
1661            rp.set_bind_group(5, &self.extra_tex_bg, &[]);
1662            rp.set_vertex_buffer(0, vbuf.slice(..));
1663            rp.set_index_buffer(ibuf.slice(..), wgpu::IndexFormat::Uint32);
1664            rp.set_vertex_buffer(1, sk_inst.slice(..));
1665            rp.draw_indexed(0..*index_count, 0, 0..1);
1666        }
1667        drop(rp);
1668
1669        // Post to surface view provided
1670        let mut pp = enc.begin_render_pass(&wgpu::RenderPassDescriptor{
1671            label: Some("post pass (external)"),
1672            color_attachments: &[Some(wgpu::RenderPassColorAttachment{
1673                view,
1674                resolve_target: None,
1675                ops: wgpu::Operations{ load: wgpu::LoadOp::Clear(wgpu::Color::BLACK), store: wgpu::StoreOp::Store },
1676            })],
1677            depth_stencil_attachment: None,
1678            timestamp_writes: None,
1679            occlusion_query_set: None,
1680        });
1681        pp.set_pipeline(&self.post_pipeline);
1682        pp.set_bind_group(0, &self.post_bind_group, &[]);
1683        pp.draw(0..3, 0..1);
1684
1685        Ok(())
1686    }

1700    pub fn render_with<F>(&mut self, f: F) -> Result<()>
1701    where
1702        F: FnOnce(
1703            &wgpu::TextureView,
1704            &mut wgpu::CommandEncoder,
1705            &wgpu::Device,
1706            &wgpu::Queue,
1707            (u32, u32),
1708        ),
1709    {
1710        let frame = self.surface.get_current_texture()?;
1711        let view = frame
1712            .texture
1713            .create_view(&wgpu::TextureViewDescriptor::default());
1714        let mut enc = self
1715            .device
1716            .create_command_encoder(&wgpu::CommandEncoderDescriptor {
1717                label: Some("encoder"),
1718            });
1719
1720    // First render the 3D scene into the frame (draw_into posts to view)
1721    self.draw_into(&view, &mut enc)?;
1722
1723        // Then allow caller to composite additional passes (e.g., egui)
1724        f(
1725            &view,
1726            &mut enc,
1727            &self.device,
1728            &self.queue,
1729            self.surface_size(),
1730        );
1731
1732        self.queue.submit(std::iter::once(enc.finish()));
1733        frame.present();
1734        Ok(())
1735    }

1738// --- Simple CPU frustum culling for instances ---
1739impl Renderer {
1740    fn build_visible_instances(&self) -> (Vec<InstanceRaw>, usize) {
1741        let m = Mat4::from_cols_array_2d(&self.camera_ubo.view_proj);
1742        let mt = m.transpose();
1743        let r0 = mt.x_axis;
1744        let r1 = mt.y_axis;
1745        let r2 = mt.z_axis;
1746        let r3 = mt.w_axis;
1747        let planes = [
1748            r3 + r0, // left
1749            r3 - r0, // right
1750            r3 + r1, // bottom
1751            r3 - r1, // top
1752            r3 + r2, // near
1753            r3 - r2, // far
1754        ];
1755        let norm_planes: Vec<(glam::Vec3, f32)> = planes
1756            .iter()
1757            .map(|p| {
1758                let n = glam::Vec3::new(p.x, p.y, p.z);
1759                let len = n.length().max(1e-6);
1760                (n / len, p.w / len)
1761            })
1762            .collect();
1763
1764        let mut out = Vec::with_capacity(self.instances.len());
1765        for inst in &self.instances {
1766            let center = inst.transform.w_axis.truncate();
1767            // approximate radius from basis vectors length (half-extents length)
1768            let sx = inst.transform.x_axis.truncate().length();
1769            let sy = inst.transform.y_axis.truncate().length();
1770            let sz = inst.transform.z_axis.truncate().length();
1771            let radius = 0.5 * glam::Vec3::new(sx, sy, sz).length();
1772            if inside_frustum_sphere(center, radius, &norm_planes) {
1773                out.push(inst.raw());
1774            }
1775        }
1776        let count = out.len();
1777        (out, count)
1778    }
1779}
1780
1781fn inside_frustum_sphere(center: glam::Vec3, radius: f32, planes: &[(glam::Vec3, f32)]) -> bool {
1782    for (n, d) in planes.iter() {
1783        if n.dot(center) + d < -radius {
1784            return false;
1785        }
1786    }
1787    true
1788}

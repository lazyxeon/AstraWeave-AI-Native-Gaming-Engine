1//! Minimal render graph scaffolding for Phase 2: establishes a deterministic, pluggable pass graph.
2//! This provides a Bevy/Fyrox-like pattern while staying optional and non-invasive to `Renderer`.
3
4use std::collections::BTreeMap;
5use anyhow::Context as _;
6
7/// Typed GPU resources passed between graph nodes.
8/// Keep this minimal for now; extend as we integrate more passes.
9pub enum Resource {
10    Texture(wgpu::Texture),
11    View(wgpu::TextureView),
12    Buffer(wgpu::Buffer),
13}
14
15/// A simple typed resource handle registry for graph nodes to pass data.
16#[derive(Default)]
17pub struct ResourceTable {
18    map: BTreeMap<String, Resource>,
19}
20
21/// Context passed to graph nodes. This will carry wgpu device/queue and shared resources.
22pub struct GraphContext<'a> {
23    /// Arbitrary user context for integration (e.g., &mut Renderer)
24    pub user: &'a mut dyn std::any::Any,
25    /// Named transient resources produced/consumed by nodes
26    pub resources: ResourceTable,
27    /// Optional GPU context for nodes that record commands
28    pub device: Option<&'a wgpu::Device>,
29    pub queue: Option<&'a wgpu::Queue>,
30    pub encoder: Option<&'a mut wgpu::CommandEncoder>,
31    /// Optional primary render target view provided by the driver (e.g., surface view)
32    pub primary_view: Option<&'a wgpu::TextureView>,
33}
34
35impl<'a> GraphContext<'a> {
36    pub fn new(user: &'a mut dyn std::any::Any) -> Self {
37        Self { user, resources: ResourceTable::default(), device: None, queue: None, encoder: None, primary_view: None }
38    }
39
40    /// Attach GPU context for nodes that perform GPU work.
41    pub fn with_gpu(
42        mut self,
43        device: &'a wgpu::Device,
44        queue: &'a wgpu::Queue,
45        encoder: &'a mut wgpu::CommandEncoder,
46    ) -> Self {
47        self.device = Some(device);
48        self.queue = Some(queue);
49        self.encoder = Some(encoder);
50        self
51    }
52
53    pub fn with_primary_view(mut self, view: &'a wgpu::TextureView) -> Self {
54        self.primary_view = Some(view);
55        self
56    }
57}
58
59impl ResourceTable {
60    pub fn insert_view(&mut self, key: impl Into<String>, view: wgpu::TextureView) {
61        self.map.insert(key.into(), Resource::View(view));
62    }
63    pub fn insert_tex(&mut self, key: impl Into<String>, tex: wgpu::Texture) {
64        self.map.insert(key.into(), Resource::Texture(tex));
65    }
66    pub fn insert_buf(&mut self, key: impl Into<String>, buf: wgpu::Buffer) {
67        self.map.insert(key.into(), Resource::Buffer(buf));
68    }
69    pub fn view(&self, key: &str) -> anyhow::Result<&wgpu::TextureView> {
70        match self.map.get(key).with_context(|| format!("resource '{}' not found", key))? {
71            Resource::View(v) => Ok(v),
72            _ => anyhow::bail!("resource '{}' is not a TextureView", key),
73        }
74    }
75    pub fn view_mut(&mut self, key: &str) -> anyhow::Result<&mut wgpu::TextureView> {
76        match self.map.get_mut(key).with_context(|| format!("resource '{}' not found", key))? {
77            Resource::View(v) => Ok(v),
78            _ => anyhow::bail!("resource '{}' is not a TextureView", key),
79        }
80    }
81    /// Get a target view by key, falling back to `primary_view` when the key is "surface".
82    pub fn target_view<'a>(&'a self, key: &str, primary_view: Option<&'a wgpu::TextureView>) -> anyhow::Result<&'a wgpu::TextureView> {
83        if key == "surface" {
84            if let Some(v) = primary_view { return Ok(v); }
85        }
86        self.view(key)
87    }
88}

90/// Trait for a render graph node. Nodes should be deterministic and side-effect free beyond GPU work.
91pub trait RenderNode {
92    fn name(&self) -> &str;
93    fn run(&mut self, ctx: &mut GraphContext) -> anyhow::Result<()>;
94}
95
96/// A very small, linear render graph. Edges are expressed as node ordering for now.
97pub struct RenderGraph {
98    nodes: Vec<Box<dyn RenderNode + Send + Sync>>, // keep Send+Sync for future parallelization
99}

103/// A node that clears a target view to a color, producing a named view resource.
104pub struct ClearNode {
105    name: String,
106    target_key: String,
107    color: wgpu::Color,
108}
109
110impl ClearNode {
111    pub fn new(name: impl Into<String>, target_key: impl Into<String>, color: wgpu::Color) -> Self {
112        Self { name: name.into(), target_key: target_key.into(), color }
113    }
114}
115
116impl RenderNode for ClearNode {
117    fn name(&self) -> &str { &self.name }
118    fn run(&mut self, ctx: &mut GraphContext) -> anyhow::Result<()> {
119        let _device = ctx.device.context("ClearNode requires device")?;
120        let view = ctx.resources.target_view(&self.target_key, ctx.primary_view)?;
121        let enc = ctx.encoder.as_deref_mut().context("ClearNode requires encoder")?;
122        let rp = enc.begin_render_pass(&wgpu::RenderPassDescriptor {
123            label: Some(&format!("clear:{}", self.name)),
124            color_attachments: &[Some(wgpu::RenderPassColorAttachment {
125                view,
126                resolve_target: None,
127                ops: wgpu::Operations { load: wgpu::LoadOp::Clear(self.color), store: wgpu::StoreOp::Store },
128            })],
129            depth_stencil_attachment: None,
130            timestamp_writes: None,
131            occlusion_query_set: None,
132        });
133        drop((_device, rp)); // rp dropped to end pass
134        Ok(())
135    }
136}

138/// A node that defers to `Renderer::draw_into` to render the 3D scene into a target view.
139pub struct RendererMainNode {
140    name: String,
141    target_key: String,
142}
143
144impl RendererMainNode {
145    pub fn new(name: impl Into<String>, target_key: impl Into<String>) -> Self {
146        Self { name: name.into(), target_key: target_key.into() }
147    }
148}
149
150impl RenderNode for RendererMainNode {
151    fn name(&self) -> &str { &self.name }
152    fn run(&mut self, ctx: &mut GraphContext) -> anyhow::Result<()> {
153        // For now, just validate that the target exists; the main scene draw is handled
154        // by the caller (e.g., Renderer::render_with). This keeps the node integration simple.
155        let _ = ctx.resources.target_view(&self.target_key, ctx.primary_view)?;
156        let _ = ctx.encoder.as_deref_mut().context("RendererMainNode requires encoder")?;
157        Ok(())
158    }
159}

161impl Default for RenderGraph {
162    fn default() -> Self { Self { nodes: Vec::new() } }
163}
164
165impl RenderGraph {
166    pub fn new() -> Self { Self::default() }
167
168    pub fn add_node<N>(&mut self, node: N) where N: RenderNode + Send + Sync + 'static {
169        self.nodes.push(Box::new(node));
170    }
171
172    /// Execute nodes in insertion order. Deterministic by construction.
173    pub fn execute(&mut self, ctx: &mut GraphContext) -> anyhow::Result<()> {
174        for n in self.nodes.iter_mut() { n.run(ctx)?; }
175        Ok(())
176    }
177}

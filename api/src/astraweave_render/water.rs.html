1//! Water rendering system with animated Gerstner waves
2//!
3//! Provides realistic ocean simulation with:
4//! - 4 summed Gerstner wave components
5//! - Fresnel-based reflections
6//! - Depth-based color blending
7//! - Animated foam on wave crests
8
9use glam::{Mat4, Vec3};
10use wgpu::util::DeviceExt;
11
12/// Water uniforms for shader
13#[repr(C)]
14#[derive(Debug, Clone, Copy, bytemuck::Pod, bytemuck::Zeroable)]
15pub struct WaterUniforms {
16    pub view_proj: [[f32; 4]; 4],      // 0-64
17    pub camera_pos: [f32; 3],          // 64-76
18    pub time: f32,                     // 76-80
19    pub water_color_deep: [f32; 3],    // 80-92
20    pub _pad0: f32,                    // 92-96
21    pub water_color_shallow: [f32; 3], // 96-108
22    pub _pad1: f32,                    // 108-112
23    pub foam_color: [f32; 3],          // 112-124
24    pub foam_threshold: f32,           // 124-128
25}
26
27impl Default for WaterUniforms {
28    fn default() -> Self {
29        Self {
30            view_proj: Mat4::IDENTITY.to_cols_array_2d(),
31            camera_pos: [0.0, 5.0, -10.0],
32            time: 0.0,
33            water_color_deep: [0.02, 0.08, 0.2], // Deep ocean blue
34            _pad0: 0.0,
35            water_color_shallow: [0.1, 0.4, 0.5], // Turquoise shallow
36            _pad1: 0.0,
37            foam_color: [0.95, 0.98, 1.0], // White foam
38            foam_threshold: 0.6,
39        }
40    }
41}

43/// Water vertex (position + UV)
44#[repr(C)]
45#[derive(Debug, Clone, Copy, bytemuck::Pod, bytemuck::Zeroable)]
46pub struct WaterVertex {
47    pub position: [f32; 3],
48    pub uv: [f32; 2],
49}
50
51impl WaterVertex {
52    pub fn desc() -> wgpu::VertexBufferLayout<'static> {
53        wgpu::VertexBufferLayout {
54            array_stride: std::mem::size_of::<WaterVertex>() as wgpu::BufferAddress,
55            step_mode: wgpu::VertexStepMode::Vertex,
56            attributes: &[
57                wgpu::VertexAttribute {
58                    offset: 0,
59                    shader_location: 0,
60                    format: wgpu::VertexFormat::Float32x3,
61                },
62                wgpu::VertexAttribute {
63                    offset: 12,
64                    shader_location: 1,
65                    format: wgpu::VertexFormat::Float32x2,
66                },
67            ],
68        }
69    }
70}

72/// Water rendering system
73pub struct WaterRenderer {
74    pipeline: wgpu::RenderPipeline,
75    _bind_group_layout: wgpu::BindGroupLayout,
76    bind_group: wgpu::BindGroup,
77    uniform_buffer: wgpu::Buffer,
78    vertex_buffer: wgpu::Buffer,
79    index_buffer: wgpu::Buffer,
80    index_count: u32,
81    uniforms: WaterUniforms,
82}
83
84impl WaterRenderer {
85    /// Create a new water renderer
86    pub fn new(
87        device: &wgpu::Device,
88        surface_format: wgpu::TextureFormat,
89        depth_format: wgpu::TextureFormat,
90    ) -> Self {
91        // Load shader
92        let shader_source = include_str!("shaders/water.wgsl");
93        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
94            label: Some("water_shader"),
95            source: wgpu::ShaderSource::Wgsl(shader_source.into()),
96        });
97
98        // Uniform buffer
99        let uniforms = WaterUniforms::default();
100        let uniform_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
101            label: Some("water_uniforms"),
102            contents: bytemuck::bytes_of(&uniforms),
103            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
104        });
105
106        // Bind group layout
107        let bind_group_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
108            label: Some("water_bind_group_layout"),
109            entries: &[wgpu::BindGroupLayoutEntry {
110                binding: 0,
111                visibility: wgpu::ShaderStages::VERTEX | wgpu::ShaderStages::FRAGMENT,
112                ty: wgpu::BindingType::Buffer {
113                    ty: wgpu::BufferBindingType::Uniform,
114                    has_dynamic_offset: false,
115                    min_binding_size: None,
116                },
117                count: None,
118            }],
119        });
120
121        // Bind group
122        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
123            label: Some("water_bind_group"),
124            layout: &bind_group_layout,
125            entries: &[wgpu::BindGroupEntry {
126                binding: 0,
127                resource: uniform_buffer.as_entire_binding(),
128            }],
129        });
130
131        // Pipeline layout
132        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
133            label: Some("water_pipeline_layout"),
134            bind_group_layouts: &[&bind_group_layout],
135            push_constant_ranges: &[],
136        });
137
138        // Render pipeline with alpha blending
139        let pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
140            label: Some("water_pipeline"),
141            layout: Some(&pipeline_layout),
142            vertex: wgpu::VertexState {
143                module: &shader,
144                entry_point: Some("vs_main"),
145                buffers: &[WaterVertex::desc()],
146                compilation_options: Default::default(),
147            },
148            fragment: Some(wgpu::FragmentState {
149                module: &shader,
150                entry_point: Some("fs_main"),
151                targets: &[Some(wgpu::ColorTargetState {
152                    format: surface_format,
153                    blend: Some(wgpu::BlendState::ALPHA_BLENDING),
154                    write_mask: wgpu::ColorWrites::ALL,
155                })],
156                compilation_options: Default::default(),
157            }),
158            primitive: wgpu::PrimitiveState {
159                topology: wgpu::PrimitiveTopology::TriangleList,
160                strip_index_format: None,
161                front_face: wgpu::FrontFace::Ccw,
162                cull_mode: None, // DEBUG: Render both sides
163                unclipped_depth: false,
164                polygon_mode: wgpu::PolygonMode::Fill,
165                conservative: false,
166            },
167            depth_stencil: Some(wgpu::DepthStencilState {
168                format: depth_format,
169                depth_write_enabled: false, // Transparent, don't write depth
170                depth_compare: wgpu::CompareFunction::LessEqual, // Normal depth testing
171                stencil: wgpu::StencilState::default(),
172                bias: wgpu::DepthBiasState::default(),
173            }),
174            multisample: wgpu::MultisampleState::default(),
175            multiview: None,
176            cache: None,
177        });
178
179        // Generate water plane mesh
180        // Generate water plane mesh (larger 500x500 area, 128x128 grid)
181        let (vertices, indices) = Self::generate_water_plane(500.0, 128);
182
183        let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
184            label: Some("water_vertex_buffer"),
185            contents: bytemuck::cast_slice(&vertices),
186            usage: wgpu::BufferUsages::VERTEX,
187        });
188
189        let index_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
190            label: Some("water_index_buffer"),
191            contents: bytemuck::cast_slice(&indices),
192            usage: wgpu::BufferUsages::INDEX,
193        });
194
195        Self {
196            pipeline,
197            _bind_group_layout: bind_group_layout,
198            bind_group,
199            uniform_buffer,
200            vertex_buffer,
201            index_buffer,
202            index_count: indices.len() as u32,
203            uniforms,
204        }
205    }
206
207    /// Generate a subdivided water plane
208    fn generate_water_plane(size: f32, subdivisions: u32) -> (Vec<WaterVertex>, Vec<u32>) {
209        let mut vertices = Vec::new();
210        let mut indices = Vec::new();
211
212        let half_size = size / 2.0;
213        let step = size / subdivisions as f32;
214
215        // Generate vertices
216        for z in 0..=subdivisions {
217            for x in 0..=subdivisions {
218                let pos_x = -half_size + x as f32 * step;
219                let pos_z = -half_size + z as f32 * step;
220                let u = x as f32 / subdivisions as f32;
221                let v = z as f32 / subdivisions as f32;
222
223                vertices.push(WaterVertex {
224                    position: [pos_x, 2.0, pos_z], // Normal Water Level (Y=2.0)
225
226                    uv: [u, v],
227                });
228            }
229        }
230
231        // Generate indices
232        for z in 0..subdivisions {
233            for x in 0..subdivisions {
234                let top_left = z * (subdivisions + 1) + x;
235                let top_right = top_left + 1;
236                let bottom_left = (z + 1) * (subdivisions + 1) + x;
237                let bottom_right = bottom_left + 1;
238
239                // First triangle
240                indices.push(top_left);
241                indices.push(bottom_left);
242                indices.push(top_right);
243
244                // Second triangle
245                indices.push(top_right);
246                indices.push(bottom_left);
247                indices.push(bottom_right);
248            }
249        }
250
251        (vertices, indices)
252    }

254    /// Update water state for animation
255    pub fn update(&mut self, queue: &wgpu::Queue, view_proj: Mat4, camera_pos: Vec3, time: f32) {
256        self.uniforms.view_proj = view_proj.to_cols_array_2d();
257        self.uniforms.camera_pos = camera_pos.into();
258        self.uniforms.time = time;
259        queue.write_buffer(&self.uniform_buffer, 0, bytemuck::bytes_of(&self.uniforms));
260    }
261
262    /// Set water level (Y position)
263    pub fn set_water_level(&mut self, _level: f32) {
264        // Water level is controlled by the uniform, already at y=0 in mesh
265    }
266
267    /// Render the water surface
268    pub fn render<'a>(&'a self, render_pass: &mut wgpu::RenderPass<'a>) {
269        render_pass.set_pipeline(&self.pipeline);
270        render_pass.set_bind_group(0, &self.bind_group, &[]);
271        render_pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
272        render_pass.set_index_buffer(self.index_buffer.slice(..), wgpu::IndexFormat::Uint32);
273        render_pass.draw_indexed(0..self.index_count, 0, 0..1);
274    }

301    #[test]
302    fn test_water_renderer_new_and_update() {
303        pollster::block_on(async {
304            let instance = wgpu::Instance::new(&wgpu::InstanceDescriptor::default());
305            let adapter = instance
306                .request_adapter(&wgpu::RequestAdapterOptions::default())
307                .await;
308            if let Ok(adapter) = adapter {
309                let (device, queue) = adapter
310                    .request_device(&wgpu::DeviceDescriptor::default())
311                    .await
312                    .unwrap();
313                let mut renderer = WaterRenderer::new(
314                    &device,
315                    wgpu::TextureFormat::Rgba8UnormSrgb,
316                    wgpu::TextureFormat::Depth32Float,
317                );
318
319                assert_eq!(renderer.index_count, 128 * 128 * 6);
320
321                let view_proj = Mat4::IDENTITY;
322                let camera_pos = Vec3::new(1.0, 2.0, 3.0);
323                let time = 10.0;
324
325                renderer.update(&queue, view_proj, camera_pos, time);
326
327                assert_eq!(renderer.uniforms.time, 10.0);
328                assert_eq!(renderer.uniforms.camera_pos, [1.0, 2.0, 3.0]);
329
330                renderer.set_water_level(5.0);
331            }
332        });
333    }

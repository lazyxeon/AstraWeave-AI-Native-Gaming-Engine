1use crate::types::InstanceRaw;
2use glam::{vec3, Mat4, Vec3};
3use rand::Rng;
4
5#[derive(Clone, Copy, Debug)]
6pub enum WeatherKind {
7    None,
8    Rain,
9    WindTrails,
10}
11
12pub struct WeatherFx {
13    kind: WeatherKind,
14    particles: Vec<Particle>,
15    buf: wgpu::Buffer,
16    max: usize,
17}
18
19#[derive(Clone, Copy, Debug)]
20struct Particle {
21    pos: Vec3,
22    vel: Vec3,
23    life: f32,
24    color: [f32; 4],
25    scale: Vec3,
26}
27
28impl WeatherFx {
29    pub fn new(device: &wgpu::Device, max: usize) -> Self {
30        let buf = device.create_buffer(&wgpu::BufferDescriptor {
31            label: Some("weather inst"),
32            size: (max * std::mem::size_of::<InstanceRaw>()) as u64,
33            usage: wgpu::BufferUsages::VERTEX | wgpu::BufferUsages::COPY_DST,
34            mapped_at_creation: false,
35        });
36        Self {
37            kind: WeatherKind::None,
38            particles: vec![],
39            buf,
40            max,
41        }
42    }
43
44    pub fn set_kind(&mut self, kind: WeatherKind) {
45        self.kind = kind;
46    }
47
48    pub fn update(&mut self, queue: &wgpu::Queue, dt: f32) {
49        match self.kind {
50            WeatherKind::None => {
51                self.particles.clear();
52            }
53            WeatherKind::Rain => self.tick_rain(dt),
54            WeatherKind::WindTrails => self.tick_wind(dt),
55        }
56        // upload
57        let raws: Vec<InstanceRaw> = self
58            .particles
59            .iter()
60            .map(|p| {
61                let m = Mat4::from_scale_rotation_translation(p.scale, glam::Quat::IDENTITY, p.pos);
62                InstanceRaw {
63                    model: m.to_cols_array_2d(),
64                    normal_matrix: [
65                        m.inverse().transpose().x_axis.truncate().to_array(),
66                        m.inverse().transpose().y_axis.truncate().to_array(),
67                        m.inverse().transpose().z_axis.truncate().to_array(),
68                    ],
69                    color: p.color,
70                    material_id: 0,
71                    _padding: [0; 3],
72                }
73            })
74            .collect();
75        queue.write_buffer(&self.buf, 0, bytemuck::cast_slice(&raws));
76    }
77
78    fn tick_rain(&mut self, dt: f32) {
79        let mut rng = rand::rng();
80        // spawn up to max
81        while self.particles.len() < self.max {
82            self.particles.push(Particle {
83                pos: vec3(
84                    rng.random_range(-25.0..25.0),
85                    rng.random_range(8.0..18.0),
86                    rng.random_range(-25.0..25.0),
87                ),
88                vel: vec3(0.0, -20.0, 0.0),
89                life: rng.random_range(0.5..1.5),
90                color: [0.7, 0.8, 1.0, 0.9],
91                scale: vec3(0.02, 0.5, 0.02),
92            });
93        }
94        // update
95        self.particles.retain_mut(|p| {
96            p.life -= dt;
97            p.pos += p.vel * dt;
98            p.pos.y > 0.0 && p.life > 0.0
99        });
100    }
101
102    fn tick_wind(&mut self, dt: f32) {
103        let mut rng = rand::rng();
104        while self.particles.len() < self.max {
105            self.particles.push(Particle {
106                pos: vec3(
107                    rng.random_range(-25.0..25.0),
108                    rng.random_range(0.5..4.0),
109                    rng.random_range(-25.0..25.0),
110                ),
111                vel: vec3(5.0, 0.0, 1.0),
112                life: rng.random_range(1.0..3.0),
113                color: [1.0, 1.0, 1.0, 0.3],
114                scale: vec3(0.05, 0.05, 0.8),
115            });
116        }
117        self.particles.retain_mut(|p| {
118            p.life -= dt;
119            p.pos += p.vel * dt;
120            p.life > 0.0
121        });
122    }
123
124    pub fn buffer(&self) -> &wgpu::Buffer {
125        &self.buf
126    }
127    pub fn count(&self) -> u32 {
128        self.particles.len() as u32
129    }
130}

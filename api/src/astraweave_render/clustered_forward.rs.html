1//! Complete Clustered Forward Rendering Implementation
2//!
3//! This module implements a full clustered forward rendering pipeline that supports
4//! 100+ dynamic lights efficiently by dividing the screen into 3D clusters and
5//! assigning lights to clusters for culling.
6
7use glam::{Mat4, Vec3};
8
9/// Configuration for clustered rendering
10#[derive(Debug, Clone, Copy)]
11pub struct ClusterConfig {
12    /// Number of clusters in X dimension (screen width)
13    pub cluster_x: u32,
14    /// Number of clusters in Y dimension (screen height)
15    pub cluster_y: u32,
16    /// Number of clusters in Z dimension (depth)
17    pub cluster_z: u32,
18    /// Near plane distance
19    pub near: f32,
20    /// Far plane distance
21    pub far: f32,
22}
23
24impl Default for ClusterConfig {
25    fn default() -> Self {
26        Self {
27            cluster_x: 16,
28            cluster_y: 16,
29            cluster_z: 32,
30            near: 0.1,
31            far: 1000.0,
32        }
33    }
34}

36/// GPU-compatible light structure
37/// Uses arrays instead of Vec4 for Pod/Zeroable compatibility
38#[repr(C)]
39#[derive(Debug, Clone, Copy, bytemuck::Pod, bytemuck::Zeroable)]
40pub struct GpuLight {
41    /// Position in world space (w = radius)
42    pub position: [f32; 4],
43    /// Color and intensity (w = intensity)
44    pub color: [f32; 4],
45}
46
47impl GpuLight {
48    pub fn new(position: Vec3, radius: f32, color: Vec3, intensity: f32) -> Self {
49        Self {
50            position: [position.x, position.y, position.z, radius],
51            color: [color.x, color.y, color.z, intensity],
52        }
53    }
54}

95impl ClusteredForwardRenderer {
96    /// Create a new clustered forward renderer
97    pub fn new(device: &wgpu::Device, config: ClusterConfig) -> Self {
98        let max_lights = 256;
99        let max_lights_per_cluster = 128;
100        let total_clusters = (config.cluster_x * config.cluster_y * config.cluster_z) as usize;
101
102        // Create buffers
103        let light_buffer = device.create_buffer(&wgpu::BufferDescriptor {
104            label: Some("Clustered Light Buffer"),
105            size: (max_lights * std::mem::size_of::<GpuLight>()) as u64,
106            usage: wgpu::BufferUsages::STORAGE | wgpu::BufferUsages::COPY_DST,
107            mapped_at_creation: false,
108        });
109
110        let cluster_buffer = device.create_buffer(&wgpu::BufferDescriptor {
111            label: Some("Cluster Buffer"),
112            size: (total_clusters * std::mem::size_of::<GpuCluster>()) as u64,
113            usage: wgpu::BufferUsages::STORAGE | wgpu::BufferUsages::COPY_DST,
114            mapped_at_creation: false,
115        });
116
117        let light_indices_buffer = device.create_buffer(&wgpu::BufferDescriptor {
118            label: Some("Light Indices Buffer"),
119            size: (total_clusters * max_lights_per_cluster * std::mem::size_of::<u32>()) as u64,
120            usage: wgpu::BufferUsages::STORAGE | wgpu::BufferUsages::COPY_DST,
121            mapped_at_creation: false,
122        });
123
124        // Create bind group layout
125        let cluster_bind_group_layout =
126            device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
127                label: Some("Cluster Bind Group Layout"),
128                entries: &[
129                    wgpu::BindGroupLayoutEntry {
130                        binding: 0,
131                        visibility: wgpu::ShaderStages::FRAGMENT,
132                        ty: wgpu::BindingType::Buffer {
133                            ty: wgpu::BufferBindingType::Storage { read_only: true },
134                            has_dynamic_offset: false,
135                            min_binding_size: None,
136                        },
137                        count: None,
138                    },
139                    wgpu::BindGroupLayoutEntry {
140                        binding: 1,
141                        visibility: wgpu::ShaderStages::FRAGMENT,
142                        ty: wgpu::BindingType::Buffer {
143                            ty: wgpu::BufferBindingType::Storage { read_only: true },
144                            has_dynamic_offset: false,
145                            min_binding_size: None,
146                        },
147                        count: None,
148                    },
149                    wgpu::BindGroupLayoutEntry {
150                        binding: 2,
151                        visibility: wgpu::ShaderStages::FRAGMENT,
152                        ty: wgpu::BindingType::Buffer {
153                            ty: wgpu::BufferBindingType::Storage { read_only: true },
154                            has_dynamic_offset: false,
155                            min_binding_size: None,
156                        },
157                        count: None,
158                    },
159                ],
160            });
161
162        // Create bind group
163        let cluster_bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
164            label: Some("Cluster Bind Group"),
165            layout: &cluster_bind_group_layout,
166            entries: &[
167                wgpu::BindGroupEntry {
168                    binding: 0,
169                    resource: light_buffer.as_entire_binding(),
170                },
171                wgpu::BindGroupEntry {
172                    binding: 1,
173                    resource: cluster_buffer.as_entire_binding(),
174                },
175                wgpu::BindGroupEntry {
176                    binding: 2,
177                    resource: light_indices_buffer.as_entire_binding(),
178                },
179            ],
180        });
181
182        Self {
183            config,
184            light_buffer,
185            cluster_buffer,
186            light_indices_buffer,
187            cluster_bind_group_layout,
188            cluster_bind_group,
189            lights: Vec::new(),
190            clusters: vec![
191                GpuCluster {
192                    min_bounds: [0.0; 4],
193                    max_bounds: [0.0; 4],
194                    light_offset: 0,
195                    light_count: 0,
196                    _padding: [0; 2],
197                };
198                total_clusters
199            ],
200            light_indices: Vec::new(),
201            _max_lights: max_lights,
202            max_lights_per_cluster,
203        }
204    }

211    /// Build clusters and assign lights
212    pub fn build_clusters(
213        &mut self,
214        queue: &wgpu::Queue,
215        view_matrix: Mat4,
216        _proj_matrix: Mat4,
217        screen_size: (u32, u32),
218    ) {
219        // Clear previous data
220        self.light_indices.clear();
221
222        let (_width, _height) = screen_size;
223        let _total_clusters =
224            (self.config.cluster_x * self.config.cluster_y * self.config.cluster_z) as usize;
225
226        // Calculate cluster bounds in view space
227        for z in 0..self.config.cluster_z {
228            for y in 0..self.config.cluster_y {
229                for x in 0..self.config.cluster_x {
230                    let cluster_idx = self.cluster_index(x, y, z);
231
232                    // Calculate cluster bounds
233                    let min_x = (x as f32 / self.config.cluster_x as f32) * 2.0 - 1.0;
234                    let max_x = ((x + 1) as f32 / self.config.cluster_x as f32) * 2.0 - 1.0;
235                    let min_y = (y as f32 / self.config.cluster_y as f32) * 2.0 - 1.0;
236                    let max_y = ((y + 1) as f32 / self.config.cluster_y as f32) * 2.0 - 1.0;
237
238                    // Exponential depth slicing for better distribution
239                    let near = self.config.near;
240                    let far = self.config.far;
241                    let min_z = near * (far / near).powf(z as f32 / self.config.cluster_z as f32);
242                    let max_z =
243                        near * (far / near).powf((z + 1) as f32 / self.config.cluster_z as f32);
244
245                    self.clusters[cluster_idx].min_bounds = [min_x, min_y, min_z, 1.0];
246                    self.clusters[cluster_idx].max_bounds = [max_x, max_y, max_z, 1.0];
247                    self.clusters[cluster_idx].light_offset = self.light_indices.len() as u32;
248                    self.clusters[cluster_idx].light_count = 0;
249
250                    // Assign lights to this cluster
251                    for (light_idx, light) in self.lights.iter().enumerate() {
252                        if self.light_intersects_cluster(
253                            light,
254                            &self.clusters[cluster_idx],
255                            view_matrix,
256                        ) {
257                            if self.clusters[cluster_idx].light_count
258                                < self.max_lights_per_cluster as u32
259                            {
260                                self.light_indices.push(light_idx as u32);
261                                self.clusters[cluster_idx].light_count += 1;
262                            }
263                        }
264                    }
265                }
266            }
267        }
268
269        // Upload to GPU
270        if !self.lights.is_empty() {
271            queue.write_buffer(&self.light_buffer, 0, bytemuck::cast_slice(&self.lights));
272        }
273        queue.write_buffer(
274            &self.cluster_buffer,
275            0,
276            bytemuck::cast_slice(&self.clusters),
277        );
278        if !self.light_indices.is_empty() {
279            queue.write_buffer(
280                &self.light_indices_buffer,
281                0,
282                bytemuck::cast_slice(&self.light_indices),
283            );
284        }
285    }

287    /// Get cluster index from 3D coordinates
288    fn cluster_index(&self, x: u32, y: u32, z: u32) -> usize {
289        (x + y * self.config.cluster_x + z * self.config.cluster_x * self.config.cluster_y) as usize
290    }

292    /// Check if a light intersects a cluster
293    fn light_intersects_cluster(
294        &self,
295        light: &GpuLight,
296        cluster: &GpuCluster,
297        view_matrix: Mat4,
298    ) -> bool {
299        // Transform light position to view space
300        let light_pos_4 =
301            Vec3::new(light.position[0], light.position[1], light.position[2]).extend(1.0);
302        let light_pos_view_4 = view_matrix * light_pos_4;
303        let light_pos_view = Vec3::new(light_pos_view_4.x, light_pos_view_4.y, light_pos_view_4.z);
304        let radius = light.position[3];
305
306        // Simple sphere-AABB intersection test
307        let closest_point = Vec3::new(
308            light_pos_view
309                .x
310                .clamp(cluster.min_bounds[0], cluster.max_bounds[0]),
311            light_pos_view
312                .y
313                .clamp(cluster.min_bounds[1], cluster.max_bounds[1]),
314            light_pos_view
315                .z
316                .clamp(cluster.min_bounds[2], cluster.max_bounds[2]),
317        );
318
319        let distance_sq = (light_pos_view - closest_point).length_squared();
320        distance_sq <= radius * radius
321    }

344/// WGSL shader code for clustered forward rendering
345pub const CLUSTERED_LIGHTING_SHADER: &str = r#"
346struct Light {
347    position: vec4<f32>,  // w = radius
348    color: vec4<f32>,     // w = intensity
349}
350
351struct Cluster {
352    min_bounds: vec4<f32>,
353    max_bounds: vec4<f32>,
354    light_offset: u32,
355    light_count: u32,
356    padding: vec2<u32>,
357}
358
359@group(2) @binding(0) var<storage, read> lights: array<Light>;
360@group(2) @binding(1) var<storage, read> clusters: array<Cluster>;
361@group(2) @binding(2) var<storage, read> light_indices: array<u32>;
362
363struct ClusterConfig {
364    cluster_x: u32,
365    cluster_y: u32,
366    cluster_z: u32,
367    near: f32,
368    far: f32,
369}
370
371fn get_cluster_index(frag_coord: vec3<f32>, config: ClusterConfig) -> u32 {
372    let x = u32(frag_coord.x / config.cluster_x);
373    let y = u32(frag_coord.y / config.cluster_y);
374    
375    // Exponential depth mapping
376    let depth = frag_coord.z;
377    let z_slice = log2(depth / config.near) / log2(config.far / config.near);
378    let z = u32(z_slice * f32(config.cluster_z));
379    
380    return x + y * config.cluster_x + z * config.cluster_x * config.cluster_y;
381}
382
383fn calculate_clustered_lighting(
384    world_pos: vec3<f32>,
385    normal: vec3<f32>,
386    view_pos: vec3<f32>,
387    albedo: vec3<f32>,
388    metallic: f32,
389    roughness: f32,
390    frag_coord: vec3<f32>,
391    config: ClusterConfig
392) -> vec3<f32> {
393    let cluster_idx = get_cluster_index(frag_coord, config);
394    let cluster = clusters[cluster_idx];
395    
396    var total_light = vec3<f32>(0.0);
397    
398    // Iterate through lights in this cluster
399    for (var i = 0u; i < cluster.light_count; i = i + 1u) {
400        let light_idx = light_indices[cluster.light_offset + i];
401        let light = lights[light_idx];
402        
403        let light_dir = light.position.xyz - world_pos;
404        let distance = length(light_dir);
405        let radius = light.position.w;
406        
407        // Skip if outside light radius
408        if (distance > radius) {
409            continue;
410        }
411        
412        let L = normalize(light_dir);
413        let V = normalize(view_pos - world_pos);
414        let H = normalize(L + V);
415        
416        let NdotL = max(dot(normal, L), 0.0);
417        let NdotV = max(dot(normal, V), 0.0);
418        let NdotH = max(dot(normal, H), 0.0);
419        
420        // Attenuation
421        let attenuation = 1.0 - pow(distance / radius, 4.0);
422        let attenuation_clamped = max(attenuation, 0.0);
423        
424        // Simple Blinn-Phong for now (can be extended to PBR)
425        let diffuse = albedo * NdotL;
426        let specular = pow(NdotH, 32.0) * (1.0 - roughness);
427        
428        let light_contribution = (diffuse + specular) * light.color.rgb * light.color.w * attenuation_clamped;
429        total_light = total_light + light_contribution;
430    }
431    
432    return total_light;
433}

82/// Configuration for the world generator
83#[derive(Debug, Clone, Serialize, Deserialize)]
84pub struct WorldConfig {
85    /// Random seed for generation
86    pub seed: u64,
87    /// Size of terrain chunks in world units
88    pub chunk_size: f32,
89    /// Resolution of heightmaps (vertices per chunk edge)
90    pub heightmap_resolution: u32,
91    /// Noise configuration for terrain generation
92    pub noise: NoiseConfig,
93    /// Climate configuration for biome assignment
94    pub climate: ClimateConfig,
95    /// Available biome configurations
96    pub biomes: Vec<BiomeConfig>,
97    /// Structure generation configuration
98    pub structures: structures::StructureConfig,
99}

120/// Main world generator that coordinates terrain, climate, and biome generation
121#[derive(Debug)]
122pub struct WorldGenerator {
123    config: WorldConfig,
124    noise: TerrainNoise,
125    climate: ClimateMap,
126    chunk_manager: ChunkManager,

130impl WorldGenerator {
131    /// Create a new world generator with the given configuration
132    pub fn new(config: WorldConfig) -> Self {
133        let noise = TerrainNoise::new(&config.noise, config.seed);
134        let climate = ClimateMap::new(&config.climate, config.seed + 1);
135        let chunk_manager = ChunkManager::new(config.chunk_size, config.heightmap_resolution);
136        let mut structure_config = config.structures.clone();
137        structure_config.seed = config.seed + 2; // Offset seed for structures
138        let structure_generator = structures::StructureGenerator::new(structure_config);
139
140        Self {
141            config,
142            noise,
143            climate,
144            chunk_manager,
145            structure_generator,
146        }
147    }

149    /// Generate a terrain chunk at the given world position with vegetation and resources
150    pub fn generate_chunk_with_scatter(
151        &mut self,
152        chunk_id: ChunkId,
153    ) -> anyhow::Result<(TerrainChunk, ScatterResult)> {
154        // Generate the basic terrain chunk (lock-free)
155        let chunk = self.generate_chunk(chunk_id)?;
156
157        // Register with chunk manager
158        self.chunk_manager.add_chunk(chunk.clone());
159
160        // Generate scatter for the chunk
161        let scatter_result = self.scatter_chunk_content(&chunk)?;
162
163        Ok((chunk, scatter_result))
164    }
165
166    /// Generate scatter content (vegetation and resources) for an existing chunk
167    pub fn scatter_chunk_content(&mut self, chunk: &TerrainChunk) -> anyhow::Result<ScatterResult> {
168        let mut result = ScatterResult::new(chunk.id());
169
170        // Create scatter system
171        let scatter_config = ScatterConfig::default();
172        let scatter = VegetationScatter::new(scatter_config);
173
174        // Sample the biome at the chunk center to determine configuration
175        let chunk_center = chunk.id().to_center_pos(self.config.chunk_size);
176        let center_biome = chunk
177            .get_biome_at_world_pos(chunk_center, self.config.chunk_size)
178            .unwrap_or(BiomeType::Grassland);
179
180        // Find the biome configuration
181        let biome_config = self
182            .config
183            .biomes
184            .iter()
185            .find(|b| b.biome_type == center_biome)
186            .unwrap_or(&self.config.biomes[0]);
187
188        // Generate vegetation
189        result.vegetation = scatter.scatter_vegetation(
190            chunk,
191            self.config.chunk_size,
192            biome_config,
193            self.config.seed + chunk.id().x as u64 * 1000 + chunk.id().z as u64,
194        )?;
195
196        // Generate resources
197        result.resources = scatter.scatter_resources(
198            chunk,
199            self.config.chunk_size,
200            biome_config,
201            self.config.seed + chunk.id().x as u64 * 2000 + chunk.id().z as u64,
202        )?;
203
204        // Generate structures
205        let structure_result = self.structure_generator.generate_structures(
206            chunk,
207            self.config.chunk_size,
208            center_biome,
209        )?;
210        result.structures = structure_result.structures;
211
212        Ok(result)
213    }

215    /// Generate a terrain chunk at the given position (lock-free, parallel-safe)
216    /// NOTE: Does NOT add to chunk_manager - caller must handle that separately
217    pub fn generate_chunk(&self, chunk_id: ChunkId) -> anyhow::Result<TerrainChunk> {
218        // Generate heightmap for this chunk (using SIMD if enabled)
219        #[cfg(feature = "simd-noise")]
220        let heightmap = noise_simd::SimdHeightmapGenerator::generate_heightmap_simd(
221            &self.noise,
222            chunk_id,
223            self.config.chunk_size,
224            self.config.heightmap_resolution,
225        )?;
226
227        #[cfg(not(feature = "simd-noise"))]
228        let heightmap = self.noise.generate_heightmap(
229            chunk_id,
230            self.config.chunk_size,
231            self.config.heightmap_resolution,
232        )?;
233
234        // Generate climate data for biome assignment
235        let climate_data = self.climate.sample_chunk(
236            chunk_id,
237            self.config.chunk_size,
238            self.config.heightmap_resolution,
239        )?;
240
241        // Assign biomes based on height and climate
242        let biome_map = self.assign_biomes(&heightmap, &climate_data)?;
243
244        // Create the terrain chunk
245        let mut chunk = TerrainChunk::new(chunk_id, heightmap, biome_map);
246
247        // Apply erosion if enabled
248        if self.config.noise.erosion_enabled {
249            chunk.apply_erosion(self.config.noise.erosion_strength)?;
250        }
251
252        // NOTE: chunk_manager.add_chunk() removed - caller handles registration
253        Ok(chunk)
254    }

256    /// Generate and register a chunk (mutable version for compatibility)
257    pub fn generate_and_register_chunk(

259        chunk_id: ChunkId,

261        let chunk = self.generate_chunk(chunk_id)?;
262        self.chunk_manager.add_chunk(chunk.clone());

264    }

271    /// Stream chunks around a center position, loading/unloading as needed
272    pub fn stream_chunks(&mut self, center: Vec3, radius: u32) -> anyhow::Result<Vec<ChunkId>> {
273        let chunks_to_load = self.chunk_manager.get_chunks_in_radius(center, radius);
274        let mut loaded = Vec::new();
275
276        for chunk_id in chunks_to_load {
277            if !self.chunk_manager.has_chunk(chunk_id) {
278                self.generate_and_register_chunk(chunk_id)?;
279                loaded.push(chunk_id);
280            }
281        }
282
283        // Unload chunks that are too far away
284        let unload_radius = radius + 2; // Keep a buffer
285        self.chunk_manager
286            .unload_distant_chunks(center, unload_radius);
287
288        Ok(loaded)
289    }

291    /// Assign biomes to heightmap points based on climate data
292    fn assign_biomes(
293        &self,
294        heightmap: &Heightmap,
295        climate_data: &[(f32, f32)], // (temperature, moisture) pairs
296    ) -> anyhow::Result<Vec<BiomeType>> {
297        let mut biome_map = Vec::with_capacity(climate_data.len());
298
299        for (i, &(temperature, moisture)) in climate_data.iter().enumerate() {
300            let height = heightmap.get_height_at_index(i);
301            let biome = self.find_best_biome(height, temperature, moisture);
302            biome_map.push(biome);
303        }
304
305        Ok(biome_map)
306    }
307
308    /// Find the best biome for given environmental conditions
309    fn find_best_biome(&self, height: f32, temperature: f32, moisture: f32) -> BiomeType {
310        let mut best_biome = BiomeType::Grassland;
311        let mut best_score = f32::NEG_INFINITY;
312
313        for biome_config in &self.config.biomes {
314            let score = biome_config.score_conditions(height, temperature, moisture);
315            if score > best_score {
316                best_score = score;
317                best_biome = biome_config.biome_type;
318            }
319        }
320
321        best_biome
322    }

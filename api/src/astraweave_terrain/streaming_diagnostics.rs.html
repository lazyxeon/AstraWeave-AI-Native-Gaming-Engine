16/// Chunk load state for diagnostics
17#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
18#[non_exhaustive]
19pub enum ChunkLoadState {
20    /// Chunk is fully loaded and rendered
21    Loaded,
22
23    /// Chunk is being generated in background
24    Loading,
25
26    /// Chunk is in load queue (not started)
27    Pending,
28
29    /// Chunk is unloaded (too far from camera)
30    Unloaded,
31}

33/// Frame hitch detection
34#[derive(Debug, Clone)]
35pub struct HitchDetector {
36    /// Recent frame times (milliseconds)
37    frame_times: VecDeque<f32>,
38
39    /// Maximum history size
40    max_history: usize,
41
42    /// Hitch threshold (ms)
43    hitch_threshold: f32,
44
45    /// Hitch count in window
46    hitch_count: usize,
47}
48
49impl HitchDetector {
50    /// Create a new hitch detector
51    pub fn new(max_history: usize, hitch_threshold: f32) -> Self {
52        Self {
53            frame_times: VecDeque::with_capacity(max_history),
54            max_history,
55            hitch_threshold,
56            hitch_count: 0,
57        }
58    }
59
60    /// Record a frame time and check for hitch
61    pub fn record_frame(&mut self, frame_time_ms: f32) -> bool {
62        let is_hitch = frame_time_ms > self.hitch_threshold;
63
64        if is_hitch {
65            self.hitch_count += 1;
66        }
67
68        self.frame_times.push_back(frame_time_ms);
69
70        // Remove oldest frame if over limit
71        if self.frame_times.len() > self.max_history {
72            let oldest = self.frame_times.pop_front().unwrap_or(0.0);
73            if oldest > self.hitch_threshold {
74                self.hitch_count = self.hitch_count.saturating_sub(1);
75            }
76        }
77
78        is_hitch
79    }
80
81    /// Get average frame time in window
82    pub fn average_frame_time(&self) -> f32 {
83        if self.frame_times.is_empty() {
84            return 0.0;
85        }
86
87        let sum: f32 = self.frame_times.iter().sum();
88        sum / self.frame_times.len() as f32
89    }
90
91    /// Get p99 frame time (99th percentile)
92    pub fn p99_frame_time(&self) -> f32 {
93        if self.frame_times.is_empty() {
94            return 0.0;
95        }
96
97        let mut sorted: Vec<f32> = self.frame_times.iter().copied().collect();
98        // Use unwrap_or for partial_cmp to handle potential NaN values gracefully
99        sorted.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
100
101        let index = ((sorted.len() as f32 * 0.99).ceil() as usize).min(sorted.len() - 1);
102        sorted[index]
103    }
104
105    /// Get hitch count in window
106    pub fn hitch_count(&self) -> usize {
107        self.hitch_count
108    }
109
110    /// Get hitch rate (percent of frames)
111    pub fn hitch_rate(&self) -> f32 {
112        if self.frame_times.is_empty() {
113            return 0.0;
114        }
115
116        (self.hitch_count as f32 / self.frame_times.len() as f32) * 100.0
117    }
118}

120/// Memory usage tracking
121#[derive(Debug, Clone, Default, Serialize, Deserialize)]
122pub struct MemoryStats {
123    /// Total bytes allocated for chunks
124    pub total_bytes: usize,
125
126    /// Bytes per chunk (average)
127    pub bytes_per_chunk: usize,
128
129    /// Number of chunks in memory
130    pub chunk_count: usize,
131
132    /// Peak memory usage
133    pub peak_bytes: usize,
134}
135
136impl MemoryStats {
137    /// Update memory stats
138    pub fn update(&mut self, chunk_count: usize, bytes_per_chunk: usize) {
139        self.chunk_count = chunk_count;
140        self.bytes_per_chunk = bytes_per_chunk;
141        self.total_bytes = chunk_count * bytes_per_chunk;
142
143        if self.total_bytes > self.peak_bytes {
144            self.peak_bytes = self.total_bytes;
145        }
146    }
147
148    /// Get memory delta from peak (percent)
149    pub fn delta_from_peak_percent(&self) -> f32 {
150        if self.peak_bytes == 0 {
151            return 0.0;
152        }
153
154        ((self.total_bytes as f32 / self.peak_bytes as f32) - 1.0) * 100.0
155    }
156
157    /// Get memory in MB
158    pub fn total_mb(&self) -> f32 {
159        self.total_bytes as f32 / (1024.0 * 1024.0)
160    }
161}

184impl StreamingDiagnostics {
185    /// Create a new diagnostics overlay
186    pub fn new(hitch_threshold_ms: f32, history_frames: usize) -> Self {
187        Self {
188            chunk_states: HashMap::new(),
189            hitch_detector: HitchDetector::new(history_frames, hitch_threshold_ms),
190            memory_stats: MemoryStats::default(),
191            streaming_stats: StreamingStats::default(),
192            lod_stats: LodStats::default(),
193            camera_pos: Vec3::ZERO,
194        }
195    }
196
197    /// Update chunk states
198    pub fn update_chunk_states(
199        &mut self,
200        loaded: &[ChunkId],
201        loading: &[ChunkId],
202        pending: &[ChunkId],
203    ) {
204        // Clear old states
205        self.chunk_states.clear();
206
207        // Mark loaded
208        for &chunk_id in loaded {
209            self.chunk_states.insert(chunk_id, ChunkLoadState::Loaded);
210        }
211
212        // Mark loading
213        for &chunk_id in loading {
214            self.chunk_states.insert(chunk_id, ChunkLoadState::Loading);
215        }
216
217        // Mark pending
218        for &chunk_id in pending {
219            self.chunk_states.insert(chunk_id, ChunkLoadState::Pending);
220        }
221    }
222
223    /// Record frame time
224    pub fn record_frame(&mut self, frame_time_ms: f32) -> bool {
225        self.hitch_detector.record_frame(frame_time_ms)
226    }
227
228    /// Update memory stats
229    pub fn update_memory(&mut self, chunk_count: usize, bytes_per_chunk: usize) {
230        self.memory_stats.update(chunk_count, bytes_per_chunk);
231    }

236    }

241    }

286    /// Generate diagnostic report
287    pub fn generate_report(&self) -> DiagnosticReport {
288        DiagnosticReport {
289            frame_stats: FrameStats {
290                average_ms: self.hitch_detector.average_frame_time(),
291                p99_ms: self.hitch_detector.p99_frame_time(),
292                hitch_count: self.hitch_detector.hitch_count(),
293                hitch_rate: self.hitch_detector.hitch_rate(),
294            },
295            memory: self.memory_stats.clone(),
296            streaming: self.streaming_stats.clone(),
297            lod: LodStatsReport {
298                total_chunks: self.lod_stats.total_chunks,
299                full_count: self.lod_stats.full_count,
300                half_count: self.lod_stats.half_count,
301                quarter_count: self.lod_stats.quarter_count,
302                skybox_count: self.lod_stats.skybox_count,
303                transitioning_count: self.lod_stats.transitioning_count,
304            },
305            chunk_counts: ChunkCounts {
306                loaded: self
307                    .chunk_states
308                    .values()
309                    .filter(|&&s| s == ChunkLoadState::Loaded)
310                    .count(),
311                loading: self
312                    .chunk_states
313                    .values()
314                    .filter(|&&s| s == ChunkLoadState::Loading)
315                    .count(),
316                pending: self
317                    .chunk_states
318                    .values()
319                    .filter(|&&s| s == ChunkLoadState::Pending)
320                    .count(),
321            },
322        }
323    }
324}

1//! # Ragdoll System
2//!
3//! Provides realistic ragdoll physics for character death animations,
4//! hit reactions, and procedural animations.
5//!
6//! ## Features
7//!
8//! - **Ragdoll Builder**: Create ragdolls from bone hierarchies
9//! - **State Management**: Smooth transitions between animated and physics states
10//! - **Impulse Propagation**: Realistic force distribution through joint chains
11//! - **Joint Limits**: Anatomically-correct constraints
12//!
13//! ## Usage
14//!
15//! ```rust
16//! use astraweave_physics::ragdoll::{RagdollBuilder, RagdollConfig, BoneShape};
17//! use glam::Vec3;
18//!
19//! // Define a simple humanoid skeleton
20//! let mut builder = RagdollBuilder::new(RagdollConfig::default());
21//!
22//! // Add bones (typically from skeleton data)
23//! builder.add_bone("pelvis", None, Vec3::ZERO, BoneShape::Capsule { radius: 0.15, half_height: 0.1 }, 2.0);
24//! builder.add_bone("spine", Some("pelvis"), Vec3::new(0.0, 0.2, 0.0), BoneShape::Capsule { radius: 0.12, half_height: 0.15 }, 1.5);
25//! builder.add_bone("head", Some("spine"), Vec3::new(0.0, 0.35, 0.0), BoneShape::Sphere { radius: 0.12 }, 1.0);
26//!
27//! // Build the ragdoll (requires PhysicsWorld)
28//! // let ragdoll = builder.build(&mut physics_world, spawn_position);
29//! ```

38/// Shape of a bone's physics collider
39#[derive(Debug, Clone, Copy)]
40pub enum BoneShape {
41    /// Capsule shape (most common for limbs)
42    Capsule { radius: f32, half_height: f32 },
43    /// Sphere shape (good for head, hands)
44    Sphere { radius: f32 },
45    /// Box shape (good for torso, pelvis)
46    Box { half_extents: Vec3 },
47}
48
49impl BoneShape {
50    /// Get the approximate volume of this shape (for mass distribution)
51    pub fn volume(&self) -> f32 {
52        match self {
53            BoneShape::Capsule { radius, half_height } => {
54                let cylinder = std::f32::consts::PI * radius * radius * (half_height * 2.0);
55                let sphere = (4.0 / 3.0) * std::f32::consts::PI * radius.powi(3);
56                cylinder + sphere
57            }
58            BoneShape::Sphere { radius } => {
59                (4.0 / 3.0) * std::f32::consts::PI * radius.powi(3)
60            }
61            BoneShape::Box { half_extents } => {
62                half_extents.x * half_extents.y * half_extents.z * 8.0
63            }
64        }
65    }
66}

68/// Joint constraint type for bones
69#[derive(Debug, Clone, Copy)]
70pub enum BoneJointType {
71    /// Ball-and-socket joint (shoulder, hip)
72    Spherical {
73        /// Cone limit in radians (how far the bone can swing)
74        swing_limit: Option<f32>,
75        /// Twist limit in radians (rotation around the bone's axis)
76        twist_limit: Option<(f32, f32)>,
77    },
78    /// Hinge joint (elbow, knee)
79    Hinge {
80        /// Axis of rotation (local to parent)
81        axis: Vec3,
82        /// Angle limits in radians (min, max)
83        limits: Option<(f32, f32)>,
84    },
85    /// Fixed joint (skull to spine, rigid connections)
86    Fixed,
87}

183/// Instance of a spawned ragdoll
184#[derive(Debug)]
185pub struct Ragdoll {
186    /// Unique ID
187    pub id: RagdollId,
188    /// Current state
189    pub state: RagdollState,
190    /// Map of bone name to body ID
191    pub bone_bodies: HashMap<String, BodyId>,
192    /// Map of bone name to joint ID (joint connecting to parent)
193    pub bone_joints: HashMap<String, JointId>,
194    /// Root bone name
195    pub root_bone: String,
196    /// Configuration used to create this ragdoll
197    pub config: RagdollConfig,
198    /// Bone definitions (for reference)
199    bone_defs: Vec<BoneDef>,
200}

202impl Ragdoll {
203    /// Get the body ID for a bone by name
204    pub fn get_bone_body(&self, name: &str) -> Option<BodyId> {
205        self.bone_bodies.get(name).copied()
206    }

208    /// Get all body IDs in this ragdoll
209    pub fn all_bodies(&self) -> impl Iterator<Item = BodyId> + '_ {
210        self.bone_bodies.values().copied()
211    }
212
213    /// Get the root body ID
214    pub fn root_body(&self) -> Option<BodyId> {
215        self.bone_bodies.get(&self.root_bone).copied()
216    }

218    /// Apply an impulse to a specific bone
219    pub fn apply_impulse_to_bone(
220        &self,
221        physics: &mut PhysicsWorld,
222        bone_name: &str,
223        impulse: Vec3,
224    ) -> bool {
225        if let Some(body_id) = self.bone_bodies.get(bone_name) {
226            physics.apply_impulse(*body_id, impulse);
227            true
228        } else {
229            false
230        }
231    }
232
233    /// Apply an impulse to the ragdoll with propagation through joints
234    /// 
235    /// The impulse is applied at the specified position and propagates
236    /// through the joint chain with falloff based on distance.
237    pub fn apply_impulse_with_propagation(
238        &self,
239        physics: &mut PhysicsWorld,
240        hit_bone: &str,
241        impulse: Vec3,
242        propagation_factor: f32,
243    ) {
244        // Apply full impulse to hit bone
245        if let Some(body_id) = self.bone_bodies.get(hit_bone) {
246            physics.apply_impulse(*body_id, impulse);
247        }
248
249        // Find connected bones and apply reduced impulse
250        if let Some(hit_def) = self.bone_defs.iter().find(|b| b.name == hit_bone) {
251            // Apply to parent (if any)
252            if let Some(ref parent_name) = hit_def.parent {
253                if let Some(parent_body) = self.bone_bodies.get(parent_name) {
254                    physics.apply_impulse(*parent_body, impulse * propagation_factor);
255                }
256            }
257
258            // Apply to children
259            for bone in &self.bone_defs {
260                if bone.parent.as_ref() == Some(&hit_def.name) {
261                    if let Some(child_body) = self.bone_bodies.get(&bone.name) {
262                        physics.apply_impulse(*child_body, impulse * propagation_factor);
263                    }
264                }
265            }
266        }
267    }
268
269    /// Get the center of mass of the ragdoll
270    pub fn center_of_mass(&self, physics: &PhysicsWorld) -> Vec3 {
271        let mut total_mass = 0.0f32;
272        let mut weighted_pos = Vec3::ZERO;
273
274        for (name, body_id) in &self.bone_bodies {
275            if let Some(transform) = physics.body_transform(*body_id) {
276                let pos = Vec3::new(transform.w_axis.x, transform.w_axis.y, transform.w_axis.z);
277                let mass = self
278                    .bone_defs
279                    .iter()
280                    .find(|b| &b.name == name)
281                    .map(|b| b.mass)
282                    .unwrap_or(1.0);
283                weighted_pos += pos * mass;
284                total_mass += mass;
285            }
286        }
287
288        if total_mass > 0.0 {
289            weighted_pos / total_mass
290        } else {
291            Vec3::ZERO
292        }
293    }
294
295    /// Check if the ragdoll is at rest (not moving significantly)
296    pub fn is_at_rest(&self, physics: &PhysicsWorld, velocity_threshold: f32) -> bool {
297        for body_id in self.bone_bodies.values() {
298            if let Some(vel) = physics.get_velocity(*body_id) {
299                if vel.length() > velocity_threshold {
300                    return false;
301                }
302            }
303        }
304        true
305    }
306}

329    /// Add a bone to the ragdoll
330    pub fn add_bone(
331        &mut self,
332        name: &str,
333        parent: Option<&str>,
334        offset: Vec3,
335        shape: BoneShape,
336        mass: f32,
337    ) -> &mut Self {
338        self.bones.push(BoneDef {
339            name: name.to_string(),
340            parent: parent.map(|s| s.to_string()),
341            offset,
342            shape,
343            mass: mass * self.config.mass_scale,
344            ..Default::default()
345        });
346        self
347    }

350    pub fn add_bone_full(&mut self, bone: BoneDef) -> &mut Self {
351        let mut bone = bone;
352        bone.mass *= self.config.mass_scale;
353        self.bones.push(bone);
354        self
355    }

405    /// Build the ragdoll and spawn it in the physics world
406    pub fn build(&mut self, physics: &mut PhysicsWorld, spawn_position: Vec3) -> Ragdoll {
407        let mut bone_bodies = HashMap::new();
408        let mut bone_joints = HashMap::new();
409        let mut root_bone = String::new();
410
411        // First pass: create all bodies
412        for bone in &self.bones {
413            let bone_pos = if let Some(ref parent_name) = bone.parent {
414                // Calculate world position based on parent
415                if let Some(&parent_body) = bone_bodies.get(parent_name) {
416                    if let Some(transform) = physics.body_transform(parent_body) {
417                        let parent_pos =
418                            Vec3::new(transform.w_axis.x, transform.w_axis.y, transform.w_axis.z);
419                        parent_pos + bone.offset
420                    } else {
421                        spawn_position + bone.offset
422                    }
423                } else {
424                    spawn_position + bone.offset
425                }
426            } else {
427                // Root bone
428                root_bone = bone.name.clone();
429                spawn_position
430            };
431
432            // Create the body based on shape
433            let body_id = match bone.shape {
434                BoneShape::Capsule { radius, half_height } => physics.add_dynamic_box(
435                    bone_pos,
436                    Vec3::new(radius, half_height, radius),
437                    bone.mass,
438                    self.config.collision_group,
439                ),
440                BoneShape::Sphere { radius } => physics.add_dynamic_box(
441                    bone_pos,
442                    Vec3::splat(radius),
443                    bone.mass,
444                    self.config.collision_group,
445                ),
446                BoneShape::Box { half_extents } => physics.add_dynamic_box(
447                    bone_pos,
448                    half_extents,
449                    bone.mass,
450                    self.config.collision_group,
451                ),
452            };
453
454            // Enable CCD if configured
455            if self.config.enable_ccd {
456                physics.enable_ccd(body_id);
457            }
458
459            bone_bodies.insert(bone.name.clone(), body_id);
460        }
461
462        // Second pass: create joints
463        for bone in &self.bones {
464            if let Some(ref parent_name) = bone.parent {
465                let Some(&parent_body) = bone_bodies.get(parent_name) else {
466                    continue;
467                };
468                let Some(&child_body) = bone_bodies.get(&bone.name) else {
469                    continue;
470                };
471
472                let joint_type = match bone.joint_type {
473                    BoneJointType::Fixed => JointType::Fixed,
474                    BoneJointType::Hinge { axis, limits } => JointType::Revolute { axis, limits },
475                    BoneJointType::Spherical { .. } => JointType::Spherical,
476                };
477
478                let joint_id = physics.add_joint(parent_body, child_body, joint_type);
479                bone_joints.insert(bone.name.clone(), joint_id);
480            }
481        }
482
483        let id = self.next_id;
484        self.next_id += 1;
485
486        Ragdoll {
487            id,
488            state: RagdollState::Active,
489            bone_bodies,
490            bone_joints,
491            root_bone,
492            config: self.config.clone(),
493            bone_defs: self.bones.clone(),
494        }
495    }
496
497    /// Clear all bones (for reuse)
498    pub fn clear(&mut self) {
499        self.bones.clear();
500    }

619                Vec3::X,

622        }

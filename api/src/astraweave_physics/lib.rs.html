289    /// Enable async physics with optional thread count (0 = auto-detect)
290    /// This configures Rayon's global thread pool, which Rapier3D uses for parallel solving
291    #[cfg(feature = "async-physics")]
292    pub fn enable_async_physics(&mut self, thread_count: usize) {
293        // Configure Rayon thread pool if not already initialized
294        if thread_count > 0 {
295            // Try to build thread pool (may fail if already initialized, which is fine)
296            let _ = rayon::ThreadPoolBuilder::new()
297                .num_threads(thread_count)
298                .build_global();
299        }
300
301        self.async_scheduler = Some(if thread_count > 0 {
302            AsyncPhysicsScheduler::with_threads(thread_count)
303        } else {
304            AsyncPhysicsScheduler::new()
305        });
306    }

320    pub fn step(&mut self) {
321        #[cfg(feature = "profiling")]
322        span!("Physics::World::step");
323
324        #[cfg(feature = "async-physics")]
325        {
326            // When async scheduler is enabled, Rapier3D automatically uses
327            // Rayon's global thread pool for parallel island solving.
328            // The thread count was configured when enable_async_physics was called.
329            if self.async_scheduler.is_some() {
330                use std::time::Instant;
331                let start = Instant::now();
332
333                self.step_internal();
334
335                let duration = start.elapsed();
336
337                // Update telemetry
338                if let Some(scheduler) = &mut self.async_scheduler {
339                    scheduler.record_step_telemetry(duration);
340                }
341                return;
342            }
343        }
344
345        // Fallback to regular step (single-threaded)
346        self.step_internal();
347    }

349    /// Internal physics step (shared by sync and async paths)
350    /// When called with async scheduler enabled, Rapier3D uses Rayon for parallel solving
351    fn step_internal(&mut self) {
352        #[cfg(feature = "profiling")]
353        {
354            span!("Physics::Rapier::pipeline");
355            plot!("Physics::collider_count", self.colliders.len() as u64);
356        }
357
358        // Apply buoyancy forces before physics step
359        self.apply_buoyancy_forces();
360
361        self.pipeline.step(
362            &self.gravity,
363            &self.integration,
364            &mut self.island_mgr,
365            &mut self.broad_phase,
366            &mut self.narrow_phase,
367            &mut self.bodies,
368            &mut self.colliders,
369            &mut self.joints,
370            &mut self.multibody_joints,
371            &mut self.ccd,
372            Some(&mut self.query_pipeline),
373            &(),
374            &self.event_handler,
375        );
376
377        // CRITICAL FIX (Week 2 Day 3): Update query pipeline after physics step
378        // Without this, raycasts in control_character() use stale geometry,
379        // causing character controller to fail ground detection
380        self.query_pipeline.update(&self.colliders);
381    }

514    pub fn jump(&mut self, id: BodyId, height: f32) {
515        if let Some(ctrl) = self.char_map.get_mut(&id) {
516            ctrl.jump_buffer_timer = ctrl.jump_buffer_limit;
517            let g = 9.81 * ctrl.gravity_scale;
518            ctrl.pending_jump_velocity = (2.0 * g * height).sqrt();
519        }
520    }

522    pub fn control_character(&mut self, id: BodyId, desired_move: Vec3, dt: f32, _climb: bool) {
523        #[cfg(feature = "profiling")]
524        span!("Physics::CharacterController::move");
525
526        let Some(mut ctrl) = self.char_map.get(&id).copied() else {
527            return;
528        };
529        let Some(h) = self.handle_of(id) else {
530            return;
531        };
532        let Some(rb) = self.bodies.get(h) else {
533            return;
534        };
535        let pos = *rb.position();
536        let start = glam::Vec3::new(pos.translation.x, pos.translation.y, pos.translation.z);
537        // Update timers
538        ctrl.jump_buffer_timer -= dt;
539        
540        // Apply gravity (if not climbing)
541        if !_climb {
542            ctrl.vertical_velocity -= 9.81 * ctrl.gravity_scale * dt;
543        } else {
544            ctrl.vertical_velocity = 0.0;
545        }
546
547        // Check Jump
548        let can_jump = ctrl.time_since_grounded < ctrl.coyote_time_limit;
549        let wants_jump = ctrl.jump_buffer_timer > 0.0;
550        
551        if can_jump && wants_jump && !_climb {
552            ctrl.vertical_velocity = ctrl.pending_jump_velocity;
553            ctrl.time_since_grounded = ctrl.coyote_time_limit + 1.0; // Invalidate coyote
554            ctrl.jump_buffer_timer = 0.0; // Consume buffer
555        }
556
557        let mut d = desired_move * dt;
558        let has_horizontal_move = d.length_squared() >= 1e-6;
559        let has_vertical_move = ctrl.vertical_velocity.abs() > 1e-4 || _climb;
560
561        if !has_horizontal_move && !has_vertical_move {
562            self.char_map.insert(id, ctrl);
563            return;
564        }
565
566        if has_horizontal_move {
567            // Basic obstacle avoidance: raycast forward; slide along hit normal
568            let dir = d.normalize();
569            let ray_origin = start + glam::Vec3::Y * (ctrl.height * 0.5);
570            let ray = rapier3d::prelude::Ray::new(
571                point![ray_origin.x, ray_origin.y, ray_origin.z],
572                vector![dir.x, dir.y, dir.z],
573            );
574            // BUG FIX (Week 2 Day 3): Exclude character's own colliders from raycasts
575            // Without this, the character detects its own capsule as an obstacle
576            let filter = QueryFilter::default().exclude_rigid_body(h);
577            if let Some((_, hit)) = self.query_pipeline.cast_ray_and_get_normal(
578                &self.bodies,
579                &self.colliders,
580                &ray,
581                d.length() + ctrl.radius + 0.05,
582                true,
583                filter,
584            ) {
585                // Deflect movement along tangent plane
586                let n = glam::Vec3::new(hit.normal.x, hit.normal.y, hit.normal.z).normalize();
587                d = d - n * d.dot(n);
588            }
589        }
590
591        // Tentative horizontal move
592        let mut new_pos = start + glam::Vec3::new(d.x, 0.0, d.z);
593
594        if _climb {
595            // Simple vertical climb
596            new_pos.y = start.y + 2.0 * dt;
597            ctrl.time_since_grounded = 0.0;
598        } else {
599            // Apply vertical velocity
600            new_pos.y = start.y + ctrl.vertical_velocity * dt;
601
602            // Ground check / Step / Slope
603            if ctrl.vertical_velocity <= 0.0 {
604                let cast_origin = new_pos + glam::Vec3::Y * (ctrl.height);
605                let ray_down = rapier3d::prelude::Ray::new(
606                    point![cast_origin.x, cast_origin.y, cast_origin.z],
607                    vector![0.0, -1.0, 0.0],
608                );
609                if let Some((_, hit)) = self.query_pipeline.cast_ray_and_get_normal(
610                    &self.bodies,
611                    &self.colliders,
612                    &ray_down,
613                    ctrl.height + ctrl.max_step + 1.0,
614                    true,
615                    QueryFilter::default().exclude_rigid_body(h),
616                ) {
617                    let ground_normal =
618                        glam::Vec3::new(hit.normal.x, hit.normal.y, hit.normal.z).normalize();
619                    let slope = ground_normal.dot(glam::Vec3::Y).acos().to_degrees();
620                    let ground_y = cast_origin.y - hit.time_of_impact;
621                    
622                    if slope <= ctrl.max_climb_angle_deg + 1e-2 {
623                        // Snap to ground if close enough
624                        if new_pos.y <= ground_y + 0.05 {
625                             new_pos.y = ground_y;
626                             ctrl.vertical_velocity = 0.0;
627                             ctrl.time_since_grounded = 0.0;
628                        }
629                    }
630                } else {
631                    ctrl.time_since_grounded += dt;
632                }
633            } else {
634                ctrl.time_since_grounded += dt;
635            }
636        }
637
638        // Commit move
639        let mut p = pos;
640        p.translation.x = new_pos.x;
641        p.translation.y = new_pos.y;
642        p.translation.z = new_pos.z;
643        if let Some(rbmut) = self.bodies.get_mut(h) {
644            // BUG FIX (Week 2 Day 3): Use set_next_kinematic_position for kinematic bodies
645            // set_position() with wake=true doesn't properly update kinematic bodies
646            // across multiple frames - position gets reset by physics step
647            rbmut.set_next_kinematic_position(p);
648        }
649    }

689    fn apply_buoyancy_forces(&mut self) {
690        for (body_id, buoyancy_data) in &self.buoyancy_bodies {
691            if let Some(handle) = self.handle_of(*body_id) {
692                if let Some(rb) = self.bodies.get_mut(handle) {
693                    let pos = rb.position();
694                    let body_y = pos.translation.y;
695
696                    // Only apply buoyancy if body is below water level
697                    if body_y < self.water_level {
698                        // Buoyancy force = volume * fluid_density * gravity (upward)
699                        let buoyancy_force = buoyancy_data.volume * self.fluid_density * 9.81;
700
701                        // Drag force = -velocity * drag coefficient
702                        let velocity = rb.linvel();
703                        let drag_force = vector![
704                            -velocity.x * buoyancy_data.drag,
705                            -velocity.y * buoyancy_data.drag,
706                            -velocity.z * buoyancy_data.drag
707                        ];
708
709                        // Total force (buoyancy up + drag)
710                        let total_force =
711                            vector![drag_force.x, buoyancy_force + drag_force.y, drag_force.z];
712
713                        rb.add_force(total_force, true);
714                    }
715                }
716            }
717        }
718    }

726    /// Apply radial explosion impulse to all dynamic bodies in range
727    ///
728    /// # Arguments
729    /// * `center` - Explosion center
730    /// * `radius` - Maximum effect radius
731    /// * `force` - Force at center (falls off with distance)
732    /// * `falloff` - Falloff curve type
733    /// * `upward_bias` - Upward impulse bias (0.0 = radial, 1.0 = fully upward)
734    ///
735    /// # Returns
736    /// Number of bodies affected
737    pub fn apply_radial_impulse(
738        &mut self,
739        center: Vec3,
740        radius: f32,
741        force: f32,
742        falloff: projectile::FalloffCurve,
743        upward_bias: f32,
744    ) -> usize {
745        // Collect body positions and IDs
746        let mut affected = Vec::new();
747
748        for (&handle, &_body_id) in &self.body_ids {
749            if let Some(rb) = self.bodies.get(handle) {
750                if !rb.is_dynamic() {
751                    continue;
752                }
753                let pos = rb.position().translation;
754                let body_pos = Vec3::new(pos.x, pos.y, pos.z);
755                let to_body = body_pos - center;
756                let distance = to_body.length();
757
758                if distance < radius {
759                    affected.push((handle, body_pos, distance, to_body));
760                }
761            }
762        }
763
764        // Apply impulses
765        let count = affected.len();
766        for (handle, _body_pos, distance, to_body) in affected {
767            let falloff_mult = falloff.calculate(distance, radius);
768            let force_mag = force * falloff_mult;
769
770            let radial_dir = if distance > 0.001 {
771                to_body.normalize()
772            } else {
773                Vec3::Y
774            };
775
776            let biased_dir =
777                (radial_dir * (1.0 - upward_bias) + Vec3::Y * upward_bias).normalize();
778            let impulse = biased_dir * force_mag;
779
780            if let Some(rb) = self.bodies.get_mut(handle) {
781                rb.apply_impulse(vector![impulse.x, impulse.y, impulse.z], true);
782            }
783        }
784
785        count
786    }

788    /// Perform a raycast and return hit info
789    ///
790    /// # Returns
791    /// (hit_position, hit_normal, body_id, distance) if hit
792    pub fn raycast(
793        &self,
794        origin: Vec3,
795        direction: Vec3,
796        max_distance: f32,
797    ) -> Option<(Vec3, Vec3, Option<BodyId>, f32)> {
798        let ray = rapier3d::prelude::Ray::new(
799            point![origin.x, origin.y, origin.z],
800            vector![direction.x, direction.y, direction.z],
801        );
802
803        self.query_pipeline
804            .cast_ray_and_get_normal(
805                &self.bodies,
806                &self.colliders,
807                &ray,
808                max_distance,
809                true,
810                QueryFilter::default(),
811            )
812            .map(|(collider_handle, hit)| {
813                let hit_pos = origin + direction * hit.time_of_impact;
814                let normal = Vec3::new(hit.normal.x, hit.normal.y, hit.normal.z);
815                
816                // Get body ID from collider
817                let body_id = self.colliders
818                    .get(collider_handle)
819                    .and_then(|c| c.parent())
820                    .and_then(|rb_handle| self.body_ids.get(&rb_handle).copied());
821
822                (hit_pos, normal, body_id, hit.time_of_impact)
823            })
824    }

876    pub fn add_joint(&mut self, body1: BodyId, body2: BodyId, joint_type: JointType) -> JointId {
877        let Some(handle1) = self.handle_of(body1) else {
878            return JointId(0);
879        };
880        let Some(handle2) = self.handle_of(body2) else {
881            return JointId(0);
882        };
883
884        let joint = match joint_type {
885            JointType::Fixed => GenericJointBuilder::new(JointAxesMask::LOCKED_FIXED_AXES).build(),
886            JointType::Revolute { axis, limits } => {
887                let local_axis = UnitVector::new_normalize(vector![axis.x, axis.y, axis.z]);
888                let mut builder = RevoluteJointBuilder::new(local_axis);
889                if let Some((min, max)) = limits {
890                    builder = builder.limits([min, max]);
891                }
892                builder.build().into()
893            }
894            JointType::Prismatic { axis, limits } => {
895                let local_axis = UnitVector::new_normalize(vector![axis.x, axis.y, axis.z]);
896                let mut builder = PrismaticJointBuilder::new(local_axis);
897                if let Some((min, max)) = limits {
898                    builder = builder.limits([min, max]);
899                }
900                builder.build().into()
901            }
902            JointType::Spherical => SphericalJointBuilder::new().build().into(),
903        };
904
905        self.joints.insert(handle1, handle2, joint, true);
906
907        let joint_id = self.next_joint_id;
908        self.next_joint_id += 1;
909        JointId(joint_id)
910    }

16/// A particle in the cloth simulation
17#[derive(Debug, Clone)]
18pub struct ClothParticle {
19    /// Current position
20    pub position: Vec3,
21    /// Previous position (for Verlet integration)
22    pub prev_position: Vec3,
23    /// Accumulated forces this frame
24    pub acceleration: Vec3,
25    /// Inverse mass (0 = pinned/infinite mass)
26    pub inv_mass: f32,
27    /// Whether this particle is pinned (fixed position)
28    pub pinned: bool,
29}
30
31impl ClothParticle {
32    /// Create a new particle
33    pub fn new(position: Vec3, mass: f32) -> Self {
34        Self {
35            position,
36            prev_position: position,
37            acceleration: Vec3::ZERO,
38            inv_mass: if mass > 0.0 { 1.0 / mass } else { 0.0 },
39            pinned: false,
40        }
41    }
42
43    /// Create a pinned particle
44    pub fn pinned(position: Vec3) -> Self {
45        Self {
46            position,
47            prev_position: position,
48            acceleration: Vec3::ZERO,
49            inv_mass: 0.0,
50            pinned: true,
51        }
52    }
53
54    /// Apply force to particle
55    pub fn apply_force(&mut self, force: Vec3) {
56        if !self.pinned {
57            self.acceleration += force * self.inv_mass;
58        }
59    }
60
61    /// Integrate using Verlet integration
62    pub fn integrate(&mut self, dt: f32, damping: f32) {
63        if self.pinned {
64            return;
65        }
66
67        let velocity = self.position - self.prev_position;
68        self.prev_position = self.position;
69        self.position += velocity * damping + self.acceleration * dt * dt;
70        self.acceleration = Vec3::ZERO;
71    }
72
73    /// Get velocity
74    pub fn velocity(&self) -> Vec3 {
75        self.position - self.prev_position
76    }
77}

79/// A distance constraint between two particles
80#[derive(Debug, Clone, Copy)]
81pub struct DistanceConstraint {
82    /// First particle index
83    pub p1: usize,
84    /// Second particle index
85    pub p2: usize,
86    /// Rest length
87    pub rest_length: f32,
88    /// Stiffness (0-1, higher = stiffer)
89    pub stiffness: f32,
90}
91
92impl DistanceConstraint {
93    /// Create a new constraint
94    pub fn new(p1: usize, p2: usize, rest_length: f32) -> Self {
95        Self {
96            p1,
97            p2,
98            rest_length,
99            stiffness: 1.0,
100        }
101    }
102
103    /// Solve the constraint
104    pub fn solve(&self, particles: &mut [ClothParticle]) {
105        let p1 = &particles[self.p1];
106        let p2 = &particles[self.p2];
107
108        let delta = p2.position - p1.position;
109        let current_length = delta.length();
110
111        if current_length < 0.0001 {
112            return;
113        }
114
115        let diff = (current_length - self.rest_length) / current_length;
116        let correction = delta * diff * 0.5 * self.stiffness;
117
118        let w1 = p1.inv_mass;
119        let w2 = p2.inv_mass;
120        let total_weight = w1 + w2;
121
122        if total_weight > 0.0 {
123            if !particles[self.p1].pinned {
124                particles[self.p1].position += correction * (w1 / total_weight);
125            }
126            if !particles[self.p2].pinned {
127                particles[self.p2].position -= correction * (w2 / total_weight);
128            }
129        }
130    }
131}

133/// Collision shape for cloth collision
134#[derive(Debug, Clone, Copy)]
135#[non_exhaustive]
136pub enum ClothCollider {
137    /// Sphere collider
138    Sphere { center: Vec3, radius: f32 },
139    /// Capsule collider
140    Capsule { start: Vec3, end: Vec3, radius: f32 },
141    /// Infinite plane
142    Plane { point: Vec3, normal: Vec3 },
143}
144
145impl ClothCollider {
146    /// Resolve collision with a particle
147    pub fn resolve_collision(&self, particle: &mut ClothParticle, friction: f32) {
148        if particle.pinned {
149            return;
150        }
151
152        match *self {
153            ClothCollider::Sphere { center, radius } => {
154                let to_particle = particle.position - center;
155                let dist = to_particle.length();
156
157                if dist < radius {
158                    let normal = to_particle.normalize_or_zero();
159                    let penetration = radius - dist;
160                    particle.position += normal * penetration;
161
162                    // Apply friction
163                    let velocity = particle.velocity();
164                    let normal_vel = velocity.dot(normal) * normal;
165                    let tangent_vel = velocity - normal_vel;
166                    particle.prev_position =
167                        particle.position - (normal_vel + tangent_vel * (1.0 - friction));
168                }
169            }
170            ClothCollider::Capsule { start, end, radius } => {
171                let axis = end - start;
172                let axis_length = axis.length();
173                if axis_length < 0.0001 {
174                    // Degenerate capsule = sphere
175                    let sphere = ClothCollider::Sphere {
176                        center: start,
177                        radius,
178                    };
179                    sphere.resolve_collision(particle, friction);
180                    return;
181                }
182
183                let axis_dir = axis / axis_length;
184                let to_particle = particle.position - start;
185                let t = to_particle.dot(axis_dir).clamp(0.0, axis_length);
186                let closest = start + axis_dir * t;
187
188                let to_particle = particle.position - closest;
189                let dist = to_particle.length();
190
191                if dist < radius {
192                    let normal = to_particle.normalize_or_zero();
193                    let penetration = radius - dist;
194                    particle.position += normal * penetration;
195
196                    let velocity = particle.velocity();
197                    let normal_vel = velocity.dot(normal) * normal;
198                    let tangent_vel = velocity - normal_vel;
199                    particle.prev_position =
200                        particle.position - (normal_vel + tangent_vel * (1.0 - friction));
201                }
202            }
203            ClothCollider::Plane { point, normal } => {
204                let to_particle = particle.position - point;
205                let dist = to_particle.dot(normal);
206
207                if dist < 0.0 {
208                    particle.position -= normal * dist;
209
210                    let velocity = particle.velocity();
211                    let normal_vel = velocity.dot(normal) * normal;
212                    let tangent_vel = velocity - normal_vel;
213                    particle.prev_position =
214                        particle.position - (normal_vel + tangent_vel * (1.0 - friction));
215                }
216            }
217        }
218    }
219}

275impl Cloth {
276    /// Create a new cloth from config
277    pub fn new(id: ClothId, config: ClothConfig, origin: Vec3) -> Self {
278        let mut particles = Vec::with_capacity(config.width * config.height);
279        let mut constraints = Vec::new();
280
281        // Create particles in a grid
282        for y in 0..config.height {
283            for x in 0..config.width {
284                let pos =
285                    origin + Vec3::new(x as f32 * config.spacing, 0.0, y as f32 * config.spacing);
286                particles.push(ClothParticle::new(pos, config.particle_mass));
287            }
288        }
289
290        // Create structural constraints (horizontal and vertical)
291        for y in 0..config.height {
292            for x in 0..config.width {
293                let idx = y * config.width + x;
294
295                // Horizontal constraint
296                if x < config.width - 1 {
297                    let mut c = DistanceConstraint::new(idx, idx + 1, config.spacing);
298                    c.stiffness = config.stiffness;
299                    constraints.push(c);
300                }
301
302                // Vertical constraint
303                if y < config.height - 1 {
304                    let mut c = DistanceConstraint::new(idx, idx + config.width, config.spacing);
305                    c.stiffness = config.stiffness;
306                    constraints.push(c);
307                }
308
309                // Shear constraints (diagonals)
310                if x < config.width - 1 && y < config.height - 1 {
311                    let diag_len = config.spacing * std::f32::consts::SQRT_2;
312
313                    let mut c1 = DistanceConstraint::new(idx, idx + config.width + 1, diag_len);
314                    c1.stiffness = config.stiffness * 0.5;
315                    constraints.push(c1);
316
317                    let mut c2 = DistanceConstraint::new(idx + 1, idx + config.width, diag_len);
318                    c2.stiffness = config.stiffness * 0.5;
319                    constraints.push(c2);
320                }
321
322                // Bend constraints (skip one particle)
323                if x < config.width - 2 {
324                    let mut c = DistanceConstraint::new(idx, idx + 2, config.spacing * 2.0);
325                    c.stiffness = config.stiffness * 0.3;
326                    constraints.push(c);
327                }
328                if y < config.height - 2 {
329                    let mut c =
330                        DistanceConstraint::new(idx, idx + config.width * 2, config.spacing * 2.0);
331                    c.stiffness = config.stiffness * 0.3;
332                    constraints.push(c);
333                }
334            }
335        }
336
337        Self {
338            id,
339            config,
340            particles,
341            constraints,
342            colliders: Vec::new(),
343            collision_friction: 0.5,
344        }
345    }

417    /// Calculate cloth normal at a particle (for wind)
418    fn particle_normal(&self, x: usize, y: usize) -> Vec3 {
419        let idx = y * self.config.width + x;
420        let center = self.particles[idx].position;
421
422        let mut normal = Vec3::ZERO;
423        let mut count = 0;
424
425        // Get neighboring particles for normal calculation
426        let neighbors = [
427            (x.wrapping_sub(1), y),
428            (x + 1, y),
429            (x, y.wrapping_sub(1)),
430            (x, y + 1),
431        ];
432
433        for i in 0..4 {
434            let (nx, ny) = neighbors[i];
435            let (nx2, ny2) = neighbors[(i + 1) % 4];
436
437            if nx < self.config.width
438                && ny < self.config.height
439                && nx2 < self.config.width
440                && ny2 < self.config.height
441            {
442                let idx1 = ny * self.config.width + nx;
443                let idx2 = ny2 * self.config.width + nx2;
444
445                let v1 = self.particles[idx1].position - center;
446                let v2 = self.particles[idx2].position - center;
447                normal += v1.cross(v2);
448                count += 1;
449            }
450        }
451
452        if count > 0 {
453            normal.normalize_or_zero()
454        } else {
455            Vec3::Y
456        }
457    }

459    /// Update cloth simulation
460    pub fn update(&mut self, dt: f32) {
461        // Pre-compute normals for wind calculation
462        let mut normals = Vec::new();
463        if self.config.wind.length_squared() > 0.001 {
464            for y in 0..self.config.height {
465                for x in 0..self.config.width {
466                    normals.push(self.particle_normal(x, y));
467                }
468            }
469        }
470
471        // Apply forces
472        for y in 0..self.config.height {
473            for x in 0..self.config.width {
474                let idx = y * self.config.width + x;
475                let inv_mass = self.particles[idx].inv_mass;
476
477                // Gravity
478                let gravity_force = self.config.gravity * (1.0 / inv_mass.max(0.001));
479                self.particles[idx].apply_force(gravity_force);
480
481                // Wind (affected by particle normal)
482                if !normals.is_empty() {
483                    let normal = normals[idx];
484                    let wind_effect = self.config.wind.dot(normal).abs();
485                    let wind_force = self.config.wind * wind_effect * (1.0 / inv_mass.max(0.001));
486                    self.particles[idx].apply_force(wind_force);
487                }
488
489                // Air resistance
490                let velocity = self.particles[idx].velocity();
491                let drag = -velocity * self.config.air_resistance * (1.0 / inv_mass.max(0.001));
492                self.particles[idx].apply_force(drag);
493            }
494        }
495
496        // Integrate
497        for particle in &mut self.particles {
498            particle.integrate(dt, self.config.damping);
499        }
500
501        // Solve constraints
502        for _ in 0..self.config.solver_iterations {
503            for constraint in &self.constraints {
504                constraint.solve(&mut self.particles);
505            }
506        }
507
508        // Resolve collisions
509        for particle in &mut self.particles {
510            for collider in &self.colliders {
511                collider.resolve_collision(particle, self.collision_friction);
512            }
513        }
514    }

556/// Manager for multiple cloth simulations

574        let id = ClothId(self.next_id);

1//! Component type metadata for type-erased BlobVec operations.
2//!
3//! This module provides the runtime type information needed to safely
4//! work with BlobVec storage without knowing concrete types at compile time.
5//!
6//! # Architecture
7//!
8//! ```text
9//! ComponentMeta<T>                         BlobVec
10//! ┌────────────────┐                      ┌────────────────┐
11//! │ layout         │ ──────────────────── │ item_layout    │
12//! │ drop_fn        │ ──────────────────── │ drop_fn        │
13//! │ clone_fn       │ (for archetype move) │ data           │
14//! └────────────────┘                      └────────────────┘
15//! ```
16//!
17//! # Safety
18//!
19//! This module uses unsafe code to provide type-erased operations.
20//! All safety invariants are documented and enforced through the API design.
21
22use std::alloc::Layout;
23use std::any::TypeId;
24use std::collections::HashMap;

26use crate::blob_vec::BlobVec;
27use crate::Component;

29/// Metadata for a component type, enabling type-erased BlobVec operations.
30///
31/// This is the key to replacing `Box<dyn Any>` with `BlobVec`:
32/// - Stores layout for proper allocation
33/// - Stores drop function for correct cleanup
34/// - Stores clone function for archetype transitions
35#[derive(Clone)]
36pub struct ComponentMeta {
37    /// Memory layout of the component type
38    pub layout: Layout,
39    /// Drop function (None for types that don't need drop)
40    pub drop_fn: Option<unsafe fn(*mut u8)>,
41    /// Clone function for copying during archetype transitions
42    /// Takes (src, dst) pointers
43    pub clone_fn: unsafe fn(*const u8, *mut u8),
44    /// Human-readable type name for debugging
45    pub type_name: &'static str,
46}
47
48impl ComponentMeta {
49    /// Create ComponentMeta for a specific component type.
50    ///
51    /// # Example
52    /// ```
53    /// use astraweave_ecs::component_meta::ComponentMeta;
54    ///
55    /// #[derive(Clone, Copy)]
56    /// struct Position { x: f32, y: f32 }
57    ///
58    /// let meta = ComponentMeta::of::<Position>();
59    /// assert_eq!(meta.layout.size(), std::mem::size_of::<Position>());
60    /// ```
61    pub fn of<T: Component + Clone>() -> Self {
62        Self {
63            layout: Layout::new::<T>(),
64            drop_fn: if std::mem::needs_drop::<T>() {
65                Some(Self::drop_fn_impl::<T>)
66            } else {
67                None
68            },
69            clone_fn: Self::clone_fn_impl::<T>,
70            type_name: std::any::type_name::<T>(),
71        }
72    }
73
74    /// Drop function implementation for type T.
75    ///
76    /// # Safety
77    /// `ptr` must point to a valid, initialized value of type T.
78    unsafe fn drop_fn_impl<T>(ptr: *mut u8) {
79        ptr.cast::<T>().drop_in_place();
80    }
81
82    /// Clone function implementation for type T.
83    ///
84    /// # Safety
85    /// - `src` must point to a valid, initialized value of type T
86    /// - `dst` must be properly aligned and have enough space for T
87    unsafe fn clone_fn_impl<T: Clone>(src: *const u8, dst: *mut u8) {
88        let src_ref = &*src.cast::<T>();
89        let cloned = src_ref.clone();
90        dst.cast::<T>().write(cloned);
91    }
92
93    /// Create a BlobVec with this component's layout and drop function.
94    pub fn create_blob_vec(&self) -> BlobVec {
95        BlobVec::from_layout(self.layout, self.drop_fn)
96    }
97
98    /// Create a BlobVec with pre-allocated capacity.
99    pub fn create_blob_vec_with_capacity(&self, capacity: usize) -> BlobVec {
100        BlobVec::from_layout_with_capacity(self.layout, self.drop_fn, capacity)
101    }
102}

104/// Registry mapping TypeId to ComponentMeta.
105///
106/// This is used by ArchetypeStorage to create properly-typed BlobVecs
107/// for any component type encountered at runtime.
108#[derive(Default)]
109pub struct ComponentMetaRegistry {
110    metas: HashMap<TypeId, ComponentMeta>,
111}
112
113impl ComponentMetaRegistry {
114    pub fn new() -> Self {
115        Self::default()
116    }
117
118    /// Register a component type.
119    ///
120    /// Returns true if the type was newly registered, false if already present.
121    pub fn register<T: Component + Clone>(&mut self) -> bool {
122        let type_id = TypeId::of::<T>();
123        if self.metas.contains_key(&type_id) {
124            return false;
125        }
126        self.metas.insert(type_id, ComponentMeta::of::<T>());
127        true
128    }
129
130    /// Get metadata for a component type.
131    ///
132    /// Returns None if the type is not registered.
133    #[inline]
134    pub fn get(&self, type_id: TypeId) -> Option<&ComponentMeta> {
135        self.metas.get(&type_id)
136    }
137
138    /// Check if a component type is registered.
139    #[inline]
140    pub fn is_registered(&self, type_id: TypeId) -> bool {
141        self.metas.contains_key(&type_id)
142    }
143}

145#[cfg(test)]
146mod tests {
147    use super::*;
148
149    #[derive(Clone, Copy, Debug, PartialEq)]
150    struct Position {
151        x: f32,
152        y: f32,
153    }
154
155    #[derive(Clone, Debug, PartialEq)]
156    struct Health {
157        current: i32,
158        max: i32,
159    }
160
161    #[derive(Clone)]
162    #[allow(dead_code)]
163    struct WithDrop {
164        value: i32,
165        data: Vec<u8>,
166    }
167
168    #[test]
169    fn test_component_meta_layout() {
170        let meta = ComponentMeta::of::<Position>();
171        assert_eq!(meta.layout.size(), std::mem::size_of::<Position>());
172        assert_eq!(meta.layout.align(), std::mem::align_of::<Position>());
173        assert!(meta.drop_fn.is_none()); // Copy type doesn't need drop
174    }
175
176    #[test]
177    fn test_component_meta_drop_fn() {
178        let meta = ComponentMeta::of::<WithDrop>();
179        assert!(meta.drop_fn.is_some()); // Has Vec, needs drop
180    }
181
182    #[test]
183    fn test_component_meta_clone_fn() {
184        let meta = ComponentMeta::of::<Health>();
185
186        let src = Health {
187            current: 80,
188            max: 100,
189        };
190        let mut dst = std::mem::MaybeUninit::<Health>::uninit();
191
192        unsafe {
193            (meta.clone_fn)(
194                &src as *const Health as *const u8,
195                dst.as_mut_ptr() as *mut u8,
196            );
197            let cloned = dst.assume_init();
198            assert_eq!(cloned, src);
199        }
200    }
201
202    #[test]
203    fn test_registry_register() {
204        let mut registry = ComponentMetaRegistry::new();
205
206        assert!(registry.register::<Position>());
207        assert!(!registry.register::<Position>()); // Already registered
208
209        assert!(registry.is_registered(TypeId::of::<Position>()));
210        assert!(!registry.is_registered(TypeId::of::<Health>()));
211    }
212
213    #[test]
214    fn test_registry_get() {
215        let mut registry = ComponentMetaRegistry::new();
216        registry.register::<Position>();
217
218        let meta = registry.get(TypeId::of::<Position>()).unwrap();
219        assert_eq!(meta.type_name, "astraweave_ecs::component_meta::tests::Position");
220
221        assert!(registry.get(TypeId::of::<Health>()).is_none());
222    }
223
224    #[test]
225    fn test_create_blob_vec() {
226        let meta = ComponentMeta::of::<Position>();
227        let blob = meta.create_blob_vec();
228        assert_eq!(blob.len(), 0);
229    }
230
231    #[test]
232    fn test_create_blob_vec_with_capacity() {
233        let meta = ComponentMeta::of::<Position>();
234        let blob = meta.create_blob_vec_with_capacity(100);
235        assert!(blob.capacity() >= 100);
236    }

1// ECS/AI/Physics State Capture & Replay (Phase 0 minimal)
2// JSON snapshot of World and tick for smoke tests and determinism checks.
3
4use crate::{sim::step, World};
5use anyhow::{Context, Result};
6
7#[derive(serde::Serialize, serde::Deserialize)]
8struct Snapshot {
9    tick: u64,
10    world: WorldSerde,
11}
12
13// We can't serialize World directly due to HashMaps with non-serializable keys;
14// provide a stable serde wrapper. For Phase 0, we capture only fields we need
15// to rehydrate a minimal world state deterministically.
16#[derive(serde::Serialize, serde::Deserialize, Default)]
17struct WorldSerde {
18    t: f32,
19    next_id: u32,
20    obstacles: Vec<(i32, i32)>,
21}
22
23impl From<&World> for WorldSerde {
24    fn from(w: &World) -> Self {
25        let mut obstacles: Vec<(i32, i32)> = w.obstacles.iter().copied().collect();
26        obstacles.sort_unstable();
27        WorldSerde {
28            t: w.t,
29            next_id: w.next_id,
30            obstacles,
31        }
32    }
33}
34
35impl World {
36    fn from_serde(ws: &WorldSerde) -> Self {
37        let mut w = World::new();
38        w.t = ws.t;
39        w.next_id = ws.next_id;
40        w.obstacles = ws.obstacles.iter().copied().collect();
41        w
42    }
43}
44
45pub fn capture_state(tick: u64, path: &str, world: &World) -> Result<()> {
46    let snap = Snapshot {
47        tick,
48        world: WorldSerde::from(world),
49    };
50    let data = serde_json::to_vec_pretty(&snap).context("serializing snapshot")?;
51    std::fs::write(path, data).context(format!("writing snapshot to {}", path))?;
52    Ok(())
53}
54
55pub fn replay_state(path: &str, steps: u32, cfg: &crate::SimConfig) -> anyhow::Result<World> {
56    let data = std::fs::read(path).context(format!("failed to read snapshot file: {}", path))?;
57    let snap: Snapshot =
58        serde_json::from_slice(&data).context("failed to deserialize snapshot JSON")?;
59    let mut w = World::from_serde(&snap.world);
60    for _ in 0..steps {
61        step(&mut w, cfg);
62    }
63    Ok(w)
64}

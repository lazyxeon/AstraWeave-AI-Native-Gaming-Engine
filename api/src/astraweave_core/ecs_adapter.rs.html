11fn sim_cooldowns(world_compat: &mut World, dt: f32) {

15fn sys_sim(world: &mut ecs::World) {
16    let dt = world.get_resource::<Dt>().map(|d| d.0).unwrap_or(0.016);
17    if let Some(w) = world.get_resource_mut::<World>() {
18        sim_cooldowns(w, dt);
19    }
20    // Phase 1: mirror basic cooldown decay into ECS components if present
21    world.each_mut::<CCooldowns>(|_, cds| {
22        for v in cds.map.values_mut() {
23            *v = (*v - dt).max(0.0);
24        }
25    });
26}

28fn sys_move(world: &mut ecs::World) {
29    // Move entities one step toward desired pos (cardinal-only 4-neighborhood) per tick
30    // Deterministic order by BTreeMap underlying storage
31    // Note: no collision here—Phase 1 minimal behavior
32    // Read positions and desired goals, mutate positions
33    // We purposely run after sim (cooldowns)
34    use std::collections::BTreeMap;
35    let goals: BTreeMap<ecs::Entity, CDesiredPos> = {
36        let mut m = BTreeMap::new();
37        let q = ecs::Query::<CDesiredPos>::new(&*world);
38        for (e, g) in q {
39            m.insert(e, *g);
40        }
41        m
42    };
43    let mut moved: Vec<(ecs::Entity, IVec2, IVec2)> = vec![];
44    world.each_mut::<CPos>(|e, p| {
45        if let Some(goal) = goals.get(&e) {
46            let dx = (goal.pos.x - p.pos.x).signum();
47            let mut dy = (goal.pos.y - p.pos.y).signum();
48            // Cardinal-only behavior: prefer moving along X this tick; if we move in X,
49            // do not also move in Y (prevents diagonal movement).
50            if dx != 0 {
51                dy = 0;
52            }
53            if dx != 0 || dy != 0 {
54                let from = IVec2 {
55                    x: p.pos.x,
56                    y: p.pos.y,
57                };
58                if dx != 0 {
59                    p.pos.x += dx;
60                } else if dy != 0 {
61                    p.pos.y += dy;
62                }
63                moved.push((
64                    e,
65                    from,
66                    IVec2 {
67                        x: p.pos.x,
68                        y: p.pos.y,
69                    },
70                ));
71            }
72        }
73    });
74    if let Some(ev) = world.get_resource_mut::<Events<MovedEvent>>() {
75        let mut w = ev.writer();
76        for (e, from, to) in moved {
77            w.send(MovedEvent {
78                entity: e,
79                from,
80                to,
81            });
82        }
83    }
84}

95fn sys_bridge_sync(world: &mut ecs::World) {
96    // Ensure any mapped ECS entities carry a CLegacyId component and
97    // remove CLegacyId from entities not present in the bridge.
98    use std::collections::BTreeSet;
99
100    // Collect all ecs entities referenced by the bridge
101    let mut referenced = BTreeSet::new();
102    if let Some(bridge) = world.get_resource::<EntityBridge>() {
103        for ecs_e in bridge.ecs_entities() {
104            referenced.insert(ecs_e);
105        }
106    }
107
108    // Add CLegacyId to referenced entities if missing
109    for &e in referenced.iter() {
110        if world.get::<crate::CLegacyId>(e).is_none() {
111            if let Some(bridge) = world.get_resource::<EntityBridge>() {
112                if let Some(legacy) = bridge.get_by_ecs(&e) {
113                    world.insert(e, crate::CLegacyId { id: legacy });
114                }
115            }
116        }
117    }
118
119    // Note: Phase 1 `astraweave_ecs::World` does not provide a component removal
120    // API. Removing CLegacyId entries would require extending the ECS. For now
121    // we only ensure referenced entities have the CLegacyId component. Stale
122    // CLegacyId components (if any) will remain until a future ECS API adds
123    // removal support.
124}

128/// Build a minimal ECS app with stages and a single simulation system that
129/// bridges into the legacy `World` struct for Phase 1.
130pub fn build_app(legacy_world: World, dt: f32) -> ecs::App {
131    let mut app = ecs::App::new();
132    // Insert base resources first
133    app.world.insert_resource(Dt(dt));
134    app.world.insert_resource(Events::<MovedEvent>::default());
135    app.world.insert_resource(EntityBridge::default());
136
137    // Auto-populate ECS entities and the entity bridge from the provided legacy World
138    // using the owned `legacy_world` to avoid borrowing app.world while also mutating it.
139    for legacy in legacy_world.entities() {
140        let e = app.world.spawn();
141        // Mirror pose if present
142        if let Some(p) = legacy_world.pose(legacy) {
143            app.world.insert(
144                e,
145                CPos {
146                    pos: IVec2 {
147                        x: p.pos.x,
148                        y: p.pos.y,
149                    },
150                },
151            );
152        }
153        if let Some(h) = legacy_world.health(legacy) {
154            app.world.insert(e, CHealth { hp: h.hp });
155        }
156        if let Some(t) = legacy_world.team(legacy) {
157            app.world.insert(e, CTeam { id: t.id });
158        }
159        if let Some(a) = legacy_world.ammo(legacy) {
160            app.world.insert(e, CAmmo { rounds: a.rounds });
161        }
162        if let Some(cds) = legacy_world.cooldowns(legacy) {
163            // convert HashMap<String,f32> -> BTreeMap<CooldownKey,f32> for CCooldowns
164            let map: crate::cooldowns::Map = cds
165                .map
166                .iter()
167                .map(|(k, v)| (crate::cooldowns::CooldownKey::from(k.as_str()), *v))
168                .collect();
169            app.world.insert(e, CCooldowns { map });
170        }
171        // populate bridge
172        if let Some(bridge) = app.world.get_resource_mut::<EntityBridge>() {
173            bridge.insert_pair(legacy, e);
174        }
175    }
176
177    // Now insert the legacy world as a resource so systems can access it.
178    app.world.insert_resource::<World>(legacy_world);
179    app.add_system("simulation", sys_sim as ecs::SystemFn);
180    app.add_system("simulation", sys_move as ecs::SystemFn);
181    // Bridge sync runs after simulation so mappings are reflected into components
182    app.add_system("sync", sys_bridge_sync as ecs::SystemFn);
183    // AI planning system is registered from astraweave-ai crate to avoid a dependency cycle.
184    app.add_system("perception", sys_refresh_los as ecs::SystemFn);
185    app
186}

286    #[test]
287    fn parity_ecs_vs_legacy_movement_and_cooldowns() {
288        // Create identical legacy and ECS worlds, run for 10 ticks, compare final state
289        let mut legacy_world = World::new();
290        let legacy_entity =
291            legacy_world.spawn("test", IVec2 { x: 0, y: 0 }, crate::Team { id: 1 }, 100, 10);
292        // Set desired position by directly modifying pose (legacy doesn't have desired pos concept)
293        if let Some(pose) = legacy_world.pose_mut(legacy_entity) {
294            pose.pos = IVec2 { x: 5, y: 3 }; // Move to target position
295        }
296        // Set cooldown
297        if let Some(cds) = legacy_world.cooldowns_mut(legacy_entity) {
298            cds.map.insert("test_cd".to_string(), 1.0);
299        }
300
301        // Create ECS world with same initial state
302        let ecs_world = World::new();
303        let mut ecs_app = build_app(ecs_world, 0.016);
304        // Create ECS entity and set up bridge
305        let ecs_entity = ecs_app.world.spawn();
306        if let Some(bridge) = ecs_app.world.get_resource_mut::<EntityBridge>() {
307            bridge.insert_pair(legacy_entity, ecs_entity);
308        }
309        // Set initial position in ECS
310        ecs_app.world.insert(
311            ecs_entity,
312            CPos {
313                pos: IVec2 { x: 0, y: 0 },
314            },
315        );
316        // Set desired position in ECS
317        ecs_app.world.insert(
318            ecs_entity,
319            CDesiredPos {
320                pos: IVec2 { x: 5, y: 3 },
321            },
322        );
323        // Set cooldown in ECS
324        ecs_app.world.insert(
325            ecs_entity,
326            CCooldowns {
327                map: std::collections::BTreeMap::from([(
328                    crate::cooldowns::CooldownKey::from("test_cd"),
329                    1.0,
330                )]),
331            },
332        );
333        // Set health in ECS
334        ecs_app.world.insert(ecs_entity, CHealth { hp: 100 });
335
336        // Run 10 ticks
337        for _ in 0..10 {
338            legacy_world.tick(0.016);
339            // For legacy, manually move toward desired position (simplified movement)
340            if let Some(pose) = legacy_world.pose_mut(legacy_entity) {
341                let current = pose.pos;
342                let target = IVec2 { x: 5, y: 3 };
343                let dx = (target.x - current.x).signum();
344                let dy = (target.y - current.y).signum();
345                pose.pos.x += dx;
346                pose.pos.y += dy;
347            }
348        }
349        ecs_app = ecs_app.run_fixed(10);
350
351        // Compare positions
352        let legacy_pos = legacy_world.pos_of(legacy_entity).unwrap();
353        let ecs_pos = ecs_app.world.get::<CPos>(ecs_entity).unwrap().pos;
354        assert_eq!(legacy_pos, ecs_pos, "Positions should match after 10 ticks");
355
356        // Compare cooldowns
357        let legacy_cd = legacy_world
358            .cooldowns(legacy_entity)
359            .unwrap()
360            .map
361            .get("test_cd")
362            .copied()
363            .unwrap_or(0.0);
364        let ecs_cd = ecs_app
365            .world
366            .get::<CCooldowns>(ecs_entity)
367            .unwrap()
368            .map
369            .get(&crate::cooldowns::CooldownKey::from("test_cd"))
370            .copied()
371            .unwrap_or(0.0);
372        assert!(
373            (legacy_cd - ecs_cd).abs() < 1e-6,
374            "Cooldowns should match: legacy={:.3}, ecs={:.3}",
375            legacy_cd,
376            ecs_cd
377        );
378
379        // Compare health (should be unchanged)
380        let legacy_hp = legacy_world.health(legacy_entity).unwrap().hp;
381        let ecs_hp = ecs_app.world.get::<CHealth>(ecs_entity).unwrap().hp;
382        assert_eq!(legacy_hp, ecs_hp, "Health should match");
383    }

1use crate::{ActionStep, EngineError};
2
3#[derive(Clone, Debug, PartialEq, Eq)]
4pub enum ToolBlockReason {
5    Cooldown,
6    LineOfSight,
7    PathBlocked,
8    Invalid,
9    Other,
10}
11
12#[derive(Clone, Debug, PartialEq, Eq)]
13pub struct ToolBlock {
14    pub reason: ToolBlockReason,
15    pub tool: &'static str,
16    pub msg: String,
17}
18
19/// Map engine validation errors to a stable tool taxonomy with light metadata.
20pub fn map_engine_error(step: &ActionStep, err: &EngineError) -> ToolBlock {
21    let tool = match step {
22        ActionStep::MoveTo { .. } => "MoveTo",
23        ActionStep::Throw { .. } => "Throw",
24        ActionStep::CoverFire { .. } => "CoverFire",
25        ActionStep::Revive { .. } => "Revive",
26    };
27    match err {
28        EngineError::Cooldown(s) => ToolBlock {
29            reason: ToolBlockReason::Cooldown,
30            tool,
31            msg: format!("cooldown: {}", s),
32        },
33        EngineError::LosBlocked => ToolBlock {
34            reason: ToolBlockReason::LineOfSight,
35            tool,
36            msg: "line of sight".into(),
37        },
38        EngineError::NoPath => ToolBlock {
39            reason: ToolBlockReason::PathBlocked,
40            tool,
41            msg: "no path".into(),
42        },
43        EngineError::InvalidAction(m) => ToolBlock {
44            reason: ToolBlockReason::Invalid,
45            tool,
46            msg: m.clone(),
47        },
48        EngineError::Resource(k) => ToolBlock {
49            reason: ToolBlockReason::Other,
50            tool,
51            msg: format!("resource:{}", k),
52        },
53    }
54}

64    #[test]
65    fn taxonomy_maps_correctly() {
66        let step = ActionStep::MoveTo { x: 0, y: 0 };
67        let b = map_engine_error(&step, &EngineError::NoPath);
68        assert_eq!(b.reason, ToolBlockReason::PathBlocked);
69        assert_eq!(b.tool, "MoveTo");
70        let b2 = map_engine_error(
71            &ActionStep::Throw {
72                item: "smoke".into(),
73                x: 0,
74                y: 0,
75            },
76            &EngineError::Cooldown("throw:smoke".into()),
77        );
78        assert_eq!(b2.reason, ToolBlockReason::Cooldown);
79        assert_eq!(b2.tool, "Throw");
80    }

93    #[test]
94    fn los_and_cooldown_map_to_taxonomy() {
95        // World: two entities with an obstacle between
96        let mut w = World::new();
97        let a = w.spawn("ally", IVec2 { x: 0, y: 0 }, Team { id: 1 }, 100, 10);
98        let e = w.spawn("enemy", IVec2 { x: 2, y: 0 }, Team { id: 2 }, 50, 0);
99        w.obstacles.insert((1, 0)); // blocks LOS
100        let cfg = ValidateCfg {
101            world_bounds: (-10, -10, 10, 10),
102        };
103        let mut log = |_s: String| {};
104        // Try cover fire -> expect LOS blocked
105        let intent = PlanIntent {
106            plan_id: "p".into(),
107            steps: vec![ActionStep::CoverFire {
108                target_id: e,
109                duration: 1.0,
110            }],
111        };
112        let err = validate_and_execute(&mut w, a, &intent, &cfg, &mut log).unwrap_err();
113        let tb = map_engine_error(intent.steps.first().unwrap(), &err);
114        assert_eq!(tb.reason, ToolBlockReason::LineOfSight);
115        // Cooldown: prime a cooldown and verify mapping
116        let intent = PlanIntent {
117            plan_id: "p2".into(),
118            steps: vec![ActionStep::Throw {
119                item: "smoke".into(),
120                x: 0,
121                y: 0,
122            }],
123        };
124        // First use sets cooldown
125        w.obstacles.clear();
126        let _ = validate_and_execute(&mut w, a, &intent, &cfg, &mut log);
127        // Second use should hit cooldown
128        let err2 = validate_and_execute(&mut w, a, &intent, &cfg, &mut log).unwrap_err();
129        let tb2 = map_engine_error(intent.steps.first().unwrap(), &err2);
130        assert_eq!(tb2.reason, ToolBlockReason::Cooldown);
131        assert!(tb2.msg.contains("throw:smoke"));
132    }

136        let mut w = World::new();
137        let a = w.spawn("ally", IVec2 { x: 0, y: 0 }, Team { id: 1 }, 100, 0);
138        let e = w.spawn("enemy", IVec2 { x: 3, y: 0 }, Team { id: 2 }, 50, 0);
139        // First step succeeds (move), second fails (cover_fire no ammo)
140        let intent = PlanIntent {
141            plan_id: "p3".into(),
142            steps: vec![
143                ActionStep::MoveTo { x: 1, y: 0 },
144                ActionStep::CoverFire {
145                    target_id: e,
146                    duration: 1.0,
147                },
148                ActionStep::MoveTo { x: 2, y: 0 }, // should not execute
149            ],
150        };
151        let cfg = ValidateCfg {
152            world_bounds: (-10, -10, 10, 10),
153        };
154        let mut log = |_s: String| {};
155        let res = validate_and_execute(&mut w, a, &intent, &cfg, &mut log);
156        assert!(matches!(res, Err(EngineError::Resource(_))));
157        // Ensure the first move happened, but not the last

159    }

1use crate::{ActionStep, EngineError};
2
3#[derive(Clone, Debug, PartialEq, Eq)]
4pub enum ToolBlockReason {
5    Cooldown,
6    LineOfSight,
7    PathBlocked,
8    Invalid,
9    Other,
10}
11
12#[derive(Clone, Debug, PartialEq, Eq)]
13pub struct ToolBlock {
14    pub reason: ToolBlockReason,
15    pub tool: &'static str,
16    pub msg: String,
17}
18
19/// Map engine validation errors to a stable tool taxonomy with light metadata.
20pub fn map_engine_error(step: &ActionStep, err: &EngineError) -> ToolBlock {
21    let tool = match step {
22        ActionStep::MoveTo { .. } => "MoveTo",
23        ActionStep::Throw { .. } => "Throw",
24        ActionStep::CoverFire { .. } => "CoverFire",
25        ActionStep::Revive { .. } => "Revive",
26    };
27    match err {
28        EngineError::Cooldown(s) => ToolBlock { reason: ToolBlockReason::Cooldown, tool, msg: format!("cooldown: {}", s) },
29        EngineError::LosBlocked => ToolBlock { reason: ToolBlockReason::LineOfSight, tool, msg: "line of sight".into() },
30        EngineError::NoPath => ToolBlock { reason: ToolBlockReason::PathBlocked, tool, msg: "no path".into() },
31        EngineError::InvalidAction(m) => ToolBlock { reason: ToolBlockReason::Invalid, tool, msg: m.clone() },
32        EngineError::Resource(k) => ToolBlock { reason: ToolBlockReason::Other, tool, msg: format!("resource:{}", k) },
33    }
34}

38    use super::*;
39    use crate::{validation::{validate_and_execute, ValidateCfg}, world::World, IVec2, Team, PlanIntent};
40    #[test]
41    fn taxonomy_maps_correctly() {
42        let step = ActionStep::MoveTo{ x: 0, y: 0 };
43        let b = map_engine_error(&step, &EngineError::NoPath);
44        assert_eq!(b.reason, ToolBlockReason::PathBlocked);
45        assert_eq!(b.tool, "MoveTo");
46        let b2 = map_engine_error(&ActionStep::Throw{ item: "smoke".into(), x:0, y:0 }, &EngineError::Cooldown("throw:smoke".into()));
47        assert_eq!(b2.reason, ToolBlockReason::Cooldown);
48        assert_eq!(b2.tool, "Throw");
49    }

51    #[test]
52    fn resource_maps_to_other() {
53        let step = ActionStep::CoverFire{ target_id: 1, duration: 1.0 };
54        let b = map_engine_error(&step, &EngineError::Resource("ammo".into()));
55        assert_eq!(b.reason, ToolBlockReason::Other);
56        assert!(b.msg.contains("ammo"));
57    }

59    #[test]
60    fn los_and_cooldown_map_to_taxonomy() {
61        // World: two entities with an obstacle between
62        let mut w = World::new();
63        let a = w.spawn("ally", IVec2{ x:0, y:0 }, Team{ id:1 }, 100, 10);
64        let e = w.spawn("enemy", IVec2{ x:2, y:0 }, Team{ id:2 }, 50, 0);
65        w.obstacles.insert((1,0)); // blocks LOS
66        let cfg = ValidateCfg{ world_bounds: (-10,-10,10,10) };
67        let mut log = |_s: String|{};
68        // Try cover fire -> expect LOS blocked
69        let intent = PlanIntent{ plan_id: "p".into(), steps: vec![ActionStep::CoverFire{ target_id: e, duration: 1.0 }] };
70        let err = validate_and_execute(&mut w, a, &intent, &cfg, &mut log).unwrap_err();
71        let tb = map_engine_error(intent.steps.first().unwrap(), &err);
72        assert_eq!(tb.reason, ToolBlockReason::LineOfSight);
73        // Cooldown: prime a cooldown and verify mapping
74        let intent = PlanIntent{ plan_id: "p2".into(), steps: vec![ActionStep::Throw{ item: "smoke".into(), x:0, y:0 }] };
75        // First use sets cooldown
76        w.obstacles.clear();
77        let _ = validate_and_execute(&mut w, a, &intent, &cfg, &mut log);
78        // Second use should hit cooldown
79        let err2 = validate_and_execute(&mut w, a, &intent, &cfg, &mut log).unwrap_err();
80        let tb2 = map_engine_error(intent.steps.first().unwrap(), &err2);
81        assert_eq!(tb2.reason, ToolBlockReason::Cooldown);
82        assert!(tb2.msg.contains("throw:smoke"));
83    }

92            ActionStep::MoveTo{ x:1, y:0 },

95        ]};
96        let cfg = ValidateCfg{ world_bounds: (-10,-10,10,10) };
97        let mut log = |_s: String|{};
98        let res = validate_and_execute(&mut w, a, &intent, &cfg, &mut log);
99        assert!(matches!(res, Err(EngineError::Resource(_))));
100        // Ensure the first move happened, but not the last
101        assert_eq!(w.pos_of(a).unwrap(), IVec2{ x:1, y:0 });
102    }

3#[derive(Clone, Debug, PartialEq, Eq)]
4pub enum ToolBlockReason {
5    Cooldown,
6    LineOfSight,
7    PathBlocked,
8    Invalid,
9    Other,
10}
11
12#[derive(Clone, Debug, PartialEq, Eq)]
13pub struct ToolBlock {
14    pub reason: ToolBlockReason,
15    pub tool: &'static str,
16    pub msg: String,
17}
18
19/// Map engine validation errors to a stable tool taxonomy with light metadata.
20pub fn map_engine_error(step: &ActionStep, err: &EngineError) -> ToolBlock {
21    let tool = match step {
22        // Movement
23        ActionStep::MoveTo { .. } => "MoveTo",
24        ActionStep::Approach { .. } => "Approach",
25        ActionStep::Retreat { .. } => "Retreat",
26        ActionStep::TakeCover { .. } => "TakeCover",
27        ActionStep::Strafe { .. } => "Strafe",
28        ActionStep::Patrol { .. } => "Patrol",
29        
30        // Offensive
31        ActionStep::Attack { .. } => "Attack",
32        ActionStep::AimedShot { .. } => "AimedShot",
33        ActionStep::QuickAttack { .. } => "QuickAttack",
34        ActionStep::HeavyAttack { .. } => "HeavyAttack",
35        ActionStep::AoEAttack { .. } => "AoEAttack",
36        ActionStep::ThrowExplosive { .. } => "ThrowExplosive",
37        ActionStep::CoverFire { .. } => "CoverFire",
38        ActionStep::Charge { .. } => "Charge",
39        
40        // Defensive
41        ActionStep::Block => "Block",
42        ActionStep::Dodge { .. } => "Dodge",
43        ActionStep::Parry => "Parry",
44        ActionStep::ThrowSmoke { .. } => "ThrowSmoke",
45        ActionStep::Heal { .. } => "Heal",
46        ActionStep::UseDefensiveAbility { .. } => "UseDefensiveAbility",
47        
48        // Equipment
49        ActionStep::EquipWeapon { .. } => "EquipWeapon",
50        ActionStep::SwitchWeapon { .. } => "SwitchWeapon",
51        ActionStep::Reload => "Reload",
52        ActionStep::UseItem { .. } => "UseItem",
53        ActionStep::DropItem { .. } => "DropItem",
54        
55        // Tactical
56        ActionStep::CallReinforcements { .. } => "CallReinforcements",
57        ActionStep::MarkTarget { .. } => "MarkTarget",
58        ActionStep::RequestCover { .. } => "RequestCover",
59        ActionStep::CoordinateAttack { .. } => "CoordinateAttack",
60        ActionStep::SetAmbush { .. } => "SetAmbush",
61        ActionStep::Distract { .. } => "Distract",
62        ActionStep::Regroup { .. } => "Regroup",
63        
64        // Utility
65        ActionStep::Scan { .. } => "Scan",
66        ActionStep::Wait { .. } => "Wait",
67        ActionStep::Interact { .. } => "Interact",
68        ActionStep::UseAbility { .. } => "UseAbility",
69        ActionStep::Taunt { .. } => "Taunt",
70        
71        // Legacy
72        ActionStep::Throw { .. } => "Throw",
73        ActionStep::Revive { .. } => "Revive",
74    };
75    
76    match err {
77        EngineError::Cooldown(s) => ToolBlock {
78            reason: ToolBlockReason::Cooldown,
79            tool,
80            msg: format!("cooldown: {}", s),
81        },
82        EngineError::LosBlocked => ToolBlock {
83            reason: ToolBlockReason::LineOfSight,
84            tool,
85            msg: "line of sight".into(),
86        },
87        EngineError::NoPath => ToolBlock {
88            reason: ToolBlockReason::PathBlocked,
89            tool,
90            msg: "no path".into(),
91        },
92        EngineError::InvalidAction(m) => ToolBlock {
93            reason: ToolBlockReason::Invalid,
94            tool,
95            msg: m.clone(),
96        },
97        EngineError::Resource(k) => ToolBlock {
98            reason: ToolBlockReason::Other,
99            tool,
100            msg: format!("resource:{}", k),
101        },
102    }
103}

125            &EngineError::Cooldown("throw:smoke".into()),
126        );

128        assert_eq!(b2.tool, "Throw");
129    }

142    #[test]
143    fn los_and_cooldown_map_to_taxonomy() {
144        // World: two entities with an obstacle between
145        let mut w = World::new();
146        let a = w.spawn("ally", IVec2 { x: 0, y: 0 }, Team { id: 1 }, 100, 10);
147        let e = w.spawn("enemy", IVec2 { x: 2, y: 0 }, Team { id: 2 }, 50, 0);
148        w.obstacles.insert((1, 0)); // blocks LOS
149        let cfg = ValidateCfg {
150            world_bounds: (-10, -10, 10, 10),
151        };
152        let mut log = |_s: String| {};
153        // Try cover fire -> expect LOS blocked
154        let intent = PlanIntent {
155            plan_id: "p".into(),
156            steps: vec![ActionStep::CoverFire {
157                target_id: e,
158                duration: 1.0,
159            }],
160        };
161        let err = validate_and_execute(&mut w, a, &intent, &cfg, &mut log).unwrap_err();
162        let tb = map_engine_error(intent.steps.first().unwrap(), &err);
163        assert_eq!(tb.reason, ToolBlockReason::LineOfSight);
164        // Cooldown: prime a cooldown and verify mapping
165        let intent = PlanIntent {
166            plan_id: "p2".into(),
167            steps: vec![ActionStep::Throw {
168                item: "smoke".into(),
169                x: 0,
170                y: 0,
171            }],
172        };
173        // First use sets cooldown
174        w.obstacles.clear();
175        let _ = validate_and_execute(&mut w, a, &intent, &cfg, &mut log);
176        // Second use should hit cooldown
177        let err2 = validate_and_execute(&mut w, a, &intent, &cfg, &mut log).unwrap_err();
178        let tb2 = map_engine_error(intent.steps.first().unwrap(), &err2);
179        assert_eq!(tb2.reason, ToolBlockReason::Cooldown);
180        assert!(tb2.msg.contains("throw:smoke"));
181    }

183    #[test]
184    fn partial_failure_multi_step_stops_on_error() {
185        let mut w = World::new();
186        let a = w.spawn("ally", IVec2 { x: 0, y: 0 }, Team { id: 1 }, 100, 0);
187        let e = w.spawn("enemy", IVec2 { x: 3, y: 0 }, Team { id: 2 }, 50, 0);
188        // First step succeeds (move), second fails (cover_fire no ammo)
189        let intent = PlanIntent {
190            plan_id: "p3".into(),
191            steps: vec![
192                ActionStep::MoveTo { x: 1, y: 0, speed: None },
193                ActionStep::CoverFire {
194                    target_id: e,
195                    duration: 1.0,
196                },
197                ActionStep::MoveTo { x: 2, y: 0, speed: None }, // should not execute
198            ],
199        };
200        let cfg = ValidateCfg {
201            world_bounds: (-10, -10, 10, 10),
202        };
203        let mut log = |_s: String| {};
204        let res = validate_and_execute(&mut w, a, &intent, &cfg, &mut log);
205        assert!(matches!(res, Err(EngineError::Resource(_))));
206        // Ensure the first move happened, but not the last
207        assert_eq!(w.pos_of(a).unwrap(), IVec2 { x: 1, y: 0 });
208    }

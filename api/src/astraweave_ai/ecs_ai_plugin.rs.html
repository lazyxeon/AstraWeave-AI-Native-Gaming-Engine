11// Helper: map a legacy companion id to an ECS entity.
12// 1) find the closest ECS entity on team 1 by position relative to snap.me
13// 2) if an EntityBridge exists prefer the bridge mapping for the legacy id
14// 3) return None if no candidate ECS companion was found
15fn map_legacy_companion_to_ecs(
16    positions: &std::collections::BTreeMap<ecs::Entity, IVec2>,
17    teams: &std::collections::BTreeMap<ecs::Entity, u8>,
18    snap: &WorldSnapshot,
19    comp: astraweave_core::Entity,
20    world: &ecs::World,
21) -> Option<ecs::Entity> {
22    // find closest companion by Manhattan distance to the legacy companion position
23    let closest = positions
24        .iter()
25        .filter(|(e, _)| teams.get(e).copied() == Some(1))
26        .map(|(e, p)| (*e, p))
27        .min_by_key(|(_, p)| (p.x - snap.me.pos.x).abs() + (p.y - snap.me.pos.y).abs())
28        .map(|(e, _)| e)?;
29
30    // prefer bridge mapping from legacy id -> ecs entity when present
31    let mapped = world
32        .resource::<EntityBridge>()
33        .and_then(|b| b.get(&comp))
34        .unwrap_or(closest);
35    Some(mapped)
36}

38fn sys_ai_planning(world: &mut ecs::World) {
39    // Build snapshots and set desired positions per companion entity
40    // Prefer legacy World + perception builder when available; fallback to ECS-only snapshot
41    // Ensure AiPlannedEvent resource exists
42    if world.resource::<Events<AiPlannedEvent>>().is_none() {
43        world.insert_resource(Events::<AiPlannedEvent>::default());
44    }
45
46    // Cache ECS component views
47    let mut positions: std::collections::BTreeMap<ecs::Entity, IVec2> = std::collections::BTreeMap::new();
48    let mut teams: std::collections::BTreeMap<ecs::Entity, u8> = std::collections::BTreeMap::new();
49    {
50        let q = ecs::Query::<CPos>::new(world);
51        for (e, p) in q { positions.insert(e, IVec2 { x: p.x, y: p.y }); }
52    }
53    {
54        let q = ecs::Query::<CTeam>::new(world);
55        for (e, t) in q { teams.insert(e, t.id); }
56    }
57
58    let orch = RuleOrchestrator;
59    let mut updates: Vec<(ecs::Entity, CDesiredPos)> = vec![];
60    let mut planned_events: Vec<AiPlannedEvent> = vec![];
61
62    // Try legacy world path
63    if let Some(w) = world.resource::<World>() {
64        // pick first player and companion if present
65        let player_opt = w.all_of_team(0).first().copied();
66        let comp_opt = w.all_of_team(1).first().copied();
67        if let (Some(player), Some(comp)) = (player_opt, comp_opt) {
68            let enemies = w.enemies_of(1);
69            let snap = build_snapshot(
70                w,
71                player,
72                comp,
73                &enemies,
74                None,
75                &PerceptionConfig { los_max: 10 },
76            );
77            let plan = orch.propose_plan(&snap);
78            if let Some(ActionStep::MoveTo { x, y }) = plan.steps.iter().find_map(|s| {
79                if let ActionStep::MoveTo { x, y } = s { Some(ActionStep::MoveTo { x: *x, y: *y }) } else { None }
80            }) {
81                if let Some(mapped) = map_legacy_companion_to_ecs(&positions, &teams, &snap, comp, world) {
82                        updates.push((mapped, CDesiredPos { pos: IVec2 { x, y } }));
83                    planned_events.push(AiPlannedEvent { entity: mapped, target: IVec2 { x, y } });
84                }
85            }
86            // Early return after legacy-based planning for now (single companion minimal)
87            if !updates.is_empty() {
88                for (e, d) in &updates { world.insert(*e, *d); }
89                if let Some(ev) = world.resource_mut::<Events<AiPlannedEvent>>() {
90                    let mut w = ev.writer();
91                    for pe in planned_events { w.send(pe); }
92                }
93                return;
94            }
95        }
96    }
97
98    // Fallback: ECS-only snapshot composition
99    let player = PlayerState { hp: 100, pos: IVec2 { x: 0, y: 0 }, stance: "stand".into(), orders: vec![] };
100    let enemies: Vec<EnemyState> = positions
101        .iter()
102        .filter_map(|(e, pos)| {
103            let team_id = teams.get(e).copied().unwrap_or(0);
104            if team_id == 2 {
105                Some(EnemyState { id: 0, pos: *pos, hp: 50, cover: "low".into(), last_seen: 0.0 })
106            } else { None }
107        })
108        .collect();
109    for (e, pos) in &positions {
110        if teams.get(e).copied() != Some(1) { continue; }
111        let ammo = world.get::<CAmmo>(*e).map(|a| a.rounds).unwrap_or(0);
112        let cds_map = world.get::<CCooldowns>(*e).map(|c| c.map.clone()).unwrap_or_default();
113        let me = CompanionState { ammo, cooldowns: cds_map, morale: 1.0, pos: *pos };
114        let snap = WorldSnapshot { t: 0.0, player: player.clone(), me, enemies: enemies.clone(), pois: vec![], objective: None };
115        let plan = orch.propose_plan(&snap);
116        if let Some(ActionStep::MoveTo { x, y }) = plan.steps.iter().find_map(|s| {
117            if let ActionStep::MoveTo { x, y } = s { Some(ActionStep::MoveTo { x: *x, y: *y }) } else { None }
118        }) {
119            updates.push((*e, CDesiredPos { pos: IVec2 { x, y } }));
120            planned_events.push(AiPlannedEvent { entity: *e, target: IVec2 { x, y } });
121        }
122    }
123    for (e, d) in updates { world.insert(e, d); }
124    if let Some(ev) = world.resource_mut::<Events<AiPlannedEvent>>() {
125        let mut w = ev.writer();
126        for pe in planned_events { w.send(pe); }
127    }
128}

130impl ecs::Plugin for AiPlanningPlugin {
131    fn build(&self, app: &mut ecs::App) {
132        app.schedule.add_system("ai_planning", sys_ai_planning as ecs::SystemFn);
133    }
134}

136/// Convenience: build an ECS app with core systems and the AI planning plugin installed.
137pub fn build_app_with_ai(legacy_world: astraweave_core::World, dt: f32) -> ecs::App {
138    let mut app = astraweave_core::ecs_adapter::build_app(legacy_world, dt);
139    app = app.add_plugin(AiPlanningPlugin);
140    app
141}

143#[cfg(test)]
144mod tests {
145    use super::*;
146    use astraweave_core::{World, IVec2, Team};
147    use anyhow::anyhow;
148    use anyhow::Result;
149
150    #[test]
151    fn ai_plugin_sets_desired_position_for_companion() -> Result<()> {
152        // Legacy world can be empty for this test
153        let w = World::new();
154        let mut app = build_app_with_ai(w, 0.016);
155        // Spawn two ECS entities: a companion and an enemy
156        let ally = app.world.spawn();
157    app.world.insert(ally, CPos { pos: IVec2 { x: 0, y: 0 } });
158        app.world.insert(ally, CTeam { id: 1 });
159        app.world.insert(ally, CAmmo { rounds: 10 });
160        app.world.insert(ally, CCooldowns { map: std::collections::BTreeMap::new() });
161
162        let enemy = app.world.spawn();
163    app.world.insert(enemy, CPos { pos: IVec2 { x: 3, y: 0 } });
164        app.world.insert(enemy, CTeam { id: 2 });
165
166        app = app.run_fixed(1);
167    let d = app.world.get::<CDesiredPos>(ally).ok_or_else(|| anyhow!("desired pos set"))?;
168        // Expect to move towards enemy along +x axis
169        assert!(d.x >= 1 && d.y == 0);
170
171        // Event should be published
172    let evs = app.world.resource_mut::<Events<AiPlannedEvent>>().ok_or_else(|| anyhow!("Events<AiPlannedEvent> resource missing"))?;
173        let mut rdr = evs.reader();
174        let v: Vec<_> = rdr.drain().collect();
175        assert_eq!(v.len(), 1);
176        assert_eq!(v[0].entity, ally);
177        Ok(())
178    }

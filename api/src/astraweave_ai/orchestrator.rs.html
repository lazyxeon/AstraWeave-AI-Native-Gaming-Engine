20/// Minimal rule-based orchestrator:
21/// If enemy in LOS-ish and "smoke" not on cooldown:
22///   throw smoke midway, move up, cover fire.
23/// Else: advance towards nearest enemy.
24pub struct RuleOrchestrator;
25
26impl Orchestrator for RuleOrchestrator {
27    fn propose_plan(&self, snap: &WorldSnapshot) -> PlanIntent {
28        let plan_id = format!("plan-{}", (snap.t * 1000.0) as i64);
29        if let Some(first) = snap.enemies.first() {
30            let m = &snap.me;
31            let _p = &snap.player;
32
33            // midpoint for smoke
34            let mid = IVec2 {
35                x: (m.pos.x + first.pos.x) / 2,
36                y: (m.pos.y + first.pos.y) / 2,
37            };
38
39            let smoke_cd = snap.me.cooldowns.get(&crate::cooldowns::CooldownKey::from("throw:smoke")).copied().unwrap_or(0.0);
40            if smoke_cd <= 0.0 {
41                return PlanIntent {
42                    plan_id,
43                    steps: vec![
44                        ActionStep::Throw {
45                            item: "smoke".into(),
46                            x: mid.x,
47                            y: mid.y,
48                        },
49                        ActionStep::MoveTo {
50                            x: m.pos.x + (first.pos.x - m.pos.x).signum() * 2,
51                            y: m.pos.y + (first.pos.y - m.pos.y).signum() * 2,
52                        },
53                        ActionStep::CoverFire {
54                            target_id: first.id,
55                            duration: 2.5,
56                        },
57                    ],
58                };
59            } else {
60                // advance cautiously
61                return PlanIntent {
62                    plan_id,
63                    steps: vec![
64                        ActionStep::MoveTo {
65                            x: m.pos.x + (first.pos.x - m.pos.x).signum(),
66                            y: m.pos.y + (first.pos.y - m.pos.y).signum(),
67                        },
68                        ActionStep::CoverFire {
69                            target_id: first.id,
70                            duration: 1.5,
71                        },
72                    ],
73                };
74            }
75        }
76
77        // fallback
78        PlanIntent {
79            plan_id,
80            steps: vec![],
81        }
82    }
83}

92/// Utility-based orchestrator: scores a few candidate plans deterministically.
93/// Heuristics:
94/// - Prefer throwing smoke if an enemy exists and cooldown is ready
95/// - Otherwise move closer to nearest enemy; if very close (<3), add brief cover fire
96pub struct UtilityOrchestrator;
97
98impl Orchestrator for UtilityOrchestrator {
99    fn propose_plan(&self, snap: &WorldSnapshot) -> PlanIntent {
100        let plan_id = format!("util-{}", (snap.t * 1000.0) as i64);
101        let me = &snap.me;
102        // Candidate A: Throw smoke at midpoint to first enemy
103        let mut cands: Vec<(f32, Vec<ActionStep>)> = vec![];
104        if let Some(enemy) = snap.enemies.first() {
105            let cd = me.cooldowns.get(&crate::cooldowns::CooldownKey::from("throw:smoke")).copied().unwrap_or(0.0);
106            if cd <= 0.0 {
107                let mid = IVec2 {
108                    x: (me.pos.x + enemy.pos.x) / 2,
109                    y: (me.pos.y + enemy.pos.y) / 2,
110                };
111                let steps = vec![
112                    ActionStep::Throw {
113                        item: "smoke".into(),
114                        x: mid.x,
115                        y: mid.y,
116                    },
117                    ActionStep::MoveTo {
118                        x: me.pos.x + (enemy.pos.x - me.pos.x).signum() * 2,
119                        y: me.pos.y + (enemy.pos.y - me.pos.y).signum() * 2,
120                    },
121                ];
122                // Score: encourage if ammo present and enemy hp > 0
123                let score = 1.0
124                    + (snap.player.hp as f32).min(100.0) * 0.0
125                    + (enemy.hp.max(0) as f32) * 0.01;
126                cands.push((score, steps));
127            }
128            // Candidate B: advance; if close, cover fire
129            let dx = (enemy.pos.x - me.pos.x).abs();
130            let dy = (enemy.pos.y - me.pos.y).abs();
131            let dist = (dx + dy) as f32;
132            let mut steps = vec![ActionStep::MoveTo {
133                x: me.pos.x + (enemy.pos.x - me.pos.x).signum(),
134                y: me.pos.y + (enemy.pos.y - me.pos.y).signum(),
135            }];
136            if dist <= 3.0 {
137                steps.push(ActionStep::CoverFire {
138                    target_id: enemy.id,
139                    duration: 1.0,
140                });
141            }
142            let score = 0.8 + (3.0 - dist).max(0.0) * 0.05;
143            cands.push((score, steps));
144        }
145    // Use total_cmp to provide a total ordering for f32 (handles NaN deterministically)
146    cands.sort_by(|a, b| b.0.total_cmp(&a.0));
147        let steps = cands.first().map(|c| c.1.clone()).unwrap_or_default();
148        PlanIntent { plan_id, steps }
149    }
150}

159/// Minimal GOAP-style orchestrator for MoveTo -> CoverFire chain towards first enemy.
160/// Preconditions: enemy exists. Goal: be within 2 cells and apply CoverFire for 1.5s.
161pub struct GoapOrchestrator;
162
163impl Orchestrator for GoapOrchestrator {
164    fn propose_plan(&self, snap: &WorldSnapshot) -> PlanIntent {
165        let plan_id = format!("goap-{}", (snap.t * 1000.0) as i64);
166        if let Some(enemy) = snap.enemies.first() {
167            let me = &snap.me;
168            let dx = (enemy.pos.x - me.pos.x).abs();
169            let dy = (enemy.pos.y - me.pos.y).abs();
170            let dist = dx + dy;
171            if dist <= 2 {
172                return PlanIntent {
173                    plan_id,
174                    steps: vec![ActionStep::CoverFire {
175                        target_id: enemy.id,
176                        duration: 1.5,
177                    }],
178                };
179            } else {
180                // Move one step closer; replan next tick
181                return PlanIntent {
182                    plan_id,
183                    steps: vec![ActionStep::MoveTo {
184                        x: me.pos.x + (enemy.pos.x - me.pos.x).signum(),
185                        y: me.pos.y + (enemy.pos.y - me.pos.y).signum(),
186                    }],
187                };
188            }
189        }
190        PlanIntent {
191            plan_id,
192            steps: vec![],
193        }
194    }
195}

220#[cfg(feature = "llm_orchestrator")]
221#[async_trait::async_trait]
222impl<C> OrchestratorAsync for LlmOrchestrator<C>
223where
224    C: astraweave_llm::LlmClient + Send + Sync,
225{
226    async fn plan(&self, snap: WorldSnapshot, _budget_ms: u32) -> Result<PlanIntent> {
227        match astraweave_llm::plan_from_llm(&self.client, &snap, &self.registry).await
228            .with_context(|| "plan_from_llm call failed")
229        {
230            Ok(p) => Ok(p),
231            Err(e) => {
232                // Log for visibility; tests can assert on the returned Result if desired
233                eprintln!("plan_from_llm failed: {:?}", e);
234                // Return an explicit fallback plan wrapped in Ok so callers get a PlanIntent
235                // even on LLM parse failures, preserving previous behavior while surfacing errors.
236                Ok(PlanIntent {
237                    plan_id: "llm-fallback".into(),
238                    steps: vec![],
239                })
240            }
241        }
242    }
243    fn name(&self) -> &'static str {
244        "LlmOrchestrator"
245    }

248/// System-wide wiring utilities for choosing an orchestrator at runtime.
249/// Set ASTRAWEAVE_USE_LLM=1 to select the local LLM (phi3:medium by default) if compiled with the llm_orchestrator feature.
250#[derive(Clone, Debug)]
251pub struct SystemOrchestratorConfig {
252    pub use_llm: bool,
253    pub ollama_url: String,
254    pub ollama_model: String,
255}
256
257impl Default for SystemOrchestratorConfig {
258    fn default() -> Self {
259        let use_llm = std::env::var("ASTRAWEAVE_USE_LLM")
260            .map(|v| v == "1" || v.eq_ignore_ascii_case("true"))
261            .unwrap_or(false);
262        let ollama_url =
263            std::env::var("OLLAMA_URL").unwrap_or_else(|_| "http://127.0.0.1:11434".to_string());
264        // Default model to phi3:medium as requested
265        let ollama_model =
266            std::env::var("OLLAMA_MODEL").unwrap_or_else(|_| "phi3:medium".to_string());
267        Self {
268            use_llm,
269            ollama_url,
270            ollama_model,
271        }
272    }
273}

275/// Create a boxed orchestrator chosen by config/env.
276/// - If use_llm is true and llm_orchestrator is compiled, constructs LlmOrchestrator with Ollama chat client.
277/// - Otherwise falls back to UtilityOrchestrator.
278pub fn make_system_orchestrator(
279    cfg: Option<SystemOrchestratorConfig>,
280) -> Box<dyn OrchestratorAsync + Send + Sync> {
281    let _cfg = cfg.unwrap_or_default();
282    #[cfg(all(feature = "llm_orchestrator"))]
283    {
284    if _cfg.use_llm {
285            // Build an Ollama-backed orchestrator (ollama feature is enabled via Cargo.toml feature wiring)
286            let client = astraweave_llm::OllamaChatClient::new(
287                _cfg.ollama_url.clone(),
288                _cfg.ollama_model.clone(),
289            );
290            // Optionally warm up model in the background to minimize first-token latency
291            let do_warm = std::env::var("OLLAMA_WARMUP")
292                .map(|v| v == "1" || v.eq_ignore_ascii_case("true"))
293                .unwrap_or(true);
294            if do_warm {
295                let client_clone = client.clone();
296                let warm_secs: u64 = std::env::var("OLLAMA_WARMUP_TIMEOUT_SECS")
297                    .ok()
298                    .and_then(|s| s.parse().ok())
299                    .unwrap_or(30);
300                // Prefer spawning on Tokio if available; otherwise use a thread
301                #[cfg(feature = "llm_orchestrator")]
302                {
303                    // We cannot assume tokio is always compiled in this crate; call into a helper function only if tokio exists.
304                }
305                // Try to use tokio if linked, else fallback
306                #[cfg(any())]
307                {
308                    let _ = (client_clone, warm_secs); // placeholder for conditional compilation
309                }
310                // Fallback: spawn a thread and perform a blocking warmup with a mini runtime if tokio is available in dependencies
311                std::thread::spawn(move || {
312                    // If tokio is linked, use a small runtime; else, do nothing (can't warmup without async runtime)
313                    #[cfg(feature = "llm_orchestrator")]
314                    {
315                        // Try building a current-thread runtime; ignore errors silently
316                        if let Ok(rt) = tokio::runtime::Builder::new_current_thread()
317                            .enable_all()
318                            .build()
319                        {
320                            let _ = rt.block_on(client_clone.warmup(warm_secs));
321                        }
322                    }
323                });
324            }
325            let orch = crate::LlmOrchestrator::new(client, Some(default_tool_registry()));
326            return Box::new(orch);
327        }
328    }
329    // Default safe fallback
330    Box::new(UtilityOrchestrator)
331}

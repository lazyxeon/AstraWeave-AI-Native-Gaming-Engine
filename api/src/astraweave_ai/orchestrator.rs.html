26/// Minimal rule-based orchestrator:
27/// If enemy in LOS-ish and "smoke" not on cooldown:
28///   throw smoke midway, move up, cover fire.
29/// Else: advance towards nearest enemy.
30pub struct RuleOrchestrator;
31
32impl Orchestrator for RuleOrchestrator {
33    fn propose_plan(&self, snap: &WorldSnapshot) -> PlanIntent {
34        #[cfg(feature = "profiling")]
35        span!("AI::RuleOrchestrator::propose_plan");
36
37        let plan_id = format!("plan-{}", (snap.t * 1000.0) as i64);
38        if let Some(first) = snap.enemies.first() {
39            let m = &snap.me;
40            let _p = &snap.player;
41
42            // midpoint for smoke
43            let mid = IVec2 {
44                x: (m.pos.x + first.pos.x) / 2,
45                y: (m.pos.y + first.pos.y) / 2,
46            };
47
48            let smoke_cd = snap
49                .me
50                .cooldowns
51                .get(COOLDOWN_THROW_SMOKE)
52                .copied()
53                .unwrap_or(0.0);
54            if smoke_cd <= 0.0 {
55                return PlanIntent {
56                    plan_id,
57                    steps: vec![
58                        ActionStep::Throw {
59                            item: "smoke".into(),
60                            x: mid.x,
61                            y: mid.y,
62                        },
63                        ActionStep::MoveTo {
64                            speed: None,
65                            // Use saturating arithmetic to prevent overflow with extreme coordinates
66                            x: m.pos
67                                .x
68                                .saturating_add(first.pos.x.saturating_sub(m.pos.x).signum() * 2),
69                            y: m.pos
70                                .y
71                                .saturating_add(first.pos.y.saturating_sub(m.pos.y).signum() * 2),
72                        },
73                        ActionStep::CoverFire {
74                            target_id: first.id,
75                            duration: 2.5,
76                        },
77                    ],
78                };
79            } else {
80                // advance cautiously
81                return PlanIntent {
82                    plan_id,
83                    steps: vec![
84                        ActionStep::MoveTo {
85                            speed: None,
86                            x: m.pos.x + (first.pos.x - m.pos.x).signum(),
87                            y: m.pos.y + (first.pos.y - m.pos.y).signum(),
88                        },
89                        ActionStep::CoverFire {
90                            target_id: first.id,
91                            duration: 1.5,
92                        },
93                    ],
94                };
95            }
96        }
97
98        // fallback
99        PlanIntent {
100            plan_id,
101            steps: vec![],
102        }
103    }
104}

113/// Utility-based orchestrator: scores a few candidate plans deterministically.
114/// Heuristics:
115/// - Prefer throwing smoke if an enemy exists and cooldown is ready
116/// - Otherwise move closer to nearest enemy; if very close (<3), add brief cover fire
117pub struct UtilityOrchestrator;
118
119impl Orchestrator for UtilityOrchestrator {
120    fn propose_plan(&self, snap: &WorldSnapshot) -> PlanIntent {
121        let plan_id = format!("util-{}", (snap.t * 1000.0) as i64);
122        let me = &snap.me;
123        // Candidate A: Throw smoke at midpoint to first enemy
124        let mut cands: Vec<(f32, Vec<ActionStep>)> = vec![];
125        if let Some(enemy) = snap.enemies.first() {
126            let cd = me
127                .cooldowns
128                .get(COOLDOWN_THROW_SMOKE)
129                .copied()
130                .unwrap_or(0.0);
131            if cd <= 0.0 {
132                let mid = IVec2 {
133                    x: (me.pos.x + enemy.pos.x) / 2,
134                    y: (me.pos.y + enemy.pos.y) / 2,
135                };
136                let steps = vec![
137                    ActionStep::Throw {
138                        item: "smoke".into(),
139                        x: mid.x,
140                        y: mid.y,
141                    },
142                    ActionStep::MoveTo {
143                        speed: None,
144                        x: me.pos.x + (enemy.pos.x - me.pos.x).signum() * 2,
145                        y: me.pos.y + (enemy.pos.y - me.pos.y).signum() * 2,
146                    },
147                ];
148                // Score: encourage if ammo present and enemy hp > 0
149                let score = 1.0
150                    + (snap.player.hp as f32).min(100.0) * 0.0
151                    + (enemy.hp.max(0) as f32) * 0.01;
152                cands.push((score, steps));
153            }
154            // Candidate B: advance; if close, cover fire
155            let dx = (enemy.pos.x - me.pos.x).abs();
156            let dy = (enemy.pos.y - me.pos.y).abs();
157            let dist = (dx + dy) as f32;
158            let mut steps = vec![ActionStep::MoveTo {
159                speed: None,
160                x: me.pos.x + (enemy.pos.x - me.pos.x).signum(),
161                y: me.pos.y + (enemy.pos.y - me.pos.y).signum(),
162            }];
163            if dist <= 3.0 {
164                steps.push(ActionStep::CoverFire {
165                    target_id: enemy.id,
166                    duration: 1.0,
167                });
168            }
169            let score = 0.8 + (3.0 - dist).max(0.0) * 0.05;
170            cands.push((score, steps));
171        }
172        // Use total_cmp to provide a total ordering for f32 (handles NaN deterministically)
173        cands.sort_by(|a, b| b.0.total_cmp(&a.0));
174        let steps = cands.first().map(|c| c.1.clone()).unwrap_or_default();
175        PlanIntent { plan_id, steps }
176    }
177}

186/// Minimal GOAP-style orchestrator for MoveTo -> CoverFire chain towards first enemy.
187/// Preconditions: enemy exists. Goal: be within 2 cells and apply CoverFire for 1.5s.
188pub struct GoapOrchestrator;
189
190impl GoapOrchestrator {
191    /// Fast-path action selection without full plan generation.
192    ///
193    /// This is optimized for instant action returns (<100 µs target) by:
194    /// - Returning single ActionStep directly (not wrapped in PlanIntent)
195    /// - Minimal distance calculation (Manhattan distance)
196    /// - No allocations (returns stack-allocated ActionStep)
197    /// - Simple heuristic: move toward closest enemy or cover fire if in range
198    ///
199    /// # Arguments
200    /// - `snap`: Current world snapshot
201    ///
202    /// # Returns
203    /// An `ActionStep` to execute this frame, or `Wait { duration: 1.0 }` if no enemies
204    ///
205    /// # Performance
206    /// - **Target**: <100 µs per call
207    /// - **Typical**: 5-30 µs (distance calc + conditional)
208    ///
209    /// # Example
210    /// ```no_run
211    /// use astraweave_ai::GoapOrchestrator;
212    ///
213    /// let goap = GoapOrchestrator;
214    /// let action = goap.next_action(&snapshot); // <100 µs
215    /// apply_action(action);
216    /// ```
217    pub fn next_action(&self, snap: &WorldSnapshot) -> ActionStep {
218        #[cfg(feature = "profiling")]
219        span!("AI::GoapOrchestrator::next_action");
220
221        // Fast path: if enemy exists, move toward or engage
222        if let Some(enemy) = snap.enemies.first() {
223            let me = &snap.me;
224            let dx = enemy.pos.x - me.pos.x;
225            let dy = enemy.pos.y - me.pos.y;
226            let dist = dx.abs() + dy.abs();
227
228            if dist <= 2 {
229                // In range: cover fire
230                ActionStep::CoverFire {
231                    target_id: enemy.id,
232                    duration: 1.5,
233                }
234            } else {
235                // Out of range: move one step closer
236                ActionStep::MoveTo {
237                    speed: None,
238                    x: me.pos.x + dx.signum(),
239                    y: me.pos.y + dy.signum(),
240                }
241            }
242        } else {
243            // No enemies: wait
244            ActionStep::Wait { duration: 1.0 }
245        }
246    }
247}

249impl Orchestrator for GoapOrchestrator {
250    fn propose_plan(&self, snap: &WorldSnapshot) -> PlanIntent {
251        let plan_id = format!("goap-{}", (snap.t * 1000.0) as i64);
252        if let Some(enemy) = snap.enemies.first() {
253            let me = &snap.me;
254            // Use saturating arithmetic to prevent overflow with extreme coordinates
255            let dx = enemy.pos.x.saturating_sub(me.pos.x).abs();
256            let dy = enemy.pos.y.saturating_sub(me.pos.y).abs();
257            let dist = dx.saturating_add(dy);
258            if dist <= 2 {
259                return PlanIntent {
260                    plan_id,
261                    steps: vec![ActionStep::CoverFire {
262                        target_id: enemy.id,
263                        duration: 1.5,
264                    }],
265                };
266            } else {
267                // Move one step closer; replan next tick
268                return PlanIntent {
269                    plan_id,
270                    steps: vec![ActionStep::MoveTo {
271                        speed: None,
272                        x: me.pos.x + (enemy.pos.x - me.pos.x).signum(),
273                        y: me.pos.y + (enemy.pos.y - me.pos.y).signum(),
274                    }],
275                };
276            }
277        }
278        PlanIntent {
279            plan_id,
280            steps: vec![],
281        }
282    }
283}

308#[cfg(feature = "llm_orchestrator")]
309#[async_trait::async_trait]
310impl<C> OrchestratorAsync for LlmOrchestrator<C>
311where
312    C: astraweave_llm::LlmClient + Send + Sync,
313{
314    async fn plan(&self, snap: WorldSnapshot, budget_ms: u32) -> Result<PlanIntent> {
315        // Read timeout from environment or use budget_ms
316        let timeout_ms = std::env::var("LLM_TIMEOUT_MS")
317            .ok()
318            .and_then(|s| s.parse().ok())
319            .unwrap_or(budget_ms.max(50)); // Use budget or minimum 50ms
320
321        let timeout_duration = std::time::Duration::from_millis(timeout_ms as u64);
322
323        // Enforce hard timeout using tokio::time::timeout
324        match tokio::time::timeout(
325            timeout_duration,
326            astraweave_llm::plan_from_llm(&self.client, &snap, &self.registry),
327        )
328        .await
329        {
330            Ok(plan_source) => {
331                // LLM completed within timeout
332                match plan_source {
333                    astraweave_llm::PlanSource::Llm(plan) => Ok(plan),
334                    astraweave_llm::PlanSource::Fallback { plan, reason } => {
335                        tracing::warn!("plan_from_llm fell back: {}", reason);
336                        Ok(PlanIntent {
337                            plan_id: "llm-fallback".into(),
338                            steps: plan.steps,
339                        })
340                    }
341                }
342            }
343            Err(_elapsed) => {
344                // Timeout exceeded - return fallback
345                tracing::warn!(
346                    "LLM planning timed out after {}ms, using fallback",
347                    timeout_ms
348                );
349                Ok(PlanIntent {
350                    plan_id: "timeout-fallback".into(),
351                    steps: astraweave_llm::fallback_heuristic_plan(&snap, &self.registry).steps,
352                })
353            }
354        }
355    }
356    fn name(&self) -> &'static str {
357        "LlmOrchestrator"
358    }
359}

361/// System-wide wiring utilities for choosing an orchestrator at runtime.
362/// Set ASTRAWEAVE_USE_LLM=1 to select the local LLM (phi3:medium by default) if compiled with the llm_orchestrator feature.
363#[derive(Clone, Debug)]
364pub struct SystemOrchestratorConfig {
365    pub use_llm: bool,
366    pub ollama_url: String,
367    pub ollama_model: String,
368}
369
370impl Default for SystemOrchestratorConfig {
371    fn default() -> Self {
372        let use_llm = std::env::var("ASTRAWEAVE_USE_LLM")
373            .map(|v| v == "1" || v.eq_ignore_ascii_case("true"))
374            .unwrap_or(false);
375        let ollama_url =
376            std::env::var("OLLAMA_URL").unwrap_or_else(|_| "http://127.0.0.1:11434".to_string());
377        // Default model to phi3:medium as requested
378        let ollama_model =
379            std::env::var("OLLAMA_MODEL").unwrap_or_else(|_| "phi3:medium".to_string());
380        Self {
381            use_llm,
382            ollama_url,
383            ollama_model,
384        }
385    }
386}

388/// Create a boxed orchestrator chosen by config/env.
389/// - If use_llm is true and llm_orchestrator is compiled, constructs LlmOrchestrator with Ollama chat client.
390/// - Otherwise falls back to UtilityOrchestrator.
391pub fn make_system_orchestrator(
392    cfg: Option<SystemOrchestratorConfig>,
393) -> Box<dyn OrchestratorAsync + Send + Sync> {
394    let _cfg = cfg.unwrap_or_default();
395    #[cfg(feature = "llm_orchestrator")]
396    {
397        if _cfg.use_llm {
398            // Build an Ollama-backed orchestrator (ollama feature is enabled via Cargo.toml feature wiring)
399            let client = astraweave_llm::OllamaChatClient::new(
400                _cfg.ollama_url.clone(),
401                _cfg.ollama_model.clone(),
402            );
403            // Optionally warm up model in the background to minimize first-token latency
404            let do_warm = std::env::var("OLLAMA_WARMUP")
405                .map(|v| v == "1" || v.eq_ignore_ascii_case("true"))
406                .unwrap_or(true);
407            if do_warm {
408                let client_clone = client.clone();
409                let warm_secs: u64 = std::env::var("OLLAMA_WARMUP_TIMEOUT_SECS")
410                    .ok()
411                    .and_then(|s| s.parse().ok())
412                    .unwrap_or(30);
413                // Prefer spawning on Tokio if available; otherwise use a thread
414                #[cfg(feature = "llm_orchestrator")]
415                {
416                    // We cannot assume tokio is always compiled in this crate; call into a helper function only if tokio exists.
417                }
418                // Try to use tokio if linked, else fallback
419                #[cfg(any())]
420                {
421                    let _ = (client_clone, warm_secs); // placeholder for conditional compilation
422                }
423                // Fallback: spawn a thread and perform a blocking warmup with a mini runtime if tokio is available in dependencies
424                std::thread::spawn(move || {
425                    // If tokio is linked, use a small runtime; else, do nothing (can't warmup without async runtime)
426                    #[cfg(feature = "llm_orchestrator")]
427                    {
428                        // Try building a current-thread runtime; ignore errors silently
429                        if let Ok(rt) = tokio::runtime::Builder::new_current_thread()
430                            .enable_all()
431                            .build()
432                        {
433                            let _ = rt.block_on(client_clone.warmup(warm_secs));
434                        }
435                    }
436                });
437            }
438            let orch = crate::LlmOrchestrator::new(client, Some(default_tool_registry()));
439            return Box::new(orch);
440        }
441    }
442    // Default safe fallback
443    Box::new(UtilityOrchestrator)
444}

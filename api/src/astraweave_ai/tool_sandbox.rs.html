11/// Enumeration of all validated action verbs
12#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
13pub enum ToolVerb {
14    MoveTo,
15    Throw,
16    CoverFire,
17    Revive,
18    Interact,
19    UseItem,
20    Stay,
21    Wander,
22    Hide,
23    Rally,
24}

26/// Validation categories for each verb
27#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
28pub enum ValidationCategory {
29    Nav,
30    Physics,
31    Resources,
32    Visibility,
33    Cooldown,
34}

36/// Error taxonomy for tool validation
37#[derive(Debug, Clone, PartialEq, Eq)]
38pub enum ToolError {
39    OutOfBounds,
40    Cooldown,
41    NoLineOfSight,
42    InsufficientResource,
43    InvalidTarget,
44    PhysicsBlocked,
45    NoPath,
46    Unknown,
47}
48
49use std::fmt;
50impl fmt::Display for ToolError {
51    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
52        let s = match self {
53            ToolError::OutOfBounds => "OutOfBounds",
54            ToolError::Cooldown => "Cooldown",
55            ToolError::NoLineOfSight => "NoLineOfSight",
56            ToolError::InsufficientResource => "InsufficientResource",
57            ToolError::InvalidTarget => "InvalidTarget",
58            ToolError::PhysicsBlocked => "PhysicsBlocked",
59            ToolError::NoPath => "NoPath",
60            ToolError::Unknown => "Unknown",
61        };
62        write!(f, "{}", s)
63    }
64}

66/// Extended validation context with navmesh and physics
67pub struct ValidationContext<'a> {
68    pub nav_mesh: Option<&'a NavMesh>,
69    pub physics_pipeline: Option<&'a PhysicsPipeline>,
70    pub rigid_body_set: Option<&'a RigidBodySet>,
71    pub collider_set: Option<&'a ColliderSet>,
72}
73
74impl<'a> Default for ValidationContext<'a> {
75    fn default() -> Self {
76        Self::new()
77    }
78}
79
80impl<'a> ValidationContext<'a> {
81    pub fn new() -> Self {
82        Self {
83            nav_mesh: None,
84            physics_pipeline: None,
85            rigid_body_set: None,
86            collider_set: None,
87        }
88    }
89
90    pub fn with_nav(mut self, nav: &'a NavMesh) -> Self {
91        self.nav_mesh = Some(nav);
92        self
93    }
94
95    pub fn with_physics(
96        mut self,
97        pipeline: &'a PhysicsPipeline,
98        bodies: &'a RigidBodySet,
99        colliders: &'a ColliderSet,
100    ) -> Self {
101        self.physics_pipeline = Some(pipeline);
102        self.rigid_body_set = Some(bodies);
103        self.collider_set = Some(colliders);
104        self
105    }
106}

108/// Validate ammo availability
109fn validate_ammo(world: &WorldSnapshot) -> Result<()> {
110    if world.me.ammo == 0 {
111        anyhow::bail!("action blocked: insufficient ammo");
112    }
113    Ok(())
114}
115/// Validate line of sight to target position
116fn validate_line_of_sight(world: &WorldSnapshot, target_pos: Option<IVec2>) -> Result<()> {
117    if let Some(pos) = target_pos {
118        if !has_line_of_sight(world.me.pos, pos, world) {
119            anyhow::bail!("action blocked: no line of sight");
120        }
121    }
122    Ok(())
123}

125/// Validate a tool action for an agent in the world
126pub fn validate_tool_action(
127    _agent_id: u32,
128    verb: ToolVerb,
129    world: &WorldSnapshot,
130    context: &ValidationContext,
131    target_pos: Option<IVec2>,
132) -> Result<()> {
133    #[cfg(feature = "profiling")]
134    span!("AI::ToolSandbox::validate");
135
136    // Cooldown checks
137    if let Some(cd) = world
138        .me
139        .cooldowns
140        .get(&format!("{:?}", verb).to_lowercase())
141    {
142        if *cd > 0.0 {
143            return Err(anyhow::anyhow!(
144                "action blocked for verb {:?}: cooldown {:.2}",
145                verb,
146                cd
147            ));
148        }
149    }
150    match verb {
151        ToolVerb::MoveTo => {
152            if let Some(pos) = target_pos {
153                // Nav check: path exists
154                if let Some(nav) = context.nav_mesh {
155                    let start = glam::Vec3::new(world.me.pos.x as f32, 0.0, world.me.pos.y as f32);
156                    let goal = glam::Vec3::new(pos.x as f32, 0.0, pos.y as f32);
157                    if nav.find_path(start, goal).is_empty() {
158                        anyhow::bail!(
159                            "action blocked: no path from {:?} to {:?}",
160                            world.me.pos,
161                            pos
162                        );
163                    }
164                }
165                // Physics check: not blocked by colliders
166                if let (Some(_pipeline), Some(bodies), Some(colliders)) = (
167                    context.physics_pipeline,
168                    context.rigid_body_set,
169                    context.collider_set,
170                ) {
171                    // Check if target position has overlapping colliders
172                    // Create a small query AABB at the target position (assuming 2D movement on Y=0 plane)
173                    let half_extents = vector![0.4, 0.1, 0.4]; // Small bounding box for agent
174                    let query_pos = point![pos.x as f32, 0.0, pos.y as f32];
175                    let query_aabb = Aabb::from_half_extents(query_pos, half_extents);
176
177                    // Query for intersecting colliders, but exclude dynamic bodies (agents)
178                    // Only consider static/environment colliders as obstacles
179                    let mut intersecting = false;
180                    for (_collider_handle, collider) in colliders.iter() {
181                        // Skip colliders attached to dynamic rigid bodies (agents)
182                        if let Some(parent_handle) = collider.parent() {
183                            if let Some(rigid_body) = bodies.get(parent_handle) {
184                                // Only consider static rigid bodies as obstacles
185                                if rigid_body.is_dynamic() {
186                                    continue;
187                                }
188                            }
189                        }
190
191                        let collider_aabb = collider.compute_aabb();
192                        if query_aabb.intersects(&collider_aabb) {
193                            intersecting = true;
194                            break;
195                        }
196                    }
197
198                    if intersecting {
199                        anyhow::bail!("action blocked: physics collision at position {:?}", pos);
200                    }
201                }
202            }
203        }
204        ToolVerb::Throw => {
205            validate_ammo(world)?;
206            validate_line_of_sight(world, target_pos)?;
207        }
208        ToolVerb::CoverFire => {
209            validate_ammo(world)?;
210            validate_line_of_sight(world, target_pos)?;
211        }
212        ToolVerb::Revive => {
213            // Resources: stamina or something
214            if world.me.morale < 0.5 {
215                anyhow::bail!(
216                    "action blocked: low morale for Revive (morale: {:.2}, required: 0.5)",
217                    world.me.morale
218                );
219            }
220            // Target valid: ally nearby
221            if let Some(pos) = target_pos {
222                let dx = pos.x - world.me.pos.x;
223                let dy = pos.y - world.me.pos.y;
224                let dist = ((dx * dx + dy * dy) as f32).sqrt();
225                if dist > 2.0 {
226                    anyhow::bail!(
227                        "action blocked: target too far for Revive (distance: {:.2}, max: 2.0)",
228                        dist
229                    );
230                }
231            }
232        }
233        _ => {} // Other actions OK for now
234    }
235    Ok(())
236}

238/// Simple LOS check (placeholder: check obstacles)
239fn has_line_of_sight(from: IVec2, to: IVec2, world: &WorldSnapshot) -> bool {
240    // Bresenham line, check if any obstacle intersects
241    let dx = (to.x - from.x).abs();
242    let dy = (to.y - from.y).abs();
243    let sx = if from.x < to.x { 1 } else { -1 };
244    let sy = if from.y < to.y { 1 } else { -1 };
245    let mut err = dx - dy;
246
247    let mut x = from.x;
248    let mut y = from.y;
249
250    while x != to.x || y != to.y {
251        if world.obstacles.iter().any(|obs| obs.x == x && obs.y == y) {
252            return false;
253        }
254        let e2 = 2 * err;
255        if e2 > -dy {
256            err -= dy;
257            x += sx;
258        }
259        if e2 < dx {
260            err += dx;
261            y += sy;
262        }
263    }
264    true
265}

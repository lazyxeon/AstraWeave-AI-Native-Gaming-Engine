19/// Planner mode selection for AI entities.
20#[derive(Clone, Debug, PartialEq, Eq)]
21pub enum PlannerMode {
22    /// Use rule-based orchestrator (smoke + advance logic).
23    Rule,
24    /// Use behavior tree planner (requires `ai-bt` feature).
25    BehaviorTree,
26    /// Use GOAP planner (requires `ai-goap` feature).
27    GOAP,
28}
29
30/// Component for AI entities to control which planner to use.
31///
32/// # Example
33/// ```
34/// use astraweave_ai::core_loop::{CAiController, PlannerMode};
35///
36/// let controller = CAiController {
37///     mode: PlannerMode::GOAP,
38///     policy: Some("gather_craft_policy".to_string()),
39/// };
40/// ```
41#[derive(Clone, Debug)]
42pub struct CAiController {
43    /// Which planner to use for this entity.
44    pub mode: PlannerMode,
45    /// Optional policy identifier for BT/GOAP configuration.
46    /// Can be used to load specific behavior trees or GOAP action sets.
47    pub policy: Option<String>,
48}
49
50impl Default for CAiController {
51    fn default() -> Self {
52        Self {
53            mode: PlannerMode::Rule,
54            policy: None,
55        }
56    }
57}

59/// Dispatch planning to the appropriate planner based on controller mode.
60///
61/// # Arguments
62/// - `controller`: The AI controller specifying which planner to use
63/// - `snapshot`: World state snapshot for the entity
64///
65/// # Returns
66/// - `Ok(PlanIntent)` with the planned action sequence
67/// - `Err` if planning fails
68///
69/// # Example
70/// ```
71/// use astraweave_ai::core_loop::{CAiController, PlannerMode, dispatch_planner};
72/// use astraweave_core::{WorldSnapshot, PlayerState, CompanionState, IVec2};
73/// use std::collections::BTreeMap;
74///
75/// let controller = CAiController {
76///     mode: PlannerMode::Rule,
77///     policy: None,
78/// };
79///
80/// let snapshot = WorldSnapshot {
81///     t: 0.0,
82///     me: CompanionState {
83///         ammo: 10,
84///         cooldowns: BTreeMap::new(),
85///         morale: 1.0,
86///         pos: IVec2 { x: 5, y: 5 },
87///     },
88///     player: PlayerState {
89///         hp: 100,
90///         pos: IVec2 { x: 5, y: 5 },
91///         stance: "stand".into(),
92///         orders: vec![],
93///     },
94///     enemies: vec![],
95///     pois: vec![],
96///     obstacles: vec![],
97///     objective: None,
98/// };
99///
100/// let plan = dispatch_planner(&controller, &snapshot).unwrap();
101/// assert!(!plan.steps.is_empty());
102/// ```
103pub fn dispatch_planner(
104    controller: &CAiController,
105    snapshot: &WorldSnapshot,
106) -> Result<PlanIntent> {
107    match controller.mode {
108        PlannerMode::Rule => {
109            let orch = RuleOrchestrator;
110            Ok(orch.propose_plan(snapshot))
111        }
112        PlannerMode::BehaviorTree => {
113            #[cfg(feature = "ai-bt")]
114            {
115                dispatch_bt(controller, snapshot)
116            }
117            #[cfg(not(feature = "ai-bt"))]
118            {
119                anyhow::bail!("BehaviorTree mode requires 'ai-bt' feature")
120            }
121        }
122        PlannerMode::GOAP => {
123            #[cfg(feature = "ai-goap")]
124            {
125                dispatch_goap(controller, snapshot)
126            }
127            #[cfg(not(feature = "ai-goap"))]
128            {
129                anyhow::bail!("GOAP mode requires 'ai-goap' feature")
130            }
131        }
132    }
133}

145/// Dispatch to GOAP planner (feature-gated).
146#[cfg(feature = "ai-goap")]
147fn dispatch_goap(controller: &CAiController, snapshot: &WorldSnapshot) -> Result<PlanIntent> {
148    // Convert WorldSnapshot to GOAP WorldState
149    let world_state = snapshot_to_goap_state(snapshot)?;
150
151    // Define goal based on policy or default
152    let goal = if let Some(policy_name) = &controller.policy {
153        // TODO: Load goal from policy configuration
154        // For now, use a simple goal based on policy name
155        if policy_name.contains("gather") || policy_name.contains("craft") {
156            WorldState {
157                has_wood: 5,
158                has_food: 1,
159                ..WorldState::default()
160            }
161        } else {
162            WorldState {
163                has_food: 1,
164                ..WorldState::default()
165            }
166        }
167    } else {
168        // Default goal: get some food
169        WorldState {
170            has_food: 1,
171            ..WorldState::default()
172        }
173    };
174
175    // Define available actions
176    let actions = create_goap_actions();
177
178    // Run GOAP planner
179    let planner = GoapPlanner::new(actions, 100);
180    let plan = planner
181        .plan(&world_state, &goal)
182        .ok_or_else(|| anyhow::anyhow!("GOAP planning failed - no plan found"))?;
183
184    // Convert GOAP action sequence to ActionSteps
185    let steps = goap_actions_to_steps(&plan)?;
186
187    Ok(PlanIntent {
188        plan_id: format!("goap_{}", plan.len()),
189        steps,
190    })
191}

193/// Convert WorldSnapshot to GOAP WorldState.
194#[cfg(feature = "ai-goap")]
195fn snapshot_to_goap_state(snapshot: &WorldSnapshot) -> Result<WorldState> {
196    // Build initial world state from snapshot
197    // For now, use simple mapping based on companion state
198    let state = WorldState {
199        has_wood: 0,                     // Would come from inventory in full system
200        has_food: 0,                     // Would come from inventory in full system
201        at_tree: false,                  // Would come from nearby resource check
202        at_campfire: false,              // Would come from nearby structure check
203        hungry: snapshot.player.hp < 50, // Use player HP as hunger proxy
204    };
205
206    Ok(state)
207}

265/// Convert GOAP action sequence to ActionSteps.
266#[cfg(feature = "ai-goap")]
267fn goap_actions_to_steps(actions: &[Action]) -> Result<Vec<ActionStep>> {
268    let mut steps = Vec::new();
269
270    for action in actions {
271        let step = match action.name.as_str() {
272            "GoToTree" => ActionStep::MoveTo { x: 10, y: 10 }, // TODO: Use actual tree positions
273            "ChopWood" => ActionStep::MoveTo { x: 10, y: 10 }, // Stay at tree for now
274            "GoToCampfire" => ActionStep::MoveTo { x: 5, y: 5 }, // TODO: Use actual campfire position
275            "CookFood" => ActionStep::MoveTo { x: 5, y: 5 },     // Stay at campfire for now
276            _ => {
277                anyhow::bail!("Unknown GOAP action: {}", action.name)
278            }
279        };
280        steps.push(step);
281    }
282
283    Ok(steps)
284}

31/// Sweep a capsule from `from` to `to`, apply damage to first hit collider body that isn't `self_id`.
32///
33/// Uses Rapier3D 0.22 API with raycast as a simplified attack sweep.
34/// Returns HitResult if a valid target is hit, considering parry windows and iframes.
35#[allow(clippy::too_many_arguments)]
36pub fn perform_attack_sweep(
37    phys: &mut PhysicsWorld,
38    self_id: u64,
39    from: Vec3,
40    to: Vec3,
41    _radius: f32,
42    base_damage: i32,
43    dtype: DamageType,
44    targets: &mut [Combatant],
45) -> Option<HitResult> {
46    // Calculate sweep direction and distance
47    let dir = to - from;
48    let distance = dir.length();
49
50    // Early exit if sweep distance is negligible
51    if distance <= 1e-3 {
52        return None;
53    }
54
55    let dir_normalized = dir / distance;
56
57    // Adjust raycast to character center height (assuming 2m tall characters)
58    let ray_from = from + Vec3::new(0.0, 1.0, 0.0);
59
60    // Use a raycast with radius check for simplified attack detection
61    let ray = Ray::new(
62        point![ray_from.x, ray_from.y, ray_from.z],
63        vector![dir_normalized.x, dir_normalized.y, dir_normalized.z],
64    );
65
66    // Create filter to exclude self from raycast
67    let filter = if let Some(self_handle) = phys.handle_of(self_id) {
68        QueryFilter::default().exclude_rigid_body(self_handle)
69    } else {
70        QueryFilter::default()
71    };
72
73    // Cast ray and check hits within radius
74    if let Some((collider_handle, hit)) = phys.query_pipeline.cast_ray_and_get_normal(
75        &phys.bodies,
76        &phys.colliders,
77        &ray,
78        distance,
79        true, // stop at first hit
80        filter,
81    ) {
82        // Get the rigid body that owns this collider
83        if let Some(collider) = phys.colliders.get(collider_handle) {
84            if let Some(body_handle) = collider.parent() {
85                // Get the body ID from PhysicsWorld's internal mapping
86                if let Some(target_id) = phys.id_of(body_handle) {
87                    // Don't hit ourselves
88                    if target_id == self_id {
89                        return None;
90                    }
91
92                    // Check if hit is within attack radius (for sweep effect)
93                    let hit_point_dist = hit.time_of_impact;
94                    if hit_point_dist > distance {
95                        return None;
96                    }
97
98                    // Filter by attack cone (check if target is within forward cone)
99                    // This prevents hitting targets behind the attacker
100                    let hit_point = ray_from + dir_normalized * hit_point_dist;
101                    let to_target = (hit_point - ray_from).normalize_or_zero();
102                    if to_target.length_squared() < 0.01 {
103                        return None;
104                    }
105                    let dot = dir_normalized.dot(to_target);
106
107                    // Only hit targets in front (dot > 0.5 = ~60 degree cone)
108                    if dot < 0.5 {
109                        return None;
110                    }
111
112                    // Find the matching combatant target
113                    if let Some(target) = targets.iter_mut().find(|t| t.body == target_id) {
114                        // Check parry window
115                        if let Some(parry) = &mut target.parry {
116                            if parry.active && parry.window > 0.0 {
117                                parry.window = 0.0;
118                                parry.active = false;
119                                return Some(HitResult {
120                                    target: target_id,
121                                    damage: 0,
122                                    parried: true,
123                                });
124                            }
125                        }
126
127                        // Check invincibility frames
128                        if let Some(iframe) = &target.iframes {
129                            if iframe.time_left > 0.0 {
130                                return Some(HitResult {
131                                    target: target_id,
132                                    damage: 0,
133                                    parried: false,
134                                });
135                            }
136                        }
137
138                        // Apply damage
139                        let damage = target.stats.apply_damage(base_damage, dtype);
140                        return Some(HitResult {
141                            target: target_id,
142                            damage,
143                            parried: false,
144                        });
145                    }
146                }
147            }
148        }
149    }
150
151    None
152}

661        let target_id = phys.add_character(Vec3::new(2.0, 0.0, 0.0), Vec3::new(0.5, 1.0, 0.5));
662        phys.step();
663
664        let mut targets = vec![create_test_combatant(target_id, 100)];
665
666        let result = perform_attack_sweep(
667            &mut phys,
668            attacker_id,
669            Vec3::ZERO,
670            Vec3::new(3.0, 0.0, 0.0),
671            0.5,
672            20,
673            DamageType::Physical,
674            &mut targets,
675        );
676
677        // With correct +y offset, ray at character center should hit.
678        // With -y offset (underground), ray would miss.
679        assert!(
680            result.is_some(),
681            "Ray with correct height offset should hit target"
682        );
683        assert_eq!(
684            targets[0].stats.hp, 80,
685            "Target HP should reflect damage taken"
686        );
